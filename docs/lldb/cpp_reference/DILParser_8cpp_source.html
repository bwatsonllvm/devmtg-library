58CompilerType ResolveTypeByName(const std::string &name,
59 ExecutionContextScope &ctx_scope) {
60 // Internally types don't have global scope qualifier in their names and
61 // LLDB doesn't support queries with it too.
62 llvm::StringRef name_ref(name);
63
64 if (name_ref.starts_with("::"))
65 name_ref = name_ref.drop_front(2);
66
67 std::vector<CompilerType> result_type_list;
68 lldb::TargetSP target_sp = ctx_scope.CalculateTarget();
69 if (!name_ref.empty() && target_sp) {
70 ModuleList &images = target_sp->GetImages();
71 TypeQuery query{ConstString(name_ref), TypeQueryOptions::e_exact_match |
72 TypeQueryOptions::e_find_one};
73 TypeResults results;
74 images.FindTypes(nullptr, query, results);
75 const lldb::TypeSP &type_sp = results.GetFirstType();
76 if (type_sp)
77 result_type_list.push_back(type_sp->GetFullCompilerType());
78 }
79
80 if (!result_type_list.empty()) {
81 CompilerType type = result_type_list[0];
82 if (type.IsValid() && type.GetTypeName().GetStringRef() == name_ref)
83 return type;
84 }
85
86 return {};
87}

89llvm::Expected<ASTNodeUP>
90DILParser::Parse(llvm::StringRef dil_input_expr, DILLexer lexer,
91 std::shared_ptr<StackFrame> frame_sp,
92 lldb::DynamicValueType use_dynamic, bool use_synthetic,
93 bool fragile_ivar, bool check_ptr_vs_member) {
94 llvm::Error error = llvm::Error::success();
95 DILParser parser(dil_input_expr, lexer, frame_sp, use_dynamic, use_synthetic,
96 fragile_ivar, check_ptr_vs_member, error);
97
98 ASTNodeUP node_up = parser.Run();
99 assert(node_up && "ASTNodeUP must not contain a nullptr");
100
101 if (error)
102 return error;
103
104 return node_up;
105}

132// Parse a cast_expression.
133//
134// cast_expression:
135// unary_expression
136// "(" type_id ")" cast_expression
137
138ASTNodeUP DILParser::ParseCastExpression() {
139 if (!CurToken().Is(Token::l_paren))
140 return ParseUnaryExpression();
141
142 // This could be a type cast, try parsing the contents as a type declaration.
143 Token token = CurToken();
144 uint32_t loc = token.GetLocation();
145
146 // Enable lexer backtracking, so that we can rollback in case it's not
147 // actually a type declaration.
148
149 // Start tentative parsing (save token location/idx, for possible rollback).
150 uint32_t save_token_idx = m_dil_lexer.GetCurrentTokenIdx();
151
152 // Consume the token only after enabling the backtracking.
153 m_dil_lexer.Advance();
154
155 // Try parsing the type declaration. If the returned value is not valid,
156 // then we should rollback and try parsing the expression.
157 auto type_id = ParseTypeId();
158 if (type_id) {
159 // Successfully parsed the type declaration. Commit the backtracked
160 // tokens and parse the cast_expression.
161
162 if (!type_id.value().IsValid())
163 return std::make_unique<ErrorNode>();
164
165 Expect(Token::r_paren);
166 m_dil_lexer.Advance();
167 auto rhs = ParseCastExpression();
168 assert(rhs && "ASTNodeUP must not contain a nullptr");
169 return std::make_unique<CastNode>(loc, type_id.value(), std::move(rhs),
170 CastKind::eNone);
171 }
172
173 // Failed to parse the contents of the parentheses as a type declaration.
174 // Rollback the lexer and try parsing it as unary_expression.
175 TentativeParsingRollback(save_token_idx);
176
177 return ParseUnaryExpression();
178}

180// Parse an unary_expression.
181//
182// unary_expression:
183// postfix_expression
184// unary_operator cast_expression
185//
186// unary_operator:
187// "&"
188// "*"
189// "+"
190// "-"
191//
192ASTNodeUP DILParser::ParseUnaryExpression() {
193 if (CurToken().IsOneOf(
194 {Token::amp, Token::star, Token::minus, Token::plus})) {
195 Token token = CurToken();
196 uint32_t loc = token.GetLocation();
197 m_dil_lexer.Advance();
198 auto rhs = ParseCastExpression();
199 assert(rhs && "ASTNodeUP must not contain a nullptr");
200 switch (token.GetKind()) {
201 case Token::star:
202 return std::make_unique<UnaryOpNode>(loc, UnaryOpKind::Deref,
203 std::move(rhs));
204 case Token::amp:
205 return std::make_unique<UnaryOpNode>(loc, UnaryOpKind::AddrOf,
206 std::move(rhs));
207 case Token::minus:
208 return std::make_unique<UnaryOpNode>(loc, UnaryOpKind::Minus,
209 std::move(rhs));
210 case Token::plus:
211 return std::make_unique<UnaryOpNode>(loc, UnaryOpKind::Plus,
212 std::move(rhs));
213 default:
214 llvm_unreachable("invalid token kind");
215 }
216 }
217 return ParsePostfixExpression();
218}
219
220// Parse a postfix_expression.
221//
222// postfix_expression:
223// primary_expression
224// postfix_expression "[" expression "]"
225// postfix_expression "[" expression ":" expression "]"
226// postfix_expression "." id_expression
227// postfix_expression "->" id_expression
228//
229ASTNodeUP DILParser::ParsePostfixExpression() {
230 ASTNodeUP lhs = ParsePrimaryExpression();
231 assert(lhs && "ASTNodeUP must not contain a nullptr");
232 while (CurToken().IsOneOf({Token::l_square, Token::period, Token::arrow})) {
233 uint32_t loc = CurToken().GetLocation();
234 Token token = CurToken();
235 switch (token.GetKind()) {
236 case Token::l_square: {
237 m_dil_lexer.Advance();
238 ASTNodeUP index = ParseExpression();
239 assert(index && "ASTNodeUP must not contain a nullptr");
240 if (CurToken().GetKind() == Token::colon) {
241 m_dil_lexer.Advance();
242 ASTNodeUP last_index = ParseExpression();
243 assert(last_index && "ASTNodeUP must not contain a nullptr");
244 lhs = std::make_unique<BitFieldExtractionNode>(
245 loc, std::move(lhs), std::move(index), std::move(last_index));
246 } else if (CurToken().GetKind() == Token::minus) {
247 BailOut("use of '-' for bitfield range is deprecated; use ':' instead",
248 CurToken().GetLocation(), CurToken().GetSpelling().length());
249 return std::make_unique<ErrorNode>();
250 } else {
251 lhs = std::make_unique<ArraySubscriptNode>(loc, std::move(lhs),
252 std::move(index));
253 }
254 Expect(Token::r_square);
255 m_dil_lexer.Advance();
256 break;
257 }
258 case Token::period:
259 case Token::arrow: {
260 m_dil_lexer.Advance();
261 Token member_token = CurToken();
262 std::string member_id = ParseIdExpression();
263 lhs = std::make_unique<MemberOfNode>(
264 member_token.GetLocation(), std::move(lhs),
265 token.GetKind() == Token::arrow, member_id);
266 break;
267 }
268 default:
269 llvm_unreachable("invalid token");
270 }
271 }
272
273 return lhs;
274}

276// Parse a primary_expression.
277//
278// primary_expression:
279// numeric_literal
280// boolean_literal
281// id_expression
282// "(" expression ")"
283//
284ASTNodeUP DILParser::ParsePrimaryExpression() {
285 if (CurToken().IsOneOf({Token::integer_constant, Token::float_constant}))
286 return ParseNumericLiteral();
287 if (CurToken().IsOneOf({Token::kw_true, Token::kw_false}))
288 return ParseBooleanLiteral();
289 if (CurToken().IsOneOf(
290 {Token::coloncolon, Token::identifier, Token::l_paren})) {
291 // Save the source location for the diagnostics message.
292 uint32_t loc = CurToken().GetLocation();
293 std::string identifier = ParseIdExpression();
294
295 if (!identifier.empty())
296 return std::make_unique<IdentifierNode>(loc, identifier);
297 }
298
299 if (CurToken().Is(Token::l_paren)) {
300 m_dil_lexer.Advance();
301 auto expr = ParseExpression();
302 Expect(Token::r_paren);
303 m_dil_lexer.Advance();
304 return expr;
305 }
306
307 BailOut(llvm::formatv("Unexpected token: {0}", CurToken()),
308 CurToken().GetLocation(), CurToken().GetSpelling().length());
309 return std::make_unique<ErrorNode>();
310}

312// Parse nested_name_specifier.
313//
314// nested_name_specifier:
315// type_name "::"
316// namespace_name "::"
317// nested_name_specifier identifier "::"
318//
319std::string DILParser::ParseNestedNameSpecifier() {
320 // The first token in nested_name_specifier is always an identifier, or
321 // '(anonymous namespace)'.
322 switch (CurToken().GetKind()) {
323 case Token::l_paren: {
324 // Anonymous namespaces need to be treated specially: They are
325 // represented the the string '(anonymous namespace)', which has a
326 // space in it (throwing off normal parsing) and is not actually
327 // proper C++> Check to see if we're looking at
328 // '(anonymous namespace)::...'
329
330 // Look for all the pieces, in order:
331 // l_paren 'anonymous' 'namespace' r_paren coloncolon
332 if (m_dil_lexer.LookAhead(1).Is(Token::identifier) &&
333 (m_dil_lexer.LookAhead(1).GetSpelling() == "anonymous") &&
334 m_dil_lexer.LookAhead(2).Is(Token::identifier) &&
335 (m_dil_lexer.LookAhead(2).GetSpelling() == "namespace") &&
336 m_dil_lexer.LookAhead(3).Is(Token::r_paren) &&
337 m_dil_lexer.LookAhead(4).Is(Token::coloncolon)) {
338 m_dil_lexer.Advance(4);
339
340 Expect(Token::coloncolon);
341 m_dil_lexer.Advance();
342 if (!CurToken().Is(Token::identifier) && !CurToken().Is(Token::l_paren)) {
343 BailOut("Expected an identifier or anonymous namespace, but not found.",
344 CurToken().GetLocation(), CurToken().GetSpelling().length());
345 }
346 // Continue parsing the nested_namespace_specifier.
347 std::string identifier2 = ParseNestedNameSpecifier();
348
349 return "(anonymous namespace)::" + identifier2;
350 }
351
352 return "";
353 } // end of special handling for '(anonymous namespace)'
354 case Token::identifier: {
355 // If the next token is scope ("::"), then this is indeed a
356 // nested_name_specifier
357 if (m_dil_lexer.LookAhead(1).Is(Token::coloncolon)) {
358 // This nested_name_specifier is a single identifier.
359 std::string identifier = CurToken().GetSpelling();
360 m_dil_lexer.Advance(1);
361 Expect(Token::coloncolon);
362 m_dil_lexer.Advance();
363 // Continue parsing the nested_name_specifier.
364 return identifier + "::" + ParseNestedNameSpecifier();
365 }
366
367 return "";
368 }
369 default:
370 return "";
371 }
372}

374// Parse a type_id.
375//
376// type_id:
377// type_specifier_seq [abstract_declarator]
378//
379// type_specifier_seq:
380// type_specifier [type_specifier]
381//
382// type_specifier:
383// ["::"] [nested_name_specifier] type_name // not handled for now!
384// builtin_typename
385//
386std::optional<CompilerType> DILParser::ParseTypeId() {
387 CompilerType type;
388 auto maybe_builtin_type = ParseBuiltinType();
389 if (maybe_builtin_type) {
390 type = *maybe_builtin_type;
391 } else {
392 // Check to see if we have a user-defined type here.
393 // First build up the user-defined type name.
394 std::string type_name;
395 ParseTypeSpecifierSeq(type_name);
396
397 if (type_name.empty())
398 return {};
399 type = ResolveTypeByName(type_name, *m_ctx_scope);
400 if (!type.IsValid())
401 return {};
402
403 // Same-name identifiers should be preferred over typenames.
404 if (LookupIdentifier(type_name, m_ctx_scope, m_use_dynamic))
405 // TODO: Make type accessible with 'class', 'struct' and 'union' keywords.
406 return {};
407
408 // Same-name identifiers should be preferred over typenames.
409 if (LookupGlobalIdentifier(type_name, m_ctx_scope,
410 m_ctx_scope->CalculateTarget(), m_use_dynamic))
411 // TODO: Make type accessible with 'class', 'struct' and 'union' keywords
412 return {};
413 }
414
415 //
416 // abstract_declarator:
417 // ptr_operator [abstract_declarator]
418 //
419 std::vector<Token> ptr_operators;
420 while (CurToken().IsOneOf({Token::star, Token::amp})) {
421 Token tok = CurToken();
422 ptr_operators.push_back(std::move(tok));
423 m_dil_lexer.Advance();
424 }
425 type = ResolveTypeDeclarators(type, ptr_operators);
426
427 return type;
428}

430// Parse a built-in type
431//
432// builtin_typename:
433// identifer_seq
434//
435// identifier_seq
436// identifer [identifier_seq]
437//
438// A built-in type can be a single identifier or a space-separated
439// list of identifiers (e.g. "short" or "long long").
440std::optional<CompilerType> DILParser::ParseBuiltinType() {
441 std::string type_name = "";
442 uint32_t save_token_idx = m_dil_lexer.GetCurrentTokenIdx();
443 bool first_word = true;
444 while (CurToken().GetKind() == Token::identifier) {
445 if (CurToken().GetSpelling() == "const" ||
446 CurToken().GetSpelling() == "volatile")
447 continue;
448 if (!first_word)
449 type_name.push_back(' ');
450 else
451 first_word = false;
452 type_name.append(CurToken().GetSpelling());
453 m_dil_lexer.Advance();
454 }
455
456 if (type_name.size() > 0) {
457 lldb::TargetSP target_sp = m_ctx_scope->CalculateTarget();
458 ConstString const_type_name(type_name.c_str());
459 for (auto type_system_sp : target_sp->GetScratchTypeSystems())
460 if (auto compiler_type =
461 type_system_sp->GetBuiltinTypeByName(const_type_name))
462 return compiler_type;
463 }
464
465 TentativeParsingRollback(save_token_idx);
466 return {};
467}

469// Parse a type_specifier_seq.
470//
471// type_specifier_seq:
472// type_specifier [type_specifier_seq]
473//
474void DILParser::ParseTypeSpecifierSeq(std::string &type_name) {
475 while (true) {
476 std::optional<std::string> err_or_string = ParseTypeSpecifier();
477 if (!err_or_string)
478 break;
479 type_name = *err_or_string;
480 }
481}

483// Parse a type_specifier.
484//
485// type_specifier:
486// ["::"] [nested_name_specifier] type_name
487//
488// Returns TRUE if a type_specifier was successfully parsed at this location.
489//
490std::optional<std::string> DILParser::ParseTypeSpecifier() {
491 // The type_specifier must be a user-defined type. Try parsing a
492 // simple_type_specifier.
493
494 // Try parsing optional global scope operator.
495 bool global_scope = false;
496 if (CurToken().Is(Token::coloncolon)) {
497 global_scope = true;
498 m_dil_lexer.Advance();
499 }
500
501 // Try parsing optional nested_name_specifier.
502 auto nested_name_specifier = ParseNestedNameSpecifier();
503
504 // Try parsing required type_name.
505 auto type_name_or_err = ParseTypeName();
506 if (!type_name_or_err)
507 return type_name_or_err;
508 std::string type_name = *type_name_or_err;
509
510 // If there is a type_name, then this is indeed a simple_type_specifier.
511 // Global and qualified (namespace/class) scopes can be empty, since they're
512 // optional. In this case type_name is type we're looking for.
513 if (!type_name.empty())
514 // User-defined typenames can't be combined with builtin keywords.
515 return llvm::formatv("{0}{1}{2}", global_scope ? "::" : "",
516 nested_name_specifier, type_name);
517
518 // No type_specifier was found here.
519 return {};
520}

522// Parse a type_name.
523//
524// type_name:
525// class_name
526// enum_name
527// typedef_name
528//
529// class_name
530// identifier
531//
532// enum_name
533// identifier
534//
535// typedef_name
536// identifier
537//
538std::optional<std::string> DILParser::ParseTypeName() {
539 // Typename always starts with an identifier.
540 if (CurToken().IsNot(Token::identifier)) {
541 return std::nullopt;
542 }
543
544 // Otherwise look for a class_name, enum_name or a typedef_name.
545 std::string identifier = CurToken().GetSpelling();
546 m_dil_lexer.Advance();
547
548 return identifier;
549}

551// Parse an id_expression.
552//
553// id_expression:
554// unqualified_id
555// qualified_id
556//
557// qualified_id:
558// ["::"] [nested_name_specifier] unqualified_id
559// ["::"] identifier
560//
561// identifier:
562// ? Token::identifier ?
563//
564std::string DILParser::ParseIdExpression() {
565 // Try parsing optional global scope operator.
566 bool global_scope = false;
567 if (CurToken().Is(Token::coloncolon)) {
568 global_scope = true;
569 m_dil_lexer.Advance();
570 }
571
572 // Try parsing optional nested_name_specifier.
573 std::string nested_name_specifier = ParseNestedNameSpecifier();
574
575 // If nested_name_specifier is present, then it's qualified_id production.
576 // Follow the first production rule.
577 if (!nested_name_specifier.empty()) {
578 // Parse unqualified_id and construct a fully qualified id expression.
579 auto unqualified_id = ParseUnqualifiedId();
580
581 return llvm::formatv("{0}{1}{2}", global_scope ? "::" : "",
582 nested_name_specifier, unqualified_id);
583 }
584
585 if (!CurToken().Is(Token::identifier))
586 return "";
587
588 // No nested_name_specifier, but with global scope -- this is also a
589 // qualified_id production. Follow the second production rule.
590 if (global_scope) {
591 Expect(Token::identifier);
592 std::string identifier = CurToken().GetSpelling();
593 m_dil_lexer.Advance();
594 return llvm::formatv("{0}{1}", global_scope ? "::" : "", identifier);
595 }
596
597 // This is unqualified_id production.
598 return ParseUnqualifiedId();
599}

601// Parse an unqualified_id.
602//
603// unqualified_id:
604// identifier
605//
606// identifier:
607// ? Token::identifier ?
608//
609std::string DILParser::ParseUnqualifiedId() {
610 Expect(Token::identifier);
611 std::string identifier = CurToken().GetSpelling();
612 m_dil_lexer.Advance();
613 return identifier;
614}

616CompilerType
617DILParser::ResolveTypeDeclarators(CompilerType type,
618 const std::vector<Token> &ptr_operators) {
619 // Resolve pointers/references.
620 for (Token tk : ptr_operators) {
621 uint32_t loc = tk.GetLocation();
622 if (tk.GetKind() == Token::star) {
623 // Pointers to reference types are forbidden.
624 if (type.IsReferenceType()) {
625 BailOut(llvm::formatv("'type name' declared as a pointer to a "
626 "reference of type {0}",
627 type.TypeDescription()),
628 loc, CurToken().GetSpelling().length());
629 return {};
630 }
631 // Get pointer type for the base type: e.g. int* -> int**.
632 type = type.GetPointerType();
633
634 } else if (tk.GetKind() == Token::amp) {
635 // References to references are forbidden.
636 // FIXME: In future we may want to allow rvalue references (i.e. &&).
637 if (type.IsReferenceType()) {
638 BailOut("type name declared as a reference to a reference", loc,
639 CurToken().GetSpelling().length());
640 return {};
641 }
642 // Get reference type for the base type: e.g. int -> int&.
643 type = type.GetLValueReferenceType();
644 }
645 }
646
647 return type;
648}

664void DILParser::BailOut(const std::string &error, uint32_t loc,
665 uint16_t err_len) {
666 if (m_error)
667 // If error is already set, then the parser is in the "bail-out" mode. Don't
668 // do anything and keep the original error.
669 return;
670
671 m_error =
672 llvm::make_error<DILDiagnosticError>(m_input_expr, error, loc, err_len);
673 // Advance the lexer token index to the end of the lexed tokens vector.
674 m_dil_lexer.ResetTokenIdx(m_dil_lexer.NumLexedTokens() - 1);
675}

677// Parse a numeric_literal.
678//
679// numeric_literal:
680// ? Token::integer_constant ?
681// ? Token::floating_constant ?
682//
683ASTNodeUP DILParser::ParseNumericLiteral() {
684 ASTNodeUP numeric_constant;
685 if (CurToken().Is(Token::integer_constant))
686 numeric_constant = ParseIntegerLiteral();
687 else
688 numeric_constant = ParseFloatingPointLiteral();
689 if (numeric_constant->GetKind() == NodeKind::eErrorNode) {
690 BailOut(llvm::formatv("Failed to parse token as numeric-constant: {0}",
691 CurToken()),
692 CurToken().GetLocation(), CurToken().GetSpelling().length());
693 return numeric_constant;
694 }
695 m_dil_lexer.Advance();
696 return numeric_constant;
697}
698
699ASTNodeUP DILParser::ParseIntegerLiteral() {
700 Token token = CurToken();
701 auto spelling = token.GetSpelling();
702 llvm::StringRef spelling_ref = spelling;
703
704 auto radix = llvm::getAutoSenseRadix(spelling_ref);
705 IntegerTypeSuffix type = IntegerTypeSuffix::None;
706 bool is_unsigned = false;
707 if (spelling_ref.consume_back_insensitive("u"))
708 is_unsigned = true;
709 if (spelling_ref.consume_back_insensitive("ll"))
710 type = IntegerTypeSuffix::LongLong;
711 else if (spelling_ref.consume_back_insensitive("l"))
712 type = IntegerTypeSuffix::Long;
713 // Suffix 'u' can be only specified only once, before or after 'l'
714 if (!is_unsigned && spelling_ref.consume_back_insensitive("u"))
715 is_unsigned = true;
716
717 llvm::APInt raw_value;
718 if (!spelling_ref.getAsInteger(radix, raw_value))
719 return std::make_unique<IntegerLiteralNode>(token.GetLocation(), raw_value,
720 radix, is_unsigned, type);
721 return std::make_unique<ErrorNode>();
722}
723
724ASTNodeUP DILParser::ParseFloatingPointLiteral() {
725 Token token = CurToken();
726 auto spelling = token.GetSpelling();
727 llvm::StringRef spelling_ref = spelling;
728
729 llvm::APFloat raw_float(llvm::APFloat::IEEEdouble());
730 if (spelling_ref.consume_back_insensitive("f"))
731 raw_float = llvm::APFloat(llvm::APFloat::IEEEsingle());
732
733 auto StatusOrErr = raw_float.convertFromString(
734 spelling_ref, llvm::APFloat::rmNearestTiesToEven);
735 if (!errorToBool(StatusOrErr.takeError()))
736 return std::make_unique<FloatLiteralNode>(token.GetLocation(), raw_float);
737 return std::make_unique<ErrorNode>();
738}
