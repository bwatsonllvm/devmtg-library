63static Status ExceptionMaskValidator(const char *string, void *unused) {
64 Status error;
65 llvm::StringRef str_ref(string);
66 llvm::SmallVector<llvm::StringRef> candidates;
67 str_ref.split(candidates, '|');
68 for (auto candidate : candidates) {
69 if (!(candidate == "EXC_BAD_ACCESS"
70 || candidate == "EXC_BAD_INSTRUCTION"
71 || candidate == "EXC_ARITHMETIC"
72 || candidate == "EXC_RESOURCE"
73 || candidate == "EXC_GUARD"
74 || candidate == "EXC_SYSCALL")) {
75 error = Status::FromErrorStringWithFormat("invalid exception type: '%s'",
76 candidate.str().c_str());
77 return error;
78 }
79 }
80 return {};
81}

203static FileSpecList LocateExecutableScriptingResourcesFromDSYM(
204 Stream &feedback_stream, FileSpec module_spec, const Target &target,
205 const FileSpec &symfile_spec) {
206 FileSpecList file_list;
207 while (module_spec.GetFilename()) {
208 std::string module_basename(module_spec.GetFilename().GetCString());
209 std::string original_module_basename(module_basename);
210
211 bool was_keyword = false;
212
213 // FIXME: for Python, don't allow certain characters in imported module
214 // filenames. Theoretically, different scripting languages may have
215 // different sets of forbidden tokens in filenames, and that should
216 // be dealt with by each ScriptInterpreter. For now, just replace dots
217 // with underscores. In order to support anything other than Python
218 // this will need to be reworked.
219 llvm::replace(module_basename, '.', '_');
220 llvm::replace(module_basename, ' ', '_');
221 llvm::replace(module_basename, '-', '_');
222 ScriptInterpreter *script_interpreter =
223 target.GetDebugger().GetScriptInterpreter();
224 if (script_interpreter &&
225 script_interpreter->IsReservedWord(module_basename.c_str())) {
226 module_basename.insert(module_basename.begin(), '_');
227 was_keyword = true;
228 }
229
230 StreamString path_string;
231 StreamString original_path_string;
232 // for OSX we are going to be in
233 // .dSYM/Contents/Resources/DWARF/<basename> let us go to
234 // .dSYM/Contents/Resources/Python/<basename>.py and see if the
235 // file exists
236 path_string.Printf("%s/../Python/%s.py",
237 symfile_spec.GetDirectory().GetCString(),
238 module_basename.c_str());
239 original_path_string.Printf("%s/../Python/%s.py",
240 symfile_spec.GetDirectory().GetCString(),
241 original_module_basename.c_str());
242 FileSpec script_fspec(path_string.GetString());
243 FileSystem::Instance().Resolve(script_fspec);
244 FileSpec orig_script_fspec(original_path_string.GetString());
245 FileSystem::Instance().Resolve(orig_script_fspec);
246
247 // if we did some replacements of reserved characters, and a
248 // file with the untampered name exists, then warn the user
249 // that the file as-is shall not be loaded
250 if (module_basename != original_module_basename &&
251 FileSystem::Instance().Exists(orig_script_fspec)) {
252 const char *reason_for_complaint = was_keyword
253 ? "conflicts with a keyword"
254 : "contains reserved characters";
255 if (FileSystem::Instance().Exists(script_fspec))
256 feedback_stream.Printf(
257 "warning: the symbol file '%s' contains a debug "
258 "script. However, its name"
259 " '%s' %s and as such cannot be loaded. LLDB will"
260 " load '%s' instead. Consider removing the file with "
261 "the malformed name to"
262 " eliminate this warning.\n",
263 symfile_spec.GetPath().c_str(), original_path_string.GetData(),
264 reason_for_complaint, path_string.GetData());
265 else
266 feedback_stream.Printf(
267 "warning: the symbol file '%s' contains a debug "
268 "script. However, its name"
269 " %s and as such cannot be loaded. If you intend"
270 " to have this script loaded, please rename '%s' to "
271 "'%s' and retry.\n",
272 symfile_spec.GetPath().c_str(), reason_for_complaint,
273 original_path_string.GetData(), path_string.GetData());
274 }
275
276 if (FileSystem::Instance().Exists(script_fspec)) {
277 file_list.Append(script_fspec);
278 break;
279 }
280
281 // If we didn't find the python file, then keep stripping the
282 // extensions and try again
283 ConstString filename_no_extension(
284 module_spec.GetFileNameStrippingExtension());
285 if (module_spec.GetFilename() == filename_no_extension)
286 break;
287
288 module_spec.SetFilename(filename_no_extension);
289 }
290
291 return file_list;
292}

294FileSpecList PlatformDarwin::LocateExecutableScriptingResources(
295 Target *target, Module &module, Stream &feedback_stream) {
296 if (!target)
297 return {};
298
299 // For now only Python scripts supported for auto-loading.
300 if (target->GetDebugger().GetScriptLanguage() != eScriptLanguagePython)
301 return {};
302
303 // NB some extensions might be meaningful and should not be stripped -
304 // "this.binary.file"
305 // should not lose ".file" but GetFileNameStrippingExtension() will do
306 // precisely that. Ideally, we should have a per-platform list of
307 // extensions (".exe", ".app", ".dSYM", ".framework") which should be
308 // stripped while leaving "this.binary.file" as-is.
309
310 const FileSpec &module_spec = module.GetFileSpec();
311
312 if (!module_spec)
313 return {};
314
315 SymbolFile *symfile = module.GetSymbolFile();
316 if (!symfile)
317 return {};
318
319 ObjectFile *objfile = symfile->GetObjectFile();
320 if (!objfile)
321 return {};
322
323 const FileSpec &symfile_spec = objfile->GetFileSpec();
324 if (symfile_spec &&
325 llvm::StringRef(symfile_spec.GetPath())
326 .contains_insensitive(".dSYM/Contents/Resources/DWARF") &&
327 FileSystem::Instance().Exists(symfile_spec))
328 return LocateExecutableScriptingResourcesFromDSYM(
329 feedback_stream, module_spec, *target, symfile_spec);
330
331 return {};
332}

345Status PlatformDarwin::GetSharedModule(
346 const ModuleSpec &module_spec, Process *process, ModuleSP &module_sp,
347 llvm::SmallVectorImpl<ModuleSP> *old_modules, bool *did_create_ptr) {
348 Status error;
349 module_sp.reset();
350
351 if (IsRemote()) {
352 // If we have a remote platform always, let it try and locate the shared
353 // module first.
354 if (m_remote_platform_sp) {
355 error = m_remote_platform_sp->GetSharedModule(
356 module_spec, process, module_sp, old_modules, did_create_ptr);
357 }
358 }
359
360 if (!module_sp) {
361 // Fall back to the local platform and find the file locally
362 error = Platform::GetSharedModule(module_spec, process, module_sp,
363 old_modules, did_create_ptr);
364
365 const FileSpec &platform_file = module_spec.GetFileSpec();
366 // Get module search paths from the target if available.
367 TargetSP target_sp = module_spec.GetTargetSP();
368 FileSpecList module_search_paths;
369 if (target_sp)
370 module_search_paths = target_sp->GetExecutableSearchPaths();
371 if (!module_sp && !module_search_paths.IsEmpty() && platform_file) {
372 // We can try to pull off part of the file path up to the bundle
373 // directory level and try any module search paths...
374 FileSpec bundle_directory;
375 if (Host::GetBundleDirectory(platform_file, bundle_directory)) {
376 if (platform_file == bundle_directory) {
377 ModuleSpec new_module_spec(module_spec);
378 new_module_spec.GetFileSpec() = bundle_directory;
379 if (Host::ResolveExecutableInBundle(new_module_spec.GetFileSpec())) {
380 Status new_error(Platform::GetSharedModule(new_module_spec, process,
381 module_sp, old_modules,
382 did_create_ptr));
383
384 if (module_sp)
385 return new_error;
386 }
387 } else {
388 char platform_path[PATH_MAX];
389 char bundle_dir[PATH_MAX];
390 platform_file.GetPath(platform_path, sizeof(platform_path));
391 const size_t bundle_directory_len =
392 bundle_directory.GetPath(bundle_dir, sizeof(bundle_dir));
393 char new_path[PATH_MAX];
394 size_t num_module_search_paths = module_search_paths.GetSize();
395 for (size_t i = 0; i < num_module_search_paths; ++i) {
396 const size_t search_path_len =
397 module_search_paths.GetFileSpecAtIndex(i).GetPath(
398 new_path, sizeof(new_path));
399 if (search_path_len < sizeof(new_path)) {
400 snprintf(new_path + search_path_len,
401 sizeof(new_path) - search_path_len, "/%s",
402 platform_path + bundle_directory_len);
403 FileSpec new_file_spec(new_path);
404 if (FileSystem::Instance().Exists(new_file_spec)) {
405 ModuleSpec new_module_spec(module_spec);
406 new_module_spec.GetFileSpec() = new_file_spec;
407 Status new_error(Platform::GetSharedModule(
408 new_module_spec, process, module_sp, old_modules,
409 did_create_ptr));
410
411 if (module_sp) {
412 module_sp->SetPlatformFileSpec(new_file_spec);
413 return new_error;
414 }
415 }
416 }
417 }
418 }
419 }
420 }
421 }
422 if (module_sp)
423 module_sp->SetPlatformFileSpec(module_spec.GetFileSpec());
424 return error;
425}
426Status PlatformDarwin::GetModuleFromSharedCaches(
427 const ModuleSpec &module_spec, Process *process, ModuleSP &module_sp,
428 llvm::SmallVectorImpl<ModuleSP> *old_modules, bool *did_create_ptr) {
429 Status err;
430
431 SymbolSharedCacheUse sc_mode =
432 ModuleList::GetGlobalModuleListProperties().GetSharedCacheBinaryLoading();
433 SharedCacheImageInfo image_info;
434 if (process && process->GetDynamicLoader()) {
435 addr_t sc_base_addr;
436 UUID sc_uuid;
437 LazyBool using_sc, private_sc;
438 FileSpec sc_path;
439 if (process->GetDynamicLoader()->GetSharedCacheInformation(
440 sc_base_addr, sc_uuid, using_sc, private_sc, sc_path)) {
441 if (module_spec.GetUUID())
442 image_info = HostInfo::GetSharedCacheImageInfo(module_spec.GetUUID(),
443 sc_uuid, sc_mode);
444 else
445 image_info = HostInfo::GetSharedCacheImageInfo(
446 module_spec.GetFileSpec().GetPathAsConstString(), sc_uuid, sc_mode);
447 }
448 }
449 // Fall back to looking for the file in lldb's own shared cache.
450 if (!image_info.GetUUID())
451 image_info = HostInfo::GetSharedCacheImageInfo(
452 module_spec.GetFileSpec().GetPathAsConstString(), sc_mode);
453
454 // If we found it and it has the correct UUID, let's proceed with
455 // creating a module from the memory contents.
456 if (image_info.GetUUID() && (!module_spec.GetUUID() ||
457 module_spec.GetUUID() == image_info.GetUUID())) {
458 ModuleSpec shared_cache_spec(module_spec.GetFileSpec(),
459 image_info.GetUUID(),
460 image_info.GetExtractor());
461 err = ModuleList::GetSharedModule(shared_cache_spec, module_sp, old_modules,
462 did_create_ptr);
463 if (module_sp) {
464 Log *log = GetLog(LLDBLog::Platform | LLDBLog::Modules);
465 LLDB_LOGF(log, "module %s was found in a shared cache",
466 module_spec.GetFileSpec().GetPath().c_str());
467 }
468 }
469 return err;
470}

472size_t
473PlatformDarwin::GetSoftwareBreakpointTrapOpcode(Target &target,
474 BreakpointSite *bp_site) {
475 const uint8_t *trap_opcode = nullptr;
476 uint32_t trap_opcode_size = 0;
477 bool bp_is_thumb = false;
478
479 llvm::Triple::ArchType machine = target.GetArchitecture().GetMachine();
480 switch (machine) {
481 case llvm::Triple::aarch64_32:
482 case llvm::Triple::aarch64: {
483 // 'brk #0' or 0xd4200000 in BE byte order
484 static const uint8_t g_arm64_breakpoint_opcode[] = {0x00, 0x00, 0x20, 0xD4};
485 trap_opcode = g_arm64_breakpoint_opcode;
486 trap_opcode_size = sizeof(g_arm64_breakpoint_opcode);
487 } break;
488
489 case llvm::Triple::thumb:
490 bp_is_thumb = true;
491 [[fallthrough]];
492 case llvm::Triple::arm: {
493 static const uint8_t g_arm_breakpoint_opcode[] = {0xFE, 0xDE, 0xFF, 0xE7};
494 static const uint8_t g_thumb_breakpooint_opcode[] = {0xFE, 0xDE};
495
496 // Auto detect arm/thumb if it wasn't explicitly specified
497 if (!bp_is_thumb) {
498 lldb::BreakpointLocationSP bp_loc_sp(bp_site->GetConstituentAtIndex(0));
499 if (bp_loc_sp)
500 bp_is_thumb = bp_loc_sp->GetAddress().GetAddressClass() ==
501 AddressClass::eCodeAlternateISA;
502 }
503 if (bp_is_thumb) {
504 trap_opcode = g_thumb_breakpooint_opcode;
505 trap_opcode_size = sizeof(g_thumb_breakpooint_opcode);
506 break;
507 }
508 trap_opcode = g_arm_breakpoint_opcode;
509 trap_opcode_size = sizeof(g_arm_breakpoint_opcode);
510 } break;
511
512 case llvm::Triple::ppc:
513 case llvm::Triple::ppc64: {
514 static const uint8_t g_ppc_breakpoint_opcode[] = {0x7F, 0xC0, 0x00, 0x08};
515 trap_opcode = g_ppc_breakpoint_opcode;
516 trap_opcode_size = sizeof(g_ppc_breakpoint_opcode);
517 } break;
518
519 default:
520 return Platform::GetSoftwareBreakpointTrapOpcode(target, bp_site);
521 }
522
523 if (trap_opcode && trap_opcode_size) {
524 if (bp_site->SetTrapOpcode(trap_opcode, trap_opcode_size))
525 return trap_opcode_size;
526 }
527 return 0;
528}

553 if (host_arch.IsExactMatch(host_arch64))
554 archs.push_back(HostInfo::GetArchitecture(HostInfo::eArchKind32));

734 const FileSpec &shell = launch_info.GetShell();

741 shell_name = shell_string.c_str();
742 else
743 shell_name++;
744

753 strcmp(shell_name, "tcsh") == 0 ||
754 strcmp(shell_name, "zsh") == 0) {
755 // csh and tcsh always seem to re-exec themselves.
756 return 2;
757 } else
758 return 1;

922llvm::Expected<StructuredData::DictionarySP>
923PlatformDarwin::FetchExtendedCrashInformation(Process &process) {
924 static constexpr llvm::StringLiteral crash_info_key("Crash-Info Annotations");
925 static constexpr llvm::StringLiteral asi_info_key(
926 "Application Specific Information");
927
928 // We cache the information we find in the process extended info dict:
929 StructuredData::DictionarySP process_dict_sp =
930 process.GetExtendedCrashInfoDict();
931 StructuredData::Array *annotations = nullptr;
932 StructuredData::ArraySP new_annotations_sp;
933 if (!process_dict_sp->GetValueForKeyAsArray(crash_info_key, annotations)) {
934 new_annotations_sp = ExtractCrashInfoAnnotations(process);
935 if (new_annotations_sp && new_annotations_sp->GetSize()) {
936 process_dict_sp->AddItem(crash_info_key, new_annotations_sp);
937 annotations = new_annotations_sp.get();
938 }
939 }
940
941 StructuredData::Dictionary *app_specific_info;
942 StructuredData::DictionarySP new_app_specific_info_sp;
943 if (!process_dict_sp->GetValueForKeyAsDictionary(asi_info_key,
944 app_specific_info)) {
945 new_app_specific_info_sp = ExtractAppSpecificInfo(process);
946 if (new_app_specific_info_sp && new_app_specific_info_sp->GetSize()) {
947 process_dict_sp->AddItem(asi_info_key, new_app_specific_info_sp);
948 app_specific_info = new_app_specific_info_sp.get();
949 }
950 }
951
952 // Now get anything else that was in the process info dict, and add it to the
953 // return here:
954 return process_dict_sp->GetSize() ? process_dict_sp : nullptr;
955}

957StructuredData::ArraySP
958PlatformDarwin::ExtractCrashInfoAnnotations(Process &process) {
959 Log *log = GetLog(LLDBLog::Process);
960
961 ConstString section_name("__crash_info");
962 Target &target = process.GetTarget();
963 StructuredData::ArraySP array_sp = std::make_shared<StructuredData::Array>();
964
965 for (ModuleSP module : target.GetImages().Modules()) {
966 SectionList *sections = module->GetSectionList();
967
968 std::string module_name = module->GetSpecificationDescription();
969
970 // The DYDL module is skipped since it's always loaded when running the
971 // binary.
972 if (module_name == "/usr/lib/dyld")
973 continue;
974
975 if (!sections) {
976 LLDB_LOG(log, "Module {0} doesn't have any section!", module_name);
977 continue;
978 }
979
980 SectionSP crash_info = sections->FindSectionByName(section_name);
981 if (!crash_info) {
982 LLDB_LOG(log, "Module {0} doesn't have section {1}!", module_name,
983 section_name);
984 continue;
985 }
986
987 addr_t load_addr = crash_info->GetLoadBaseAddress(&target);
988
989 if (load_addr == LLDB_INVALID_ADDRESS) {
990 LLDB_LOG(log, "Module {0} has an invalid '{1}' section load address: {2}",
991 module_name, section_name, load_addr);
992 continue;
993 }
994
995 Status error;
996 CrashInfoAnnotations annotations;
997 size_t expected_size = sizeof(CrashInfoAnnotations);
998 size_t bytes_read = process.ReadMemoryFromInferior(load_addr, &annotations,
999 expected_size, error);
1000
1001 if (expected_size != bytes_read || error.Fail()) {
1002 LLDB_LOG(log, "Failed to read {0} section from memory in module {1}: {2}",
1003 section_name, module_name, error);
1004 continue;
1005 }
1006
1007 // initial support added for version 5
1008 if (annotations.version < 5) {
1009 LLDB_LOG(log,
1010 "Annotation version lower than 5 unsupported! Module {0} has "
1011 "version {1} instead.",
1012 module_name, annotations.version);
1013 continue;
1014 }
1015
1016 if (!annotations.message) {
1017 LLDB_LOG(log, "No message available for module {0}.", module_name);
1018 continue;
1019 }
1020
1021 std::string message;
1022 bytes_read =
1023 process.ReadCStringFromMemory(annotations.message, message, error);
1024
1025 if (message.empty() || bytes_read != message.size() || error.Fail()) {
1026 LLDB_LOG(log, "Failed to read the message from memory in module {0}: {1}",
1027 module_name, error);
1028 continue;
1029 }
1030
1031 // Remove trailing newline from message
1032 if (message.back() == '\n')
1033 message.pop_back();
1034
1035 if (!annotations.message2)
1036 LLDB_LOG(log, "No message2 available for module {0}.", module_name);
1037
1038 std::string message2;
1039 bytes_read =
1040 process.ReadCStringFromMemory(annotations.message2, message2, error);
1041
1042 if (!message2.empty() && bytes_read == message2.size() && error.Success())
1043 if (message2.back() == '\n')
1044 message2.pop_back();
1045
1046 StructuredData::DictionarySP entry_sp =
1047 std::make_shared<StructuredData::Dictionary>();
1048
1049 entry_sp->AddStringItem("image", module->GetFileSpec().GetPath(false));
1050 entry_sp->AddStringItem("uuid", module->GetUUID().GetAsString());
1051 entry_sp->AddStringItem("message", message);
1052 entry_sp->AddStringItem("message2", message2);
1053 entry_sp->AddIntegerItem("abort-cause", annotations.abort_cause);
1054
1055 array_sp->AddItem(entry_sp);
1056 }
1057
1058 return array_sp;
1059}

1061StructuredData::DictionarySP
1062PlatformDarwin::ExtractAppSpecificInfo(Process &process) {
1063 StructuredData::DictionarySP metadata_sp = process.GetMetadata();
1064
1065 if (!metadata_sp || !metadata_sp->GetSize() || !metadata_sp->HasKey("asi"))
1066 return {};
1067
1068 StructuredData::Dictionary *asi;
1069 if (!metadata_sp->GetValueForKeyAsDictionary("asi", asi))
1070 return {};
1071
1072 StructuredData::DictionarySP dict_sp =
1073 std::make_shared<StructuredData::Dictionary>();
1074
1075 auto flatten_asi_dict = [&dict_sp](llvm::StringRef key,
1076 StructuredData::Object *val) -> bool {
1077 if (!val)
1078 return false;
1079
1080 StructuredData::Array *arr = val->GetAsArray();
1081 if (!arr || !arr->GetSize())
1082 return false;
1083
1084 dict_sp->AddItem(key, arr->GetItemAtIndex(0));
1085 return true;
1086 };
1087
1088 asi->ForEach(flatten_asi_dict);
1089
1090 return dict_sp;
1091}

1093static llvm::Expected<lldb_private::FileSpec>
1094ResolveSDKPathFromDebugInfo(lldb_private::Target *target) {
1095
1096 ModuleSP exe_module_sp = target->GetExecutableModule();
1097 if (!exe_module_sp)
1098 return llvm::createStringError("Failed to get module from target");
1099
1100 SymbolFile *sym_file = exe_module_sp->GetSymbolFile();
1101 if (!sym_file)
1102 return llvm::createStringError("Failed to get symbol file from executable");
1103
1104 if (sym_file->GetNumCompileUnits() == 0)
1105 return llvm::createStringError(
1106 "Failed to resolve SDK for target: executable's symbol file has no "
1107 "compile units");
1108
1109 XcodeSDK merged_sdk;
1110 for (unsigned i = 0; i < sym_file->GetNumCompileUnits(); ++i) {
1111 if (auto cu_sp = sym_file->GetCompileUnitAtIndex(i)) {
1112 auto cu_sdk = sym_file->ParseXcodeSDK(*cu_sp);
1113 merged_sdk.Merge(cu_sdk);
1114 }
1115 }
1116
1117 // TODO: The result of this loop is almost equivalent to deriving the SDK
1118 // from the target triple, which would be a lot cheaper.
1119 FileSpec sdk_path = merged_sdk.GetSysroot();
1120 if (FileSystem::Instance().Exists(sdk_path)) {
1121 return sdk_path;
1122 }
1123 auto path_or_err = HostInfo::GetSDKRoot(HostInfo::SDKOptions{merged_sdk});
1124 if (!path_or_err)
1125 return llvm::createStringError(
1126 llvm::formatv("Failed to resolve SDK path: {0}",
1127 llvm::toString(path_or_err.takeError())));
1128
1129 return FileSpec(*path_or_err);
1130}

1132void PlatformDarwin::AddClangModuleCompilationOptionsForSDKType(
1133 Target *target, std::vector<std::string> &options, XcodeSDK::Type sdk_type) {
1134 const std::vector<std::string> apple_arguments = {
1135 "-x", "objective-c++", "-fobjc-arc",
1136 "-fblocks", "-D_ISO646_H", "-D__ISO646_H",
1137 "-fgnuc-version=4.2.1"};
1138
1139 options.insert(options.end(), apple_arguments.begin(), apple_arguments.end());
1140
1141 StreamString minimum_version_option;
1142 bool use_current_os_version = false;
1143 // If the SDK type is for the host OS, use its version number.
1144 auto get_host_os = []() { return HostInfo::GetTargetTriple().getOS(); };
1145 switch (sdk_type) {
1146 case XcodeSDK::Type::MacOSX:
1147 use_current_os_version = get_host_os() == llvm::Triple::MacOSX;
1148 break;
1149 case XcodeSDK::Type::iPhoneOS:
1150 use_current_os_version = get_host_os() == llvm::Triple::IOS;
1151 break;
1152 case XcodeSDK::Type::AppleTVOS:
1153 use_current_os_version = get_host_os() == llvm::Triple::TvOS;
1154 break;
1155 case XcodeSDK::Type::watchOS:
1156 use_current_os_version = get_host_os() == llvm::Triple::WatchOS;
1157 break;
1158 case XcodeSDK::Type::XROS:
1159 use_current_os_version = get_host_os() == llvm::Triple::XROS;
1160 break;
1161 default:
1162 break;
1163 }
1164
1165 llvm::VersionTuple version;
1166 if (use_current_os_version)
1167 version = GetOSVersion();
1168 else if (target) {
1169 // Our OS doesn't match our executable so we need to get the min OS version
1170 // from the object file
1171 ModuleSP exe_module_sp = target->GetExecutableModule();
1172 if (exe_module_sp) {
1173 ObjectFile *object_file = exe_module_sp->GetObjectFile();
1174 if (object_file)
1175 version = object_file->GetMinimumOSVersion();
1176 }
1177 }
1178 // Only add the version-min options if we got a version from somewhere.
1179 // clang has no version-min clang flag for XROS.
1180 if (!version.empty() && sdk_type != XcodeSDK::Type::Linux &&
1181 sdk_type != XcodeSDK::Type::XROS) {
1182#define OPTION(PREFIX_OFFSET, NAME_OFFSET, VAR, ...) \
1183 llvm::StringRef opt_##VAR = OptionStrTable[NAME_OFFSET]; \
1184 (void)opt_##VAR;
1185#include "clang/Options/Options.inc"
1186#undef OPTION
1187 minimum_version_option << '-';
1188 switch (sdk_type) {
1189 case XcodeSDK::Type::MacOSX:
1190 minimum_version_option << opt_mmacos_version_min_EQ;
1191 break;
1192 case XcodeSDK::Type::iPhoneSimulator:
1193 minimum_version_option << opt_mios_simulator_version_min_EQ;
1194 break;
1195 case XcodeSDK::Type::iPhoneOS:
1196 minimum_version_option << opt_mios_version_min_EQ;
1197 break;
1198 case XcodeSDK::Type::AppleTVSimulator:
1199 minimum_version_option << opt_mtvos_simulator_version_min_EQ;
1200 break;
1201 case XcodeSDK::Type::AppleTVOS:
1202 minimum_version_option << opt_mtvos_version_min_EQ;
1203 break;
1204 case XcodeSDK::Type::WatchSimulator:
1205 minimum_version_option << opt_mwatchos_simulator_version_min_EQ;
1206 break;
1207 case XcodeSDK::Type::watchOS:
1208 minimum_version_option << opt_mwatchos_version_min_EQ;
1209 break;
1210 case XcodeSDK::Type::XRSimulator:
1211 case XcodeSDK::Type::XROS:
1212 // FIXME: Pass the right argument once it exists.
1213 case XcodeSDK::Type::BridgeOS:
1214 case XcodeSDK::Type::Linux:
1215 case XcodeSDK::Type::unknown:
1216 if (Log *log = GetLog(LLDBLog::Host)) {
1217 XcodeSDK::Info info;
1218 info.type = sdk_type;
1219 LLDB_LOGF(log, "Clang modules on %s are not supported",
1220 XcodeSDK::GetCanonicalName(info).c_str());
1221 }
1222 return;
1223 }
1224 minimum_version_option << version.getAsString();
1225 options.emplace_back(std::string(minimum_version_option.GetString()));
1226 }
1227
1228 FileSpec sysroot_spec;
1229
1230 if (target) {
1231 auto sysroot_spec_or_err = ::ResolveSDKPathFromDebugInfo(target);
1232 if (!sysroot_spec_or_err) {
1233 LLDB_LOG_ERROR(GetLog(LLDBLog::Types | LLDBLog::Host),
1234 sysroot_spec_or_err.takeError(),
1235 "Failed to resolve sysroot: {0}");
1236 } else {
1237 sysroot_spec = *sysroot_spec_or_err;
1238 }
1239 }
1240
1241 if (!FileSystem::Instance().IsDirectory(sysroot_spec.GetPath())) {
1242 std::lock_guard<std::mutex> guard(m_mutex);
1243 sysroot_spec = GetSDKDirectoryForModules(sdk_type);
1244 }
1245
1246 if (FileSystem::Instance().IsDirectory(sysroot_spec.GetPath())) {
1247 options.push_back("-isysroot");
1248 options.push_back(sysroot_spec.GetPath());
1249 }
1250}

1292lldb_private::FileSpec PlatformDarwin::LocateExecutable(const char *basename) {
1293 // A collection of SBFileSpec whose SBFileSpec.m_directory members are filled
1294 // in with any executable directories that should be searched.
1295 static std::vector<FileSpec> g_executable_dirs;
1296
1297 // Find the global list of directories that we will search for executables
1298 // once so we don't keep doing the work over and over.
1299 static llvm::once_flag g_once_flag;
1300 llvm::call_once(g_once_flag, []() {
1301
1302 // When locating executables, trust the DEVELOPER_DIR first if it is set
1303 FileSpec xcode_contents_dir = HostInfo::GetXcodeContentsDirectory();
1304 if (xcode_contents_dir) {
1305 FileSpec xcode_lldb_resources = xcode_contents_dir;
1306 xcode_lldb_resources.AppendPathComponent("SharedFrameworks");
1307 xcode_lldb_resources.AppendPathComponent("LLDB.framework");
1308 xcode_lldb_resources.AppendPathComponent("Resources");
1309 if (FileSystem::Instance().Exists(xcode_lldb_resources)) {
1310 FileSpec dir;
1311 dir.SetDirectory(xcode_lldb_resources.GetPathAsConstString());
1312 g_executable_dirs.push_back(dir);
1313 }
1314 }
1315 // Xcode might not be installed so we also check for the Command Line Tools.
1316 FileSpec command_line_tools_dir = GetCommandLineToolsLibraryPath();
1317 if (command_line_tools_dir) {
1318 FileSpec cmd_line_lldb_resources = command_line_tools_dir;
1319 cmd_line_lldb_resources.AppendPathComponent("PrivateFrameworks");
1320 cmd_line_lldb_resources.AppendPathComponent("LLDB.framework");
1321 cmd_line_lldb_resources.AppendPathComponent("Resources");
1322 if (FileSystem::Instance().Exists(cmd_line_lldb_resources)) {
1323 FileSpec dir;
1324 dir.SetDirectory(cmd_line_lldb_resources.GetPathAsConstString());
1325 g_executable_dirs.push_back(dir);
1326 }
1327 }
1328 });
1329
1330 // Now search the global list of executable directories for the executable we
1331 // are looking for
1332 for (const auto &executable_dir : g_executable_dirs) {
1333 FileSpec executable_file;
1334 executable_file.SetDirectory(executable_dir.GetDirectory());
1335 executable_file.SetFilename(basename);
1336 if (FileSystem::Instance().Exists(executable_file))
1337 return executable_file;
1338 }
1339
1340 return FileSpec();
1341}

1344PlatformDarwin::LaunchProcess(lldb_private::ProcessLaunchInfo &launch_info) {
1345 // Starting in Fall 2016 OSes, NSLog messages only get mirrored to stderr if
1346 // the OS_ACTIVITY_DT_MODE environment variable is set. (It doesn't require
1347 // any specific value; rather, it just needs to exist). We will set it here
1348 // as long as the IDE_DISABLED_OS_ACTIVITY_DT_MODE flag is not set. Xcode
1349 // makes use of IDE_DISABLED_OS_ACTIVITY_DT_MODE to tell
1350 // LLDB *not* to muck with the OS_ACTIVITY_DT_MODE flag when they
1351 // specifically want it unset.
1352 const char *disable_env_var = "IDE_DISABLED_OS_ACTIVITY_DT_MODE";
1353 auto &env_vars = launch_info.GetEnvironment();
1354 if (!env_vars.count(disable_env_var)) {
1355 // We want to make sure that OS_ACTIVITY_DT_MODE is set so that we get
1356 // os_log and NSLog messages mirrored to the target process stderr.
1357 env_vars.try_emplace("OS_ACTIVITY_DT_MODE", "enable");
1358 }
1359
1360 // Let our parent class do the real launching.
1361 return PlatformPOSIX::LaunchProcess(launch_info);

1364lldb_private::Status PlatformDarwin::FindBundleBinaryInExecSearchPaths(
1365 const ModuleSpec &module_spec, Process *process, ModuleSP &module_sp,
1366 llvm::SmallVectorImpl<ModuleSP> *old_modules, bool *did_create_ptr) {
1367 const FileSpec &platform_file = module_spec.GetFileSpec();
1368 TargetSP target_sp = module_spec.GetTargetSP();
1369 FileSpecList module_search_paths;
1370 if (target_sp)
1371 module_search_paths = target_sp->GetExecutableSearchPaths();
1372 // See if the file is present in any of the module_search_paths
1373 // directories.
1374 if (!module_sp && !module_search_paths.IsEmpty() && platform_file) {
1375 // create a vector of all the file / directory names in platform_file e.g.
1376 // this might be
1377 // /System/Library/PrivateFrameworks/UIFoundation.framework/UIFoundation
1378 //
1379 // We'll need to look in the module_search_paths_ptr directories for both
1380 // "UIFoundation" and "UIFoundation.framework" -- most likely the latter
1381 // will be the one we find there.
1382
1383 std::vector<llvm::StringRef> path_parts = platform_file.GetComponents();
1384 // We want the components in reverse order.
1385 std::reverse(path_parts.begin(), path_parts.end());
1386 const size_t path_parts_size = path_parts.size();
1387
1388 size_t num_module_search_paths = module_search_paths.GetSize();
1389 for (size_t i = 0; i < num_module_search_paths; ++i) {
1390 Log *log_verbose = GetLog(LLDBLog::Host);
1391 LLDB_LOGF(
1392 log_verbose,
1393 "PlatformRemoteDarwinDevice::GetSharedModule searching for binary in "
1394 "search-path %s",
1395 module_search_paths.GetFileSpecAtIndex(i).GetPath().c_str());
1396 // Create a new FileSpec with this module_search_paths_ptr plus just the
1397 // filename ("UIFoundation"), then the parent dir plus filename
1398 // ("UIFoundation.framework/UIFoundation") etc - up to four names (to
1399 // handle "Foo.framework/Contents/MacOS/Foo")
1400
1401 for (size_t j = 0; j < 4 && j < path_parts_size - 1; ++j) {
1402 FileSpec path_to_try(module_search_paths.GetFileSpecAtIndex(i));
1403
1404 // Add the components backwards. For
1405 // .../PrivateFrameworks/UIFoundation.framework/UIFoundation path_parts
1406 // is
1407 // [0] UIFoundation
1408 // [1] UIFoundation.framework
1409 // [2] PrivateFrameworks
1410 //
1411 // and if 'j' is 2, we want to append path_parts[1] and then
1412 // path_parts[0], aka 'UIFoundation.framework/UIFoundation', to the
1413 // module_search_paths_ptr path.
1414
1415 for (int k = j; k >= 0; --k) {
1416 path_to_try.AppendPathComponent(path_parts[k]);
1417 }
1418
1419 if (FileSystem::Instance().Exists(path_to_try)) {
1420 ModuleSpec new_module_spec(module_spec);
1421 new_module_spec.GetFileSpec() = path_to_try;
1422 Status new_error(Platform::GetSharedModule(new_module_spec, process,
1423 module_sp, old_modules,
1424 did_create_ptr));
1425
1426 if (module_sp) {
1427 module_sp->SetPlatformFileSpec(path_to_try);
1428 return new_error;
1429 }
1430 }
1431 }
1432 }
1433 }
1434 return Status();
1435}

1460PlatformDarwin::GetSDKPathFromDebugInfo(Module &module) {
1461 SymbolFile *sym_file = module.GetSymbolFile();
1462 if (!sym_file)
1463 return llvm::createStringError(
1464 llvm::inconvertibleErrorCode(),
1465 llvm::formatv("No symbol file available for module '{0}'",
1466 module.GetFileSpec().GetFilename().AsCString("")));
1467

1470 llvm::formatv("Could not resolve SDK for module '{0}'. Symbol file has "

1472 module.GetFileSpec()));
1473
1474 bool found_public_sdk = false;
1475 bool found_internal_sdk = false;
1476 XcodeSDK merged_sdk;
1477 for (unsigned i = 0; i < sym_file->GetNumCompileUnits(); ++i) {
1478 if (auto cu_sp = sym_file->GetCompileUnitAtIndex(i)) {
1479 auto cu_sdk = sym_file->ParseXcodeSDK(*cu_sp);
1480 bool is_internal_sdk = cu_sdk.IsAppleInternalSDK();
1481 found_public_sdk |= !is_internal_sdk;
1482 found_internal_sdk |= is_internal_sdk;
1483
1484 merged_sdk.Merge(cu_sdk);
1485 }
1486 }
1487
1488 const bool found_mismatch = found_internal_sdk && found_public_sdk;
1489
1490 return std::pair{std::move(merged_sdk), found_mismatch};
1491}
