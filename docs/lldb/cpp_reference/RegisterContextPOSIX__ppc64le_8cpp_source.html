157size_t RegisterContextPOSIX_ppc64le::GetRegisterSetCount() {
158 size_t sets = 0;
159 for (size_t set = 0; set < k_num_register_sets; ++set) {
160 if (IsRegisterSetAvailable(set))
161 ++sets;
162 }
163
164 return sets;
165}
166
167const RegisterSet *RegisterContextPOSIX_ppc64le::GetRegisterSet(size_t set) {
168 if (IsRegisterSetAvailable(set))
169 return &g_reg_sets_ppc64le[set];
170 else
171 return nullptr;
172}
