56template <typename Head>
57inline void stringify_helper(llvm::raw_string_ostream &ss, const Head &head) {

61template <typename Head, typename... Tail>
62inline void stringify_helper(llvm::raw_string_ostream &ss, const Head &head,
63 const Tail &...tail) {
64 stringify_append(ss, head);
65 ss << ", ";
66 stringify_helper(ss, tail...);
67}

69template <typename... Ts> inline std::string stringify_args(const Ts &...ts) {
70 std::string buffer;
71 llvm::raw_string_ostream ss(buffer);
72 stringify_helper(ss, ts...);
