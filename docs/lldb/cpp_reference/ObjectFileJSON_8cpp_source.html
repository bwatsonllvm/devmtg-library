39ObjectFile *ObjectFileJSON::CreateInstance(const ModuleSP &module_sp,
40 DataExtractorSP extractor_sp,
41 offset_t data_offset,
42 const FileSpec *file,
43 offset_t file_offset,
44 offset_t length) {
45 if (!extractor_sp || !extractor_sp->HasData()) {
46 DataBufferSP data_sp = MapFileData(*file, length, file_offset);
47 if (!data_sp)
48 return nullptr;
49 extractor_sp = std::make_shared<DataExtractor>(data_sp);
50 data_offset = 0;
51 }
52 if (!extractor_sp->HasData())
53 return nullptr;
54
55 if (!MagicBytesMatch(extractor_sp->GetSubsetExtractorSP(data_offset)))
56 return nullptr;
57
58 // Update the data to contain the entire file if it doesn't already.
59 if (extractor_sp->GetByteSize() < length) {
60 DataBufferSP data_sp = MapFileData(*file, length, file_offset);
61 if (!data_sp)
62 return nullptr;
63 extractor_sp->SetData(data_sp);
64 data_offset = 0;
65 }
66
67 Log *log = GetLog(LLDBLog::Symbols);
68
69 auto text = llvm::StringRef((const char *)extractor_sp->GetData().data());
70
71 Expected<json::Value> json = json::parse(text);
72 if (!json) {
73 LLDB_LOG_ERROR(log, json.takeError(),
74 "failed to parse JSON object file: {0}");
75 return nullptr;
76 }
77
78 json::Path::Root root;
79 Header header;
80 if (!fromJSON(*json, header, root)) {
81 LLDB_LOG_ERROR(log, root.getError(),
82 "failed to parse JSON object file header: {0}");
83 return nullptr;
84 }
85
86 ArchSpec arch(header.triple);
87 UUID uuid;
88 uuid.SetFromStringRef(header.uuid);
89 Type type = header.type.value_or(eTypeDebugInfo);
90
91 Body body;
92 if (!fromJSON(*json, body, root)) {
93 LLDB_LOG_ERROR(log, root.getError(),
94 "failed to parse JSON object file body: {0}");
95 return nullptr;
96 }
97
98 return new ObjectFileJSON(module_sp, extractor_sp, data_offset, file,
99 file_offset, length, std::move(arch),
100 std::move(uuid), type, std::move(body.symbols),
101 std::move(body.sections));
102}

111size_t ObjectFileJSON::GetModuleSpecifications(
112 const FileSpec &file, DataExtractorSP &extractor_sp, offset_t data_offset,
113 offset_t file_offset, offset_t length, ModuleSpecList &specs) {
114 if (!extractor_sp ||
115 !MagicBytesMatch(extractor_sp->GetSubsetExtractorSP(data_offset)))
116 return 0;
117
118 // Update the data to contain the entire file if it doesn't already.
119 if (extractor_sp->GetByteSize() < length) {
120 if (DataBufferSP file_data_sp = MapFileData(file, length, file_offset))
121 extractor_sp->SetData(file_data_sp);
122 if (!extractor_sp->HasData())
123 return 0;
124 data_offset = 0;
125 }
126
127 Log *log = GetLog(LLDBLog::Symbols);
128
129 auto text = llvm::StringRef((const char *)extractor_sp->GetData().data());
130
131 Expected<json::Value> json = json::parse(text);
132 if (!json) {
133 LLDB_LOG_ERROR(log, json.takeError(),
134 "failed to parse JSON object file: {0}");
135 return 0;
136 }
137
138 json::Path::Root root;
139 Header header;
140 if (!fromJSON(*json, header, root)) {
141 LLDB_LOG_ERROR(log, root.getError(),
142 "failed to parse JSON object file header: {0}");
143 return 0;
144 }
145
146 ArchSpec arch(header.triple);
147 UUID uuid;
148 uuid.SetFromStringRef(header.uuid);
149
150 ModuleSpec spec(file, std::move(arch));
151 spec.GetUUID() = std::move(uuid);
152 specs.Append(spec);
153 return 1;
154}

172void ObjectFileJSON::ParseSymtab(Symtab &symtab) {
173 Log *log = GetLog(LLDBLog::Symbols);
174 SectionList *section_list = GetModule()->GetSectionList();
175 for (JSONSymbol json_symbol : m_symbols) {
176 llvm::Expected<Symbol> symbol = Symbol::FromJSON(json_symbol, section_list);
177 if (!symbol) {
178 LLDB_LOG_ERROR(log, symbol.takeError(), "invalid symbol: {0}");
179 continue;
180 }
181 symtab.AddSymbol(*symbol);
182 }
183 symtab.Finalize();
184}

186void ObjectFileJSON::CreateSections(SectionList &unified_section_list) {
187 if (m_sections_up)
188 return;
189 m_sections_up = std::make_unique<SectionList>();
190
191 lldb::user_id_t id = 0;
192 for (const auto &json_section : m_sections) {
193 auto make_section = [this, &id](const JSONSection &section,
194 SectionSP parent_section_sp =
195 nullptr) -> SectionSP {
196 SectionSP section_sp;
197 auto sect_id = section.user_id.value_or(id + 1);
198 if (!section.user_id.has_value())
199 ++id;
200 const auto name = ConstString(section.name);
201 const auto sect_type = section.type.value_or(eSectionTypeCode);
202 const auto vm_addr = section.address.value_or(0);
203 const auto vm_size = section.size.value_or(0);
204 const auto file_offset = section.file_offset.value_or(0);
205 const auto file_size = section.file_size.value_or(0);
206 const auto log2align = section.log2align.value_or(0);
207 const auto flags = section.flags.value_or(0);
208 if (parent_section_sp) {
209 section_sp = std::make_shared<Section>(
210 parent_section_sp, GetModule(), this, sect_id, name, sect_type,
211 vm_addr - parent_section_sp->GetFileAddress(), vm_size, file_offset,
212 file_size, log2align, flags);
213
214 } else {
215 section_sp = std::make_shared<Section>(
216 GetModule(), this, sect_id, name, sect_type, vm_addr, vm_size,
217 file_offset, file_size, log2align, flags);
218 }
219 // Set permissions
220 uint32_t permissions = 0;
221 if (section.read.value_or(0))
222 permissions |= lldb::ePermissionsReadable;
223 if (section.write.value_or(0))
224 permissions |= lldb::ePermissionsWritable;
225 if (section.execute.value_or(0))
226 permissions |= lldb::ePermissionsExecutable;
227 if (permissions)
228 section_sp->SetPermissions(permissions);
229 section_sp->SetIsFake(section.fake.value_or(false));
230 section_sp->SetIsEncrypted(section.encrypted.value_or(false));
231 section_sp->SetIsThreadSpecific(section.thread_specific.value_or(false));
232 return section_sp;
233 };
234 auto section_sp = make_section(json_section);
235 for (const auto &subsection : json_section.subsections) {
236 SectionSP subsection_sp = make_section(subsection, section_sp);
237 section_sp->GetChildren().AddSection(subsection_sp);
238 }
239
240 m_sections_up->AddSection(section_sp);
241 unified_section_list.AddSection(section_sp);
242 }
243}

245bool ObjectFileJSON::SetLoadAddress(Target &target, lldb::addr_t value,
246 bool value_is_offset) {
247 Log *log(GetLog(LLDBLog::DynamicLoader));
248 if (!m_sections_up)
249 return true;
250
251 addr_t slide = value;
252 if (!value_is_offset) {
253 addr_t lowest_addr = LLDB_INVALID_ADDRESS;
254 for (const SectionSP &section_sp : *m_sections_up) {
255 addr_t section_load_addr = section_sp->GetFileAddress();
256 lowest_addr = std::min(lowest_addr, section_load_addr);
257 }
258 if (lowest_addr == LLDB_INVALID_ADDRESS)
259 return false;
260 slide = value - lowest_addr;
261 }
262
263 // Apply slide to each section's file address.
264 for (const SectionSP &section_sp : *m_sections_up) {
265 addr_t section_load_addr = section_sp->GetFileAddress();
266 if (section_load_addr != LLDB_INVALID_ADDRESS) {
267 LLDB_LOGF(
268 log,
269 "ObjectFileJSON::SetLoadAddress section %s to load addr 0x%" PRIx64,
270 section_sp->GetName().AsCString(), section_load_addr + slide);
271 target.SetSectionLoadAddress(section_sp, section_load_addr + slide,
272 /*warn_multiple=*/true);
273 }
274 }
275
276 return true;
277}

279bool ObjectFileJSON::MagicBytesMatch(DataExtractorSP extractor_sp) {
280 lldb::offset_t offset = 0;
281 char magic = extractor_sp->GetU8(&offset);
282 return magic == '{';

294bool fromJSON(const json::Value &value, ObjectFileJSON::Body &body,
295 json::Path path) {
296 json::ObjectMapper o(value, path);
297 return o && o.mapOptional("symbols", body.symbols) &&
298 o.mapOptional("sections", body.sections);
