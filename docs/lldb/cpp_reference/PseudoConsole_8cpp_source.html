29struct Kernel32 {
30 Kernel32() {
31 hModule = LoadLibraryW(L"kernel32.dll");
32 if (!hModule) {
33 llvm::Error err = llvm::errorCodeToError(
34 std::error_code(GetLastError(), std::system_category()));
35 LLDB_LOG_ERROR(GetLog(LLDBLog::Host), std::move(err),
36 "Could not load kernel32: {0}");
37 return;
38 }
39 CreatePseudoConsole_ =
40 (CreatePseudoConsole_t)GetProcAddress(hModule, "CreatePseudoConsole");
41 ClosePseudoConsole_ =
42 (ClosePseudoConsole_t)GetProcAddress(hModule, "ClosePseudoConsole");
43 isAvailable = (CreatePseudoConsole_ && ClosePseudoConsole_);
44 }

73llvm::Error PseudoConsole::OpenPseudoConsole() {
74 if (!kernel32.IsConPTYAvailable())
75 return llvm::make_error<llvm::StringError>("ConPTY is not available",
76 llvm::errc::io_error);
77
78 assert(m_conpty_handle == INVALID_HANDLE_VALUE &&
79 "ConPTY has already been opened");
80
81 HRESULT hr;
82 HANDLE hInputRead = INVALID_HANDLE_VALUE;
83 HANDLE hInputWrite = INVALID_HANDLE_VALUE;
84 HANDLE hOutputRead = INVALID_HANDLE_VALUE;
85 HANDLE hOutputWrite = INVALID_HANDLE_VALUE;
86
87 wchar_t pipe_name[MAX_PATH];
88 swprintf(pipe_name, MAX_PATH, L"\\\\.\\pipe\\conpty-lldb-%d-%p",
89 GetCurrentProcessId(), this);
90
91 // A 4096 bytes buffer should be large enough for the majority of console
92 // burst outputs.
93 hOutputRead =
94 CreateNamedPipeW(pipe_name, PIPE_ACCESS_INBOUND | FILE_FLAG_OVERLAPPED,
95 PIPE_TYPE_BYTE | PIPE_WAIT, 1, 4096, 4096, 0, NULL);
96 hOutputWrite = CreateFileW(pipe_name, GENERIC_WRITE, 0, NULL, OPEN_EXISTING,
97 FILE_ATTRIBUTE_NORMAL, NULL);
98
99 if (!CreatePipe(&hInputRead, &hInputWrite, NULL, 0))
100 return llvm::errorCodeToError(
101 std::error_code(GetLastError(), std::system_category()));
102
103 COORD consoleSize{80, 25};
104 HPCON hPC = INVALID_HANDLE_VALUE;
105 hr = kernel32.CreatePseudoConsole(consoleSize, hInputRead, hOutputWrite, 0,
106 &hPC);
107 CloseHandle(hInputRead);
108 CloseHandle(hOutputWrite);
109
110 if (FAILED(hr)) {
111 CloseHandle(hInputWrite);
112 CloseHandle(hOutputRead);
113 return llvm::make_error<llvm::StringError>(
114 "Failed to create Windows ConPTY pseudo terminal",
115 llvm::errc::io_error);
116 }
117
118 DWORD mode = PIPE_NOWAIT;
119 SetNamedPipeHandleState(hOutputRead, &mode, NULL, NULL);
120
121 m_conpty_handle = hPC;
122 m_conpty_output = hOutputRead;
123 m_conpty_input = hInputWrite;
124
125 if (auto error = DrainInitSequences()) {
126 Log *log = GetLog(LLDBLog::Host);
127 LLDB_LOG_ERROR(log, std::move(error),
128 "failed to finalize ConPTY's setup: {0}");
129 }
130
131 return llvm::Error::success();
132}

140void PseudoConsole::Close() {
141 SetStopping(true);
142 std::unique_lock<std::mutex> guard(m_mutex);
143 if (m_conpty_handle != INVALID_HANDLE_VALUE)
144 kernel32.ClosePseudoConsole(m_conpty_handle);
145 m_conpty_handle = INVALID_HANDLE_VALUE;
146 SetStopping(false);
147 m_cv.notify_all();
148}

150void PseudoConsole::ClosePipes() {
151 if (m_conpty_input != INVALID_HANDLE_VALUE)
152 CloseHandle(m_conpty_input);
153 if (m_conpty_output != INVALID_HANDLE_VALUE)
154 CloseHandle(m_conpty_output);
155
156 m_conpty_input = INVALID_HANDLE_VALUE;
157 m_conpty_output = INVALID_HANDLE_VALUE;
158}

160llvm::Error PseudoConsole::DrainInitSequences() {
161 STARTUPINFOEXW startupinfoex = {};
162 startupinfoex.StartupInfo.cb = sizeof(STARTUPINFOEXW);
163 startupinfoex.StartupInfo.dwFlags |= STARTF_USESTDHANDLES;
164
165 auto attributelist_or_err = ProcThreadAttributeList::Create(startupinfoex);
166 if (!attributelist_or_err)
167 return llvm::errorCodeToError(attributelist_or_err.getError());
168 ProcThreadAttributeList attributelist = std::move(*attributelist_or_err);
169 if (auto error = attributelist.SetupPseudoConsole(m_conpty_handle))
170 return error;
171
172 PROCESS_INFORMATION pi = {};
173
174 wchar_t comspec[MAX_PATH];
175 DWORD comspecLen = GetEnvironmentVariableW(L"COMSPEC", comspec, MAX_PATH);
176 if (comspecLen == 0 || comspecLen >= MAX_PATH)
177 return llvm::createStringError(
178 std::error_code(GetLastError(), std::system_category()),
179 "Failed to get the 'COMSPEC' environment variable");
180
181 std::wstring cmdline_str = std::wstring(comspec) + L" /c 'echo foo && exit'";
182 std::vector<wchar_t> cmdline(cmdline_str.begin(), cmdline_str.end());
183 cmdline.push_back(L'\0');
184
185 if (!CreateProcessW(/*lpApplicationName=*/comspec, cmdline.data(),
186 /*lpProcessAttributes=*/NULL, /*lpThreadAttributes=*/NULL,
187 /*bInheritHandles=*/TRUE,
188 /*dwCreationFlags=*/EXTENDED_STARTUPINFO_PRESENT |
189 CREATE_UNICODE_ENVIRONMENT,
190 /*lpEnvironment=*/NULL, /*lpCurrentDirectory=*/NULL,
191 /*lpStartupInfo=*/
192 reinterpret_cast<STARTUPINFOW *>(&startupinfoex),
193 /*lpProcessInformation=*/&pi))
194 return llvm::errorCodeToError(
195 std::error_code(GetLastError(), std::system_category()));
196
197 char buf[4096];
198 OVERLAPPED ov = {};
199 ov.hEvent = CreateEvent(NULL, TRUE, FALSE, NULL);
200
201 DWORD read;
202 ReadFile(m_conpty_output, buf, sizeof(buf), &read, &ov);
203
204 WaitForSingleObject(pi.hProcess, INFINITE);
205
206 if (GetOverlappedResult(m_conpty_output, &ov, &read, FALSE) && read > 0) {
207 ResetEvent(ov.hEvent);
208 ReadFile(m_conpty_output, buf, sizeof(buf), &read, &ov);
209 }
210
211 CancelIo(m_conpty_output);
212 CloseHandle(ov.hEvent);
213 CloseHandle(pi.hProcess);
214 CloseHandle(pi.hThread);
215
216 return llvm::Error::success();
217}
