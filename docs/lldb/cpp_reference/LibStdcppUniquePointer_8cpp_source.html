59ValueObjectSP LibStdcppUniquePtrSyntheticFrontEnd::GetTuple() {
60 ValueObjectSP valobj_backend_sp = m_backend.GetSP();
61
62 if (!valobj_backend_sp)
63 return nullptr;
64
65 ValueObjectSP valobj_sp = valobj_backend_sp->GetNonSyntheticValue();
66 if (!valobj_sp)
67 return nullptr;
68
69 ValueObjectSP obj_child_sp = valobj_sp->GetChildMemberWithName("_M_t");
70 if (!obj_child_sp)
71 return nullptr;
72
73 ValueObjectSP obj_subchild_sp = obj_child_sp->GetChildMemberWithName("_M_t");
74
75 // if there is a _M_t subchild, the tuple is found in the obj_subchild_sp
76 // (for libstdc++ 6.0.23).
77 if (obj_subchild_sp) {
78 return obj_subchild_sp;
79 }
80
81 return obj_child_sp;
82}

84lldb::ChildCacheState LibStdcppUniquePtrSyntheticFrontEnd::Update() {
85 ValueObjectSP tuple_sp = GetTuple();
86
87 if (!tuple_sp)
88 return lldb::ChildCacheState::eRefetch;
89
90 std::unique_ptr<SyntheticChildrenFrontEnd> tuple_frontend(
91 LibStdcppTupleSyntheticFrontEndCreator(nullptr, tuple_sp));
92
93 ValueObjectSP ptr_obj = tuple_frontend->GetChildAtIndex(0);
94 if (!ptr_obj)
95 return lldb::ChildCacheState::eRefetch;
96
97 m_ptr_obj = ptr_obj->Clone(ConstString("pointer")).get();
98
99 // Add a 'deleter' child if there was a non-empty deleter type specified.
100 //
101 // The object might have size=1 in the TypeSystem but occupies no dedicated
102 // storage due to no_unique_address, so infer the actual size from the total
103 // size of the unique_ptr class. If sizeof(unique_ptr) == sizeof(void*) then
104 // the deleter is empty and should be hidden.
105 if (llvm::expectedToOptional(tuple_sp->GetByteSize()).value_or(0) >
106 llvm::expectedToOptional(ptr_obj->GetByteSize()).value_or(0)) {
107 ValueObjectSP del_obj = tuple_frontend->GetChildAtIndex(1);
108 if (del_obj)
109 m_del_obj = del_obj->Clone(ConstString("deleter")).get();
110 }
111
112 return lldb::ChildCacheState::eRefetch;
113}

115lldb::ValueObjectSP
116LibStdcppUniquePtrSyntheticFrontEnd::GetChildAtIndex(uint32_t idx) {
117 if (idx == 0 && m_ptr_obj)
118 return m_ptr_obj->GetSP();
119 if (idx == 1 && m_del_obj)
120 return m_del_obj->GetSP();
121 if (idx == 2) {
122 if (m_ptr_obj) {
123 Status status;
124 auto value_sp = m_ptr_obj->Dereference(status);
125 if (status.Success()) {
126 return value_sp;
127 }
128 }
129 }
130 return lldb::ValueObjectSP();
131}

140llvm::Expected<size_t>
141LibStdcppUniquePtrSyntheticFrontEnd::GetIndexOfChildWithName(ConstString name) {
142 if (name == "ptr" || name == "pointer")
143 return 0;
144 if (name == "del" || name == "deleter")
145 return 1;
146 if (name == "obj" || name == "object" || name == "$$dereference$$")
147 return 2;
148 return llvm::createStringError("Type has no child named '%s'",
149 name.AsCString());
150}
