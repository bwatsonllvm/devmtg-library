129 void moduleImport(SourceLocation import_location, clang::ModuleIdPath path,
130 const clang::Module * /*null*/) override {
131 // Ignore modules that are imported in the wrapper code as these are not
132 // loaded by the user.
133 llvm::StringRef filename =
134 m_source_mgr.getPresumedLoc(import_location).getFilename();
135 if (filename == ClangExpressionSourceCode::g_prefix_file_name)
136 return;
137
138 SourceModule module;
139
140 for (const IdentifierLoc &component : path)
141 module.path.push_back(
142 ConstString(component.getIdentifierInfo()->getName()));
143
144 ClangModulesDeclVendor::ModuleVector exported_modules;
145 if (auto err = m_decl_vendor.AddModule(module, &exported_modules)) {
146 m_has_errors = true;
147 m_error_stream.PutCString(llvm::toString(std::move(err)));
148 m_error_stream.PutChar('\n');
149 }
150
151 for (ClangModulesDeclVendor::ModuleID module : exported_modules)
152 m_persistent_vars.AddHandLoadedClangModule(module);
153 }
154
155 bool hasErrors() { return m_has_errors; }
156
157 llvm::StringRef getErrorString() { return m_error_stream.GetString(); }
158};

183 /// Returns the last error ClangDiagnostic message that the
184 /// DiagnosticManager received or a nullptr.
185 ClangDiagnostic *MaybeGetLastClangDiag() const {
186 if (m_manager->Diagnostics().empty())
187 return nullptr;
188 auto &diags = m_manager->Diagnostics();
189 for (auto it = diags.rbegin(); it != diags.rend(); it++) {
190 lldb_private::Diagnostic *diag = it->get();
191 if (ClangDiagnostic *clang_diag = dyn_cast<ClangDiagnostic>(diag)) {
192 if (clang_diag->GetSeverity() == lldb::eSeverityWarning)
193 return nullptr;
194 if (clang_diag->GetSeverity() == lldb::eSeverityError)
195 return clang_diag;
196 }
197 }
198 return nullptr;
199 }

201 void HandleDiagnostic(DiagnosticsEngine::Level DiagLevel,
202 const clang::Diagnostic &Info) override {
203 if (!m_manager) {
204 // We have no DiagnosticManager before/after parsing but we still could
205 // receive diagnostics (e.g., by the ASTImporter failing to copy decls
206 // when we move the expression result ot the ScratchASTContext). Let's at
207 // least log these diagnostics until we find a way to properly render
208 // them and display them to the user.
209 Log *log = GetLog(LLDBLog::Expressions);
210 if (log) {
211 llvm::SmallVector<char, 32> diag_str;
212 Info.FormatDiagnostic(diag_str);
213 diag_str.push_back('\0');
214 const char *plain_diag = diag_str.data();
215 LLDB_LOG(log, "Received diagnostic outside parsing: {0}", plain_diag);
216 }
217 return;
218 }
219
220 // Update error/warning counters.
221 DiagnosticConsumer::HandleDiagnostic(DiagLevel, Info);
222
223 // Render diagnostic message to m_output.
224 m_output.clear();
225 m_passthrough->HandleDiagnostic(DiagLevel, Info);
226
227 DiagnosticDetail detail;
228 switch (DiagLevel) {
229 case DiagnosticsEngine::Level::Fatal:
230 case DiagnosticsEngine::Level::Error:
231 detail.severity = lldb::eSeverityError;
232 break;
233 case DiagnosticsEngine::Level::Warning:
234 detail.severity = lldb::eSeverityWarning;
235 break;
236 case DiagnosticsEngine::Level::Remark:
237 case DiagnosticsEngine::Level::Ignored:
238 detail.severity = lldb::eSeverityInfo;
239 break;
240 case DiagnosticsEngine::Level::Note:
241 // 'note:' diagnostics for errors and warnings can also contain Fix-Its.
242 // We add these Fix-Its to the last error diagnostic to make sure
243 // that we later have all Fix-Its related to an 'error' diagnostic when
244 // we apply them to the user expression.
245 auto *clang_diag = MaybeGetLastClangDiag();
246 // If we don't have a previous diagnostic there is nothing to do.
247 // If the previous diagnostic already has its own Fix-Its, assume that
248 // the 'note:' Fix-It is just an alternative way to solve the issue and
249 // ignore these Fix-Its.
250 if (!clang_diag || clang_diag->HasFixIts())
251 break;
252 // Ignore all Fix-Its that are not associated with an error.
253 if (clang_diag->GetSeverity() != lldb::eSeverityError)
254 break;
255 AddAllFixIts(clang_diag, Info);
256 break;
257 }
258 // ClangDiagnostic messages are expected to have no whitespace/newlines
259 // around them.
260 std::string stripped_output =
261 std::string(llvm::StringRef(m_output).trim());
262
263 // Translate the source location.
264 if (Info.hasSourceManager()) {
265 DiagnosticDetail::SourceLocation loc;
266 clang::SourceManager &sm = Info.getSourceManager();
267 const clang::SourceLocation sloc = Info.getLocation();
268 if (sloc.isValid()) {
269 const clang::FullSourceLoc fsloc(sloc, sm);
270 clang::PresumedLoc PLoc = fsloc.getPresumedLoc(true);
271 StringRef filename =
272 PLoc.isValid() ? PLoc.getFilename() : StringRef{};
273 loc.file = FileSpec(filename);
274 loc.line = fsloc.getSpellingLineNumber();
275 loc.column = fsloc.getSpellingColumnNumber();
276 loc.in_user_input = filename == m_filename;
277 loc.hidden = filename.starts_with("<lldb wrapper ");
278
279 // Find the range of the primary location.
280 for (const auto &range : Info.getRanges()) {
281 if (range.getBegin() == sloc) {
282 // FIXME: This is probably not handling wide characters correctly.
283 unsigned end_col = sm.getSpellingColumnNumber(range.getEnd());
284 if (end_col > loc.column)
285 loc.length = end_col - loc.column;
286 break;
287 }
288 }
289 detail.source_location = loc;
290 }
291 }
292 llvm::SmallString<0> msg;
293 Info.FormatDiagnostic(msg);
294 detail.message = msg.str();
295 detail.rendered = stripped_output;
296 auto new_diagnostic =
297 std::make_unique<ClangDiagnostic>(detail, Info.getID());
298
299 // Don't store away warning fixits, since the compiler doesn't have
300 // enough context in an expression for the warning to be useful.
301 // FIXME: Should we try to filter out FixIts that apply to our generated
302 // code, and not the user's expression?
303 if (detail.severity == lldb::eSeverityError)
304 AddAllFixIts(new_diagnostic.get(), Info);
305
306 m_manager->AddDiagnostic(std::move(new_diagnostic));
307 }

349/// Iff the given identifier is a C++ keyword, remove it from the
350/// identifier table (i.e., make the token a normal identifier).
351static void RemoveCppKeyword(IdentifierTable &idents, llvm::StringRef token) {
352 // FIXME: 'using' is used by LLDB for local variables, so we can't remove
353 // this keyword without breaking this functionality.
354 if (token == "using")
355 return;
356 // GCC's '__null' is used by LLDB to define NULL/Nil/nil.
357 if (token == "__null")
358 return;
359
360 LangOptions cpp_lang_opts;
361 cpp_lang_opts.CPlusPlus = true;
362 cpp_lang_opts.CPlusPlus11 = true;
363 cpp_lang_opts.CPlusPlus20 = true;
364
365 clang::IdentifierInfo &ii = idents.get(token);
366 // The identifier has to be a C++-exclusive keyword. if not, then there is
367 // nothing to do.
368 if (!ii.isCPlusPlusKeyword(cpp_lang_opts))
369 return;
370 // If the token is already an identifier, then there is nothing to do.
371 if (ii.getTokenID() == clang::tok::identifier)
372 return;
373 // Otherwise the token is a C++ keyword, so turn it back into a normal
374 // identifier.
375 ii.revertTokenIDToIdentifier();
376}

399/// Returns a string representing current ABI.
400///
401/// \param[in] target_arch
402/// The target architecture.
403///
404/// \return
405/// A string representing target ABI for the current architecture.
406static std::string GetClangTargetABI(const ArchSpec &target_arch) {
407 if (target_arch.IsMIPS()) {
408 switch (target_arch.GetFlags() & ArchSpec::eMIPSABI_mask) {
409 case ArchSpec::eMIPSABI_N64:
410 return "n64";
411 case ArchSpec::eMIPSABI_N32:
412 return "n32";
413 case ArchSpec::eMIPSABI_O32:
414 return "o32";
415 default:
416 return {};
417 }
418 }
419
420 if (target_arch.GetTriple().isRISCV64()) {
421 switch (target_arch.GetFlags() & ArchSpec::eRISCV_float_abi_mask) {
422 case ArchSpec::eRISCV_float_abi_soft:
423 return "lp64";
424 case ArchSpec::eRISCV_float_abi_single:
425 return "lp64f";
426 case ArchSpec::eRISCV_float_abi_double:
427 return "lp64d";
428 case ArchSpec::eRISCV_float_abi_quad:
429 return "lp64q";
430 default:
431 return {};
432 }
433 }
434
435 if (target_arch.GetTriple().isRISCV32()) {
436 switch (target_arch.GetFlags() & ArchSpec::eRISCV_float_abi_mask) {
437 case ArchSpec::eRISCV_float_abi_soft:
438 return "ilp32";
439 case ArchSpec::eRISCV_float_abi_single:
440 return "ilp32f";
441 case ArchSpec::eRISCV_float_abi_double:
442 return "ilp32d";
443 case ArchSpec::eRISCV_float_abi_soft | ArchSpec::eRISCV_rve:
444 return "ilp32e";
445 default:
446 return {};
447 }
448 }
449
450 if (target_arch.GetTriple().isLoongArch64()) {
451 switch (target_arch.GetFlags() & ArchSpec::eLoongArch_abi_mask) {
452 case ArchSpec::eLoongArch_abi_soft_float:
453 return "lp64s";
454 case ArchSpec::eLoongArch_abi_single_float:
455 return "lp64f";
456 case ArchSpec::eLoongArch_abi_double_float:
457 return "lp64d";
458 default:
459 return {};
460 }
461 }
462
463 return {};
464}

466static void SetupTargetOpts(CompilerInstance &compiler,
467 lldb_private::Target const &target) {
468 Log *log = GetLog(LLDBLog::Expressions);
469 ArchSpec target_arch = target.GetArchitecture();
470
471 const auto target_machine = target_arch.GetMachine();
472 if (target_arch.IsValid()) {
473 std::string triple = target_arch.GetTriple().str();
474 compiler.getTargetOpts().Triple = triple;
475 LLDB_LOGF(log, "Using %s as the target triple",
476 compiler.getTargetOpts().Triple.c_str());
477 } else {
478 // If we get here we don't have a valid target and just have to guess.
479 // Sometimes this will be ok to just use the host target triple (when we
480 // evaluate say "2+3", but other expressions like breakpoint conditions and
481 // other things that _are_ target specific really shouldn't just be using
482 // the host triple. In such a case the language runtime should expose an
483 // overridden options set (3), below.
484 compiler.getTargetOpts().Triple = llvm::sys::getDefaultTargetTriple();
485 LLDB_LOGF(log, "Using default target triple of %s",
486 compiler.getTargetOpts().Triple.c_str());
487 }
488 // Now add some special fixes for known architectures: Any arm32 iOS
489 // environment, but not on arm64
490 if (compiler.getTargetOpts().Triple.find("arm64") == std::string::npos &&
491 compiler.getTargetOpts().Triple.find("arm") != std::string::npos &&
492 compiler.getTargetOpts().Triple.find("ios") != std::string::npos) {
493 compiler.getTargetOpts().ABI = "apcs-gnu";
494 }
495 // Supported subsets of x86
496 if (target_machine == llvm::Triple::x86 ||
497 target_machine == llvm::Triple::x86_64) {
498 compiler.getTargetOpts().FeaturesAsWritten.push_back("+sse");
499 compiler.getTargetOpts().FeaturesAsWritten.push_back("+sse2");
500 }
501
502 // Set the target CPU to generate code for. This will be empty for any CPU
503 // that doesn't really need to make a special
504 // CPU string.
505 compiler.getTargetOpts().CPU = target_arch.GetClangTargetCPU();
506
507 // Set the target ABI
508 if (std::string abi = GetClangTargetABI(target_arch); !abi.empty())
509 compiler.getTargetOpts().ABI = std::move(abi);
510
511 if ((target_machine == llvm::Triple::riscv64 &&
512 compiler.getTargetOpts().ABI == "lp64f") ||
513 (target_machine == llvm::Triple::riscv32 &&
514 compiler.getTargetOpts().ABI == "ilp32f"))
515 compiler.getTargetOpts().FeaturesAsWritten.emplace_back("+f");
516
517 if ((target_machine == llvm::Triple::riscv64 &&
518 compiler.getTargetOpts().ABI == "lp64d") ||
519 (target_machine == llvm::Triple::riscv32 &&
520 compiler.getTargetOpts().ABI == "ilp32d"))
521 compiler.getTargetOpts().FeaturesAsWritten.emplace_back("+d");
522
523 if ((target_machine == llvm::Triple::loongarch64 &&
524 compiler.getTargetOpts().ABI == "lp64f"))
525 compiler.getTargetOpts().FeaturesAsWritten.emplace_back("+f");
526
527 if ((target_machine == llvm::Triple::loongarch64 &&
528 compiler.getTargetOpts().ABI == "lp64d"))
529 compiler.getTargetOpts().FeaturesAsWritten.emplace_back("+d");
530}

532static void SetupLangOpts(CompilerInstance &compiler,
533 ExecutionContextScope &exe_scope,
534 const Expression &expr,
535 DiagnosticManager &diagnostic_manager) {
536 Log *log = GetLog(LLDBLog::Expressions);
537
538 // If the expression is being evaluated in the context of an existing stack
539 // frame, we introspect to see if the language runtime is available.
540
541 lldb::StackFrameSP frame_sp = exe_scope.CalculateStackFrame();
542 lldb::ProcessSP process_sp = exe_scope.CalculateProcess();
543
544 lldb::LanguageType language = expr.Language().AsLanguageType();
545
546 if (process_sp)
547 LLDB_LOG(
548 log,
549 "Frame has language of type {0}\nPicked {1} for expression evaluation.",
550 lldb_private::Language::GetNameForLanguageType(
551 frame_sp ? frame_sp->GetLanguage().AsLanguageType()
552 : lldb::eLanguageTypeUnknown),
553 lldb_private::Language::GetNameForLanguageType(language));
554
555 lldb::LanguageType language_for_note = language;
556 std::string language_fallback_reason;
557
558 LangOptions &lang_opts = compiler.getLangOpts();
559
560 switch (language) {
561 case lldb::eLanguageTypeC:
562 case lldb::eLanguageTypeC89:
563 case lldb::eLanguageTypeC99:
564 case lldb::eLanguageTypeC11:
565 // FIXME: the following language option is a temporary workaround,
566 // to "ask for C, get C++."
567 // For now, the expression parser must use C++ anytime the language is a C
568 // family language, because the expression parser uses features of C++ to
569 // capture values.
570 lang_opts.CPlusPlus = true;
571
572 language_for_note = lldb::eLanguageTypeC_plus_plus;
573 language_fallback_reason =
574 "Expression evaluation in pure C not supported. ";
575 break;
576 case lldb::eLanguageTypeObjC:
577 lang_opts.ObjC = true;
578 // FIXME: the following language option is a temporary workaround,
579 // to "ask for ObjC, get ObjC++" (see comment above).
580 lang_opts.CPlusPlus = true;
581
582 language_for_note = lldb::eLanguageTypeObjC_plus_plus;
583 language_fallback_reason =
584 "Expression evaluation in pure Objective-C not supported. ";
585
586 // Clang now sets as default C++14 as the default standard (with
587 // GNU extensions), so we do the same here to avoid mismatches that
588 // cause compiler error when evaluating expressions (e.g. nullptr not found
589 // as it's a C++11 feature). Currently lldb evaluates C++14 as C++11 (see
590 // two lines below) so we decide to be consistent with that, but this could
591 // be re-evaluated in the future.
592 lang_opts.CPlusPlus11 = true;
593 break;
594 case lldb::eLanguageTypeC_plus_plus_20:
595 lang_opts.CPlusPlus20 = true;
596 [[fallthrough]];
597 case lldb::eLanguageTypeC_plus_plus_17:
598 // FIXME: add a separate case for CPlusPlus14. Currently folded into C++17
599 // because C++14 is the default standard for Clang but enabling CPlusPlus14
600 // expression evaluatino doesn't pass the test-suite cleanly.
601 lang_opts.CPlusPlus14 = true;
602 lang_opts.CPlusPlus17 = true;
603 [[fallthrough]];
604 case lldb::eLanguageTypeC_plus_plus:
605 case lldb::eLanguageTypeC_plus_plus_11:
606 case lldb::eLanguageTypeC_plus_plus_14:
607 lang_opts.CPlusPlus11 = true;
608 compiler.getHeaderSearchOpts().UseLibcxx = true;
609 [[fallthrough]];
610 case lldb::eLanguageTypeC_plus_plus_03: {
611 lang_opts.CPlusPlus = true;
612 if (process_sp
613 // We're stopped in a frame without debug-info. The user probably
614 // intends to make global queries (which should include Objective-C).
615 && !(frame_sp && frame_sp->HasDebugInformation())) {
616 lang_opts.ObjC =
617 process_sp->GetLanguageRuntime(lldb::eLanguageTypeObjC) != nullptr;
618 if (lang_opts.ObjC) {
619 language_for_note = lldb::eLanguageTypeObjC_plus_plus;
620 language_fallback_reason = "Possibly stopped inside system library, so "
621 "speculatively enabled Objective-C. ";
622 }
623 }
624 } break;
625 case lldb::eLanguageTypeObjC_plus_plus:
626 case lldb::eLanguageTypeUnknown:
627 default:
628 lang_opts.ObjC = true;
629 lang_opts.CPlusPlus = true;
630 lang_opts.CPlusPlus11 = true;
631 compiler.getHeaderSearchOpts().UseLibcxx = true;
632
633 language_for_note = lldb::eLanguageTypeObjC_plus_plus;
634 if (language != language_for_note) {
635 if (language != lldb::eLanguageTypeUnknown)
636 language_fallback_reason = llvm::formatv(
637 "Expression evaluation in {0} not supported. ",
638 lldb_private::Language::GetDisplayNameForLanguageType(language));
639
640 language_fallback_reason +=
641 llvm::formatv("Falling back to default language. ");
642 }
643 break;
644 }
645
646 diagnostic_manager.AddDiagnostic(
647 llvm::formatv("{0}Ran expression as '{1}'.", language_fallback_reason,
648 lldb_private::Language::GetDisplayNameForLanguageType(
649 language_for_note))
650 .str(),
651 lldb::Severity::eSeverityInfo, DiagnosticOrigin::eDiagnosticOriginLLDB);
652
653 lang_opts.Bool = true;
654 lang_opts.WChar = true;
655 lang_opts.Blocks = true;
656 lang_opts.DebuggerSupport =
657 true; // Features specifically for debugger clients
658 if (expr.DesiredResultType() == Expression::eResultTypeId)
659 lang_opts.DebuggerCastResultToId = true;
660
661 lang_opts.CharIsSigned =
662 ArchSpec(compiler.getTargetOpts().Triple.c_str()).CharIsSignedByDefault();
663
664 // Spell checking is a nice feature, but it ends up completing a lot of types
665 // that we didn't strictly speaking need to complete. As a result, we spend a
666 // long time parsing and importing debug information.
667 lang_opts.SpellChecking = false;
668
669 if (process_sp && lang_opts.ObjC) {
670 if (auto *runtime = ObjCLanguageRuntime::Get(*process_sp)) {
671 switch (runtime->GetRuntimeVersion()) {
672 case ObjCLanguageRuntime::ObjCRuntimeVersions::eAppleObjC_V2:
673 lang_opts.ObjCRuntime.set(ObjCRuntime::MacOSX, VersionTuple(10, 7));
674 break;
675 case ObjCLanguageRuntime::ObjCRuntimeVersions::eObjC_VersionUnknown:
676 case ObjCLanguageRuntime::ObjCRuntimeVersions::eAppleObjC_V1:
677 lang_opts.ObjCRuntime.set(ObjCRuntime::FragileMacOSX,
678 VersionTuple(10, 7));
679 break;
680 case ObjCLanguageRuntime::ObjCRuntimeVersions::eGNUstep_libobjc2:
681 lang_opts.ObjCRuntime.set(ObjCRuntime::GNUstep, VersionTuple(2, 0));
682 break;
683 }
684
685 if (runtime->HasNewLiteralsAndIndexing())
686 lang_opts.DebuggerObjCLiteral = true;
687 }
688 }

690 lang_opts.ThreadsafeStatics = false;
691 lang_opts.AccessControl = false; // Debuggers get universal access
692 lang_opts.DollarIdents = true; // $ indicates a persistent variable name
693 // We enable all builtin functions beside the builtins from libc/libm (e.g.
694 // 'fopen'). Those libc functions are already correctly handled by LLDB, and
695 // additionally enabling them as expandable builtins is breaking Clang.
696 lang_opts.NoBuiltin = true;
697}

728ClangExpressionParser::ClangExpressionParser(
729 ExecutionContextScope *exe_scope, Expression &expr,
730 bool generate_debug_info, DiagnosticManager &diagnostic_manager,
731 std::vector<std::string> include_directories, std::string filename)
732 : ExpressionParser(exe_scope, expr, generate_debug_info), m_compiler(),
733 m_pp_callbacks(nullptr),
734 m_include_directories(std::move(include_directories)),
735 m_filename(std::move(filename)) {
736 Log *log = GetLog(LLDBLog::Expressions);
737
738 // We can't compile expressions without a target. So if the exe_scope is
739 // null or doesn't have a target, then we just need to get out of here. I'll
740 // lldbassert and not make any of the compiler objects since
741 // I can't return errors directly from the constructor. Further calls will
742 // check if the compiler was made and
743 // bag out if it wasn't.
744
745 if (!exe_scope) {
746 lldbassert(exe_scope &&
747 "Can't make an expression parser with a null scope.");
748 return;
749 }
750
751 lldb::TargetSP target_sp;
752 target_sp = exe_scope->CalculateTarget();
753 if (!target_sp) {
754 lldbassert(target_sp.get() &&
755 "Can't make an expression parser with a null target.");
756 return;
757 }
758
759 // 1. Create a new compiler instance.
760 m_compiler = std::make_unique<CompilerInstance>();
761
762 // Make sure clang uses the same VFS as LLDB.
763 m_compiler->setVirtualFileSystem(
764 FileSystem::Instance().GetVirtualFileSystem());
765
766 // 2. Configure the compiler with a set of default options that are
767 // appropriate for most situations.
768 SetupTargetOpts(*m_compiler, *target_sp);
769
770 // 3. Create and install the target on the compiler.
771 m_compiler->createDiagnostics();
772 // Limit the number of error diagnostics we emit.
773 // A value of 0 means no limit for both LLDB and Clang.
774 m_compiler->getDiagnostics().setErrorLimit(target_sp->GetExprErrorLimit());
775
776 if (auto *target_info = TargetInfo::CreateTargetInfo(
777 m_compiler->getDiagnostics(),
778 m_compiler->getInvocation().getTargetOpts())) {
779 if (log) {
780 LLDB_LOGF(log, "Target datalayout string: '%s'",
781 target_info->getDataLayoutString());
782 LLDB_LOGF(log, "Target ABI: '%s'", target_info->getABI().str().c_str());
783 LLDB_LOGF(log, "Target vector alignment: %d",
784 target_info->getMaxVectorAlign());
785 }
786 m_compiler->setTarget(target_info);
787 } else {
788 if (log)
789 LLDB_LOGF(log, "Failed to create TargetInfo for '%s'",
790 m_compiler->getTargetOpts().Triple.c_str());
791
792 lldbassert(false && "Failed to create TargetInfo.");
793 }
794
795 // 4. Set language options.
796 SetupLangOpts(*m_compiler, *exe_scope, expr, diagnostic_manager);
797 auto *clang_expr = dyn_cast<ClangUserExpression>(&m_expr);
798 if (clang_expr && clang_expr->DidImportCxxModules()) {
799 LLDB_LOG(log, "Adding lang options for importing C++ modules");
800 SetupImportStdModuleLangOpts(*m_compiler, *target_sp);
801 SetupModuleHeaderPaths(m_compiler.get(), m_include_directories, target_sp);
802 }
803
804 // Set CodeGen options
805 m_compiler->getCodeGenOpts().EmitDeclMetadata = true;
806 m_compiler->getCodeGenOpts().InstrumentFunctions = false;
807 m_compiler->getCodeGenOpts().setFramePointer(
808 CodeGenOptions::FramePointerKind::All);
809 if (generate_debug_info)
810 m_compiler->getCodeGenOpts().setDebugInfo(codegenoptions::FullDebugInfo);
811 else
812 m_compiler->getCodeGenOpts().setDebugInfo(codegenoptions::NoDebugInfo);
813
814 // Disable some warnings.
815 SetupDefaultClangDiagnostics(*m_compiler);
816
817 // Inform the target of the language options
818 //
819 // FIXME: We shouldn't need to do this, the target should be immutable once
820 // created. This complexity should be lifted elsewhere.
821 m_compiler->getTarget().adjust(m_compiler->getDiagnostics(),
822 m_compiler->getLangOpts(),
823 /*AuxTarget=*/nullptr);
824
825 // 5. Set up the diagnostic buffer for reporting errors
826 auto diag_mgr = new ClangDiagnosticManagerAdapter(
827 m_compiler->getDiagnostics().getDiagnosticOptions(),
828 clang_expr ? clang_expr->GetFilename() : StringRef());
829 m_compiler->getDiagnostics().setClient(diag_mgr);
830
831 // 6. Set up the source management objects inside the compiler
832 m_compiler->createFileManager();
833 if (!m_compiler->hasSourceManager())
834 m_compiler->createSourceManager();
835 m_compiler->createPreprocessor(TU_Complete);
836
837 switch (expr.Language().AsLanguageType()) {
838 case lldb::eLanguageTypeC:
839 case lldb::eLanguageTypeC89:
840 case lldb::eLanguageTypeC99:
841 case lldb::eLanguageTypeC11:
842 case lldb::eLanguageTypeObjC:
843 // This is not a C++ expression but we enabled C++ as explained above.
844 // Remove all C++ keywords from the PP so that the user can still use
845 // variables that have C++ keywords as names (e.g. 'int template;').
846 RemoveAllCppKeywords(m_compiler->getPreprocessor().getIdentifierTable());
847 break;
848 default:
849 break;
850 }
851
852 if (auto *clang_persistent_vars = llvm::cast<ClangPersistentVariables>(
853 target_sp->GetPersistentExpressionStateForLanguage(
854 lldb::eLanguageTypeC))) {
855 if (std::shared_ptr<ClangModulesDeclVendor> decl_vendor =
856 clang_persistent_vars->GetClangModulesDeclVendor()) {
857 std::unique_ptr<PPCallbacks> pp_callbacks(
858 new LLDBPreprocessorCallbacks(*decl_vendor, *clang_persistent_vars,
859 m_compiler->getSourceManager()));
860 m_pp_callbacks =
861 static_cast<LLDBPreprocessorCallbacks *>(pp_callbacks.get());
862 m_compiler->getPreprocessor().addPPCallbacks(std::move(pp_callbacks));
863 }
864 }
865
866 // 7. Most of this we get from the CompilerInstance, but we also want to give
867 // the context an ExternalASTSource.
868
869 auto &PP = m_compiler->getPreprocessor();
870 auto &builtin_context = PP.getBuiltinInfo();
871 builtin_context.initializeBuiltins(PP.getIdentifierTable(),
872 m_compiler->getLangOpts());
873
874 m_compiler->createASTContext();
875 clang::ASTContext &ast_context = m_compiler->getASTContext();
876
877 m_ast_context = std::make_shared<TypeSystemClang>(
878 "Expression ASTContext for '" + m_filename + "'", ast_context);
879
880 std::string module_name("$__lldb_module");
881
882 m_llvm_context = std::make_unique<LLVMContext>();
883 m_code_generator =
884 CreateLLVMCodeGen(*m_compiler, module_name, *m_llvm_context);
885}

891/// \class CodeComplete
892///
893/// A code completion consumer for the clang Sema that is responsible for
894/// creating the completion suggestions when a user requests completion
895/// of an incomplete `expr` invocation.
896class CodeComplete : public CodeCompleteConsumer {
897 CodeCompletionTUInfo m_info;
898
899 std::string m_expr;
900 unsigned m_position = 0;
901 /// The printing policy we use when printing declarations for our completion
902 /// descriptions.
903 clang::PrintingPolicy m_desc_policy;
904
905 struct CompletionWithPriority {
906 CompletionResult::Completion completion;
907 /// See CodeCompletionResult::Priority;
908 unsigned Priority;
909
910 /// Establishes a deterministic order in a list of CompletionWithPriority.
911 /// The order returned here is the order in which the completions are
912 /// displayed to the user.
913 bool operator<(const CompletionWithPriority &o) const {
914 // High priority results should come first.
915 if (Priority != o.Priority)
916 return Priority > o.Priority;
917
918 // Identical priority, so just make sure it's a deterministic order.
919 return completion.GetUniqueKey() < o.completion.GetUniqueKey();
920 }
921 };
922
923 /// The stored completions.
924 /// Warning: These are in a non-deterministic order until they are sorted
925 /// and returned back to the caller.
926 std::vector<CompletionWithPriority> m_completions;
927
928 /// Returns true if the given character can be used in an identifier.
929 /// This also returns true for numbers because for completion we usually
930 /// just iterate backwards over iterators.
931 ///
932 /// Note: lldb uses '$' in its internal identifiers, so we also allow this.
933 static bool IsIdChar(char c) {
934 return c == '_' || std::isalnum(c) || c == '$';
935 }
936
937 /// Returns true if the given character is used to separate arguments
938 /// in the command line of lldb.
939 static bool IsTokenSeparator(char c) { return c == ' ' || c == '\t'; }
940
941 /// Drops all tokens in front of the expression that are unrelated for
942 /// the completion of the cmd line. 'unrelated' means here that the token
943 /// is not interested for the lldb completion API result.
944 StringRef dropUnrelatedFrontTokens(StringRef cmd) const {
945 if (cmd.empty())
946 return cmd;
947
948 // If we are at the start of a word, then all tokens are unrelated to
949 // the current completion logic.
950 if (IsTokenSeparator(cmd.back()))
951 return StringRef();
952
953 // Remove all previous tokens from the string as they are unrelated
954 // to completing the current token.
955 StringRef to_remove = cmd;
956 while (!to_remove.empty() && !IsTokenSeparator(to_remove.back())) {
957 to_remove = to_remove.drop_back();
958 }
959 cmd = cmd.drop_front(to_remove.size());
960
961 return cmd;
962 }
963
964 /// Removes the last identifier token from the given cmd line.
965 StringRef removeLastToken(StringRef cmd) const {
966 while (!cmd.empty() && IsIdChar(cmd.back())) {
967 cmd = cmd.drop_back();
968 }
969 return cmd;
970 }
971
972 /// Attempts to merge the given completion from the given position into the
973 /// existing command. Returns the completion string that can be returned to
974 /// the lldb completion API.
975 std::string mergeCompletion(StringRef existing, unsigned pos,
976 StringRef completion) const {
977 StringRef existing_command = existing.substr(0, pos);
978 // We rewrite the last token with the completion, so let's drop that
979 // token from the command.
980 existing_command = removeLastToken(existing_command);
981 // We also should remove all previous tokens from the command as they
982 // would otherwise be added to the completion that already has the
983 // completion.
984 existing_command = dropUnrelatedFrontTokens(existing_command);
985 return existing_command.str() + completion.str();
986 }
987
988public:
989 /// Constructs a CodeComplete consumer that can be attached to a Sema.
990 ///
991 /// \param[out] expr
992 /// The whole expression string that we are currently parsing. This
993 /// string needs to be equal to the input the user typed, and NOT the
994 /// final code that Clang is parsing.
995 /// \param[out] position
996 /// The character position of the user cursor in the `expr` parameter.
997 ///
998 CodeComplete(clang::LangOptions ops, std::string expr, unsigned position)
999 : CodeCompleteConsumer(CodeCompleteOptions()),
1000 m_info(std::make_shared<GlobalCodeCompletionAllocator>()), m_expr(expr),
1001 m_position(position), m_desc_policy(ops) {
1002
1003 // Ensure that the printing policy is producing a description that is as
1004 // short as possible.
1005 m_desc_policy.SuppressScope = true;
1006 m_desc_policy.SuppressTagKeyword = true;
1007 m_desc_policy.FullyQualifiedName = false;
1008 m_desc_policy.TerseOutput = true;
1009 m_desc_policy.IncludeNewlines = false;
1010 m_desc_policy.UseVoidForZeroParams = false;
1011 m_desc_policy.Bool = true;
1012 }
1013
1014 /// \name Code-completion filtering
1015 /// Check if the result should be filtered out.
1016 bool isResultFilteredOut(StringRef Filter,
1017 CodeCompletionResult Result) override {
1018 // This code is mostly copied from CodeCompleteConsumer.
1019 switch (Result.Kind) {
1020 case CodeCompletionResult::RK_Declaration:
1021 return !(
1022 Result.Declaration->getIdentifier() &&
1023 Result.Declaration->getIdentifier()->getName().starts_with(Filter));
1024 case CodeCompletionResult::RK_Keyword:
1025 return !StringRef(Result.Keyword).starts_with(Filter);
1026 case CodeCompletionResult::RK_Macro:
1027 return !Result.Macro->getName().starts_with(Filter);
1028 case CodeCompletionResult::RK_Pattern:
1029 return !StringRef(Result.Pattern->getAsString()).starts_with(Filter);
1030 }
1031 // If we trigger this assert or the above switch yields a warning, then
1032 // CodeCompletionResult has been enhanced with more kinds of completion
1033 // results. Expand the switch above in this case.
1034 assert(false && "Unknown completion result type?");
1035 // If we reach this, then we should just ignore whatever kind of unknown
1036 // result we got back. We probably can't turn it into any kind of useful
1037 // completion suggestion with the existing code.
1038 return true;
1039 }
1040
1041private:
1042 /// Generate the completion strings for the given CodeCompletionResult.
1043 /// Note that this function has to process results that could come in
1044 /// non-deterministic order, so this function should have no side effects.
1045 /// To make this easier to enforce, this function and all its parameters
1046 /// should always be const-qualified.
1047 /// \return Returns std::nullopt if no completion should be provided for the
1048 /// given CodeCompletionResult.
1049 std::optional<CompletionWithPriority>
1050 getCompletionForResult(const CodeCompletionResult &R) const {
1051 std::string ToInsert;
1052 std::string Description;
1053 // Handle the different completion kinds that come from the Sema.
1054 switch (R.Kind) {
1055 case CodeCompletionResult::RK_Declaration: {
1056 const NamedDecl *D = R.Declaration;
1057 ToInsert = R.Declaration->getNameAsString();
1058 // If we have a function decl that has no arguments we want to
1059 // complete the empty parantheses for the user. If the function has
1060 // arguments, we at least complete the opening bracket.
1061 if (const FunctionDecl *F = dyn_cast<FunctionDecl>(D)) {
1062 if (F->getNumParams() == 0)
1063 ToInsert += "()";
1064 else
1065 ToInsert += "(";
1066 raw_string_ostream OS(Description);
1067 F->print(OS, m_desc_policy, false);
1068 } else if (const VarDecl *V = dyn_cast<VarDecl>(D)) {
1069 Description = V->getType().getAsString(m_desc_policy);
1070 } else if (const FieldDecl *F = dyn_cast<FieldDecl>(D)) {
1071 Description = F->getType().getAsString(m_desc_policy);
1072 } else if (const NamespaceDecl *N = dyn_cast<NamespaceDecl>(D)) {
1073 // If we try to complete a namespace, then we can directly append
1074 // the '::'.
1075 if (!N->isAnonymousNamespace())
1076 ToInsert += "::";
1077 }
1078 break;
1079 }
1080 case CodeCompletionResult::RK_Keyword:
1081 ToInsert = R.Keyword;
1082 break;
1083 case CodeCompletionResult::RK_Macro:
1084 ToInsert = R.Macro->getName().str();
1085 break;
1086 case CodeCompletionResult::RK_Pattern:
1087 ToInsert = R.Pattern->getTypedText();
1088 break;
1089 }
1090 // We also filter some internal lldb identifiers here. The user
1091 // shouldn't see these.
1092 if (llvm::StringRef(ToInsert).starts_with("$__lldb_"))
1093 return std::nullopt;
1094 if (ToInsert.empty())
1095 return std::nullopt;
1096 // Merge the suggested Token into the existing command line to comply
1097 // with the kind of result the lldb API expects.
1098 std::string CompletionSuggestion =
1099 mergeCompletion(m_expr, m_position, ToInsert);
1100
1101 CompletionResult::Completion completion(CompletionSuggestion, Description,
1102 CompletionMode::Normal);
1103 return {{completion, R.Priority}};
1104 }
1105
1106public:
1107 /// Adds the completions to the given CompletionRequest.
1108 void GetCompletions(CompletionRequest &request) {
1109 // Bring m_completions into a deterministic order and pass it on to the
1110 // CompletionRequest.
1111 llvm::sort(m_completions);
1112
1113 for (const CompletionWithPriority &C : m_completions)
1114 request.AddCompletion(C.completion.GetCompletion(),
1115 C.completion.GetDescription(),
1116 C.completion.GetMode());
1117 }
1118
1119 /// \name Code-completion callbacks
1120 /// Process the finalized code-completion results.
1121 void ProcessCodeCompleteResults(Sema &SemaRef, CodeCompletionContext Context,
1122 CodeCompletionResult *Results,
1123 unsigned NumResults) override {
1124
1125 // The Sema put the incomplete token we try to complete in here during
1126 // lexing, so we need to retrieve it here to know what we are completing.
1127 StringRef Filter = SemaRef.getPreprocessor().getCodeCompletionFilter();
1128
1129 // Iterate over all the results. Filter out results we don't want and
1130 // process the rest.
1131 for (unsigned I = 0; I != NumResults; ++I) {
1132 // Filter the results with the information from the Sema.
1133 if (!Filter.empty() && isResultFilteredOut(Filter, Results[I]))
1134 continue;
1135
1136 CodeCompletionResult &R = Results[I];
1137 std::optional<CompletionWithPriority> CompletionAndPriority =
1138 getCompletionForResult(R);
1139 if (!CompletionAndPriority)
1140 continue;
1141 m_completions.push_back(*CompletionAndPriority);
1142 }
1143 }
1144
1145 /// \param S the semantic-analyzer object for which code-completion is being
1146 /// done.
1147 ///
1148 /// \param CurrentArg the index of the current argument.
1149 ///
1150 /// \param Candidates an array of overload candidates.
1151 ///
1152 /// \param NumCandidates the number of overload candidates
1153 void ProcessOverloadCandidates(Sema &S, unsigned CurrentArg,
1154 OverloadCandidate *Candidates,
1155 unsigned NumCandidates,
1156 SourceLocation OpenParLoc,
1157 bool Braced) override {
1158 // At the moment we don't filter out any overloaded candidates.
1159 }
1160
1161 CodeCompletionAllocator &getAllocator() override {
1162 return m_info.getAllocator();
1163 }
1164
1165 CodeCompletionTUInfo &getCodeCompletionTUInfo() override { return m_info; }
1166};

1169bool ClangExpressionParser::Complete(CompletionRequest &request, unsigned line,
1170 unsigned pos, unsigned typed_pos) {
1171 DiagnosticManager mgr;
1172 // We need the raw user expression here because that's what the CodeComplete
1173 // class uses to provide completion suggestions.
1174 // However, the `Text` method only gives us the transformed expression here.
1175 // To actually get the raw user input here, we have to cast our expression to
1176 // the LLVMUserExpression which exposes the right API. This should never fail
1177 // as we always have a ClangUserExpression whenever we call this.
1178 ClangUserExpression *llvm_expr = cast<ClangUserExpression>(&m_expr);
1179 CodeComplete CC(m_compiler->getLangOpts(), llvm_expr->GetUserText(),
1180 typed_pos);
1181 // We don't need a code generator for parsing.
1182 m_code_generator.reset();
1183 // Start parsing the expression with our custom code completion consumer.
1184 ParseInternal(mgr, &CC, line, pos);
1185 CC.GetCompletions(request);
1186 return true;
1187}

1193unsigned
1194ClangExpressionParser::ParseInternal(DiagnosticManager &diagnostic_manager,
1195 CodeCompleteConsumer *completion_consumer,
1196 unsigned completion_line,
1197 unsigned completion_column) {
1198 ClangDiagnosticManagerAdapter *adapter =
1199 static_cast<ClangDiagnosticManagerAdapter *>(
1200 m_compiler->getDiagnostics().getClient());
1201
1202 adapter->ResetManager(&diagnostic_manager);
1203
1204 const char *expr_text = m_expr.Text();
1205
1206 clang::SourceManager &source_mgr = m_compiler->getSourceManager();
1207 bool created_main_file = false;
1208
1209 // Clang wants to do completion on a real file known by Clang's file manager,
1210 // so we have to create one to make this work.
1211 // TODO: We probably could also simulate to Clang's file manager that there
1212 // is a real file that contains our code.
1213 bool should_create_file = completion_consumer != nullptr;
1214
1215 // We also want a real file on disk if we generate full debug info.
1216 should_create_file |= m_compiler->getCodeGenOpts().getDebugInfo() ==
1217 codegenoptions::FullDebugInfo;
1218
1219 if (should_create_file) {
1220 int temp_fd = -1;
1221 llvm::SmallString<128> result_path;
1222 if (FileSpec tmpdir_file_spec = HostInfo::GetProcessTempDir()) {
1223 tmpdir_file_spec.AppendPathComponent("lldb-%%%%%%.expr");
1224 std::string temp_source_path = tmpdir_file_spec.GetPath();
1225 llvm::sys::fs::createUniqueFile(temp_source_path, temp_fd, result_path);
1226 } else {
1227 llvm::sys::fs::createTemporaryFile("lldb", "expr", temp_fd, result_path);
1228 }
1229
1230 if (temp_fd != -1) {
1231 lldb_private::NativeFile file(temp_fd, File::eOpenOptionWriteOnly, true);
1232 const size_t expr_text_len = strlen(expr_text);
1233 size_t bytes_written = expr_text_len;
1234 if (file.Write(expr_text, bytes_written).Success()) {
1235 if (bytes_written == expr_text_len) {
1236 file.Close();

1238 result_path)) {

1240 *fileEntry, SourceLocation(), SrcMgr::C_User));
1241 created_main_file = true;
1242 }
1243 }
1244 }

1246 }

1250 MemoryBuffer::getMemBufferCopy(expr_text, m_filename);

1252 }

1255 &m_compiler->getPreprocessor());

1258 dyn_cast<ClangExpressionHelper>(m_expr.GetTypeSystemHelper());

1261 // completion consumer to the Sema and specify a completion position.
1262 // While parsing the Sema will call this consumer with the provided
1263 // completion suggestions.

1269 // indexed from 0, so we need to add 1 to the line and column here.
1270 ++completion_line;
1271 ++completion_column;
1272 PP.SetCodeCompletionPoint(*main_file, completion_line, completion_column);
1273 }
1274
1275 ASTConsumer *ast_transformer =
1276 type_system_helper->ASTTransformer(m_code_generator.get());

1279 if (ast_transformer) {
1280 Consumer = std::make_unique<ASTConsumerForwarder>(ast_transformer);
1281 } else if (m_code_generator) {
1282 Consumer = std::make_unique<ASTConsumerForwarder>(m_code_generator.get());
1283 } else {
1284 Consumer = std::make_unique<ASTConsumer>();
1285 }

1288
1289 m_compiler->setSema(new Sema(m_compiler->getPreprocessor(), ast_context,
1290 *Consumer, TU_Complete, completion_consumer));
1291 m_compiler->setASTConsumer(std::move(Consumer));
1292
1293 if (ast_context.getLangOpts().Modules) {
1294 m_compiler->createASTReader();
1295 m_ast_context->setSema(&m_compiler->getSema());
1296 }
1297

1299 if (decl_map) {

1301 decl_map->InstallDiagnosticManager(diagnostic_manager);

1304 decl_map->CreateProxy();

1307 llvm::makeIntrusiveRefCnt<ExternalASTSourceWrapper>(ast_source);
1308

1310 auto module_wrapper = llvm::makeIntrusiveRefCnt<ExternalASTSourceWrapper>(
1311 ast_context.getExternalSourcePtr());

1314 module_wrapper, ast_source_wrapper);
1315
1316 ast_context.setExternalSource(multiplexer);
1317 } else {
1318 ast_context.setExternalSource(ast_source);
1319 }
1320 m_compiler->getSema().addExternalSource(ast_source_wrapper);
1321 decl_map->InstallASTContext(*m_ast_context);
1322 }
1323
1324 // Check that the ASTReader is properly attached to ASTContext and Sema.
1325 if (ast_context.getLangOpts().Modules) {
1326 assert(m_compiler->getASTContext().getExternalSource() &&
1327 "ASTContext doesn't know about the ASTReader?");
1328 assert(m_compiler->getSema().getExternalSource() &&
1329 "Sema doesn't know about the ASTReader?");
1330 }
1331
1332 {
1333 llvm::CrashRecoveryContextCleanupRegistrar<Sema> CleanupSema(
1334 &m_compiler->getSema());
1335 ParseAST(m_compiler->getSema(), false, false);
1336 }
1337
1338 // Make sure we have no pointer to the Sema we are about to destroy.
1339 if (ast_context.getLangOpts().Modules)
1340 m_ast_context->setSema(nullptr);
1341 // Destroy the Sema. This is necessary because we want to emulate the
1342 // original behavior of ParseAST (which also destroys the Sema after parsing).
1343 m_compiler->setSema(nullptr);
1344
1345 adapter->EndSourceFile();
1346 // Creating persistent variables can trigger diagnostic emission.
1347 // Make sure we reset the manager so we don't get asked to handle
1348 // diagnostics after we finished parsing.
1349 adapter->ResetManager();
1350
1351 unsigned num_errors = adapter->getNumErrors();
1352
1353 if (m_pp_callbacks && m_pp_callbacks->hasErrors()) {
1354 num_errors++;
1355 diagnostic_manager.PutString(lldb::eSeverityError,
1356 "while importing modules:");
1357 diagnostic_manager.AppendMessageToDiagnostic(
1358 m_pp_callbacks->getErrorString());
1359 }
1360
1361 if (!num_errors) {
1362 type_system_helper->CommitPersistentDecls();
1363 }
1364
1365 return num_errors;
1366}

1368/// Applies the given Fix-It hint to the given commit.
1369static void ApplyFixIt(const FixItHint &fixit, clang::edit::Commit &commit) {
1370 // This is cobbed from clang::Rewrite::FixItRewriter.
1371 if (fixit.CodeToInsert.empty()) {
1372 if (fixit.InsertFromRange.isValid()) {
1373 commit.insertFromRange(fixit.RemoveRange.getBegin(),
1374 fixit.InsertFromRange, /*afterToken=*/false,
1375 fixit.BeforePreviousInsertions);
1376 return;
1377 }
1378 commit.remove(fixit.RemoveRange);
1379 return;
1380 }
1381 if (fixit.RemoveRange.isTokenRange() ||
1382 fixit.RemoveRange.getBegin() != fixit.RemoveRange.getEnd()) {
1383 commit.replace(fixit.RemoveRange, fixit.CodeToInsert);
1384 return;
1385 }
1386 commit.insert(fixit.RemoveRange.getBegin(), fixit.CodeToInsert,
1387 /*afterToken=*/false, fixit.BeforePreviousInsertions);
1388}

1390bool ClangExpressionParser::RewriteExpression(
1391 DiagnosticManager &diagnostic_manager) {
1392 clang::SourceManager &source_manager = m_compiler->getSourceManager();
1393 clang::edit::EditedSource editor(source_manager, m_compiler->getLangOpts(),
1394 nullptr);
1395 clang::edit::Commit commit(editor);
1396 clang::Rewriter rewriter(source_manager, m_compiler->getLangOpts());
1397
1398 class RewritesReceiver : public edit::EditsReceiver {
1399 Rewriter &rewrite;
1400
1401 public:
1402 RewritesReceiver(Rewriter &in_rewrite) : rewrite(in_rewrite) {}
1403
1404 void insert(SourceLocation loc, StringRef text) override {
1405 rewrite.InsertText(loc, text);
1406 }
1407 void replace(CharSourceRange range, StringRef text) override {
1408 rewrite.ReplaceText(range.getBegin(), rewrite.getRangeSize(range), text);
1409 }
1410 };
1411
1412 RewritesReceiver rewrites_receiver(rewriter);
1413
1414 const DiagnosticList &diagnostics = diagnostic_manager.Diagnostics();
1415 size_t num_diags = diagnostics.size();
1416 if (num_diags == 0)
1417 return false;
1418
1419 for (const auto &diag : diagnostic_manager.Diagnostics()) {
1420 const auto *diagnostic = llvm::dyn_cast<ClangDiagnostic>(diag.get());
1421 if (!diagnostic)
1422 continue;
1423 if (!diagnostic->HasFixIts())
1424 continue;
1425 for (const FixItHint &fixit : diagnostic->FixIts())
1426 ApplyFixIt(fixit, commit);
1427 }
1428
1429 // FIXME - do we want to try to propagate specific errors here?
1430 if (!commit.isCommitable())
1431 return false;
1432 else if (!editor.commit(commit))
1433 return false;
1434
1435 // Now play all the edits, and stash the result in the diagnostic manager.
1436 editor.applyRewrites(rewrites_receiver);
1437 RewriteBuffer &main_file_buffer =
1438 rewriter.getEditBuffer(source_manager.getMainFileID());
1439
1440 std::string fixed_expression;
1441 llvm::raw_string_ostream out_stream(fixed_expression);
1442
1443 main_file_buffer.write(out_stream);
1444 diagnostic_manager.SetFixedExpression(fixed_expression);
1445
1446 return true;
1447}

1462lldb_private::Status ClangExpressionParser::DoPrepareForExecution(
1463 lldb::addr_t &func_addr, lldb::addr_t &func_end,
1464 lldb::IRExecutionUnitSP &execution_unit_sp, ExecutionContext &exe_ctx,
1465 bool &can_interpret, ExecutionPolicy execution_policy) {
1466 func_addr = LLDB_INVALID_ADDRESS;
1467 func_end = LLDB_INVALID_ADDRESS;
1468 Log *log = GetLog(LLDBLog::Expressions);
1469
1470 lldb_private::Status err;
1471
1472 std::unique_ptr<llvm::Module> llvm_module_up(
1473 m_code_generator->ReleaseModule());
1474
1475 if (!llvm_module_up) {
1476 err = Status::FromErrorString("IR doesn't contain a module");
1477 return err;
1478 }
1479
1480 ConstString function_name;
1481
1482 if (execution_policy != eExecutionPolicyTopLevel) {
1483 // Find the actual name of the function (it's often mangled somehow)
1484
1485 if (!FindFunctionInModule(function_name, llvm_module_up.get(),
1486 m_expr.FunctionName())) {
1487 err = Status::FromErrorStringWithFormat(
1488 "Couldn't find %s() in the module", m_expr.FunctionName());
1489 return err;
1490 } else {
1491 LLDB_LOGF(log, "Found function %s for %s", function_name.AsCString(),
1492 m_expr.FunctionName());
1493 }
1494 }
1495
1496 SymbolContext sc;
1497
1498 if (lldb::StackFrameSP frame_sp = exe_ctx.GetFrameSP()) {
1499 sc = frame_sp->GetSymbolContext(lldb::eSymbolContextEverything);
1500 } else if (lldb::TargetSP target_sp = exe_ctx.GetTargetSP()) {
1501 sc.target_sp = target_sp;
1502 }
1503
1504 LLVMUserExpression::IRPasses custom_passes;
1505 {
1506 auto lang = m_expr.Language();
1507 LLDB_LOGF(log, "%s - Current expression language is %s\n", __FUNCTION__,
1508 lang.GetDescription().data());
1509 lldb::ProcessSP process_sp = exe_ctx.GetProcessSP();
1510 if (process_sp && lang) {
1511 auto runtime = process_sp->GetLanguageRuntime(lang.AsLanguageType());
1512 if (runtime)
1513 runtime->GetIRPasses(custom_passes);
1514 }
1515 }
1516
1517 if (custom_passes.EarlyPasses) {
1518 LLDB_LOGF(log,
1519 "%s - Running Early IR Passes from LanguageRuntime on "
1520 "expression module '%s'",
1521 __FUNCTION__, m_expr.FunctionName());
1522
1523 custom_passes.EarlyPasses->run(*llvm_module_up);
1524 }
1525
1526 execution_unit_sp = std::make_shared<IRExecutionUnit>(
1527 m_llvm_context, // handed off here
1528 llvm_module_up, // handed off here
1529 function_name, exe_ctx.GetTargetSP(), sc,
1530 m_compiler->getTargetOpts().Features);
1531
1532 if (auto *options = m_expr.GetOptions())
1533 execution_unit_sp->AppendPreferredSymbolContexts(
1534 options->GetPreferredSymbolContexts());
1535
1536 ClangExpressionHelper *type_system_helper =
1537 dyn_cast<ClangExpressionHelper>(m_expr.GetTypeSystemHelper());
1538 ClangExpressionDeclMap *decl_map =
1539 type_system_helper->DeclMap(); // result can be NULL
1540
1541 if (decl_map) {
1542 StreamString error_stream;
1543 IRForTarget ir_for_target(decl_map, m_expr.NeedsVariableResolution(),
1544 *execution_unit_sp, error_stream,
1545 function_name.AsCString());
1546
1547 if (!ir_for_target.runOnModule(*execution_unit_sp->GetModule())) {
1548 err = Status(error_stream.GetString().str());
1549 return err;
1550 }
1551
1552 Process *process = exe_ctx.GetProcessPtr();
1553
1554 if (execution_policy != eExecutionPolicyAlways &&
1555 execution_policy != eExecutionPolicyTopLevel) {
1556 lldb_private::Status interpret_error;
1557
1558 bool interpret_function_calls =
1559 !process ? false : process->CanInterpretFunctionCalls();
1560 can_interpret = IRInterpreter::CanInterpret(
1561 *execution_unit_sp->GetModule(), *execution_unit_sp->GetFunction(),
1562 interpret_error, interpret_function_calls);
1563
1564 if (!can_interpret && execution_policy == eExecutionPolicyNever) {
1565 err = Status::FromErrorStringWithFormat(
1566 "Can't evaluate the expression without a running target due to: %s",
1567 interpret_error.AsCString());
1568 return err;
1569 }
1570 }
1571
1572 if (!process && execution_policy == eExecutionPolicyAlways) {
1573 err = Status::FromErrorString(
1574 "Expression needed to run in the target, but the "
1575 "target can't be run");
1576 return err;
1577 }
1578
1579 if (!process && execution_policy == eExecutionPolicyTopLevel) {
1580 err = Status::FromErrorString(
1581 "Top-level code needs to be inserted into a runnable "
1582 "target, but the target can't be run");
1583 return err;
1584 }
1585
1586 if (execution_policy == eExecutionPolicyAlways ||
1587 (execution_policy != eExecutionPolicyTopLevel && !can_interpret)) {
1588 if (m_expr.NeedsValidation() && process) {
1589 if (!process->GetDynamicCheckers()) {
1590 ClangDynamicCheckerFunctions *dynamic_checkers =
1591 new ClangDynamicCheckerFunctions();
1592
1593 DiagnosticManager install_diags;
1594 if (Error Err = dynamic_checkers->Install(install_diags, exe_ctx))
1595 return Status::FromError(install_diags.GetAsError(
1596 lldb::eExpressionSetupError, "couldn't install checkers:"));
1597
1598 process->SetDynamicCheckers(dynamic_checkers);
1599
1600 LLDB_LOGF(log, "== [ClangExpressionParser::PrepareForExecution] "
1601 "Finished installing dynamic checkers ==");
1602 }
1603
1604 if (auto *checker_funcs = llvm::dyn_cast<ClangDynamicCheckerFunctions>(
1605 process->GetDynamicCheckers())) {
1606 IRDynamicChecks ir_dynamic_checks(*checker_funcs,
1607 function_name.AsCString());
1608
1609 llvm::Module *module = execution_unit_sp->GetModule();
1610 if (!module || !ir_dynamic_checks.runOnModule(*module)) {
1611 err = Status::FromErrorString(
1612 "Couldn't add dynamic checks to the expression");
1613 return err;
1614 }
1615
1616 if (custom_passes.LatePasses) {
1617 LLDB_LOGF(log,
1618 "%s - Running Late IR Passes from LanguageRuntime on "
1619 "expression module '%s'",
1620 __FUNCTION__, m_expr.FunctionName());
1621
1622 custom_passes.LatePasses->run(*module);
1623 }
1624 }
1625 }
1626 }
1627
1628 if (execution_policy == eExecutionPolicyAlways ||
1629 execution_policy == eExecutionPolicyTopLevel || !can_interpret) {
1630 execution_unit_sp->GetRunnableInfo(err, func_addr, func_end);
1631 }
1632 } else {
1633 execution_unit_sp->GetRunnableInfo(err, func_addr, func_end);
1634 }
1635
1636 return err;
1637}
