76/// The implementation of lldb::Type's m_payload field for TypeSystemClang.
77class TypePayloadClang {
78 /// The payload is used for typedefs and ptrauth types.
79 /// For typedefs, the Layout is as follows:
80 /// \verbatim
81 /// bit 0..30 ... Owning Module ID.
82 /// bit 31 ...... IsCompleteObjCClass.
83 /// \endverbatim
84 /// For ptrauth types, we store the PointerAuthQualifier as an opaque value.
85 Type::Payload m_payload = 0;
86
87public:
88 TypePayloadClang() = default;
89 explicit TypePayloadClang(OptionalClangModuleID owning_module,
90 bool is_complete_objc_class = false);
91 explicit TypePayloadClang(uint32_t opaque_payload) : m_payload(opaque_payload) {}
92 operator Type::Payload() { return m_payload; }
93
94 static constexpr unsigned ObjCClassBit = 1 << 31;
95 bool IsCompleteObjCClass() { return Flags(m_payload).Test(ObjCClassBit); }
96 void SetIsCompleteObjCClass(bool is_complete_objc_class) {
97 m_payload = is_complete_objc_class ? Flags(m_payload).Set(ObjCClassBit)
98 : Flags(m_payload).Clear(ObjCClassBit);
99 }
100 OptionalClangModuleID GetOwningModule() {
101 return OptionalClangModuleID(Flags(m_payload).Clear(ObjCClassBit));
102 }
103 void SetOwningModule(OptionalClangModuleID id);
104 /// \}
105};

238 /// Creates a CompilerType from the given QualType with the current
239 /// TypeSystemClang instance as the CompilerType's typesystem.
240 /// \param qt The QualType for a type that belongs to the ASTContext of this
241 /// TypeSystemClang.
242 /// \return The CompilerType representing the given QualType. If the
243 /// QualType's type pointer is a nullptr then the function returns an
244 /// invalid CompilerType.
245 CompilerType GetType(clang::QualType qt) {
246 if (qt.getTypePtrOrNull() == nullptr)
247 return CompilerType();
248 // Check that the type actually belongs to this TypeSystemClang.
249 assert(qt->getAsTagDecl() == nullptr ||
250 &qt->getAsTagDecl()->getASTContext() == &getASTContext());
251 return CompilerType(weak_from_this(), qt.getAsOpaquePtr());
252 }

262 template <typename RecordDeclType>
263 CompilerType
264 GetTypeForIdentifier(const clang::ASTContext &Ctx, llvm::StringRef type_name,
265 clang::DeclContext *decl_context = nullptr) {
266 CompilerType compiler_type;
267 if (type_name.empty())
268 return compiler_type;
269
270 clang::ASTContext &ast = getASTContext();
271 if (!decl_context)
272 decl_context = ast.getTranslationUnitDecl();
273
274 clang::IdentifierInfo &myIdent = ast.Idents.get(type_name);
275 clang::DeclarationName myName =
276 ast.DeclarationNames.getIdentifier(&myIdent);
277 clang::DeclContext::lookup_result result = decl_context->lookup(myName);
278 if (result.empty())
279 return compiler_type;
280
281 clang::NamedDecl *named_decl = *result.begin();
282 if (const auto *type_decl = llvm::dyn_cast<clang::TypeDecl>(named_decl);
283 llvm::isa_and_nonnull<RecordDeclType>(type_decl))
284 compiler_type = CompilerType(
285 weak_from_this(), Ctx.getTypeDeclType(type_decl).getAsOpaquePtr());
286
287 return compiler_type;
288 }

335 class TemplateParameterInfos {
336 public:
337 TemplateParameterInfos() = default;
338 TemplateParameterInfos(llvm::ArrayRef<const char *> names_in,
339 llvm::ArrayRef<clang::TemplateArgument> args_in)
340 : names(names_in), args(args_in) {
341 assert(names.size() == args_in.size());
342 }
343
344 TemplateParameterInfos(TemplateParameterInfos const &) = delete;
345 TemplateParameterInfos(TemplateParameterInfos &&) = delete;
346
347 TemplateParameterInfos &operator=(TemplateParameterInfos const &) = delete;
348 TemplateParameterInfos &operator=(TemplateParameterInfos &&) = delete;
349
350 ~TemplateParameterInfos() = default;
351
352 bool IsValid() const {
353 // Having a pack name but no packed args doesn't make sense, so mark
354 // these template parameters as invalid.
355 if (pack_name && !packed_args)
356 return false;
357 return args.size() == names.size() &&
358 (!packed_args || !packed_args->packed_args);
359 }
360
361 bool IsEmpty() const { return args.empty(); }
362 size_t Size() const { return args.size(); }
363
364 llvm::ArrayRef<clang::TemplateArgument> GetArgs() const { return args; }
365 llvm::ArrayRef<const char *> GetNames() const { return names; }
366
367 clang::TemplateArgument const &Front() const {
368 assert(!args.empty());
369 return args.front();
370 }
371
372 void InsertArg(char const *name, clang::TemplateArgument arg) {
373 args.emplace_back(std::move(arg));
374 names.push_back(name);
375 }
376
377 // Parameter pack related
378
379 bool hasParameterPack() const { return static_cast<bool>(packed_args); }
380
381 TemplateParameterInfos const &GetParameterPack() const {
382 assert(packed_args != nullptr);
383 return *packed_args;
384 }
385
386 TemplateParameterInfos &GetParameterPack() {
387 assert(packed_args != nullptr);
388 return *packed_args;
389 }
390
391 llvm::ArrayRef<clang::TemplateArgument> GetParameterPackArgs() const {
392 assert(packed_args != nullptr);
393 return packed_args->GetArgs();
394 }
395
396 bool HasPackName() const { return pack_name && pack_name[0]; }
397
398 llvm::StringRef GetPackName() const {
399 assert(HasPackName());
400 return pack_name;
401 }
402
403 void SetPackName(char const *name) { pack_name = name; }
404
405 void SetParameterPack(std::unique_ptr<TemplateParameterInfos> args) {
406 packed_args = std::move(args);
407 }
408
409 private:
410 /// Element 'names[i]' holds the template argument name
411 /// of 'args[i]'
412 llvm::SmallVector<const char *, 2> names;
413 llvm::SmallVector<clang::TemplateArgument, 2> args;
414
415 const char * pack_name = nullptr;
416 std::unique_ptr<TemplateParameterInfos> packed_args;
417 };
