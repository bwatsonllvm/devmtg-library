28HistoryThread::HistoryThread(lldb_private::Process &process, lldb::tid_t tid,
29 std::vector<lldb::addr_t> pcs,

63lldb::StackFrameListSP HistoryThread::GetStackFrameList() {
64 // FIXME do not throw away the lock after we acquire it..
65 std::unique_lock<std::mutex> lock(m_framelist_mutex);
66 lock.unlock();
67 if (m_framelist.get() == nullptr) {
68 m_framelist =
69 std::make_shared<StackFrameList>(*this, StackFrameListSP(), true);
70 }
71
72 return m_framelist;
73}

75uint32_t HistoryThread::GetExtendedBacktraceOriginatingIndexID() {
76 if (m_originating_unique_thread_id != LLDB_INVALID_THREAD_ID) {
77 if (GetProcess()->HasAssignedIndexIDToThread(
78 m_originating_unique_thread_id)) {
79 return GetProcess()->AssignIndexIDToThread(
80 m_originating_unique_thread_id);
81 }
82 }
83 return LLDB_INVALID_THREAD_ID;
