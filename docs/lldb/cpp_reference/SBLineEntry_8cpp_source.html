181bool SBLineEntry::GetDescription(SBStream &description) {
182 LLDB_INSTRUMENT_VA(this, description);
183
184 Stream &strm = description.ref();
185
186 if (m_opaque_up) {
187 char file_path[PATH_MAX * 2];
188 m_opaque_up->GetFile().GetPath(file_path, sizeof(file_path));
189 strm.Printf("%s:%u", file_path, GetLine());
190 if (GetColumn() > 0)
191 strm.Printf(":%u", GetColumn());
192 } else
193 strm.PutCString("No value");
194
195 return true;
196}
