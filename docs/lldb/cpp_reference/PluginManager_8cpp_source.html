97static constexpr llvm::StringLiteral g_plugin_prefix = "liblldbPlugin";
98struct PluginDir {
99 enum LoadPolicy {
100 /// Try to load anything that looks like a shared library.
101 LoadAnyDylib,
102
103 /// Only load shared libraries who's filename start with g_plugin_prefix.
104 LoadOnlyWithLLDBPrefix,
105 };
106
107 PluginDir(FileSpec path, LoadPolicy policy)
108 : path(std::move(path)), policy(policy) {}
109
110 explicit operator bool() const { return FileSystem::Instance().Exists(path); }
111
112 /// The path to the plugin directory.
113 const FileSpec path;
114
115 /// Filter when looking for plugins.
116 const LoadPolicy policy;
117};

119llvm::Expected<PluginInfo> PluginInfo::Create(const FileSpec &path) {
120 PluginInfo plugin_info;
121 std::string error;
122 plugin_info.library = llvm::sys::DynamicLibrary::getPermanentLibrary(
123 path.GetPath().c_str(), &error);
124 if (!plugin_info.library.isValid())
125 return llvm::createStringError(error);
126
127 // Look for files that follow the convention <g_plugin_prefix><name>.<ext>, in
128 // which case we need to call lldb_initialize_<name> and
129 // lldb_terminate_<name>.
130 llvm::StringRef file_name =
131 path.GetFileNameStrippingExtension().GetStringRef();
132 if (file_name.starts_with(g_plugin_prefix)) {
133 llvm::StringRef plugin_name = file_name.substr(g_plugin_prefix.size());
134 std::string init_symbol =
135 llvm::Twine("lldb_initialize_" + plugin_name).str();
136
137 if (auto *init_fn = CastToFPtr<PluginInitCallback>(
138 plugin_info.library.getAddressOfSymbol(init_symbol.c_str()))) {
139 if (!init_fn())
140 return llvm::createStringErrorV("initializer '{0}' returned false",
141 init_symbol);
142 const std::string term_symbol =
143 llvm::Twine("lldb_terminate_" + plugin_name).str();
144 plugin_info.plugin_term_callback = CastToFPtr<PluginTermCallback>(
145 plugin_info.library.getAddressOfSymbol(term_symbol.c_str()));
146 }
147 return plugin_info;
148 }
149
150 // Look for the legacy LLDBPluginInitialize/LLDBPluginTerminate symbols.
151 if (auto *init_fn = CastToFPtr<PluginInitCallback>(
152 plugin_info.library.getAddressOfSymbol("LLDBPluginInitialize"))) {
153 if (!init_fn())
154 return llvm::createStringError(
155 "initializer 'LLDBPluginInitialize' returned false");
156
157 plugin_info.plugin_init_callback = init_fn;
158 plugin_info.plugin_term_callback = CastToFPtr<PluginTermCallback>(
159 plugin_info.library.getAddressOfSymbol("LLDBPluginTerminate"));
160 return plugin_info;
161 }
162
163 return llvm::createStringError("no initialize symbol found");
164}

166static FileSystem::EnumerateDirectoryResult
167LoadPluginCallback(void *baton, llvm::sys::fs::file_type ft,
168 llvm::StringRef path) {
169 namespace fs = llvm::sys::fs;
170
171 static constexpr std::array<llvm::StringLiteral, 3>
172 g_shared_library_extension = {".dylib", ".so", ".dll"};
173
174 // If we have a regular file, a symbolic link or unknown file type, try and
175 // process the file. We must handle unknown as sometimes the directory
176 // enumeration might be enumerating a file system that doesn't have correct
177 // file type information.
178 if (ft == fs::file_type::regular_file || ft == fs::file_type::symlink_file ||
179 ft == fs::file_type::type_unknown) {
180 FileSpec plugin_file_spec(path);
181 FileSystem::Instance().Resolve(plugin_file_spec);
182
183 // Don't try to load unknown extensions.
184 if (!llvm::is_contained(g_shared_library_extension,
185 plugin_file_spec.GetFileNameExtension()))
186 return FileSystem::eEnumerateDirectoryResultNext;
187
188 // Don't try to load libraries that don't start with g_plugin_prefix if so
189 // requested.
190 PluginDir::LoadPolicy *policy = (PluginDir::LoadPolicy *)baton;
191 if (*policy == PluginDir::LoadOnlyWithLLDBPrefix &&
192 !plugin_file_spec.GetFilename().GetStringRef().starts_with(
193 g_plugin_prefix))
194 return FileSystem::eEnumerateDirectoryResultNext;
195
196 // Don't try to load an already loaded plugin again.
197 if (PluginIsLoaded(plugin_file_spec))
198 return FileSystem::eEnumerateDirectoryResultNext;
199
200 llvm::Expected<PluginInfo> plugin_info =
201 PluginInfo::Create(plugin_file_spec);
202 if (plugin_info) {
203 SetPluginInfo(plugin_file_spec, std::move(*plugin_info));
204 } else {
205 // Cache an empty plugin info so we don't try to load it again and again.
206 SetPluginInfo(plugin_file_spec, PluginInfo());
207
208 LLDB_LOG_ERROR(GetLog(LLDBLog::Host), plugin_info.takeError(),
209 "could not load plugin: {0}");
210 }
211
212 return FileSystem::eEnumerateDirectoryResultNext;
213 }
214
215 if (ft == fs::file_type::directory_file ||
216 ft == fs::file_type::symlink_file || ft == fs::file_type::type_unknown) {
217 // Try and recurse into anything that a directory or symbolic link. We must
218 // also do this for unknown as sometimes the directory enumeration might be
219 // enumerating a file system that doesn't have correct file type
220 // information.
221 return FileSystem::eEnumerateDirectoryResultEnter;
222 }
223
224 return FileSystem::eEnumerateDirectoryResultNext;
225}

227void PluginManager::Initialize() {
228 static const bool find_directories = true;
229 static const bool find_files = true;
230 static const bool find_other = true;
231
232 // Directories to scan for plugins. Unlike the plugin directories, which are
233 // meant exclusively for LLDB, the shared library directory is likely to
234 // contain unrelated shared libraries that we do not want to load. Therefore,
235 // limit the scan to libraries that start with g_plugin_prefix.
236 const std::array<PluginDir, 3> plugin_dirs = {
237 PluginDir(HostInfo::GetShlibDir(), PluginDir::LoadOnlyWithLLDBPrefix),
238 PluginDir(HostInfo::GetSystemPluginDir(), PluginDir::LoadAnyDylib),
239 PluginDir(HostInfo::GetUserPluginDir(), PluginDir::LoadAnyDylib)};
240
241 for (const PluginDir &plugin_dir : plugin_dirs) {
242 if (plugin_dir) {
243 FileSystem::Instance().EnumerateDirectory(
244 plugin_dir.path.GetPath().c_str(), find_directories, find_files,
245 find_other, LoadPluginCallback, (void *)&plugin_dir.policy);
246 }
247 }
248}

430llvm::json::Object PluginManager::GetJSON(llvm::StringRef pattern) {
431 llvm::json::Object plugin_stats;
432
433 for (const PluginNamespace &plugin_ns : GetPluginNamespaces()) {
434 llvm::json::Array namespace_stats;
435
436 for (const RegisteredPluginInfo &plugin : plugin_ns.get_info()) {
437 if (MatchPluginName(pattern, plugin_ns, plugin)) {
438 llvm::json::Object plugin_json;
439 plugin_json.try_emplace("name", plugin.name);
440 plugin_json.try_emplace("enabled", plugin.enabled);
441 namespace_stats.emplace_back(std::move(plugin_json));
442 }
443 }
444 if (!namespace_stats.empty())
445 plugin_stats.try_emplace(plugin_ns.name, std::move(namespace_stats));
446 }
447
448 return plugin_stats;
449}

451bool PluginManager::MatchPluginName(llvm::StringRef pattern,
452 const PluginNamespace &plugin_ns,
453 const RegisteredPluginInfo &plugin_info) {
454 // The empty pattern matches all plugins.
455 if (pattern.empty())
456 return true;
457
458 // Check if the pattern matches the namespace.
459 if (pattern == plugin_ns.name)
460 return true;
461
462 // Check if the pattern matches the qualified name.
463 std::string qualified_name = (plugin_ns.name + "." + plugin_info.name).str();
464 return pattern == qualified_name;
465}

485template <typename Instance> class PluginInstances {
486public:
487 template <typename... Args>
488 bool RegisterPlugin(llvm::StringRef name, llvm::StringRef description,
489 typename Instance::CallbackType callback,
490 Args &&...args) {
491 if (!callback)
492 return false;
493 assert(!name.empty());
494 m_instances.emplace_back(name, description, callback,
495 std::forward<Args>(args)...);
496 return true;
497 }
498
499 bool UnregisterPlugin(typename Instance::CallbackType callback) {
500 if (!callback)
501 return false;
502 auto pos = m_instances.begin();
503 auto end = m_instances.end();
504 for (; pos != end; ++pos) {
505 if (pos->create_callback == callback) {
506 m_instances.erase(pos);
507 return true;
508 }
509 }
510 return false;
511 }
512
513 typename Instance::CallbackType GetCallbackAtIndex(uint32_t idx) {
514 if (const Instance *instance = GetInstanceAtIndex(idx))
515 return instance->create_callback;
516 return nullptr;
517 }
518
519 llvm::StringRef GetDescriptionAtIndex(uint32_t idx) {
520 if (const Instance *instance = GetInstanceAtIndex(idx))
521 return instance->description;
522 return "";
523 }
524
525 llvm::StringRef GetNameAtIndex(uint32_t idx) {
526 if (const Instance *instance = GetInstanceAtIndex(idx))
527 return instance->name;
528 return "";
529 }
530
531 typename Instance::CallbackType GetCallbackForName(llvm::StringRef name) {
532 if (const Instance *instance = GetInstanceForName(name))
533 return instance->create_callback;
534 return nullptr;
535 }
536
537 void PerformDebuggerCallback(Debugger &debugger) {
538 for (const auto &instance : m_instances) {
539 if (!instance.enabled)
540 continue;
541 if (instance.debugger_init_callback)
542 instance.debugger_init_callback(debugger);
543 }
544 }
545
546 // Return a copy of all the enabled instances.
547 // Note that this is a copy of the internal state so modifications
548 // to the returned instances will not be reflected back to instances
549 // stored by the PluginInstances object.
550 std::vector<Instance> GetSnapshot() {
551 std::vector<Instance> enabled_instances;
552 for (const auto &instance : m_instances) {
553 if (instance.enabled)
554 enabled_instances.push_back(instance);
555 }
556 return enabled_instances;
557 }
558
559 const Instance *GetInstanceAtIndex(uint32_t idx) {
560 uint32_t count = 0;
561
562 return FindEnabledInstance(
563 [&](const Instance &instance) { return count++ == idx; });
564 }
565
566 const Instance *GetInstanceForName(llvm::StringRef name) {
567 if (name.empty())
568 return nullptr;
569
570 return FindEnabledInstance(
571 [&](const Instance &instance) { return instance.name == name; });
572 }
573
574 const Instance *
575 FindEnabledInstance(std::function<bool(const Instance &)> predicate) const {
576 for (const auto &instance : m_instances) {
577 if (!instance.enabled)
578 continue;
579 if (predicate(instance))
580 return &instance;
581 }
582 return nullptr;
583 }
584
585 // Return a list of all the registered plugin instances. This includes both
586 // enabled and disabled instances. The instances are listed in the order they
587 // were registered which is the order they would be queried if they were all
588 // enabled.
589 std::vector<RegisteredPluginInfo> GetPluginInfoForAllInstances() {
590 // Lookup the plugin info for each instance in the sorted order.
591 std::vector<RegisteredPluginInfo> plugin_infos;
592 plugin_infos.reserve(m_instances.size());
593 for (const Instance &instance : m_instances)
594 plugin_infos.push_back(
595 {instance.name, instance.description, instance.enabled});
596
597 return plugin_infos;
598 }
599
600 bool SetInstanceEnabled(llvm::StringRef name, bool enable) {
601 auto it = llvm::find_if(m_instances, [&](const Instance &instance) {
602 return instance.name == name;
603 });
604
605 if (it == m_instances.end())
606 return false;
607
608 it->enabled = enable;
609 return true;
610 }
611
612private:
613 std::vector<Instance> m_instances;
614};

1025Status PluginManager::SaveCore(lldb_private::SaveCoreOptions &options) {
1026 Status error;
1027 if (!options.GetOutputFile()) {
1028 error = Status::FromErrorString("No output file specified");
1029 return error;
1030 }
1031
1032 if (!options.GetProcess()) {
1033 error = Status::FromErrorString("Invalid process");
1034 return error;
1035 }
1036
1037 error = options.EnsureValidConfiguration();
1038 if (error.Fail())
1039 return error;
1040
1041 if (!options.GetPluginName().has_value()) {
1042 // Try saving core directly from the process plugin first.
1043 llvm::Expected<bool> ret =
1044 options.GetProcess()->SaveCore(options.GetOutputFile()->GetPath());
1045 if (!ret)
1046 return Status::FromError(ret.takeError());
1047 if (ret.get())
1048 return Status();
1049 }
1050
1051 // Fall back to object plugins.
1052 const auto &plugin_name = options.GetPluginName().value_or("");
1053 auto instances = GetObjectFileInstances().GetSnapshot();
1054 for (auto &instance : instances) {
1055 if (plugin_name.empty() || instance.name == plugin_name) {
1056 // TODO: Refactor the instance.save_core() to not require a process and
1057 // get it from options instead.
1058 if (instance.save_core &&
1059 instance.save_core(options.GetProcess(), options, error))
1060 return error;
1061 }
1062 }
1063
1064 // Check to see if any of the object file plugins tried and failed to save.
1065 // if any failure, return the error message.
1066 if (error.Fail())
1067 return error;
1068
1069 // Report only for the plugin that was specified.
1070 if (!plugin_name.empty())
1071 return Status::FromErrorStringWithFormatv(
1072 "The \"{}\" plugin is not able to save a core for this process.",
1073 plugin_name);
1074
1075 return Status::FromErrorString(
1076 "no ObjectFile plugins were able to save a core for this process");
1077}

1079std::vector<llvm::StringRef> PluginManager::GetSaveCorePluginNames() {
1080 std::vector<llvm::StringRef> plugin_names;
1081 auto instances = GetObjectFileInstances().GetSnapshot();
1082 for (auto &instance : instances) {
1083 if (instance.save_core)
1084 plugin_names.emplace_back(instance.name);
1085 }
1086 return plugin_names;
1087}

2657void PluginManager::AutoCompletePluginName(llvm::StringRef name,
2658 CompletionRequest &request) {
2659 // Split the name into the namespace and the plugin name.
2660 // If there is no dot then the ns_name will be equal to name and
2661 // plugin_prefix will be empty.
2662 llvm::StringRef ns_name, plugin_prefix;
2663 std::tie(ns_name, plugin_prefix) = name.split('.');
2664
2665 for (const PluginNamespace &plugin_ns : GetPluginNamespaces()) {
2666 // If the plugin namespace matches exactly then
2667 // add all the plugins in this namespace as completions if the
2668 // plugin names starts with the plugin_prefix. If the plugin_prefix
2669 // is empty then it will match all the plugins (empty string is a
2670 // prefix of everything).
2671 if (plugin_ns.name == ns_name) {
2672 for (const RegisteredPluginInfo &plugin : plugin_ns.get_info()) {
2673 llvm::SmallString<128> buf;
2674 if (plugin.name.starts_with(plugin_prefix))
2675 request.AddCompletion(
2676 (plugin_ns.name + "." + plugin.name).toStringRef(buf));
2677 }
2678 } else if (plugin_ns.name.starts_with(name) &&
2679 !plugin_ns.get_info().empty()) {
2680 // Otherwise check if the namespace is a prefix of the full name.
2681 // Use a partial completion here so that we can either operate on the full
2682 // namespace or tab-complete to the next level.
2683 request.AddCompletion(plugin_ns.name, "", CompletionMode::Partial);
2684 }
2685 }
