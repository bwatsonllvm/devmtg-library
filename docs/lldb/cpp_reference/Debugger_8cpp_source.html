235Status Debugger::SetPropertyValue(const ExecutionContext *exe_ctx,
236 VarSetOperationType op,
237 llvm::StringRef property_path,
238 llvm::StringRef value) {
239 bool is_load_script =
240 (property_path == "target.load-script-from-symbol-file");
241 // These properties might change how we visualize data.
242 bool invalidate_data_vis = (property_path == "escape-non-printables");
243 invalidate_data_vis |=
244 (property_path == "target.max-zero-padding-in-float-format");
245 if (invalidate_data_vis) {
246 DataVisualization::ForceUpdate();
247 }
248
249 TargetSP target_sp;
250 LoadScriptFromSymFile load_script_old_value = eLoadScriptFromSymFileFalse;
251 if (is_load_script && exe_ctx && exe_ctx->GetTargetSP()) {
252 target_sp = exe_ctx->GetTargetSP();
253 load_script_old_value =
254 target_sp->TargetProperties::GetLoadScriptFromSymbolFile();
255 }
256 Status error(Properties::SetPropertyValue(exe_ctx, op, property_path, value));
257 if (error.Success()) {
258 // FIXME it would be nice to have "on-change" callbacks for properties
259 if (property_path == g_debugger_properties[ePropertyPrompt].name) {
260 llvm::StringRef new_prompt = GetPrompt();
261 std::string str = lldb_private::ansi::FormatAnsiTerminalCodes(
262 new_prompt, GetUseColor());
263 if (str.length())
264 new_prompt = str;
265 GetCommandInterpreter().UpdatePrompt(new_prompt);
266 auto bytes = std::make_unique<EventDataBytes>(new_prompt);
267 auto prompt_change_event_sp = std::make_shared<Event>(
268 CommandInterpreter::eBroadcastBitResetPrompt, bytes.release());
269 GetCommandInterpreter().BroadcastEvent(prompt_change_event_sp);
270 } else if (property_path == g_debugger_properties[ePropertyUseColor].name) {
271 // use-color changed. set use-color, this also pings the prompt so it can
272 // reset the ansi terminal codes.
273 SetUseColor(GetUseColor());
274 } else if (property_path ==
275 g_debugger_properties[ePropertyPromptAnsiPrefix].name ||
276 property_path ==
277 g_debugger_properties[ePropertyPromptAnsiSuffix].name) {
278 // Prompt color changed. set use-color, this also pings the prompt so it
279 // can reset the ansi terminal codes.
280 SetUseColor(GetUseColor());
281 } else if (property_path ==
282 g_debugger_properties[ePropertyShowStatusline].name) {
283 // Statusline setting changed. If we have a statusline instance, update it
284 // now. Otherwise it will get created in the default event handler.
285 std::lock_guard<std::mutex> guard(m_statusline_mutex);
286 if (StatuslineSupported()) {
287 m_statusline.emplace(*this);
288 m_statusline->Enable(GetSelectedExecutionContextRef());
289 } else {
290 m_statusline.reset();
291 }
292 } else if (property_path ==
293 g_debugger_properties[ePropertyStatuslineFormat].name ||
294 property_path ==
295 g_debugger_properties[ePropertySeparator].name) {
296 // Statusline format changed. Redraw the statusline.
297 RedrawStatusline(std::nullopt);
298 } else if (property_path ==
299 g_debugger_properties[ePropertyUseSourceCache].name) {
300 // use-source-cache changed. Wipe out the cache contents if it was
301 // disabled.
302 if (!GetUseSourceCache()) {
303 m_source_file_cache.Clear();
304 }
305 } else if (is_load_script && target_sp &&
306 load_script_old_value == eLoadScriptFromSymFileWarn) {
307 if (target_sp->TargetProperties::GetLoadScriptFromSymbolFile() ==
308 eLoadScriptFromSymFileTrue) {
309 std::list<Status> errors;
310 StreamString feedback_stream;
311 if (!target_sp->LoadScriptingResources(errors, feedback_stream)) {
312 lldb::StreamUP s = GetAsyncErrorStream();
313 for (auto &error : errors)
314 s->Printf("%s\n", error.AsCString());
315 if (feedback_stream.GetSize())
316 s->PutCString(feedback_stream.GetString());
317 }
318 }
319 }
320 }
321 return error;
322}

807static FileSystem::EnumerateDirectoryResult
808LoadPluginCallback(void *baton, llvm::sys::fs::file_type ft,
809 llvm::StringRef path) {
810 Status error;
811
812 static constexpr llvm::StringLiteral g_dylibext(".dylib");
813 static constexpr llvm::StringLiteral g_solibext(".so");
814
815 if (!baton)
816 return FileSystem::eEnumerateDirectoryResultQuit;
817
818 Debugger *debugger = (Debugger *)baton;
819
820 namespace fs = llvm::sys::fs;
821 // If we have a regular file, a symbolic link or unknown file type, try and
822 // process the file. We must handle unknown as sometimes the directory
823 // enumeration might be enumerating a file system that doesn't have correct
824 // file type information.
825 if (ft == fs::file_type::regular_file || ft == fs::file_type::symlink_file ||
826 ft == fs::file_type::type_unknown) {
827 FileSpec plugin_file_spec(path);
828 FileSystem::Instance().Resolve(plugin_file_spec);
829
830 if (plugin_file_spec.GetFileNameExtension() != g_dylibext &&
831 plugin_file_spec.GetFileNameExtension() != g_solibext) {
832 return FileSystem::eEnumerateDirectoryResultNext;
833 }
834
835 Status plugin_load_error;
836 debugger->LoadPlugin(plugin_file_spec, plugin_load_error);
837
838 return FileSystem::eEnumerateDirectoryResultNext;
839 } else if (ft == fs::file_type::directory_file ||
840 ft == fs::file_type::symlink_file ||
841 ft == fs::file_type::type_unknown) {
842 // Try and recurse into anything that a directory or symbolic link. We must
843 // also do this for unknown as sometimes the directory enumeration might be
844 // enumerating a file system that doesn't have correct file type
845 // information.
846 return FileSystem::eEnumerateDirectoryResultEnter;
847 }
848
849 return FileSystem::eEnumerateDirectoryResultNext;
850}

902void Debugger::HandleDestroyCallback() {
903 const lldb::user_id_t user_id = GetID();
904 // Invoke and remove all the callbacks in an FIFO order. Callbacks which are
905 // added during this loop will be appended, invoked and then removed last.
906 // Callbacks which are removed during this loop will not be invoked.
907 while (true) {
908 DestroyCallbackInfo callback_info;
909 {
910 std::lock_guard<std::mutex> guard(m_destroy_callback_mutex);
911 if (m_destroy_callbacks.empty())
912 break;
913 // Pop the first item in the list
914 callback_info = m_destroy_callbacks.front();
915 m_destroy_callbacks.erase(m_destroy_callbacks.begin());
916 }
917 // Call the destroy callback with user id and baton
918 callback_info.callback(user_id, callback_info.baton);
919 }
920}

922void Debugger::Destroy(DebuggerSP &debugger_sp) {
923 if (!debugger_sp)
924 return;
925
926 debugger_sp->HandleDestroyCallback();
927 CommandInterpreter &cmd_interpreter = debugger_sp->GetCommandInterpreter();
928
929 if (cmd_interpreter.GetSaveSessionOnQuit()) {
930 CommandReturnObject result(debugger_sp->GetUseColor());
931 cmd_interpreter.SaveTranscript(result);
932 if (result.Succeeded())
933 (*debugger_sp->GetAsyncOutputStream())
934 << result.GetOutputString() << '\n';
935 else
936 (*debugger_sp->GetAsyncErrorStream()) << result.GetErrorString() << '\n';
937 }
938
939 debugger_sp->Clear();
940
941 if (g_debugger_list_ptr && g_debugger_list_mutex_ptr) {
942 std::lock_guard<std::recursive_mutex> guard(*g_debugger_list_mutex_ptr);
943 DebuggerList::iterator pos, end = g_debugger_list_ptr->end();
944 for (pos = g_debugger_list_ptr->begin(); pos != end; ++pos) {
945 if ((*pos).get() == debugger_sp.get()) {
946 g_debugger_list_ptr->erase(pos);
947 return;
948 }
949 }
950 }

1116 // chain having a list of debuggers and with code that can be running on
1117 // other threads, we need to ensure this doesn't happen multiple times.

1124 telemetry::ScopedDispatcher<telemetry::DebuggerInfo> helper(

1126 assert(this == info->debugger);
1127 (void)this;

1130 this);

1133 StopEventHandlerThread();
1134 m_listener_sp->Clear();

1173Status Debugger::SetInputString(const char *data) {
1174 Status result;
1175 enum PIPES { READ, WRITE }; // Indexes for the read and write fds
1176 int fds[2] = {-1, -1};
1177
1178 if (data == nullptr) {
1179 result = Status::FromErrorString("String data is null");
1180 return result;
1181 }
1182
1183 size_t size = strlen(data);
1184 if (size == 0) {
1185 result = Status::FromErrorString("String data is empty");
1186 return result;
1187 }
1188
1189 if (OpenPipe(fds, size) != 0) {
1190 result = Status::FromErrorString(
1191 "can't create pipe file descriptors for LLDB commands");
1192 return result;
1193 }
1194
1195 int r = write(fds[WRITE], data, size);
1196 (void)r;
1197 // Close the write end of the pipe, so that the command interpreter will exit
1198 // when it consumes all the data.
1199 llvm::sys::Process::SafelyCloseFileDescriptor(fds[WRITE]);
1200
1201 // Open the read file descriptor as a FILE * that we can return as an input
1202 // handle.
1203 FILE *commands_file = fdopen(fds[READ], "rb");
1204 if (commands_file == nullptr) {
1205 result = Status::FromErrorStringWithFormat(
1206 "fdopen(%i, \"rb\") failed (errno = %i) "
1207 "when trying to open LLDB commands pipe",
1208 fds[READ], errno);
1209 llvm::sys::Process::SafelyCloseFileDescriptor(fds[READ]);
1210 return result;
1211 }
1212
1213 SetInputFile((FileSP)std::make_shared<NativeFile>(
1214 commands_file, File::eOpenOptionReadOnly, true));
1215 return result;
1216}

1295void Debugger::ClearIOHandlers() {

1297 // not want to close that one here.
1298 std::lock_guard<std::recursive_mutex> guard(m_io_handler_stack.GetMutex());
1299 while (m_io_handler_stack.GetSize() > 1) {
1300 IOHandlerSP reader_sp(m_io_handler_stack.Top());
1301 if (reader_sp)
1302 PopIOHandler(reader_sp);
1303 }
1304}
1305
1306void Debugger::RunIOHandlers() {
1307 IOHandlerSP reader_sp = m_io_handler_stack.Top();
1308 while (true) {
1309 if (!reader_sp)
1310 break;
1311
1312 reader_sp->Run();
1313 {
1314 std::lock_guard<std::recursive_mutex> guard(
1315 m_io_handler_synchronous_mutex);
1316
1317 // Remove all input readers that are done from the top of the stack
1318 while (true) {
1319 IOHandlerSP top_reader_sp = m_io_handler_stack.Top();
1320 if (top_reader_sp && top_reader_sp->GetIsDone())
1321 PopIOHandler(top_reader_sp);
1322 else
1323 break;
1324 }
1325 reader_sp = m_io_handler_stack.Top();
1326 }
1327 }
1328 ClearIOHandlers();
1329}
1330
1331void Debugger::RunIOHandlerSync(const IOHandlerSP &reader_sp) {
1332 std::lock_guard<std::recursive_mutex> guard(m_io_handler_synchronous_mutex);
1333
1334 PushIOHandler(reader_sp);
1335 IOHandlerSP top_reader_sp = reader_sp;
1336
1337 while (top_reader_sp) {
1338 top_reader_sp->Run();
1339
1340 // Don't unwind past the starting point.
1341 if (top_reader_sp.get() == reader_sp.get()) {
1342 if (PopIOHandler(reader_sp))
1343 break;
1344 }
1345
1346 // If we pushed new IO handlers, pop them if they're done or restart the
1347 // loop to run them if they're not.
1348 while (true) {
1349 top_reader_sp = m_io_handler_stack.Top();
1350 if (top_reader_sp && top_reader_sp->GetIsDone()) {
1351 PopIOHandler(top_reader_sp);
1352 // Don't unwind past the starting point.
1353 if (top_reader_sp.get() == reader_sp.get())
1354 return;
1355 } else {
1356 break;
1357 }
1358 }
1359 }
1360}

1402void Debugger::AdoptTopIOHandlerFilesIfInvalid(FileSP &in,
1403 LockableStreamFileSP &out,
1404 LockableStreamFileSP &err) {
1405 // Before an IOHandler runs, it must have in/out/err streams. This function
1406 // is called when one ore more of the streams are nullptr. We use the top
1407 // input reader's in/out/err streams, or fall back to the debugger file
1408 // handles, or we fall back onto stdin/stdout/stderr as a last resort.
1409
1410 std::lock_guard<std::recursive_mutex> guard(m_io_handler_stack.GetMutex());
1411 IOHandlerSP top_reader_sp(m_io_handler_stack.Top());
1412 // If no STDIN has been set, then set it appropriately
1413 if (!in || !in->IsValid()) {
1414 if (top_reader_sp)
1415 in = top_reader_sp->GetInputFileSP();
1416 else
1417 in = GetInputFileSP();
1418 // If there is nothing, use stdin
1419 if (!in)
1420 in = std::make_shared<NativeFile>(stdin, File::eOpenOptionReadOnly,
1421 NativeFile::Unowned);
1422 }
1423 // If no STDOUT has been set, then set it appropriately
1424 if (!out || !out->GetUnlockedFile().IsValid()) {
1425 if (top_reader_sp)
1426 out = top_reader_sp->GetOutputStreamFileSP();
1427 else
1428 out = GetOutputStreamSP();
1429 // If there is nothing, use stdout
1430 if (!out)
1431 out = std::make_shared<LockableStreamFile>(stdout, NativeFile::Unowned,
1432 m_output_mutex);
1433 }
1434 // If no STDERR has been set, then set it appropriately
1435 if (!err || !err->GetUnlockedFile().IsValid()) {
1436 if (top_reader_sp)
1437 err = top_reader_sp->GetErrorStreamFileSP();
1438 else
1439 err = GetErrorStreamSP();
1440 // If there is nothing, use stderr
1441 if (!err)
1442 err = std::make_shared<LockableStreamFile>(stderr, NativeFile::Unowned,
1443 m_output_mutex);
1444 }
1445}
1446
1447void Debugger::PushIOHandler(const IOHandlerSP &reader_sp,
1448 bool cancel_top_handler) {
1449 if (!reader_sp)
1450 return;
1451
1452 std::lock_guard<std::recursive_mutex> guard(m_io_handler_stack.GetMutex());
1453
1454 // Get the current top input reader...
1455 IOHandlerSP top_reader_sp(m_io_handler_stack.Top());
1456
1457 // Don't push the same IO handler twice...
1458 if (reader_sp == top_reader_sp)
1459 return;
1460
1461 // Push our new input reader
1462 m_io_handler_stack.Push(reader_sp);
1463 reader_sp->Activate();
1464
1465 // Interrupt the top input reader to it will exit its Run() function and let
1466 // this new input reader take over
1467 if (top_reader_sp) {
1468 top_reader_sp->Deactivate();
1469 if (cancel_top_handler)
1470 top_reader_sp->Cancel();
1471 }
1472}
1473
1474bool Debugger::PopIOHandler(const IOHandlerSP &pop_reader_sp) {
1475 if (!pop_reader_sp)
1476 return false;
1477
1478 std::lock_guard<std::recursive_mutex> guard(m_io_handler_stack.GetMutex());
1479
1480 // The reader on the stop of the stack is done, so let the next read on the
1481 // stack refresh its prompt and if there is one...
1482 if (m_io_handler_stack.IsEmpty())
1483 return false;
1484
1485 IOHandlerSP reader_sp(m_io_handler_stack.Top());
1486
1487 if (pop_reader_sp != reader_sp)
1488 return false;
1489
1490 reader_sp->Deactivate();
1491 reader_sp->Cancel();
1492 m_io_handler_stack.Pop();
1493
1494 reader_sp = m_io_handler_stack.Top();
1495 if (reader_sp)
1496 reader_sp->Activate();
1497
1498 return true;
1499}

1604bool Debugger::FormatDisassemblerAddress(const FormatEntity::Entry *format,
1605 const SymbolContext *sc,
1606 const SymbolContext *prev_sc,
1607 const ExecutionContext *exe_ctx,
1608 const Address *addr, Stream &s) {
1609 FormatEntity::Entry format_entry;
1610
1611 if (format == nullptr) {
1612 if (exe_ctx != nullptr && exe_ctx->HasTargetScope()) {
1613 format_entry =
1614 exe_ctx->GetTargetRef().GetDebugger().GetDisassemblyFormat();
1615 format = &format_entry;
1616 }
1617 if (format == nullptr) {
1618 FormatEntity::Parse("${addr}: ", format_entry);
1619 format = &format_entry;
1620 }
1621 }
1622 bool function_changed = false;
1623 bool initial_function = false;
1624 if (prev_sc && (prev_sc->function || prev_sc->symbol)) {
1625 if (sc && (sc->function || sc->symbol)) {
1626 if (prev_sc->symbol && sc->symbol) {
1627 if (!sc->symbol->Compare(prev_sc->symbol->GetName(),
1628 prev_sc->symbol->GetType())) {
1629 function_changed = true;
1630 }
1631 } else if (prev_sc->function && sc->function) {
1632 if (prev_sc->function->GetMangled() != sc->function->GetMangled()) {
1633 function_changed = true;
1634 }
1635 }
1636 }
1637 }
1638 // The first context on a list of instructions will have a prev_sc that has
1639 // no Function or Symbol -- if SymbolContext had an IsValid() method, it
1640 // would return false. But we do get a prev_sc pointer.
1641 if ((sc && (sc->function || sc->symbol)) && prev_sc &&
1642 (prev_sc->function == nullptr && prev_sc->symbol == nullptr)) {
1643 initial_function = true;
1644 }
1645 return FormatEntity::Formatter(sc, exe_ctx, addr, function_changed,
1646 initial_function)
1647 .Format(*format, s);
1648}

1740static void PrivateReportDiagnostic(Debugger &debugger, Severity severity,
1741 std::string message,
1742 bool debugger_specific) {
1743 uint32_t event_type = 0;
1744 switch (severity) {
1745 case eSeverityInfo:
1746 assert(false && "eSeverityInfo should not be broadcast");
1747 return;
1748 case eSeverityWarning:
1749 event_type = lldb::eBroadcastBitWarning;
1750 break;
1751 case eSeverityError:
1752 event_type = lldb::eBroadcastBitError;
1753 break;
1754 }
1755

1758 // Diagnostics are too important to drop. If nobody is listening, print the
1759 // diagnostic directly to the debugger's error stream.
1760 DiagnosticEventData event_data(severity, std::move(message),
1761 debugger_specific);
1762 event_data.Dump(debugger.GetAsyncErrorStream().get());
1763 return;
1764 }
1765 EventSP event_sp = std::make_shared<Event>(
1766 event_type,
1767 new DiagnosticEventData(severity, std::move(message), debugger_specific));
1768 broadcaster.BroadcastEvent(event_sp);

1771void Debugger::ReportDiagnosticImpl(Severity severity, std::string message,
1772 std::optional<lldb::user_id_t> debugger_id,
1773 std::once_flag *once) {
1774 auto ReportDiagnosticLambda = [&]() {
1775 // Always log diagnostics to the system log.
1776 Host::SystemLog(severity, message);
1777
1778 // The diagnostic subsystem is optional but we still want to broadcast
1779 // events when it's disabled.
1780 if (Diagnostics::Enabled())
1781 Diagnostics::Instance().Report(message);
1782
1783 // We don't broadcast info events.
1784 if (severity == lldb::eSeverityInfo)
1785 return;
1786
1787 // Check if this diagnostic is for a specific debugger.
1788 if (debugger_id) {
1789 // It is debugger specific, grab it and deliver the event if the debugger
1790 // still exists.
1791 DebuggerSP debugger_sp = FindDebuggerWithID(*debugger_id);
1792 if (debugger_sp)
1793 PrivateReportDiagnostic(*debugger_sp, severity, std::move(message),
1794 true);
1795 return;
1796 }
1797 // The diagnostic event is not debugger specific, iterate over all debuggers
1798 // and deliver a diagnostic event to each one.
1799 if (g_debugger_list_ptr && g_debugger_list_mutex_ptr) {
1800 std::lock_guard<std::recursive_mutex> guard(*g_debugger_list_mutex_ptr);
1801 for (const auto &debugger : *g_debugger_list_ptr)
1802 PrivateReportDiagnostic(*debugger, severity, message, false);
1803 }
1804 };
1805
1806 if (once)
1807 std::call_once(*once, ReportDiagnosticLambda);
1808 else
1809 ReportDiagnosticLambda();
1810}

1901 if (log_options == 0)
1902 log_options = LLDB_LOG_OPTION_PREPEND_THREAD_NAME;

1905 error_stream);

2316 ProgressReport progress_report{data->GetID(), data->GetCompleted(),

2321
2322 // Do some bookkeeping regardless of whether we're going to display

2328 const bool complete = data->GetCompleted() == data->GetTotal();
