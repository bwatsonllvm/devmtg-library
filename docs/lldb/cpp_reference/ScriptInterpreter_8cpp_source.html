186Status ScriptInterpreter::SetBreakpointCommandCallback(
187 std::vector<std::reference_wrapper<BreakpointOptions>> &bp_options_vec,
188 const char *callback_text) {
189 Status error;
190 for (BreakpointOptions &bp_options : bp_options_vec) {
191 error = SetBreakpointCommandCallback(bp_options, callback_text,
192 /*is_callback=*/false);
193 if (!error.Success())
194 break;
195 }
196 return error;
197}

217static void ReadThreadBytesReceived(void *baton, const void *src,
218 size_t src_len) {
219 if (src && src_len) {
220 Stream *strm = (Stream *)baton;
221 strm->Write(src, src_len);
222 strm->Flush();
223 }

226llvm::Expected<std::unique_ptr<ScriptInterpreterIORedirect>>
227ScriptInterpreterIORedirect::Create(bool enable_io, Debugger &debugger,
228 CommandReturnObject *result) {
229 if (enable_io)
230 return std::unique_ptr<ScriptInterpreterIORedirect>(
231 new ScriptInterpreterIORedirect(debugger, result));
232
233 auto nullin = FileSystem::Instance().Open(FileSpec(FileSystem::DEV_NULL),
234 File::eOpenOptionReadOnly);
235 if (!nullin)
236 return nullin.takeError();
237
238 auto nullout = FileSystem::Instance().Open(FileSpec(FileSystem::DEV_NULL),
239 File::eOpenOptionWriteOnly);
240 if (!nullout)
241 return nullin.takeError();
242
243 return std::unique_ptr<ScriptInterpreterIORedirect>(
244 new ScriptInterpreterIORedirect(std::move(*nullin), std::move(*nullout)));
245}

247ScriptInterpreterIORedirect::ScriptInterpreterIORedirect(
248 std::unique_ptr<File> input, std::unique_ptr<File> output)
249 : m_input_file_sp(std::move(input)),
250 m_output_file_sp(std::make_shared<LockableStreamFile>(std::move(output),
251 m_output_mutex)),
252 m_error_file_sp(m_output_file_sp),
253 m_communication("lldb.ScriptInterpreterIORedirect.comm"),
254 m_disconnect(false) {}
255
256ScriptInterpreterIORedirect::ScriptInterpreterIORedirect(
257 Debugger &debugger, CommandReturnObject *result)
258 : m_communication("lldb.ScriptInterpreterIORedirect.comm"),
259 m_disconnect(false) {
260
261 if (result) {
262 m_input_file_sp = debugger.GetInputFileSP();
263
264 Pipe pipe;
265 Status pipe_result = pipe.CreateNew();
266#if defined(_WIN32)
267 lldb::file_t read_file = pipe.GetReadNativeHandle();
268 pipe.ReleaseReadFileDescriptor();
269 std::unique_ptr<ConnectionGenericFile> conn_up =
270 std::make_unique<ConnectionGenericFile>(read_file, true);
271#else
272 std::unique_ptr<ConnectionFileDescriptor> conn_up =
273 std::make_unique<ConnectionFileDescriptor>(
274 pipe.ReleaseReadFileDescriptor(), true);
275#endif
276
277 if (conn_up->IsConnected()) {
278 m_communication.SetConnection(std::move(conn_up));
279 m_communication.SetReadThreadBytesReceivedCallback(
280 ReadThreadBytesReceived, &result->GetOutputStream());
281 m_communication.StartReadThread();
282 m_disconnect = true;
283
284 FILE *outfile_handle = fdopen(pipe.ReleaseWriteFileDescriptor(), "w");
285 m_output_file_sp = std::make_shared<LockableStreamFile>(
286 std::make_shared<StreamFile>(outfile_handle, NativeFile::Owned),
287 m_output_mutex);
288 m_error_file_sp = m_output_file_sp;
289 if (outfile_handle)
290 ::setbuf(outfile_handle, nullptr);
291
292 result->SetImmediateOutputFile(debugger.GetOutputFileSP());
293 result->SetImmediateErrorFile(debugger.GetErrorFileSP());
294 }
295 }
296
297 if (!m_input_file_sp || !m_output_file_sp || !m_error_file_sp)
298 debugger.AdoptTopIOHandlerFilesIfInvalid(m_input_file_sp, m_output_file_sp,
299 m_error_file_sp);
300}

302void ScriptInterpreterIORedirect::Flush() {
303 if (m_output_file_sp)
304 m_output_file_sp->Lock().Flush();
305 if (m_error_file_sp)
306 m_error_file_sp->Lock().Flush();
307}

309ScriptInterpreterIORedirect::~ScriptInterpreterIORedirect() {
310 if (!m_disconnect)
311 return;
312
313 assert(m_output_file_sp);
314 assert(m_error_file_sp);
315 assert(m_output_file_sp == m_error_file_sp);
316
317 // Close the write end of the pipe since we are done with our one line
318 // script. This should cause the read thread that output_comm is using to
319 // exit.
320 m_output_file_sp->GetUnlockedFile().Close();
321 // The close above should cause this thread to exit when it gets to the end
322 // of file, so let it get all its data.
323 m_communication.JoinReadThread();
324 // Now we can close the read end of the pipe.
325 m_communication.Disconnect();
326}
