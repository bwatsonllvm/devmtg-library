149uint64_t SBSection::GetFileOffset() {
150 LLDB_INSTRUMENT_VA(this);
151
152 SectionSP section_sp(GetSP());
153 if (section_sp) {
154 ModuleSP module_sp(section_sp->GetModule());
155 if (module_sp) {
156 ObjectFile *objfile = module_sp->GetObjectFile();

244bool SBSection::GetDescription(SBStream &description) {
245 LLDB_INSTRUMENT_VA(this, description);
246
247 Stream &strm = description.ref();
248
249 SectionSP section_sp(GetSP());
250 if (section_sp) {
251 const addr_t file_addr = section_sp->GetFileAddress();
252 strm.Printf("[0x%16.16" PRIx64 "-0x%16.16" PRIx64 ") ", file_addr,
253 file_addr + section_sp->GetByteSize());
254 section_sp->DumpName(strm.AsRawOstream());
255 } else {
256 strm.PutCString("No value");
257 }
258
259 return true;
260}
