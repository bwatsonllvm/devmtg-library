31struct SharedCacheImageInfo {
32 SharedCacheImageInfo()
33 : m_filename(), m_uuid(), m_extractor_sp(),
34 m_create_data_extractor(nullptr), m_image_baton(nullptr) {}
35 SharedCacheImageInfo(ConstString filename, UUID uuid,
36 lldb::DataExtractorSP extractor_sp)
37 : m_filename(filename), m_uuid(uuid), m_extractor_sp(extractor_sp),
38 m_create_data_extractor(nullptr), m_image_baton(nullptr) {}
39 SharedCacheImageInfo(
40 ConstString filename, UUID uuid,
41 lldb::DataExtractorSP (*create_data_extractor)(void *image),
42 void *image_baton)
43 : m_filename(filename), m_uuid(uuid), m_extractor_sp(),
44 m_create_data_extractor(create_data_extractor),
45 m_image_baton(image_baton) {}
46
47 lldb::DataExtractorSP GetExtractor() {
48 if (!m_extractor_sp && m_image_baton)
49 m_extractor_sp = m_create_data_extractor(m_image_baton);
50 return m_extractor_sp;
51 }
52 ConstString GetFilename() const { return m_filename; }
53 const UUID &GetUUID() const { return m_uuid; }
54 void *GetImageBaton();
55 void SetExtractor(lldb::DataExtractorSP extractor_sp) {
56 m_extractor_sp = extractor_sp;
57 }
58 void SetImageBaton(void *image_baton) { m_image_baton = image_baton; }
59 void SetDataExtractorCreateFunction(
60 lldb::DataExtractorSP (*create_data_extractor)(void *image));
61
62private:
63 ConstString m_filename;
64 UUID m_uuid;
65 lldb::DataExtractorSP m_extractor_sp;
66 lldb::DataExtractorSP (*m_create_data_extractor)(void *image);
67 void *m_image_baton;
68};
