273 struct FieldInfo {
274 /// Size in bits that this field occupies. Can but
275 /// need not be the DW_AT_bit_size of the field.
276 uint64_t bit_size = 0;
277
278 /// Offset of this field in bits from the beginning
279 /// of the containing struct. Can but need not
280 /// be the DW_AT_data_bit_offset of the field.
281 uint64_t bit_offset = 0;
282
283 /// In case this field is folded into the storage
284 /// of a previous member's storage (for example
285 /// with [[no_unique_address]]), the effective field
286 /// end is the offset in bits from the beginning of
287 /// the containing struct where the field we were
288 /// folded into ended.
289 std::optional<uint64_t> effective_field_end;
290
291 /// Set to 'true' if this field is a bit-field.
292 bool is_bitfield = false;
293
294 /// Set to 'true' if this field is DW_AT_artificial.
295 bool is_artificial = false;
296
297 FieldInfo() = default;
298
299 void SetIsBitfield(bool flag) { is_bitfield = flag; }
300 bool IsBitfield() const { return is_bitfield; }
301
302 void SetIsArtificial(bool flag) { is_artificial = flag; }
303 bool IsArtificial() const { return is_artificial; }
304
305 bool NextBitfieldOffsetIsValid(const uint64_t next_bit_offset) const {
306 // Any subsequent bitfields must not overlap and must be at a higher
307 // bit offset than any previous bitfield + size.
308 return (bit_size + bit_offset) <= next_bit_offset;
309 }
310
311 /// Returns the offset in bits of where the storage this field
312 /// occupies ends.
313 uint64_t GetFieldEnd() const { return bit_size + bit_offset; }
314
315 void SetEffectiveFieldEnd(uint64_t val) { effective_field_end = val; }
316
317 /// If this field was folded into storage of a previous field,
318 /// returns the offset in bits of where that storage ends. Otherwise,
319 /// returns the regular field end (see \ref GetFieldEnd).
320 uint64_t GetEffectiveFieldEnd() const {
321 return effective_field_end.value_or(GetFieldEnd());
322 }
323 };

364 /// Tries to detect whether \ref class_clang_type contained an unnamed
365 /// bit-field between \ref previous_field and \ref current_field, and if
366 /// so, adds a clang::FieldDecl representing that bit-field to

368 ///
369 /// This is necessary because Clang (and GCC) doesn't emit a DW_TAG_member

373 ///

375 /// current parser. Will contain an entry for
376 /// the unnamed bit-field if this function created
377 /// one.
378 ///
379 /// \param[in] class_clang_type The RecordType to which the unnamed bit-field
380 /// will be added (if any).

382 /// \param[in] previous_field FieldInfo of the previous DW_TAG_member

384 ///
385 /// \param[in] current_field FieldInfo of the current DW_TAG_member
386 /// being parsed.
387 ///
