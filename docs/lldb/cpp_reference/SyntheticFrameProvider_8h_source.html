25/// This struct contains the metadata needed to instantiate a frame provider
26/// and optional filters to control which threads it applies to.
27struct ScriptedFrameProviderDescriptor {
28 /// Metadata for instantiating the provider (e.g. script class name and args).
29 lldb::ScriptedMetadataSP scripted_metadata_sp;
30
31 /// Interface for calling static methods on the provider class.
32 lldb::ScriptedFrameProviderInterfaceSP interface_sp;
33
34 /// Optional list of thread specifications to which this provider applies.
35 /// If empty, the provider applies to all threads. A thread matches if it
36 /// satisfies ANY of the specs in this vector (OR logic).
37 std::vector<ThreadSpec> thread_specs;
38
39 ScriptedFrameProviderDescriptor() = default;
40
41 ScriptedFrameProviderDescriptor(lldb::ScriptedMetadataSP metadata_sp)
42 : scripted_metadata_sp(metadata_sp) {}
43
44 ScriptedFrameProviderDescriptor(lldb::ScriptedMetadataSP metadata_sp,
45 const std::vector<ThreadSpec> &specs)
46 : scripted_metadata_sp(metadata_sp), thread_specs(specs) {}
47
48 /// Get the name of this descriptor (the scripted class name).
49 llvm::StringRef GetName() const {
50 return scripted_metadata_sp ? scripted_metadata_sp->GetClassName() : "";
51 }
52
53 /// Get the description of this frame provider.
54 ///
55 /// \return A string describing what this frame provider does, or an
56 /// empty string if no description is available.
57 std::string GetDescription() const;
58
59 /// Get the priority of this frame provider.
60 ///
61 /// Priority determines the order in which providers are evaluated when
62 /// multiple providers could apply to the same thread. Lower numbers indicate
63 /// higher priority (like Unix nice values).
64 ///
65 /// \return Priority value where 0 is highest priority, or std::nullopt for
66 /// default priority (UINT32_MAX - lowest priority).
67 std::optional<uint32_t> GetPriority() const;
68
69 /// Check if this descriptor applies to the given thread.
70 bool AppliesToThread(Thread &thread) const {
71 // If no thread specs specified, applies to all threads.
72 if (thread_specs.empty())
73 return true;
74
75 // Check if the thread matches any of the specs (OR logic).
76 for (const auto &spec : thread_specs) {
77 if (spec.ThreadPassesBasicTests(thread))
78 return true;
79 }
80 return false;
81 }
82
83 /// Check if this descriptor has valid metadata for script-based providers.
84 bool IsValid() const { return scripted_metadata_sp != nullptr; }
85
86 /// Get a unique identifier for this descriptor based on its contents.
87 /// The ID is computed from the class name and arguments dictionary,
88 /// not from the pointer address, so two descriptors with the same
89 /// contents will have the same ID.
90 uint32_t GetID() const;
91
92 /// Dump a description of this descriptor to the given stream.
93 void Dump(Stream *s) const;
94};

96/// Base class for all synthetic frame providers.
97///
98/// Synthetic frame providers allow modifying or replacing the stack frames
99/// shown for a thread. This is useful for:
100/// - Providing frames for custom calling conventions or languages.
101/// - Reconstructing missing frames from crash dumps or core files.
102/// - Adding diagnostic or synthetic frames for debugging.
103/// - Visualizing state machines or async execution contexts.
104class SyntheticFrameProvider : public PluginInterface {
105public:
106 /// Try to create a SyntheticFrameProvider instance for the given input
107 /// frames and descriptor.
108 ///
109 /// This method iterates through all registered SyntheticFrameProvider
110 /// plugins and returns the first one that can handle the given descriptor.
111 ///
112 /// \param[in] input_frames
113 /// The input stack frame list that this provider will transform.
114 /// This could be real unwound frames or output from another provider.
115 ///
116 /// \param[in] descriptor
117 /// The descriptor containing metadata for the provider.
118 ///
119 /// \return
120 /// A shared pointer to a SyntheticFrameProvider if one could be created,
121 /// otherwise an \a llvm::Error.
122 static llvm::Expected<lldb::SyntheticFrameProviderSP>
123 CreateInstance(lldb::StackFrameListSP input_frames,
124 const ScriptedFrameProviderDescriptor &descriptor);
125
126 /// Try to create a SyntheticFrameProvider instance for the given input
127 /// frames using a specific C++ plugin.
128 ///
129 /// This method directly invokes a specific SyntheticFrameProvider plugin
130 /// by name, bypassing the descriptor-based plugin iteration. This is useful
131 /// for C++ plugins that don't require scripted metadata.
132 ///
133 /// \param[in] input_frames
134 /// The input stack frame list that this provider will transform.
135 /// This could be real unwound frames or output from another provider.
136 ///
137 /// \param[in] plugin_name
138 /// The name of the plugin to use for creating the provider.
139 ///
140 /// \param[in] thread_specs
141 /// Optional list of thread specifications to which this provider applies.
142 /// If empty, the provider applies to all threads.
143 ///
144 /// \return
145 /// A shared pointer to a SyntheticFrameProvider if one could be created,
146 /// otherwise an \a llvm::Error.
147 static llvm::Expected<lldb::SyntheticFrameProviderSP>
148 CreateInstance(lldb::StackFrameListSP input_frames,
149 llvm::StringRef plugin_name,
150 const std::vector<ThreadSpec> &thread_specs = {});
151
152 ~SyntheticFrameProvider() override;
153
154 virtual std::string GetDescription() const = 0;
155
156 /// Get the priority of this frame provider.
157 ///
158 /// Priority determines the order in which providers are evaluated when
159 /// multiple providers could apply to the same thread. Lower numbers indicate
160 /// higher priority (like Unix nice values).
161 ///
162 /// \return
163 /// Priority value where 0 is highest priority, or std::nullopt for
164 /// default priority (UINT32_MAX - lowest priority).
165 virtual std::optional<uint32_t> GetPriority() const { return std::nullopt; }
166
167 /// Get a single stack frame at the specified index.
168 ///
169 /// This method is called lazily - frames are only created when requested.
170 /// The provider can access its input frames via GetInputFrames() if needed.
171 ///
172 /// \param[in] idx
173 /// The index of the frame to create.
174 ///
175 /// \return
176 /// An Expected containing the StackFrameSP if successful. Returns an
177 /// error when the index is beyond the last frame to signal the end of
178 /// the frame list.
179 virtual llvm::Expected<lldb::StackFrameSP> GetFrameAtIndex(uint32_t idx) = 0;
180
181 /// Get the thread associated with this provider.
182 Thread &GetThread() { return m_input_frames->GetThread(); }
183
184 /// Get the input frames that this provider transforms.
185 lldb::StackFrameListSP GetInputFrames() const { return m_input_frames; }
186
187protected:
188 SyntheticFrameProvider(lldb::StackFrameListSP input_frames);
189
190 lldb::StackFrameListSP m_input_frames;
191};
