100/// These are the three values deciding instruction control flow kind.
101/// InstructionLengthDecode function decodes an instruction and get this struct.
102///
103/// primary_opcode
104/// Primary opcode of the instruction.
105/// For one-byte opcode instruction, it's the first byte after prefix.
106/// For two- and three-byte opcodes, it's the second byte.
107///
108/// opcode_len
109/// The length of opcode in bytes. Valid opcode lengths are 1, 2, or 3.
110///
111/// modrm
112/// ModR/M byte of the instruction.
113/// Bits[7:6] indicate MOD. Bits[5:3] specify a register and R/M bits[2:0]
114/// may contain a register or specify an addressing mode, depending on MOD.
115struct InstructionOpcodeAndModrm {
116 uint8_t primary_opcode;
117 uint8_t opcode_len;
118 uint8_t modrm;
119};
120
121/// Determine the InstructionControlFlowKind based on opcode and modrm bytes.
122/// Refer to http://ref.x86asm.net/coder.html for the full list of opcode and
123/// instruction set.
124///
125/// \param[in] opcode_and_modrm
126/// Contains primary_opcode byte, its length, and ModR/M byte.
127/// Refer to the struct InstructionOpcodeAndModrm for details.
128///
129/// \return
130/// The control flow kind of the instruction or
131/// eInstructionControlFlowKindOther if the instruction doesn't affect
132/// the control flow of the program.
133lldb::InstructionControlFlowKind
134MapOpcodeIntoControlFlowKind(InstructionOpcodeAndModrm opcode_and_modrm) {
135 uint8_t opcode = opcode_and_modrm.primary_opcode;
136 uint8_t opcode_len = opcode_and_modrm.opcode_len;
137 uint8_t modrm = opcode_and_modrm.modrm;
138
139 if (opcode_len > 2)
140 return lldb::eInstructionControlFlowKindOther;
141
142 if (opcode >= 0x70 && opcode <= 0x7F) {
143 if (opcode_len == 1)
144 return lldb::eInstructionControlFlowKindCondJump;
145 else
146 return lldb::eInstructionControlFlowKindOther;
147 }
148
149 if (opcode >= 0x80 && opcode <= 0x8F) {
150 if (opcode_len == 2)
151 return lldb::eInstructionControlFlowKindCondJump;
152 else
153 return lldb::eInstructionControlFlowKindOther;
154 }
155
156 switch (opcode) {
157 case 0x9A:
158 if (opcode_len == 1)
159 return lldb::eInstructionControlFlowKindFarCall;
160 break;
161 case 0xFF:
162 if (opcode_len == 1) {
163 uint8_t modrm_reg = (modrm >> 3) & 7;
164 if (modrm_reg == 2)
165 return lldb::eInstructionControlFlowKindCall;
166 else if (modrm_reg == 3)
167 return lldb::eInstructionControlFlowKindFarCall;
168 else if (modrm_reg == 4)
169 return lldb::eInstructionControlFlowKindJump;
170 else if (modrm_reg == 5)
171 return lldb::eInstructionControlFlowKindFarJump;
172 }
173 break;
174 case 0xE8:
175 if (opcode_len == 1)
176 return lldb::eInstructionControlFlowKindCall;
177 break;
178 case 0xCD:
179 case 0xCC:
180 case 0xCE:
181 case 0xF1:
182 if (opcode_len == 1)
183 return lldb::eInstructionControlFlowKindFarCall;
184 break;
185 case 0xCF:
186 if (opcode_len == 1)
187 return lldb::eInstructionControlFlowKindFarReturn;
188 break;
189 case 0xE9:
190 case 0xEB:
191 if (opcode_len == 1)
192 return lldb::eInstructionControlFlowKindJump;
193 break;
194 case 0xEA:
195 if (opcode_len == 1)
196 return lldb::eInstructionControlFlowKindFarJump;
197 break;
198 case 0xE3:
199 case 0xE0:
200 case 0xE1:
201 case 0xE2:
202 if (opcode_len == 1)
203 return lldb::eInstructionControlFlowKindCondJump;
204 break;
205 case 0xC3:
206 case 0xC2:
207 if (opcode_len == 1)
208 return lldb::eInstructionControlFlowKindReturn;
209 break;
210 case 0xCB:
211 case 0xCA:
212 if (opcode_len == 1)
213 return lldb::eInstructionControlFlowKindFarReturn;
214 break;
215 case 0x05:
216 case 0x34:
217 if (opcode_len == 2)
218 return lldb::eInstructionControlFlowKindFarCall;
219 break;
220 case 0x35:
221 case 0x07:
222 if (opcode_len == 2)
223 return lldb::eInstructionControlFlowKindFarReturn;
224 break;
225 case 0x01:
226 if (opcode_len == 2) {
227 switch (modrm) {
228 case 0xc1:
229 return lldb::eInstructionControlFlowKindFarCall;
230 case 0xc2:
231 case 0xc3:
232 return lldb::eInstructionControlFlowKindFarReturn;
233 default:
234 break;
235 }
236 }
237 break;
238 default:
239 break;
240 }
241
242 return lldb::eInstructionControlFlowKindOther;
243}

245/// Decode an instruction into opcode, modrm and opcode_len.
246/// Refer to http://ref.x86asm.net/coder.html for the instruction bytes layout.
247/// Opcodes in x86 are generally the first byte of instruction, though two-byte
248/// instructions and prefixes exist. ModR/M is the byte following the opcode
249/// and adds additional information for how the instruction is executed.
250///
251/// \param[in] inst_bytes
252/// Raw bytes of the instruction
253///
254///
255/// \param[in] bytes_len
256/// The length of the inst_bytes array.
257///
258/// \param[in] is_exec_mode_64b
259/// If true, the execution mode is 64 bit.
260///
261/// \return
262/// Returns decoded instruction as struct InstructionOpcodeAndModrm, holding
263/// primary_opcode, opcode_len and modrm byte. Refer to the struct definition
264/// for more details.
265/// Otherwise if the given instruction is invalid, returns std::nullopt.
266std::optional<InstructionOpcodeAndModrm>
267InstructionLengthDecode(const uint8_t *inst_bytes, int bytes_len,
268 bool is_exec_mode_64b) {
269 int op_idx = 0;
270 bool prefix_done = false;
271 InstructionOpcodeAndModrm ret = {0, 0, 0};
272
273 // In most cases, the primary_opcode is the first byte of the instruction
274 // but some instructions have a prefix to be skipped for these calculations.
275 // The following mapping is inspired from libipt's instruction decoding logic
276 // in `src/pt_ild.c`
277 while (!prefix_done) {
278 if (op_idx >= bytes_len)
279 return std::nullopt;
280
281 ret.primary_opcode = inst_bytes[op_idx];
282 switch (ret.primary_opcode) {
283 // prefix_ignore
284 case 0x26:
285 case 0x2e:
286 case 0x36:
287 case 0x3e:
288 case 0x64:
289 case 0x65:
290 // prefix_osz, prefix_asz
291 case 0x66:
292 case 0x67:
293 // prefix_lock, prefix_f2, prefix_f3
294 case 0xf0:
295 case 0xf2:
296 case 0xf3:
297 op_idx++;
298 break;
299
300 // prefix_rex
301 case 0x40:
302 case 0x41:
303 case 0x42:
304 case 0x43:
305 case 0x44:
306 case 0x45:
307 case 0x46:
308 case 0x47:
309 case 0x48:
310 case 0x49:
311 case 0x4a:
312 case 0x4b:
313 case 0x4c:
314 case 0x4d:
315 case 0x4e:
316 case 0x4f:
317 if (is_exec_mode_64b)
318 op_idx++;
319 else
320 prefix_done = true;
321 break;
322
323 // prefix_vex_c4, c5
324 case 0xc5:
325 if (!is_exec_mode_64b && (inst_bytes[op_idx + 1] & 0xc0) != 0xc0) {
326 prefix_done = true;
327 break;
328 }
329
330 ret.opcode_len = 2;
331 ret.primary_opcode = inst_bytes[op_idx + 2];
332 ret.modrm = inst_bytes[op_idx + 3];
333 return ret;
334
335 case 0xc4:
336 if (!is_exec_mode_64b && (inst_bytes[op_idx + 1] & 0xc0) != 0xc0) {
337 prefix_done = true;
338 break;
339 }
340 ret.opcode_len = inst_bytes[op_idx + 1] & 0x1f;
341 ret.primary_opcode = inst_bytes[op_idx + 3];
342 ret.modrm = inst_bytes[op_idx + 4];
343 return ret;
344
345 // prefix_evex
346 case 0x62:
347 if (!is_exec_mode_64b && (inst_bytes[op_idx + 1] & 0xc0) != 0xc0) {
348 prefix_done = true;
349 break;
350 }
351 ret.opcode_len = inst_bytes[op_idx + 1] & 0x03;
352 ret.primary_opcode = inst_bytes[op_idx + 4];
353 ret.modrm = inst_bytes[op_idx + 5];
354 return ret;
355
356 default:
357 prefix_done = true;
358 break;
359 }
360 } // prefix done
361
362 ret.primary_opcode = inst_bytes[op_idx];
363 ret.modrm = inst_bytes[op_idx + 1];
364 ret.opcode_len = 1;
365
366 // If the first opcode is 0F, it's two- or three- byte opcodes.
367 if (ret.primary_opcode == 0x0F) {
368 ret.primary_opcode = inst_bytes[++op_idx]; // get the next byte
369
370 if (ret.primary_opcode == 0x38) {
371 ret.opcode_len = 3;
372 ret.primary_opcode = inst_bytes[++op_idx]; // get the next byte
373 ret.modrm = inst_bytes[op_idx + 1];
374 } else if (ret.primary_opcode == 0x3A) {
375 ret.opcode_len = 3;
376 ret.primary_opcode = inst_bytes[++op_idx];
377 ret.modrm = inst_bytes[op_idx + 1];
378 } else if ((ret.primary_opcode & 0xf8) == 0x38) {
379 ret.opcode_len = 0;
380 ret.primary_opcode = inst_bytes[++op_idx];
381 ret.modrm = inst_bytes[op_idx + 1];
382 } else if (ret.primary_opcode == 0x0F) {
383 ret.opcode_len = 3;
384 // opcode is 0x0F, no needs to update
385 ret.modrm = inst_bytes[op_idx + 1];
386 } else {
387 ret.opcode_len = 2;
388 ret.modrm = inst_bytes[op_idx + 1];
389 }
390 }
391
392 return ret;
393}

395lldb::InstructionControlFlowKind GetControlFlowKind(bool is_exec_mode_64b,
396 Opcode m_opcode) {
397 std::optional<InstructionOpcodeAndModrm> ret;
398
399 if (m_opcode.GetOpcodeBytes() == nullptr || m_opcode.GetByteSize() <= 0) {
400 // x86_64 and i386 instructions are categorized as Opcode::Type::eTypeBytes
401 return lldb::eInstructionControlFlowKindUnknown;
402 }
403
404 // Opcode bytes will be decoded into primary_opcode, modrm and opcode length.
405 // These are the three values deciding instruction control flow kind.
406 ret = InstructionLengthDecode((const uint8_t *)m_opcode.GetOpcodeBytes(),
407 m_opcode.GetByteSize(), is_exec_mode_64b);
408 if (!ret)
409 return lldb::eInstructionControlFlowKindUnknown;
410 else
411 return MapOpcodeIntoControlFlowKind(*ret);
412}

457 size_t Decode(const lldb_private::Disassembler &disassembler,
458 const lldb_private::DataExtractor &data,
459 lldb::offset_t data_offset) override {
460 // All we have to do is read the opcode which can be easy for some
461 // architectures
462 bool got_op = false;
463 DisassemblerScope disasm(*this);
464 if (disasm) {
465 const ArchSpec &arch = disasm->GetArchitecture();
466 const lldb::ByteOrder byte_order = data.GetByteOrder();
467
468 const uint32_t min_op_byte_size = arch.GetMinimumOpcodeByteSize();
469 const uint32_t max_op_byte_size = arch.GetMaximumOpcodeByteSize();
470 if (min_op_byte_size == max_op_byte_size) {
471 // Fixed size instructions, just read that amount of data.
472 if (!data.ValidOffsetForDataOfSize(data_offset, min_op_byte_size))
473 return false;
474
475 switch (min_op_byte_size) {
476 case 1:
477 m_opcode.SetOpcode8(data.GetU8(&data_offset), byte_order);
478 got_op = true;
479 break;
480
481 case 2:
482 m_opcode.SetOpcode16(data.GetU16(&data_offset), byte_order);
483 got_op = true;
484 break;
485
486 case 4:
487 m_opcode.SetOpcode32(data.GetU32(&data_offset), byte_order);
488 got_op = true;
489 break;
490
491 case 8:
492 m_opcode.SetOpcode64(data.GetU64(&data_offset), byte_order);
493 got_op = true;
494 break;
495
496 default:
497 if (arch.GetTriple().isRISCV())
498 m_opcode.SetOpcode16_32TupleBytes(
499 data.PeekData(data_offset, min_op_byte_size), min_op_byte_size,
500 byte_order);
501 else
502 m_opcode.SetOpcodeBytes(
503 data.PeekData(data_offset, min_op_byte_size), min_op_byte_size);
504 got_op = true;
505 break;
506 }
507 }
508 if (!got_op) {
509 bool is_alternate_isa = false;
510 DisassemblerLLVMC::MCDisasmInstance *mc_disasm_ptr =
511 GetDisasmToUse(is_alternate_isa, disasm);
512
513 const llvm::Triple::ArchType machine = arch.GetMachine();
514 if (machine == llvm::Triple::arm || machine == llvm::Triple::thumb) {
515 if (machine == llvm::Triple::thumb || is_alternate_isa) {
516 uint32_t thumb_opcode = data.GetU16(&data_offset);
517 if ((thumb_opcode & 0xe000) != 0xe000 ||
518 ((thumb_opcode & 0x1800u) == 0)) {
519 m_opcode.SetOpcode16(thumb_opcode, byte_order);
520 m_is_valid = true;
521 } else {
522 thumb_opcode <<= 16;
523 thumb_opcode |= data.GetU16(&data_offset);
524 m_opcode.SetOpcode16_2(thumb_opcode, byte_order);
525 m_is_valid = true;
526 }
527 } else {
528 m_opcode.SetOpcode32(data.GetU32(&data_offset), byte_order);
529 m_is_valid = true;
530 }
531 } else {
532 // The opcode isn't evenly sized, so we need to actually use the llvm
533 // disassembler to parse it and get the size.
534 uint8_t *opcode_data =
535 const_cast<uint8_t *>(data.PeekData(data_offset, 1));
536 const size_t opcode_data_len = data.BytesLeft(data_offset);
537 const addr_t pc = m_address.GetFileAddress();
538 llvm::MCInst inst;
539
540 uint64_t inst_size = 0;
541 m_is_valid = mc_disasm_ptr->GetMCInst(opcode_data, opcode_data_len,
542 pc, inst, inst_size);
543 m_opcode.Clear();
544 if (inst_size != 0) {
545 if (arch.GetTriple().isRISCV())
546 m_opcode.SetOpcode16_32TupleBytes(opcode_data, inst_size,
547 byte_order);
548 else
549 m_opcode.SetOpcodeBytes(opcode_data, inst_size);
550 }
551 }
552 }
553 return m_opcode.GetByteSize();
554 }
555 return 0;
556 }

580 void CalculateMnemonicOperandsAndComment(
581 const lldb_private::ExecutionContext *exe_ctx) override {
582 DataExtractor data;
583 const AddressClass address_class = GetAddressClass();
584
585 if (m_opcode.GetData(data)) {
586 std::string out_string;
587 std::string markup_out_string;
588 std::string comment_string;
589 std::string markup_comment_string;
590
591 DisassemblerScope disasm(*this, exe_ctx);
592 if (disasm) {
593 DisassemblerLLVMC::MCDisasmInstance *mc_disasm_ptr;
594
595 if (address_class == AddressClass::eCodeAlternateISA)
596 mc_disasm_ptr = disasm->m_alternate_disasm_up.get();
597 else
598 mc_disasm_ptr = disasm->m_disasm_up.get();
599
600 lldb::addr_t pc = m_address.GetFileAddress();
601 m_using_file_addr = true;
602
603 bool use_hex_immediates = true;
604 Disassembler::HexImmediateStyle hex_style = Disassembler::eHexStyleC;
605
606 if (exe_ctx) {
607 Target *target = exe_ctx->GetTargetPtr();
608 if (target) {
609 use_hex_immediates = target->GetUseHexImmediates();
610 hex_style = target->GetHexImmediateStyle();
611
612 const lldb::addr_t load_addr = m_address.GetLoadAddress(target);
613 if (load_addr != LLDB_INVALID_ADDRESS) {
614 pc = load_addr;
615 m_using_file_addr = false;
616 }
617 }
618 }
619
620 const uint8_t *opcode_data = data.GetDataStart();
621 const size_t opcode_data_len = data.GetByteSize();
622 llvm::MCInst inst;
623 uint64_t inst_size = 0;
624 bool valid = mc_disasm_ptr->GetMCInst(opcode_data, opcode_data_len, pc,
625 inst, inst_size);
626
627 if (valid && inst_size > 0) {
628 mc_disasm_ptr->SetStyle(use_hex_immediates, hex_style);
629
630 const bool saved_use_color = mc_disasm_ptr->GetUseColor();
631 mc_disasm_ptr->SetUseColor(false);
632 mc_disasm_ptr->PrintMCInst(inst, pc, out_string, comment_string);
633 mc_disasm_ptr->SetUseColor(true);
634 mc_disasm_ptr->PrintMCInst(inst, pc, markup_out_string,
635 markup_comment_string);
636 mc_disasm_ptr->SetUseColor(saved_use_color);
637
638 if (!comment_string.empty()) {
639 AppendComment(comment_string);
640 }
641 }
642
643 if (inst_size == 0) {
644 m_comment.assign("unknown opcode");
645 inst_size = m_opcode.GetByteSize();
646 StreamString mnemonic_strm;
647 lldb::offset_t offset = 0;
648 lldb::ByteOrder byte_order = data.GetByteOrder();
649 switch (inst_size) {
650 case 1: {
651 const uint8_t uval8 = data.GetU8(&offset);
652 m_opcode.SetOpcode8(uval8, byte_order);
653 m_opcode_name.assign(".byte");
654 mnemonic_strm.Printf("0x%2.2x", uval8);
655 } break;
656 case 2: {
657 const uint16_t uval16 = data.GetU16(&offset);
658 m_opcode.SetOpcode16(uval16, byte_order);
659 m_opcode_name.assign(".short");
660 mnemonic_strm.Printf("0x%4.4x", uval16);
661 } break;
662 case 4: {
663 const uint32_t uval32 = data.GetU32(&offset);
664 m_opcode.SetOpcode32(uval32, byte_order);
665 m_opcode_name.assign(".long");
666 mnemonic_strm.Printf("0x%8.8x", uval32);
667 } break;
668 case 8: {
669 const uint64_t uval64 = data.GetU64(&offset);
670 m_opcode.SetOpcode64(uval64, byte_order);
671 m_opcode_name.assign(".quad");
672 mnemonic_strm.Printf("0x%16.16" PRIx64, uval64);
673 } break;
674 default:
675 if (inst_size == 0)
676 return;
677 else {
678 const uint8_t *bytes = data.PeekData(offset, inst_size);
679 if (bytes == nullptr)
680 return;
681 m_opcode_name.assign(".byte");
682 m_opcode.SetOpcodeBytes(bytes, inst_size);
683 mnemonic_strm.Printf("0x%2.2x", bytes[0]);
684 for (uint32_t i = 1; i < inst_size; ++i)
685 mnemonic_strm.Printf(" 0x%2.2x", bytes[i]);
686 }
687 break;
688 }
689 m_mnemonics = std::string(mnemonic_strm.GetString());
690 return;
691 }

768 static std::pair<Operand, llvm::StringRef::const_iterator>
769 ParseRegisterName(llvm::StringRef::const_iterator osi,
770 llvm::StringRef::const_iterator ose) {
771 Operand ret;
772 ret.m_type = Operand::Type::Register;
773 std::string str;
774
775 osi = ConsumeWhitespace(osi, ose);
776
777 while (osi != ose) {
778 if (*osi >= '0' && *osi <= '9') {
779 if (str.empty()) {
780 return std::make_pair(Operand(), osi);
781 } else {
782 str.push_back(*osi);
783 }
784 } else if (*osi >= 'a' && *osi <= 'z') {
785 str.push_back(*osi);
786 } else {
787 switch (*osi) {
788 default:
789 if (str.empty()) {
790 return std::make_pair(Operand(), osi);
791 } else {
792 ret.m_register = ConstString(str);
793 return std::make_pair(ret, osi);
794 }
795 case '%':
796 if (!str.empty()) {
797 return std::make_pair(Operand(), osi);
798 }
799 break;
800 }
801 }
802 ++osi;
803 }
804
805 ret.m_register = ConstString(str);
806 return std::make_pair(ret, osi);

809 static std::pair<Operand, llvm::StringRef::const_iterator>
810 ParseImmediate(llvm::StringRef::const_iterator osi,
811 llvm::StringRef::const_iterator ose) {
812 Operand ret;
813 ret.m_type = Operand::Type::Immediate;
814 std::string str;
815 bool is_hex = false;
816
817 osi = ConsumeWhitespace(osi, ose);
818
819 while (osi != ose) {
820 if (*osi >= '0' && *osi <= '9') {
821 str.push_back(*osi);
822 } else if (*osi >= 'a' && *osi <= 'f') {
823 if (is_hex) {
824 str.push_back(*osi);
825 } else {
826 return std::make_pair(Operand(), osi);
827 }
828 } else {
829 switch (*osi) {
830 default:
831 if (str.empty()) {
832 return std::make_pair(Operand(), osi);
833 } else {
834 ret.m_immediate = strtoull(str.c_str(), nullptr, 0);
835 return std::make_pair(ret, osi);
836 }
837 case 'x':
838 if (str == "0") {
839 is_hex = true;
840 str.push_back(*osi);
841 } else {
842 return std::make_pair(Operand(), osi);
843 }
844 break;
845 case '#':
846 case '$':
847 if (!str.empty()) {
848 return std::make_pair(Operand(), osi);
849 }
850 break;
851 case '-':
852 if (str.empty()) {
853 ret.m_negative = true;
854 } else {
855 return std::make_pair(Operand(), osi);
856 }
857 }
858 }
859 ++osi;
860 }
861
862 ret.m_immediate = strtoull(str.c_str(), nullptr, 0);
863 return std::make_pair(ret, osi);
864 }

943 deref.m_type = Operand::Type::Dereference;
944 deref.m_children.push_back(index);
945 return std::make_pair(deref, osi);

1070 static void DumpOperand(const Operand &op, Stream &s) {
1071 switch (op.m_type) {
1072 case Operand::Type::Dereference:
1073 s.PutCString("*");
1074 DumpOperand(op.m_children[0], s);
1075 break;
1076 case Operand::Type::Immediate:
1077 if (op.m_negative) {
1078 s.PutCString("-");
1079 }
1080 s.PutCString(llvm::to_string(op.m_immediate));
1081 break;
1082 case Operand::Type::Invalid:
1083 s.PutCString("Invalid");
1084 break;
1085 case Operand::Type::Product:
1086 s.PutCString("(");
1087 DumpOperand(op.m_children[0], s);
1088 s.PutCString("*");
1089 DumpOperand(op.m_children[1], s);
1090 s.PutCString(")");
1091 break;
1092 case Operand::Type::Register:
1093 s.PutCString(op.m_register.GetStringRef());
1094 break;
1095 case Operand::Type::Sum:
1096 s.PutCString("(");
1097 DumpOperand(op.m_children[0], s);
1098 s.PutCString("+");
1099 DumpOperand(op.m_children[1], s);
1100 s.PutCString(")");
1101 break;

1105 bool ParseOperands(
1106 llvm::SmallVectorImpl<Instruction::Operand> &operands) override {
1107 const char *operands_string = GetOperands(nullptr);
1108
1109 if (!operands_string) {
1110 return false;
1111 }
1112
1113 llvm::StringRef operands_ref(operands_string);
1114
1115 llvm::StringRef::const_iterator osi = operands_ref.begin();
1116 llvm::StringRef::const_iterator ose = operands_ref.end();
1117
1118 while (osi != ose) {
1119 Operand operand;
1120 llvm::StringRef::const_iterator iter;
1121
1122 if ((std::tie(operand, iter) = ParseIntelIndexedAccess(osi, ose),
1123 operand.IsValid()) ||
1124 (std::tie(operand, iter) = ParseIntelDerefAccess(osi, ose),
1125 operand.IsValid()) ||
1126 (std::tie(operand, iter) = ParseARMOffsetAccess(osi, ose),
1127 operand.IsValid()) ||
1128 (std::tie(operand, iter) = ParseARMDerefAccess(osi, ose),
1129 operand.IsValid()) ||
1130 (std::tie(operand, iter) = ParseRegisterName(osi, ose),
1131 operand.IsValid()) ||
1132 (std::tie(operand, iter) = ParseImmediate(osi, ose),
1133 operand.IsValid())) {
1134 osi = iter;
1135 operands.push_back(operand);
1136 } else {
1137 return false;
1138 }
1139
1140 std::pair<bool, llvm::StringRef::const_iterator> found_and_iter =
1141 ConsumeChar(osi, ',', ose);
1142 if (found_and_iter.first) {
1143 osi = found_and_iter.second;
1144 }
1145
1146 osi = ConsumeWhitespace(osi, ose);
1147 }
1148
1149 DisassemblerSP disasm_sp = m_disasm_wp.lock();
1150
1151 if (disasm_sp && operands.size() > 1) {
1152 // TODO tie this into the MC Disassembler's notion of clobbers.
1153 switch (disasm_sp->GetArchitecture().GetMachine()) {
1154 default:
1155 break;
1156 case llvm::Triple::x86:
1157 case llvm::Triple::x86_64:
1158 operands[operands.size() - 1].m_clobbered = true;
1159 break;
1160 case llvm::Triple::arm:
1161 operands[0].m_clobbered = true;
1162 break;
1163 }
1164 }
1165
1166 if (Log *log = GetLog(LLDBLog::Process | LLDBLog::Disassembler)) {
1167 StreamString ss;
1168
1169 ss.Printf("[%s] expands to %zu operands:\n", operands_string,
1170 operands.size());
1171 for (const Operand &operand : operands) {
1172 ss.PutCString(" ");
1173 DumpOperand(operand, ss);
1174 ss.PutCString("\n");
1175 }
1176
1177 log->PutString(ss.GetString());
1178 }
1179
1180 return true;
1181 }

1189 std::weak_ptr<DisassemblerLLVMC> m_disasm_wp;

1192 bool m_using_file_addr = false;
1193 bool m_has_visited_instruction = false;

1195 // Be conservative. If we didn't understand the instruction, say it:
1196 // - Might branch
1197 // - Does not have a delay slot
1198 // - Is not a call
1199 // - Is not a load
1200 // - Is not an authenticated instruction
1201 bool m_does_branch = true;
1202 bool m_has_delay_slot = false;
1203 bool m_is_call = false;
1204 bool m_is_load = false;
1205 bool m_is_authenticated = false;
1206 bool m_is_barrier = false;

1208 void VisitInstruction() {
1209 if (m_has_visited_instruction)
1210 return;
1211
1212 DisassemblerScope disasm(*this);
1213 if (!disasm)
1214 return;
1215
1216 DataExtractor data;
1217 if (!m_opcode.GetData(data))
1218 return;
1219
1220 bool is_alternate_isa;
1221 lldb::addr_t pc = m_address.GetFileAddress();
1222 DisassemblerLLVMC::MCDisasmInstance *mc_disasm_ptr =
1223 GetDisasmToUse(is_alternate_isa, disasm);
1224 const uint8_t *opcode_data = data.GetDataStart();
1225 const size_t opcode_data_len = data.GetByteSize();
1226 llvm::MCInst inst;
1227 uint64_t inst_size = 0;
1228 const bool valid = mc_disasm_ptr->GetMCInst(opcode_data, opcode_data_len,
1229 pc, inst, inst_size);
1230 if (!valid)
1231 return;
1232
1233 m_has_visited_instruction = true;
1234 m_does_branch = mc_disasm_ptr->CanBranch(inst);
1235 m_has_delay_slot = mc_disasm_ptr->HasDelaySlot(inst);
1236 m_is_call = mc_disasm_ptr->IsCall(inst);
1237 m_is_load = mc_disasm_ptr->IsLoad(inst);
1238 m_is_authenticated = mc_disasm_ptr->IsAuthenticated(inst);
1239 m_is_barrier = mc_disasm_ptr->IsBarrier(inst);
1240 }

1316 triple, nullptr, DisassemblerLLVMC::SymbolLookupCallback, &owner,
1317 context_up.get(), std::move(rel_info_up)));
1318 disasm_up->setSymbolizer(std::move(symbolizer_up));
1319

1321 flavor == ~0U ? asm_info_up->getAssemblerDialect() : flavor;
1322

1331

1333 std::unique_ptr<llvm::MCInstrAnalysis> instr_analysis_up(
1334 curr_target->createMCInstrAnalysis(instr_info_up.get()));
1335
1336 return Instance(new MCDisasmInstance(
1337 std::move(instr_info_up), std::move(reg_info_up),
1338 std::move(subtarget_info_up), MCOptions, std::move(asm_info_up),
1339 std::move(context_up), std::move(disasm_up), std::move(instr_printer_up),
1340 std::move(instr_analysis_up)));

1451bool DisassemblerLLVMC::MCDisasmInstance::IsAuthenticated(
1452 llvm::MCInst &mc_inst) const {
1453 const auto &InstrDesc = m_instr_info_up->get(mc_inst.getOpcode());
1454
1455 // Treat software auth traps (brk 0xc470 + aut key, where 0x70 == 'p', 0xc4
1456 // == 'a' + 'c') as authenticated instructions for reporting purposes, in
1457 // addition to the standard authenticated instructions specified in ARMv8.3.
1458 bool IsBrkC47x = false;
1459 if (InstrDesc.isTrap() && mc_inst.getNumOperands() == 1) {
1460 const llvm::MCOperand &Op0 = mc_inst.getOperand(0);
1461 if (Op0.isImm() && Op0.getImm() >= 0xc470 && Op0.getImm() <= 0xc474)
1462 IsBrkC47x = true;
1463 }
1464
1465 return InstrDesc.isAuthenticated() || IsBrkC47x;

1468DisassemblerLLVMC::DisassemblerLLVMC(const ArchSpec &arch,
1469 const char *flavor_string,
1470 const char *cpu_string,
1471 const char *features_string)
1472 : Disassembler(arch, flavor_string), m_exe_ctx(nullptr), m_inst(nullptr),
1473 m_data_from_file(false), m_adrp_address(LLDB_INVALID_ADDRESS),

1475 if (!FlavorValidForArchSpec(arch, m_flavor.c_str())) {
1476 m_flavor.assign("default");
1477 }
1478
1479 const bool cpu_or_features_overriden = cpu_string || features_string;
1480 unsigned flavor = ~0U;
1481 llvm::Triple triple = arch.GetTriple();
1482
1483 // So far the only supported flavor is "intel" on x86. The base class will
1484 // set this correctly coming in.
1485 if (triple.getArch() == llvm::Triple::x86 ||
1486 triple.getArch() == llvm::Triple::x86_64) {
1487 if (m_flavor == "intel") {
1488 flavor = 1;
1489 } else if (m_flavor == "att") {
1490 flavor = 0;
1491 }
1492 }
1493
1494 ArchSpec thumb_arch(arch);
1495 if (triple.getArch() == llvm::Triple::arm) {
1496 std::string thumb_arch_name(thumb_arch.GetTriple().getArchName().str());
1497 // Replace "arm" with "thumb" so we get all thumb variants correct
1498 if (thumb_arch_name.size() > 3) {
1499 thumb_arch_name.erase(0, 3);
1500 thumb_arch_name.insert(0, "thumb");
1501 } else {
1502 thumb_arch_name = "thumbv9.3a";
1503 }
1504 thumb_arch.GetTriple().setArchName(llvm::StringRef(thumb_arch_name));
1505 }
1506
1507 // If no sub architecture specified then use the most recent arm architecture
1508 // so the disassembler will return all instructions. Without it we will see a
1509 // lot of unknown opcodes if the code uses instructions which are not
1510 // available in the oldest arm version (which is used when no sub architecture
1511 // is specified).

1513 triple.getSubArch() == llvm::Triple::NoSubArch)

1517 features_string ? std::string(features_string) : "";

1524 features_str += "+fp-armv8,";

1536 case ArchSpec::eCore_mips32r2el:
1537 cpu = "mips32r2";
1538 break;
1539 case ArchSpec::eCore_mips32r3:
1540 case ArchSpec::eCore_mips32r3el:

1546 break;
1547 case ArchSpec::eCore_mips32r6:
1548 case ArchSpec::eCore_mips32r6el:
1549 cpu = "mips32r6";
1550 break;
1551 case ArchSpec::eCore_mips64:
1552 case ArchSpec::eCore_mips64el:

1573 break;

1667size_t DisassemblerLLVMC::DecodeInstructions(const Address &base_addr,
1668 const DataExtractor &data,
1669 lldb::offset_t data_offset,
1670 size_t num_instructions,
1671 bool append, bool data_from_file) {
1672 if (!append)
1673 m_instruction_list.Clear();
1674
1675 if (!IsValid())
1676 return 0;
1677
1678 m_data_from_file = data_from_file;
1679 uint32_t data_cursor = data_offset;
1680 const size_t data_byte_size = data.GetByteSize();
1681 uint32_t instructions_parsed = 0;
1682 Address inst_addr(base_addr);
1683
1684 while (data_cursor < data_byte_size &&
1685 instructions_parsed < num_instructions) {
1686
1687 AddressClass address_class = AddressClass::eCode;
1688
1689 if (m_alternate_disasm_up)
1690 address_class = inst_addr.GetAddressClass();
1691
1692 InstructionSP inst_sp(
1693 new InstructionLLVMC(*this, inst_addr, address_class));
1694
1695 if (!inst_sp)
1696 break;
1697
1698 uint32_t inst_size = inst_sp->Decode(*this, data, data_cursor);
1699
1700 if (inst_size == 0)
1701 break;
1702
1703 m_instruction_list.Append(inst_sp);
1704 data_cursor += inst_size;
1705 inst_addr.Slide(inst_size);
1706 instructions_parsed++;
1707 }
1708
1709 return data_cursor - data_offset;
1710}

1770const char *DisassemblerLLVMC::SymbolLookup(uint64_t value, uint64_t *type_ptr,
1771 uint64_t pc, const char **name) {
1772 if (*type_ptr) {
1773 if (m_exe_ctx && m_inst) {
1774 // std::string remove_this_prior_to_checkin;
1775 Target *target = m_exe_ctx ? m_exe_ctx->GetTargetPtr() : nullptr;
1776 Address value_so_addr;
1777 Address pc_so_addr;
1778 if (target->GetArchitecture().GetMachine() == llvm::Triple::aarch64 ||
1779 target->GetArchitecture().GetMachine() == llvm::Triple::aarch64_be ||
1780 target->GetArchitecture().GetMachine() == llvm::Triple::aarch64_32) {
1781 if (*type_ptr == LLVMDisassembler_ReferenceType_In_ARM64_ADRP) {
1782 m_adrp_address = pc;
1783 m_adrp_insn = value;
1784 *name = nullptr;
1785 *type_ptr = LLVMDisassembler_ReferenceType_InOut_None;
1786 return nullptr;
1787 }
1788 // If this instruction is an ADD and
1789 // the previous instruction was an ADRP and
1790 // the ADRP's register and this ADD's register are the same,
1791 // then this is a pc-relative address calculation.
1792 if (*type_ptr == LLVMDisassembler_ReferenceType_In_ARM64_ADDXri &&
1793 m_adrp_insn && m_adrp_address == pc - 4 &&
1794 (*m_adrp_insn & 0x1f) == ((value >> 5) & 0x1f)) {
1795 uint32_t addxri_inst;
1796 uint64_t adrp_imm, addxri_imm;
1797 // Get immlo and immhi bits, OR them together to get the ADRP imm
1798 // value.
1799 adrp_imm =
1800 ((*m_adrp_insn & 0x00ffffe0) >> 3) | ((*m_adrp_insn >> 29) & 0x3);
1801 // if high bit of immhi after right-shifting set, sign extend
1802 if (adrp_imm & (1ULL << 20))
1803 adrp_imm |= ~((1ULL << 21) - 1);
1804
1805 addxri_inst = value;
1806 addxri_imm = (addxri_inst >> 10) & 0xfff;
1807 // check if 'sh' bit is set, shift imm value up if so
1808 // (this would make no sense, ADRP already gave us this part)
1809 if ((addxri_inst >> (12 + 5 + 5)) & 1)
1810 addxri_imm <<= 12;
1811 value = (m_adrp_address & 0xfffffffffffff000LL) + (adrp_imm << 12) +
1812 addxri_imm;
1813 }
1814 m_adrp_address = LLDB_INVALID_ADDRESS;
1815 m_adrp_insn.reset();
1816 }
1817
1818 if (m_inst->UsingFileAddress()) {
1819 ModuleSP module_sp(m_inst->GetAddress().GetModule());
1820 if (module_sp) {
1821 module_sp->ResolveFileAddress(value, value_so_addr);
1822 module_sp->ResolveFileAddress(pc, pc_so_addr);
1823 }
1824 } else if (target && target->HasLoadedSections()) {
1825 target->ResolveLoadAddress(value, value_so_addr);
1826 target->ResolveLoadAddress(pc, pc_so_addr);
1827 }
1828
1829 SymbolContext sym_ctx;
1830 const SymbolContextItem resolve_scope =
1831 eSymbolContextFunction | eSymbolContextSymbol;
1832 if (pc_so_addr.IsValid() && pc_so_addr.GetModule()) {
1833 pc_so_addr.GetModule()->ResolveSymbolContextForAddress(
1834 pc_so_addr, resolve_scope, sym_ctx);
1835 }
1836
1837 if (value_so_addr.IsValid() && value_so_addr.GetSection()) {
1838 StreamString ss;
1839
1840 bool format_omitting_current_func_name = false;
1841 if (sym_ctx.symbol || sym_ctx.function) {
1842 AddressRange range;
1843 for (uint32_t idx = 0;
1844 sym_ctx.GetAddressRange(resolve_scope, idx, false, range);
1845 ++idx) {
1846 if (range.ContainsLoadAddress(value_so_addr, target)) {
1847 format_omitting_current_func_name = true;
1848 break;
1849 }
1850 }
1851 }
1852
1853 // If the "value" address (the target address we're symbolicating) is
1854 // inside the same SymbolContext as the current instruction pc
1855 // (pc_so_addr), don't print the full function name - just print it
1856 // with DumpStyleNoFunctionName style, e.g. "<+36>".
1857 if (format_omitting_current_func_name) {
1858 value_so_addr.Dump(&ss, target, Address::DumpStyleNoFunctionName,
1859 Address::DumpStyleSectionNameOffset);
1860 } else {
1861 value_so_addr.Dump(
