53ProcessTrace::~ProcessTrace() {
54 Clear();
55 // We need to call finalize on the process before destroying ourselves to
56 // make sure all of the broadcaster cleanup goes as planned. If we destruct
57 // this class, then Process::~Process() might have problems trying to fully
58 // destroy the broadcaster.
59 Finalize(true /* destructing */);
60}

62void ProcessTrace::DidAttach(ArchSpec &process_arch) {
63 ListenerSP listener_sp(
64 Listener::MakeListener("lldb.process_trace.did_attach_listener"));
65 HijackProcessEvents(listener_sp);
66
67 SetCanJIT(false);
68 StartPrivateStateThread(lldb::eStateStopped, false);
69 if (!m_current_private_state_thread) {
70 LLDB_LOG(GetLog(LLDBLog::Process), "ProcessTrace: failed to start private "
71 "state thread.");
72 return;
73 }
74
75 EventSP event_sp;
76 WaitForProcessToStop(std::nullopt, &event_sp, true, listener_sp);
77
78 RestoreProcessEvents();
79
80 Process::DidAttach(process_arch);
81}

92size_t ProcessTrace::ReadMemory(addr_t addr, void *buf, size_t size,
93 Status &error) {
94 if (const ABISP &abi = GetABI())
95 addr = abi->FixAnyAddress(addr);
96
97 // Don't allow the caching that lldb_private::Process::ReadMemory does since
98 // we have it all cached in the trace files.
99 return DoReadMemory(addr, buf, size, error);
100}

117bool ProcessTrace::GetProcessInfo(ProcessInstanceInfo &info) {
118 info.Clear();
119 info.SetProcessID(GetID());
120 info.SetArchitecture(GetArchitecture());
121 ModuleSP module_sp = GetTarget().GetExecutableModule();
122 if (module_sp) {
123 const bool add_exe_file_as_first_arg = false;
124 info.SetExecutableFile(GetTarget().GetExecutableModule()->GetFileSpec(),
125 add_exe_file_as_first_arg);
126 }
127 return true;
128}

130size_t ProcessTrace::DoReadMemory(addr_t addr, void *buf, size_t size,
131 Status &error) {
132 Address resolved_address;
133 GetTarget().ResolveLoadAddress(addr, resolved_address);
134
135 return GetTarget().ReadMemoryFromFileCache(resolved_address, buf, size,
136 error);
