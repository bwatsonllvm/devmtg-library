68bool ProcessWasm::CanDebug(lldb::TargetSP target_sp,
69 bool plugin_specified_by_name) {
70 if (plugin_specified_by_name)
71 return true;
72
73 if (Module *exe_module = target_sp->GetExecutableModulePointer()) {
74 if (ObjectFile *exe_objfile = exe_module->GetObjectFile())
75 return exe_objfile->GetArchitecture().GetMachine() ==
76 llvm::Triple::wasm32;
77 }
78
79 // However, if there is no wasm module, we return false, otherwise,
80 // we might use ProcessWasm to attach gdb remote.
81 return false;
82}

88size_t ProcessWasm::ReadMemory(lldb::addr_t vm_addr, void *buf, size_t size,
89 Status &error) {
90 wasm_addr_t wasm_addr(vm_addr);
91
92 switch (wasm_addr.GetType()) {
93 case WasmAddressType::Memory:
94 case WasmAddressType::Object:
95 return ProcessGDBRemote::ReadMemory(vm_addr, buf, size, error);
96 case WasmAddressType::Invalid:
97 break;
98 }
99
100 error.FromErrorStringWithFormatv(
101 "Wasm read failed for invalid address {0:x} (type = {1:x}, module = "
102 "{2:x}, offset = {3:x})",
103 vm_addr, wasm_addr.GetType(), wasm_addr.GetModuleID(),
104 wasm_addr.GetOffset());
105 return 0;
106}

108llvm::Expected<std::vector<lldb::addr_t>>
109ProcessWasm::GetWasmCallStack(lldb::tid_t tid) {
110 StreamString packet;
111 packet.Printf("qWasmCallStack:");
112 packet.Printf("%" PRIx64, tid);
113
114 StringExtractorGDBRemote response;
115 if (m_gdb_comm.SendPacketAndWaitForResponse(packet.GetString(), response) !=
116 GDBRemoteCommunication::PacketResult::Success)
117 return llvm::createStringError("failed to send qWasmCallStack");
118
119 if (!response.IsNormalResponse())
120 return llvm::createStringError("failed to get response for qWasmCallStack");
121
122 WritableDataBufferSP data_buffer_sp =
123 std::make_shared<DataBufferHeap>(response.GetStringRef().size() / 2, 0);
124 const size_t bytes = response.GetHexBytes(data_buffer_sp->GetData(), '\xcc');
125 if (bytes == 0 || bytes % sizeof(uint64_t) != 0)
126 return llvm::createStringError("invalid response for qWasmCallStack");
127
128 // To match the Wasm specification, the addresses are encoded in little endian
129 // byte order.
130 DataExtractor data(data_buffer_sp, lldb::eByteOrderLittle,
131 GetAddressByteSize());
132 lldb::offset_t offset = 0;
133 std::vector<lldb::addr_t> call_stack_pcs;
134 while (offset < bytes)
135 call_stack_pcs.push_back(data.GetU64(&offset));
136
137 return call_stack_pcs;
138}

140llvm::Expected<lldb::DataBufferSP>
141ProcessWasm::GetWasmVariable(WasmVirtualRegisterKinds kind, int frame_index,
142 int index) {
143 StreamString packet;
144 switch (kind) {
145 case eWasmTagLocal:
146 packet.Printf("qWasmLocal:");
147 break;
148 case eWasmTagGlobal:
149 packet.Printf("qWasmGlobal:");
150 break;
151 case eWasmTagOperandStack:
152 packet.PutCString("qWasmStackValue:");
153 break;
154 case eWasmTagNotAWasmLocation:
155 return llvm::createStringError("not a Wasm location");
156 }
157 packet.Printf("%d;%d", frame_index, index);
158
159 StringExtractorGDBRemote response;
160 if (m_gdb_comm.SendPacketAndWaitForResponse(packet.GetString(), response) !=
161 GDBRemoteCommunication::PacketResult::Success)
162 return llvm::createStringError("failed to send Wasm variable");
163
164 if (!response.IsNormalResponse())
165 return llvm::createStringError("failed to get response for Wasm variable");
166
167 WritableDataBufferSP buffer_sp(
168 new DataBufferHeap(response.GetStringRef().size() / 2, 0));
169 response.GetHexBytes(buffer_sp->GetData(), '\xcc');
170 return buffer_sp;
171}
