38/// \class DataExtractor DataExtractor.h "lldb/Core/DataExtractor.h" An data
39/// extractor class.
40///
41/// DataExtractor is a class that can extract data (swapping if needed) from a
42/// data buffer. The data buffer can be caller owned, or can be shared data
43/// that can be shared between multiple DataExtractor instances. Multiple
44/// DataExtractor objects can share the same data, yet extract values in
45/// different address sizes and byte order modes. Each object can have a
46/// unique position in the shared data and extract data from different
47/// offsets.
48///
49/// \see DataBuffer
50class DataExtractor {
51public:
52 /// \typedef DataExtractor::Type
53 /// Type enumerations used in the dump routines.
54 enum Type {
55 TypeUInt8, ///< Format output as unsigned 8 bit integers
56 TypeChar, ///< Format output as characters
57 TypeUInt16, ///< Format output as unsigned 16 bit integers
58 TypeUInt32, ///< Format output as unsigned 32 bit integers
59 TypeUInt64, ///< Format output as unsigned 64 bit integers
60 TypePointer, ///< Format output as pointers
61 TypeULEB128, ///< Format output as ULEB128 numbers
62 TypeSLEB128 ///< Format output as SLEB128 numbers
63 };

118 /// Construct with a subset of \a data.
119 ///
120 /// Initialize this object with a subset of the data bytes in \a data. If \a
121 /// data contains shared data, then a reference to the shared data will be
122 /// added to ensure the shared data stays around as long as any objects have
123 /// references to the shared data. The byte order value and the address size
124 /// settings are copied from \a data. If \a offset is not a valid offset in
125 /// \a data, then no reference to the shared data will be added. If there
126 /// are not \a length bytes available in \a data starting at \a offset, the
127 /// length will be truncated to contain as many bytes as possible.
128 ///
129 /// \param[in] data
130 /// Another DataExtractor object that contains data.
131 ///
132 /// \param[in] offset
133 /// The offset into \a data at which the subset starts.
134 ///
135 /// \param[in] length
136 /// The length in bytes of the subset of data.

317 /// Extract \a length bytes from \a *offset_ptr.
318 ///
319 /// Returns a pointer to a bytes in this object's data at the offset pointed
320 /// to by \a offset_ptr. If \a length is zero or too large, then the offset
321 /// pointed to by \a offset_ptr will not be updated and nullptr will be
322 /// returned.
323 ///
324 /// \param[in,out] offset_ptr
325 /// A pointer to an offset within the data that will be advanced
326 /// by the appropriate number of bytes if the value is extracted
327 /// correctly. If the offset is out of bounds or there are not
328 /// enough bytes to extract this value, the offset will be left
329 /// unmodified.
330 ///
331 /// \param[in] length
332 /// The optional length of a string to extract. If the value is
333 /// zero, a NULL terminated C string will be extracted.
334 ///
335 /// \return
336 /// A pointer to the bytes in this object's data if the offset
337 /// and length are valid, or nullptr otherwise.
338 virtual const void *GetData(lldb::offset_t *offset_ptr,
339 lldb::offset_t length) const {
340 const uint8_t *ptr = PeekData(*offset_ptr, length);
341 if (ptr)
342 *offset_ptr += length;
343 return ptr;
344 }

363 /// Copy \a dst_len bytes from \a *offset_ptr and ensure the copied data is
364 /// treated as a value that can be swapped to match the specified byte
365 /// order.
366 ///
367 /// For values that are larger than the supported integer sizes, this
368 /// function can be used to extract data in a specified byte order. It can
369 /// also be used to copy a smaller integer value from to a larger value. The
370 /// extra bytes left over will be padded correctly according to the byte
371 /// order of this object and the \a dst_byte_order. This can be very handy
372 /// when say copying a partial data value into a register.
373 ///
374 /// \param[in] src_offset
375 /// The offset into this data from which to start copying an endian
376 /// entity
377 ///
378 /// \param[in] src_len
379 /// The length of the endian data to copy from this object into the \a
380 /// dst object
381 ///
382 /// \param[out] dst
383 /// The buffer where to place the endian data. The data might need to be
384 /// byte swapped (and appropriately padded with zeroes if \a src_len !=
385 /// \a dst_len) if \a dst_byte_order does not match the byte order in
386 /// this object.
387 ///
388 /// \param[in] dst_len
389 /// The length number of bytes that the endian value will occupy is \a
390 /// dst.
391 ///
392 /// \param[in] dst_byte_order
393 /// The byte order that the endian value should be in the \a dst buffer.
394 ///
395 /// \return
396 /// Returns the number of bytes that were copied, or zero if anything
397 /// goes wrong.
398 lldb::offset_t CopyByteOrderedData(lldb::offset_t src_offset,
399 lldb::offset_t src_len, void *dst,
400 lldb::offset_t dst_len,
401 lldb::ByteOrder dst_byte_order) const;
