43lldb::RegisterContextSP
44ThreadFreeBSDKernelCore::CreateRegisterContextForFrame(StackFrame *frame) {
45 RegisterContextSP reg_ctx_sp;
46 uint32_t concrete_frame_idx = 0;
47
48 if (frame)
49 concrete_frame_idx = frame->GetConcreteFrameIndex();
50
51 if (concrete_frame_idx == 0) {
52 if (m_thread_reg_ctx_sp)
53 return m_thread_reg_ctx_sp;
54
55 ProcessFreeBSDKernelCore *process =
56 static_cast<ProcessFreeBSDKernelCore *>(GetProcess().get());
57 ArchSpec arch = process->GetTarget().GetArchitecture();
58
59 switch (arch.GetMachine()) {
60 case llvm::Triple::aarch64:
61 m_thread_reg_ctx_sp =
62 std::make_shared<RegisterContextFreeBSDKernelCore_arm64>(
63 *this, std::make_unique<RegisterInfoPOSIX_arm64>(arch, 0),
64 m_pcb_addr);
65 break;
66 case llvm::Triple::x86:
67 m_thread_reg_ctx_sp =
68 std::make_shared<RegisterContextFreeBSDKernelCore_i386>(
69 *this, new RegisterContextFreeBSD_i386(arch), m_pcb_addr);
70 break;
71 case llvm::Triple::x86_64:
72 m_thread_reg_ctx_sp =
73 std::make_shared<RegisterContextFreeBSDKernelCore_x86_64>(
74 *this, new RegisterContextFreeBSD_x86_64(arch), m_pcb_addr);
75 break;
76 default:
77 assert(false &&
78 "Unsupported architecture passed to ThreadFreeBSDKernelCore");
79 break;
80 }
81
82 reg_ctx_sp = m_thread_reg_ctx_sp;
83 } else {
84 reg_ctx_sp = GetUnwinder().CreateRegisterContextForFrame(frame);
85 }
86 return reg_ctx_sp;
87}

89bool ThreadFreeBSDKernelCore::CalculateStopInfo() {
90 if (m_is_crashed) {
91 // Set a stop reason for crashing threads only so that they get selected
92 // preferentially.
93 SetStopInfo(StopInfo::CreateStopReasonWithException(*this, "kernel panic"));
94 return true;
95 }
96 return false;
97}
