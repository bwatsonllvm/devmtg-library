65void Statusline::Draw(std::string str) {
66 lldb::LockableStreamFileSP stream_sp = m_debugger.GetOutputStreamSP();
67 if (!stream_sp)
68 return;
69
70 str = ansi::TrimAndPad(str, m_terminal_width);
71
72 LockedStreamFile locked_stream = stream_sp->Lock();
73 locked_stream << ANSI_SAVE_CURSOR;
74 locked_stream.Printf(ANSI_TO_START_OF_ROW,
75 static_cast<unsigned>(m_terminal_height));
76
77 // Use "reverse video" to make sure the statusline has a background. Only do
78 // this when colors are disabled, and rely on the statusline format otherwise.
79 if (!m_debugger.GetUseColor())
80 locked_stream << ANSI_REVERSE_VIDEO;
81
82 locked_stream << str;
83 locked_stream << ANSI_NORMAL;
84 locked_stream << ANSI_RESTORE_CURSOR;
85}

87void Statusline::UpdateScrollWindow(ScrollWindowMode mode) {
88 assert(m_terminal_width != 0 && m_terminal_height != 0);
89
90 lldb::LockableStreamFileSP stream_sp = m_debugger.GetOutputStreamSP();
91 if (!stream_sp)
92 return;
93
94 const unsigned reduced_scroll_rows = m_terminal_height - 1;
95 { // Scope for locked_stream:
96 LockedStreamFile locked_stream = stream_sp->Lock();
97
98 switch (mode) {
99 case EnableStatusline:
100 // Move everything on the screen up.
101 locked_stream << '\n';
102 locked_stream.Printf(ANSI_UP_ROWS, 1);
103 // Reduce the scroll window.
104 locked_stream << ANSI_SAVE_CURSOR;
105 locked_stream.Printf(ANSI_SET_SCROLL_ROWS, reduced_scroll_rows);
106 locked_stream << ANSI_RESTORE_CURSOR;
107 break;
108 case DisableStatusline:
109 // Reset the scroll window.
110 locked_stream << ANSI_SAVE_CURSOR;
111 locked_stream.Printf(ANSI_SET_SCROLL_ROWS,
112 static_cast<unsigned>(m_terminal_height));
113 locked_stream << ANSI_RESTORE_CURSOR;
114 // Clear the screen below to hide the old statusline.
115 locked_stream << ANSI_CLEAR_BELOW;
116 break;
117 case ResizeStatusline:
118 // Clear the screen and update the scroll window.
119 // FIXME: Find a better solution (#146919).
120 locked_stream << ANSI_CLEAR_SCREEN;
121 locked_stream.Printf(ANSI_SET_SCROLL_ROWS, reduced_scroll_rows);
122 break;
123 }
124 }
125 m_debugger.RefreshIOHandler();
126}

128void Statusline::Redraw(std::optional<ExecutionContextRef> exe_ctx_ref) {
129 // Update the cached execution context.
130 if (exe_ctx_ref)
131 m_exe_ctx_ref = *exe_ctx_ref;
132
133 // Lock the execution context.
134 ExecutionContext exe_ctx =
135 m_exe_ctx_ref.Lock(/*thread_and_frame_only_if_stopped=*/false);
136
137 // Compute the symbol context if we're stopped.
138 SymbolContext sym_ctx;
139 llvm::Expected<StoppedExecutionContext> stopped_exe_ctx =
140 GetStoppedExecutionContext(&m_exe_ctx_ref);
141 if (stopped_exe_ctx) {
142 // The StoppedExecutionContext only ensures that we hold the run lock.
143 // The process could be in an exited or unloaded state and have no frame.
144 if (auto frame_sp = stopped_exe_ctx->GetFrameSP())
145 sym_ctx = frame_sp->GetSymbolContext(eSymbolContextEverything);
146 } else {
147 // We can draw the statusline without being stopped.
148 llvm::consumeError(stopped_exe_ctx.takeError());
149 }
150
151 StreamString stream;
152 FormatEntity::Entry format = m_debugger.GetStatuslineFormat();
153 FormatEntity::Formatter(&sym_ctx, &exe_ctx, nullptr, false, false)
154 .Format(format, stream);
155
156 Draw(stream.GetString().str());
157}
