19static Status fileLock(int fd, int cmd, int lock_type, const uint64_t start,
20 const uint64_t len) {
21 struct flock fl;
22
23 fl.l_type = lock_type;
24 fl.l_whence = SEEK_SET;
25 fl.l_start = start;
26 fl.l_len = len;
27 fl.l_pid = ::getpid();
28
29 if (llvm::sys::RetryAfterSignal(-1, ::fcntl, fd, cmd, &fl) == -1)
30 return Status::FromErrno();
31
32 return Status();
33}
