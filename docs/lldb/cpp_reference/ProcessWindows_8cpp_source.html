56std::string GetProcessExecutableName(HANDLE process_handle) {
57 std::vector<wchar_t> file_name;
58 DWORD file_name_size = MAX_PATH; // first guess, not an absolute limit
59 DWORD copied = 0;
60 do {
61 file_name_size *= 2;
62 file_name.resize(file_name_size);
63 copied = ::GetModuleFileNameExW(process_handle, NULL, file_name.data(),
64 file_name_size);
65 } while (copied >= file_name_size);
66 file_name.resize(copied);
67 std::string result;
68 llvm::convertWideToUTF8(file_name.data(), result);
69 return result;
70}
71
72std::string GetProcessExecutableName(DWORD pid) {
73 std::string file_name;
74 HANDLE process_handle =
75 ::OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_READ, FALSE, pid);
76 if (process_handle != NULL) {
77 file_name = GetProcessExecutableName(process_handle);
78 ::CloseHandle(process_handle);
79 }
80 return file_name;
81}

158Status ProcessWindows::DoDetach(bool keep_stopped) {
159 Status error;
160 Log *log = GetLog(WindowsLog::Process);
161 StateType private_state = GetPrivateState();
162 if (private_state != eStateExited && private_state != eStateDetached) {
163 if (!keep_stopped) {
164 // if the thread is suspended by lldb, we have to resume threads before
165 // detaching process. When we do after DetachProcess(), thread handles
166 // become invalid so we do before detach.
167 if (private_state == eStateStopped || private_state == eStateCrashed) {
168 LLDB_LOG(log, "process {0} is in state {1}. Resuming for detach...",
169 m_session_data->m_debugger->GetProcess().GetProcessId(),
170 GetPrivateState());
171
172 LLDB_LOG(log, "resuming {0} threads for detach.",
173 m_thread_list.GetSize());
174
175 bool failed = false;
176 for (uint32_t i = 0; i < m_thread_list.GetSize(); ++i) {
177 auto thread = std::static_pointer_cast<TargetThreadWindows>(
178 m_thread_list.GetThreadAtIndex(i));
179 Status result = thread->DoResume();
180 if (result.Fail()) {
181 failed = true;
182 LLDB_LOG(log,
183 "Trying to resume thread at index {0}, but failed with "
184 "error {1}.",
185 i, result);
186 }
187 }
188
189 if (failed) {
190 error = Status::FromErrorString("Resuming Threads for Detach failed");
191 }
192 }
193 }
194
195 error = DetachProcess();
196 if (error.Success())
197 SetPrivateState(eStateDetached);
198 else
199 LLDB_LOG(log, "Detaching process error: {0}", error);
200 } else {
201 error = Status::FromErrorStringWithFormatv(
202 "error: process {0} in state = {1}, but "
203 "cannot detach it in this state.",
204 GetID(), private_state);
205 LLDB_LOG(log, "error: {0}", error);
206 }
207 return error;
208}

231Status ProcessWindows::DoResume(RunDirection direction) {
232 Log *log = GetLog(WindowsLog::Process);
233 llvm::sys::ScopedLock lock(m_mutex);
234
235 if (direction == RunDirection::eRunReverse) {
236 return Status::FromErrorStringWithFormatv(
237 "{0} does not support reverse execution of processes", GetPluginName());
238 }
239
240 Status error;
241
242 StateType private_state = GetPrivateState();
243 if (private_state == eStateStopped || private_state == eStateCrashed) {
244 LLDB_LOG(log, "process {0} is in state {1}. Resuming...",
245 m_session_data->m_debugger->GetProcess().GetProcessId(),
246 GetPrivateState());
247
248 LLDB_LOG(log, "resuming {0} threads.", m_thread_list.GetSize());
249
250 bool failed = false;
251 for (uint32_t i = 0; i < m_thread_list.GetSize(); ++i) {
252 auto thread = std::static_pointer_cast<TargetThreadWindows>(
253 m_thread_list.GetThreadAtIndex(i));
254 Status result = thread->DoResume();
255 if (result.Fail()) {
256 failed = true;
257 LLDB_LOG(
258 log,
259 "Trying to resume thread at index {0}, but failed with error {1}.",
260 i, result);
261 }
262 }
263
264 if (failed) {
265 error = Status::FromErrorString("ProcessWindows::DoResume failed");
266 } else {
267 SetPrivateState(eStateRunning);
268 }
269
270 ExceptionRecordSP active_exception =
271 m_session_data->m_debugger->GetActiveException().lock();
272 if (active_exception) {
273 // Resume the process and continue processing debug events. Mask the
274 // exception so that from the process's view, there is no indication that
275 // anything happened.
276 m_session_data->m_debugger->ContinueAsyncException(
277 ExceptionResult::MaskException);
278 }
279 } else {
280 LLDB_LOG(log, "error: process {0} is in state {1}. Returning...",
281 m_session_data->m_debugger->GetProcess().GetProcessId(),
282 GetPrivateState());
283 }
284 return error;
285}

315static void
316DumpAdditionalExceptionInformation(llvm::raw_ostream &stream,
317 const ExceptionRecordSP &exception) {
318 // Decode additional exception information for specific exception types based
319 // on
320 // https://docs.microsoft.com/en-us/windows/desktop/api/winnt/ns-winnt-_exception_record
321
322 const int addr_min_width = 2 + 8; // "0x" + 4 address bytes
323
324 const std::vector<ULONG_PTR> &args = exception->GetExceptionArguments();
325 switch (exception->GetExceptionCode()) {
326 case EXCEPTION_ACCESS_VIOLATION: {
327 if (args.size() < 2)
328 break;
329
330 stream << ": ";
331 const int access_violation_code = args[0];
332 const lldb::addr_t access_violation_address = args[1];
333 switch (access_violation_code) {
334 case 0:
335 stream << "Access violation reading";
336 break;
337 case 1:
338 stream << "Access violation writing";
339 break;
340 case 8:
341 stream << "User-mode data execution prevention (DEP) violation at";
342 break;
343 default:
344 stream << "Unknown access violation (code " << access_violation_code
345 << ") at";
346 break;
347 }
348 stream << " location "
349 << llvm::format_hex(access_violation_address, addr_min_width);
350 break;
351 }
352 case EXCEPTION_IN_PAGE_ERROR: {
353 if (args.size() < 3)
354 break;
355
356 stream << ": ";
357 const int page_load_error_code = args[0];
358 const lldb::addr_t page_load_error_address = args[1];
359 const DWORD underlying_code = args[2];
360 switch (page_load_error_code) {
361 case 0:
362 stream << "In page error reading";
363 break;
364 case 1:
365 stream << "In page error writing";
366 break;
367 case 8:
368 stream << "User-mode data execution prevention (DEP) violation at";
369 break;
370 default:
371 stream << "Unknown page loading error (code " << page_load_error_code
372 << ") at";
373 break;
374 }
375 stream << " location "
376 << llvm::format_hex(page_load_error_address, addr_min_width)
377 << " (status code " << llvm::format_hex(underlying_code, 8) << ")";
378 break;
379 }
380 }

383void ProcessWindows::RefreshStateAfterStop() {
384 Log *log = GetLog(WindowsLog::Exception);
385 llvm::sys::ScopedLock lock(m_mutex);
386
387 if (!m_session_data) {
388 LLDB_LOG(log, "no active session. Returning...");
389 return;
390 }
391
392 m_thread_list.RefreshStateAfterStop();
393
394 std::weak_ptr<ExceptionRecord> exception_record =
395 m_session_data->m_debugger->GetActiveException();
396 ExceptionRecordSP active_exception = exception_record.lock();
397 if (!active_exception) {
398 LLDB_LOG(log,
399 "there is no active exception in process {0}. Why is the "
400 "process stopped?",
401 m_session_data->m_debugger->GetProcess().GetProcessId());
402 return;
403 }
404
405 StopInfoSP stop_info;
406 m_thread_list.SetSelectedThreadByID(active_exception->GetThreadID());
407 ThreadSP stop_thread = m_thread_list.GetSelectedThread();
408 if (!stop_thread)
409 return;
410
411 RegisterContextSP register_context = stop_thread->GetRegisterContext();
412 uint64_t pc = register_context->GetPC();
413
414 // If we're at a BreakpointSite, mark this as an Unexecuted Breakpoint.
415 // We'll clear that state if we've actually executed the breakpoint.
416 BreakpointSiteSP site(GetBreakpointSiteList().FindByAddress(pc));
417 if (site && site->IsEnabled())
418 stop_thread->SetThreadStoppedAtUnexecutedBP(pc);
419
420 switch (active_exception->GetExceptionCode()) {
421 case EXCEPTION_SINGLE_STEP: {
422 auto *reg_ctx = static_cast<RegisterContextWindows *>(
423 stop_thread->GetRegisterContext().get());
424 uint32_t slot_id = reg_ctx->GetTriggeredHardwareBreakpointSlotId();
425 if (slot_id != LLDB_INVALID_INDEX32) {
426 int id = m_watchpoint_ids[slot_id];
427 LLDB_LOG(log,
428 "Single-stepped onto a watchpoint in process {0} at address "
429 "{1:x} with watchpoint {2}",
430 m_session_data->m_debugger->GetProcess().GetProcessId(), pc, id);
431
432 stop_info = StopInfo::CreateStopReasonWithWatchpointID(*stop_thread, id);
433 stop_thread->SetStopInfo(stop_info);
434
435 return;
436 }
437
438 LLDB_LOG(log, "single stepping thread {0}", stop_thread->GetID());
439 stop_info = StopInfo::CreateStopReasonToTrace(*stop_thread);
440 stop_thread->SetStopInfo(stop_info);
441
442 return;
443 }
444
445 case EXCEPTION_BREAKPOINT: {
446 int breakpoint_size = 1;
447 switch (GetTarget().GetArchitecture().GetMachine()) {
448 case llvm::Triple::aarch64:
449 breakpoint_size = 4;
450 break;
451
452 case llvm::Triple::arm:
453 case llvm::Triple::thumb:
454 breakpoint_size = 2;
455 break;
456
457 case llvm::Triple::x86:
458 case llvm::Triple::x86_64:
459 breakpoint_size = 1;
460 break;
461
462 default:
463 LLDB_LOG(log, "Unknown breakpoint size for architecture");
464 break;
465 }
466
467 // The current PC is AFTER the BP opcode, on all architectures.
468 pc = register_context->GetPC() - breakpoint_size;
469
470 site = GetBreakpointSiteList().FindByAddress(pc);
471 if (site) {
472 LLDB_LOG(log,
473 "detected breakpoint in process {0} at address {1:x} with "
474 "breakpoint site {2}",
475 m_session_data->m_debugger->GetProcess().GetProcessId(), pc,
476 site->GetID());
477
478 stop_thread->SetThreadHitBreakpointSite();
479 if (site->ValidForThisThread(*stop_thread)) {
480 LLDB_LOG(log,
481 "Breakpoint site {0} is valid for this thread ({1:x}), "
482 "creating stop info.",
483 site->GetID(), stop_thread->GetID());
484
485 stop_info = StopInfo::CreateStopReasonWithBreakpointSiteID(
486 *stop_thread, site->GetID());
487 register_context->SetPC(pc);
488 } else {
489 LLDB_LOG(log,
490 "Breakpoint site {0} is not valid for this thread, "
491 "creating empty stop info.",
492 site->GetID());
493 }
494 stop_thread->SetStopInfo(stop_info);
495 return;
496 } else {
497 // The thread hit a hard-coded breakpoint like an `int 3` or
498 // `__debugbreak()`.
499 LLDB_LOG(log,
500 "No breakpoint site matches for this thread. __debugbreak()? "
501 "Creating stop info with the exception.");
502 // FALLTHROUGH: We'll treat this as a generic exception record in the
503 // default case.
504 [[fallthrough]];
505 }
506 }
507
508 default: {
509 std::string desc;
510 llvm::raw_string_ostream desc_stream(desc);
511 desc_stream << "Exception "
512 << llvm::format_hex(active_exception->GetExceptionCode(), 8)
513 << " encountered at address "
514 << llvm::format_hex(active_exception->GetExceptionAddress(), 8);
515 DumpAdditionalExceptionInformation(desc_stream, active_exception);
516
517 stop_info =
518 StopInfo::CreateStopReasonWithException(*stop_thread, desc.c_str());
519 stop_thread->SetStopInfo(stop_info);
520 LLDB_LOG(log, "{0}", desc);
521 return;
522 }
523 }

540bool ProcessWindows::DoUpdateThreadList(ThreadList &old_thread_list,
541 ThreadList &new_thread_list) {
542 Log *log = GetLog(WindowsLog::Thread);
543 // Add all the threads that were previously running and for which we did not
544 // detect a thread exited event.
545 int new_size = 0;
546 int continued_threads = 0;
547 int exited_threads = 0;
548 int new_threads = 0;
549
550 for (ThreadSP old_thread : old_thread_list.Threads()) {
551 lldb::tid_t old_thread_id = old_thread->GetID();
552 auto exited_thread_iter =
553 m_session_data->m_exited_threads.find(old_thread_id);
554 if (exited_thread_iter == m_session_data->m_exited_threads.end()) {
555 new_thread_list.AddThread(old_thread);
556 ++new_size;
557 ++continued_threads;
558 LLDB_LOGV(log, "Thread {0} was running and is still running.",
559 old_thread_id);
560 } else {
561 LLDB_LOGV(log, "Thread {0} was running and has exited.", old_thread_id);
562 ++exited_threads;
563 }
564 }
565
566 // Also add all the threads that are new since the last time we broke into
567 // the debugger.
568 for (const auto &thread_info : m_session_data->m_new_threads) {
569 new_thread_list.AddThread(thread_info.second);
570 ++new_size;
571 ++new_threads;
572 LLDB_LOGV(log, "Thread {0} is new since last update.", thread_info.first);
573 }
574
575 LLDB_LOG(log, "{0} new threads, {1} old threads, {2} exited threads.",
576 new_threads, continued_threads, exited_threads);
577
578 m_session_data->m_new_threads.clear();
579 m_session_data->m_exited_threads.clear();
580
581 return new_size > 0;
582}

649void ProcessWindows::OnExitProcess(uint32_t exit_code) {
650 // No need to acquire the lock since m_session_data isn't accessed.
651 Log *log = GetLog(WindowsLog::Process);
652 LLDB_LOG(log, "Process {0} exited with code {1}", GetID(), exit_code);
653
654 if (m_pty) {

656 m_stdio_communication.InterruptRead();
657 m_pty->Close();

659
660 TargetSP target = CalculateTarget();
661 if (target) {
662 ModuleSP executable_module = target->GetExecutableModule();
663 ModuleList unloaded_modules;
664 unloaded_modules.Append(executable_module);
665 target->ModulesDidUnload(unloaded_modules, true);
666 }
667
668 SetExitStatus(exit_code, /*exit_string=*/"");
669 SetPrivateState(eStateExited);
670
671 ProcessDebugger::OnExitProcess(exit_code);
672}

674void ProcessWindows::OnDebuggerConnected(lldb::addr_t image_base) {
675 DebuggerThreadSP debugger = m_session_data->m_debugger;
676 Log *log = GetLog(WindowsLog::Process);
677 LLDB_LOG(log, "Debugger connected to process {0}. Image base = {1:x}",
678 debugger->GetProcess().GetProcessId(), image_base);
679
680 ModuleSP module;
681 // During attach, we won't have the executable module, so find it now.
682 const DWORD pid = debugger->GetProcess().GetProcessId();
683 const std::string file_name = GetProcessExecutableName(pid);
684 if (file_name.empty()) {
685 return;
686 }
687
688 FileSpec executable_file(file_name);
689 FileSystem::Instance().Resolve(executable_file);
690 ModuleSpec module_spec(executable_file);
691 Status error;
692 module =
693 GetTarget().GetOrCreateModule(module_spec, true /* notify */, &error);
694 if (!module) {
695 return;
696 }
697
698 GetTarget().SetExecutableModule(module, eLoadDependentsNo);
699
700 if (auto dyld = GetDynamicLoader())
701 dyld->OnLoadModule(module, ModuleSpec(), image_base);
702
703 // Add the main executable module to the list of pending module loads. We
704 // can't call GetTarget().ModulesDidLoad() here because we still haven't

719ExceptionResult
720ProcessWindows::OnDebugException(bool first_chance,
721 const ExceptionRecord &record) {
722 Log *log = GetLog(WindowsLog::Exception);
723 llvm::sys::ScopedLock lock(m_mutex);
724
725 // FIXME: Without this check, occasionally when running the test suite there
726 // is
727 // an issue where m_session_data can be null. It's not clear how this could
728 // happen but it only surfaces while running the test suite. In order to
729 // properly diagnose this, we probably need to first figure allow the test
730 // suite to print out full lldb logs, and then add logging to the process
731 // plugin.
732 if (!m_session_data) {
733 LLDB_LOG(log,
734 "Debugger thread reported exception {0:x} at address {1:x}, "
735 "but there is no session.",
736 record.GetExceptionCode(), record.GetExceptionAddress());
737 return ExceptionResult::SendToApplication;
738 }
739
740 if (!first_chance) {
741 // Not any second chance exception is an application crash by definition.
742 // It may be an expression evaluation crash.
743 SetPrivateState(eStateStopped);
744 }
745
746 ExceptionResult result = ExceptionResult::SendToApplication;
747 switch (record.GetExceptionCode()) {
748 case EXCEPTION_BREAKPOINT:
749 // Handle breakpoints at the first chance.
750 result = ExceptionResult::BreakInDebugger;
751
752 if (!m_session_data->m_initial_stop_received) {
753 LLDB_LOG(
754 log,
755 "Hit loader breakpoint at address {0:x}, setting initial stop event.",
756 record.GetExceptionAddress());
757 m_session_data->m_initial_stop_received = true;
758 ::SetEvent(m_session_data->m_initial_stop_event);
759 } else {
760 LLDB_LOG(log, "Hit non-loader breakpoint at address {0:x}.",
761 record.GetExceptionAddress());
762 }
763 SetPrivateState(eStateStopped);
764 break;
765 case EXCEPTION_SINGLE_STEP:
766 result = ExceptionResult::BreakInDebugger;
767 SetPrivateState(eStateStopped);
768 break;
769 default:
770 LLDB_LOG(log,
771 "Debugger thread reported exception {0:x} at address {1:x} "
772 "(first_chance={2})",
773 record.GetExceptionCode(), record.GetExceptionAddress(),
774 first_chance);
775 // For non-breakpoints, give the application a chance to handle the
776 // exception first.
777 if (first_chance)
778 result = ExceptionResult::SendToApplication;
779 else
780 result = ExceptionResult::BreakInDebugger;
781 }
782
783 return result;
784}

866Status ProcessWindows::EnableWatchpoint(WatchpointSP wp_sp, bool notify) {
867 Status error;
868
869 if (wp_sp->IsEnabled()) {
870 wp_sp->SetEnabled(true, notify);
871 return error;
872 }
873
874 WatchpointInfo info;
875 for (info.slot_id = 0;
876 info.slot_id < RegisterContextWindows::GetNumHardwareBreakpointSlots();
877 info.slot_id++)
878 if (m_watchpoint_ids[info.slot_id] == LLDB_INVALID_BREAK_ID)
879 break;
880 if (info.slot_id == RegisterContextWindows::GetNumHardwareBreakpointSlots()) {
881 error = Status::FromErrorStringWithFormat(
882 "Can't find free slot for watchpoint %i", wp_sp->GetID());
883 return error;
884 }
885 info.address = wp_sp->GetLoadAddress();
886 info.size = wp_sp->GetByteSize();
887 info.read = wp_sp->WatchpointRead();
888 info.write = wp_sp->WatchpointWrite() || wp_sp->WatchpointModify();
889
890 for (unsigned i = 0U; i < m_thread_list.GetSize(); i++) {
891 Thread *thread = m_thread_list.GetThreadAtIndex(i).get();
892 auto *reg_ctx = static_cast<RegisterContextWindows *>(
893 thread->GetRegisterContext().get());
894 if (!reg_ctx->AddHardwareBreakpoint(info.slot_id, info.address, info.size,
895 info.read, info.write)) {
896 error = Status::FromErrorStringWithFormat(
897 "Can't enable watchpoint %i on thread 0x%llx", wp_sp->GetID(),
898 thread->GetID());
899 break;
900 }
901 }
902 if (error.Fail()) {
903 for (unsigned i = 0U; i < m_thread_list.GetSize(); i++) {
904 Thread *thread = m_thread_list.GetThreadAtIndex(i).get();
905 auto *reg_ctx = static_cast<RegisterContextWindows *>(
906 thread->GetRegisterContext().get());
907 reg_ctx->RemoveHardwareBreakpoint(info.slot_id);
908 }
909 return error;
910 }
911
912 m_watchpoints[wp_sp->GetID()] = info;
913 m_watchpoint_ids[info.slot_id] = wp_sp->GetID();
914
915 wp_sp->SetEnabled(true, notify);
916
917 return error;
918}

920Status ProcessWindows::DisableWatchpoint(WatchpointSP wp_sp, bool notify) {
921 Status error;
922
923 if (!wp_sp->IsEnabled()) {
924 wp_sp->SetEnabled(false, notify);
925 return error;
926 }
927
928 auto it = m_watchpoints.find(wp_sp->GetID());
929 if (it == m_watchpoints.end()) {
930 error = Status::FromErrorStringWithFormat(
931 "Info about watchpoint %i is not found", wp_sp->GetID());
932 return error;
933 }
934
935 for (unsigned i = 0U; i < m_thread_list.GetSize(); i++) {
936 Thread *thread = m_thread_list.GetThreadAtIndex(i).get();
937 auto *reg_ctx = static_cast<RegisterContextWindows *>(
938 thread->GetRegisterContext().get());
939 if (!reg_ctx->RemoveHardwareBreakpoint(it->second.slot_id)) {
940 error = Status::FromErrorStringWithFormat(
941 "Can't disable watchpoint %i on thread 0x%llx", wp_sp->GetID(),
942 thread->GetID());
943 break;
944 }
945 }
946 if (error.Fail())
947 return error;
948
949 m_watchpoint_ids[it->second.slot_id] = LLDB_INVALID_BREAK_ID;
950 m_watchpoints.erase(it);
951
952 wp_sp->SetEnabled(false, notify);
953
954 return error;
955}

980 /// Peek the console for input. If it has any, drain the pipe until text input
981 /// is found or the pipe is empty.
982 ///
983 /// \param hStdin
984 /// The handle to the standard input's pipe.
985 ///
986 /// \return
987 /// true if the pipe has text input.
988 llvm::Expected<bool> ConsoleHasTextInput(const HANDLE hStdin) {
989 // Check if there are already characters buffered. Pressing enter counts as
990 // 2 characters '\r\n' and only one of them is a keyDown event.
991 DWORD bytesAvailable = 0;
992 if (PeekNamedPipe(hStdin, NULL, 0, NULL, &bytesAvailable, NULL)) {
993 if (bytesAvailable > 0)
994 return true;
995 }
996
997 while (true) {
998 INPUT_RECORD inputRecord;
999 DWORD numRead = 0;
1000 if (!PeekConsoleInput(hStdin, &inputRecord, 1, &numRead))
1001 return llvm::createStringError("Failed to peek standard input.");
1002
1003 if (numRead == 0)
1004 return false;
1005
1006 if (inputRecord.EventType == KEY_EVENT &&
1007 inputRecord.Event.KeyEvent.bKeyDown &&
1008 inputRecord.Event.KeyEvent.uChar.AsciiChar != 0)
1009 return true;
1010
1011 if (!ReadConsoleInput(hStdin, &inputRecord, 1, &numRead))
1012 return llvm::createStringError("Failed to read standard input.");
1013 }
1014 }
1015
1016 void Run() override {
1017 if (!m_read_file.IsValid() || m_write_file == INVALID_HANDLE_VALUE) {
1018 SetIsDone(true);
1019 return;
1020 }
1021
1022 SetIsDone(false);
1023 SetIsRunning(true);
1024
1025 HANDLE hStdin = m_read_file.GetWaitableHandle();
1026 HANDLE waitHandles[2] = {hStdin, m_interrupt_event};
1027
1028 DWORD consoleMode;
1029 bool isConsole = GetConsoleMode(hStdin, &consoleMode) != 0;
1030 // With ENABLE_LINE_INPUT, ReadFile returns only when a carriage return is
1031 // read. This will block lldb in ReadFile until the user hits enter. Save
1032 // the previous console mode to restore it later and remove
1033 // ENABLE_LINE_INPUT.
1034 DWORD oldConsoleMode = consoleMode;
1035 SetConsoleMode(hStdin,
1036 consoleMode & ~ENABLE_LINE_INPUT & ~ENABLE_ECHO_INPUT);
1037
1038 while (true) {
1039 {
1040 std::lock_guard<std::mutex> guard(m_mutex);
1041 if (GetIsDone())
1042 goto exit_loop;
1043 }
1044
1045 DWORD result = WaitForMultipleObjects(2, waitHandles, FALSE, INFINITE);
1046 switch (result) {
1047 case WAIT_FAILED:
1048 goto exit_loop;
1049 case WAIT_OBJECT_0: {
1050 if (isConsole) {
1051 auto hasInputOrErr = ConsoleHasTextInput(hStdin);
1052 if (!hasInputOrErr) {
1053 Log *log = GetLog(WindowsLog::Process);
1054 LLDB_LOG_ERROR(log, hasInputOrErr.takeError(),
1055 "failed to process debuggee's IO: {0}");
1056 goto exit_loop;
1057 }
1058
1059 // If no text input is ready, go back to waiting.
1060 if (!*hasInputOrErr)
1061 continue;
1062 }
1063
1064 char ch = 0;
1065 DWORD read = 0;
1066 if (!ReadFile(hStdin, &ch, 1, &read, nullptr) || read != 1)
1067 goto exit_loop;
1068
1069 DWORD written = 0;
1070 if (!WriteFile(m_write_file, &ch, 1, &written, nullptr) || written != 1)
1071 goto exit_loop;
1072 break;
1073 }
1074 case WAIT_OBJECT_0 + 1: {
1075 ControlOp op = m_pending_op.exchange(eControlOpNone);
1076 if (op == eControlOpQuit)
1077 goto exit_loop;
1078 if (op == eControlOpInterrupt &&
1079 StateIsRunningState(m_process->GetState()))
1080 m_process->SendAsyncInterrupt();
1081 break;
1082 }
1083 default:
1084 goto exit_loop;
1085 }
1086 }
1087
1088 exit_loop:;
1089 SetIsRunning(false);
1090 SetIsDone(true);
1091 SetConsoleMode(hStdin, oldConsoleMode);
1092 }
1093
1094 void Cancel() override {
1095 std::lock_guard<std::mutex> guard(m_mutex);
1096 SetIsDone(true);
1097 if (m_is_running) {
1098 m_pending_op.store(eControlOpQuit);
1099 ::SetEvent(m_interrupt_event);
1100 }
1101 }
1102
1103 bool Interrupt() override {
1104 if (m_active) {
1105 m_pending_op.store(eControlOpInterrupt);
1106 ::SetEvent(m_interrupt_event);
1107 return true;
1108 }
1109 if (StateIsRunningState(m_process->GetState())) {
1110 m_process->SendAsyncInterrupt();
1111 return true;
1112 }
1113 return false;
1114 }
