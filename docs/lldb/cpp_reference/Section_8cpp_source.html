194addr_t Section::GetFileAddress() const {
195 SectionSP parent_sp(GetParent());
196 if (parent_sp) {
197 // This section has a parent which means m_file_addr is an offset into the
198 // parent section, so the file address for this section is the file address
199 // of the parent plus the offset
200 return parent_sp->GetFileAddress() + m_file_addr;
201 }
202 // This section has no parent, so m_file_addr is the file base address
203 return m_file_addr;
204}

206bool Section::SetFileAddress(lldb::addr_t file_addr) {
207 SectionSP parent_sp(GetParent());
208 if (parent_sp) {
209 if (m_file_addr >= file_addr)
210 return parent_sp->SetFileAddress(m_file_addr - file_addr);
211 return false;
212 } else {
213 // This section has no parent, so m_file_addr is the file base address
214 m_file_addr = file_addr;
215 return true;
216 }
217}

229addr_t Section::GetLoadBaseAddress(Target *target) const {
230 addr_t load_base_addr = LLDB_INVALID_ADDRESS;
231 SectionSP parent_sp(GetParent());
232 if (parent_sp) {
233 load_base_addr = parent_sp->GetLoadBaseAddress(target);
234 if (load_base_addr != LLDB_INVALID_ADDRESS)
235 load_base_addr += GetOffset();
236 }
237 if (load_base_addr == LLDB_INVALID_ADDRESS) {
238 load_base_addr = target->GetSectionLoadAddress(
239 const_cast<Section *>(this)->shared_from_this());
240 }
241 return load_base_addr;
242}

244bool Section::ResolveContainedAddress(addr_t offset, Address &so_addr,
245 bool allow_section_end) const {
246 const size_t num_children = m_children.GetSize();
247 for (size_t i = 0; i < num_children; i++) {
248 Section *child_section = m_children.GetSectionAtIndex(i).get();
249
250 addr_t child_offset = child_section->GetOffset();
251 if (child_offset <= offset &&
252 offset - child_offset <
253 child_section->GetByteSize() + (allow_section_end ? 1 : 0))
254 return child_section->ResolveContainedAddress(offset - child_offset,
255 so_addr, allow_section_end);
256 }
257 so_addr.SetOffset(offset);
258 so_addr.SetSection(const_cast<Section *>(this)->shared_from_this());
259
260 // Ensure that there are no orphaned (i.e., moduleless) sections.
261 assert(GetModule().get());
262 return true;
263}

265bool Section::ContainsFileAddress(addr_t vm_addr) const {
266 const addr_t file_addr = GetFileAddress();
267 if (file_addr != LLDB_INVALID_ADDRESS && !IsThreadSpecific()) {
268 if (file_addr <= vm_addr) {
269 const addr_t offset = vm_addr - file_addr;
270 return offset < GetByteSize();
271 }
272 }
273 return false;
274}

276void Section::Dump(llvm::raw_ostream &s, unsigned indent, Target *target,
277 uint32_t depth) const {
278 s.indent(indent);
279 s << llvm::format("0x%16.16" PRIx64 " %-22s ", GetID(), GetTypeAsCString());
280 bool resolved = true;
281 addr_t addr = LLDB_INVALID_ADDRESS;
282
283 if (GetByteSize() == 0)
284 s.indent(39);
285 else {
286 if (target)
287 addr = GetLoadBaseAddress(target);
288
289 if (addr == LLDB_INVALID_ADDRESS) {
290 if (target)
291 resolved = false;
292 addr = GetFileAddress();
293 }
294
295 VMRange range(addr, addr + m_byte_size);
296 range.Dump(s, 0);
297 }
298
299 s << llvm::format("%c %c%c%c 0x%8.8" PRIx64 " 0x%8.8" PRIx64 " 0x%8.8x ",
300 resolved ? ' ' : '*', m_readable ? 'r' : '-',
301 m_writable ? 'w' : '-', m_executable ? 'x' : '-',
302 m_file_offset, m_file_size, Get());
303
304 DumpName(s);
305
306 s << "\n";
307
308 if (depth > 0)
309 m_children.Dump(s, indent, target, false, depth - 1);
310}

312void Section::DumpName(llvm::raw_ostream &s) const {
313 SectionSP parent_sp(GetParent());
314 if (parent_sp) {
315 parent_sp->DumpName(s);
316 s << '.';
317 } else {
318 // The top most section prints the module basename
319 const char *name = nullptr;
320 ModuleSP module_sp(GetModule());
321
322 if (m_obj_file) {
323 const FileSpec &file_spec = m_obj_file->GetFileSpec();
324 name = file_spec.GetFilename().AsCString();
325 }
326 if ((!name || !name[0]) && module_sp)
327 name = module_sp->GetFileSpec().GetFilename().AsCString();
328 if (name && name[0])
329 s << name << '.';
330 }
331 s << m_name;
332}

343bool Section::Slide(addr_t slide_amount, bool slide_children) {
344 if (m_file_addr != LLDB_INVALID_ADDRESS) {
345 if (slide_amount == 0)
346 return true;
347
348 m_file_addr += slide_amount;
349
350 if (slide_children)
351 m_children.Slide(slide_amount, slide_children);
352
353 return true;
354 }
355 return false;

358/// Get the permissions as OR'ed bits from lldb::Permissions
359uint32_t Section::GetPermissions() const {
360 uint32_t permissions = 0;
361 if (m_readable)
362 permissions |= ePermissionsReadable;
363 if (m_writable)
364 permissions |= ePermissionsWritable;
365 if (m_executable)
366 permissions |= ePermissionsExecutable;
367 return permissions;
368}

370/// Set the permissions using bits OR'ed from lldb::Permissions
371void Section::SetPermissions(uint32_t permissions) {
372 m_readable = (permissions & ePermissionsReadable) != 0;
373 m_writable = (permissions & ePermissionsWritable) != 0;
374 m_executable = (permissions & ePermissionsExecutable) != 0;
375}

503size_t SectionList::FindSectionIndex(const Section *sect) {
504 iterator sect_iter;
505 iterator begin = m_sections.begin();
506 iterator end = m_sections.end();
507 for (sect_iter = begin; sect_iter != end; ++sect_iter) {
508 if (sect_iter->get() == sect) {
509 // The secton was already in this section list
510 return std::distance(begin, sect_iter);
511 }
512 }
513 return UINT32_MAX;

524bool SectionList::ReplaceSection(user_id_t sect_id,
525 const lldb::SectionSP &sect_sp,
526 uint32_t depth) {
527 iterator sect_iter, end = m_sections.end();
528 for (sect_iter = m_sections.begin(); sect_iter != end; ++sect_iter) {
529 if ((*sect_iter)->GetID() == sect_id) {
530 *sect_iter = sect_sp;
531 return true;
532 } else if (depth > 0) {
533 if ((*sect_iter)
534 ->GetChildren()
535 .ReplaceSection(sect_id, sect_sp, depth - 1))
536 return true;
537 }
538 }
539 return false;

560SectionSP SectionList::FindSectionByName(ConstString section_dstr) const {
561 SectionSP sect_sp;
562 // Check if we have a valid section string
563 if (section_dstr && !m_sections.empty()) {
564 const_iterator sect_iter;
565 const_iterator end = m_sections.end();
566 for (sect_iter = m_sections.begin();
567 sect_iter != end && sect_sp.get() == nullptr; ++sect_iter) {
568 Section *child_section = sect_iter->get();
569 if (child_section) {
570 if (child_section->GetName() == section_dstr) {
571 sect_sp = *sect_iter;
572 } else {
573 sect_sp =
574 child_section->GetChildren().FindSectionByName(section_dstr);
575 }
576 }
577 }
578 }
579 return sect_sp;
580}

600SectionSP SectionList::FindSectionByType(SectionType sect_type,
601 bool check_children,
602 size_t start_idx) const {
603 SectionSP sect_sp;
604 size_t num_sections = m_sections.size();
605 for (size_t idx = start_idx; idx < num_sections; ++idx) {
606 if (m_sections[idx]->GetType() == sect_type) {
607 sect_sp = m_sections[idx];
608 break;
609 } else if (check_children) {
610 sect_sp = m_sections[idx]->GetChildren().FindSectionByType(
611 sect_type, check_children, 0);
612 if (sect_sp)
613 break;
614 }
615 }
616 return sect_sp;
617}

619SectionSP SectionList::FindSectionContainingFileAddress(addr_t vm_addr,
620 uint32_t depth) const {
621 SectionSP sect_sp;
622 const_iterator sect_iter;
623 const_iterator end = m_sections.end();
624 for (sect_iter = m_sections.begin();
625 sect_iter != end && sect_sp.get() == nullptr; ++sect_iter) {
626 Section *sect = sect_iter->get();
627 if (sect->ContainsFileAddress(vm_addr)) {
628 // The file address is in this section. We need to make sure one of our
629 // child sections doesn't contain this address as well as obeying the
630 // depth limit that was passed in.
631 if (depth > 0)
632 sect_sp = sect->GetChildren().FindSectionContainingFileAddress(
633 vm_addr, depth - 1);
634
635 if (sect_sp.get() == nullptr && !sect->IsFake())
636 sect_sp = *sect_iter;
637 }
638 }
639 return sect_sp;
640}

646void SectionList::Dump(llvm::raw_ostream &s, unsigned indent, Target *target,
647 bool show_header, uint32_t depth) const {
648 bool target_has_loaded_sections = target && target->HasLoadedSections();
649 if (show_header && !m_sections.empty()) {
650 s.indent(indent);
651 s << llvm::formatv(
652 "SectID Type {0} Address "
653 " Perm File Off. File Size Flags Section Name\n",
654 target_has_loaded_sections ? "Load" : "File");
655 s.indent(indent);
656 s << "------------------ ---------------------- "
657 "--------------------------------------- ---- ---------- ---------- "
658 "---------- ----------------------------\n";
659 }
660
661 for (const auto &section_sp : m_sections)
662 section_sp->Dump(s, indent, target_has_loaded_sections ? target : nullptr,
663 depth);
664}

678 for (const auto &section : m_sections) {
679 const SectionList &sub_sections = section->GetChildren();
680 if (sub_sections.GetSize() > 0)

687
688SectionList SectionList::Merge(SectionList &lhs, SectionList &rhs,
689 MergeCallback filter) {
690 SectionList output_list;
691
692 // Iterate through all the sections in lhs and see if we have matches in
693 // the rhs list.
694 for (const auto &lhs_section : lhs) {
695 auto rhs_section = rhs.FindSectionByName(lhs_section->GetName());
696 if (rhs_section)
697 output_list.AddSection(filter(lhs_section, rhs_section));
698 else
699 output_list.AddSection(lhs_section);
700 }
701
702 // Now that we've visited all possible duplicates, we can iterate over
703 // the rhs and take any values not in lhs.
704 for (const auto &rhs_section : rhs) {
705 auto lhs_section = lhs.FindSectionByName(rhs_section->GetName());
706 // Because we already visited everything overlapping between rhs
707 // and lhs, any section not in lhs is unique and can be output.
708 if (!lhs_section)
709 output_list.AddSection(rhs_section);
710 }
711
712 return output_list;
713}
