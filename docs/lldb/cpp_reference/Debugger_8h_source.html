410 /// Interruption in LLDB:
411 ///
412 /// This is a voluntary interruption mechanism, not preemptive. Parts of lldb
413 /// that do work that can be safely interrupted call
414 /// Debugger::InterruptRequested and if that returns true, they should return
415 /// at a safe point, shortcutting the rest of the work they were to do.
416 ///
417 /// lldb clients can both offer a CommandInterpreter (through
418 /// RunCommandInterpreter) and use the SB API's for their own purposes, so it
419 /// is convenient to separate "interrupting the CommandInterpreter execution"
420 /// and interrupting the work it is doing with the SB API's. So there are two
421 /// ways to cause an interrupt:
422 /// * CommandInterpreter::InterruptCommand: Interrupts the command currently
423 /// running in the command interpreter IOHandler thread
424 /// * Debugger::RequestInterrupt: Interrupts are active on anything but the
425 /// CommandInterpreter thread till CancelInterruptRequest is called.
426 ///
427 /// Since the two checks are mutually exclusive, however, it's also convenient
428 /// to have just one function to check the interrupt state.
429
430 /// Bump the "interrupt requested" count on the debugger to support
431 /// cooperative interruption. If this is non-zero, InterruptRequested will
432 /// return true. Interruptible operations are expected to query the
433 /// InterruptRequested API periodically, and interrupt what they were doing
434 /// if it returns \b true.
435 ///
436 void RequestInterrupt();
437
438 /// Decrement the "interrupt requested" counter.
439 void CancelInterruptRequest();
440
441 /// Redraw the statusline if enabled.
442 void RedrawStatusline(std::optional<ExecutionContextRef> exe_ctx_ref);
443
444 /// This is the correct way to query the state of Interruption.
445 /// If you are on the RunCommandInterpreter thread, it will check the
446 /// command interpreter state, and if it is on another thread it will
447 /// check the debugger Interrupt Request state.
448 /// \param[in] cur_func
449 /// For reporting if the interruption was requested. Don't provide this by
450 /// hand, use INTERRUPT_REQUESTED so this gets done consistently.
451 ///
452 /// \param[in] formatv
453 /// A formatv string for the interrupt message. If the elements of the
454 /// message are expensive to compute, you can use the no-argument form of
455 /// InterruptRequested, then make up the report using REPORT_INTERRUPTION.
456 ///
457 /// \return
458 /// A boolean value, if \b true an interruptible operation should interrupt
459 /// itself.
460 template <typename... Args>
461 bool InterruptRequested(const char *cur_func, const char *formatv,
462 Args &&...args) {
463 bool ret_val = InterruptRequested();
464 if (ret_val) {
465 if (!formatv)
466 formatv = "Unknown message";
467 if (!cur_func)
468 cur_func = "<UNKNOWN>";
469 ReportInterruption(InterruptionReport(
470 cur_func, llvm::formatv(formatv, std::forward<Args>(args)...)));
471 }
472 return ret_val;
473 }

475 /// This handy define will keep you from having to generate a report for the
476 /// interruption by hand. Use this except in the case where the arguments to
477 /// the message description are expensive to compute.
478#define INTERRUPT_REQUESTED(debugger, ...) \
479 (debugger).InterruptRequested(__func__, __VA_ARGS__)
