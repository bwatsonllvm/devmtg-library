63static bool IsLetter(char c) {
64 return ('a' <= c && c <= 'z') || ('A' <= c && c <= 'Z');
65}

67static bool IsDigit(char c) { return '0' <= c && c <= '9'; }

69// A word starts with a letter, underscore, or dollar sign, followed by
70// letters ('a'..'z','A'..'Z'), digits ('0'..'9'), and/or underscores.
71static std::optional<llvm::StringRef> IsWord(llvm::StringRef expr,
72 llvm::StringRef &remainder) {
73 // Find the longest prefix consisting of letters, digits, underscors and
74 // '$'. If it doesn't start with a digit, then it's a word.
75 llvm::StringRef candidate = remainder.take_while(
76 [](char c) { return IsDigit(c) || IsLetter(c) || c == '_' || c == '$'; });
77 if (candidate.empty() || IsDigit(candidate[0]))
78 return std::nullopt;
79 remainder = remainder.drop_front(candidate.size());
80 return candidate;
81}

83static bool IsNumberBodyChar(char ch) {
84 return IsDigit(ch) || IsLetter(ch) || ch == '.';

87static std::optional<llvm::StringRef> IsNumber(llvm::StringRef &remainder,
88 bool &isFloat) {
89 llvm::StringRef tail = remainder;
90 llvm::StringRef body = tail.take_while(IsNumberBodyChar);
91 size_t dots = body.count('.');
92 if (dots > 1 || dots == body.size())
93 return std::nullopt;
94 if (IsDigit(body.front()) || (body[0] == '.' && IsDigit(body[1]))) {
95 isFloat = dots == 1;
96 tail = tail.drop_front(body.size());
97 bool isHex = body.contains_insensitive('x');
98 bool hasExp = !isHex && body.contains_insensitive('e');
99 bool hasHexExp = isHex && body.contains_insensitive('p');
100 if (hasExp || hasHexExp) {
101 isFloat = true; // This marks numbers like 0x1p1 and 1e1 as float
102 if (body.ends_with_insensitive("e") || body.ends_with_insensitive("p"))
103 if (tail.consume_front("+") || tail.consume_front("-"))
104 tail = tail.drop_while(IsNumberBodyChar);
105 }
106 size_t number_length = remainder.size() - tail.size();
107 llvm::StringRef number = remainder.take_front(number_length);
108 remainder = remainder.drop_front(number_length);
109 return number;
110 }
111 return std::nullopt;
112}

114static llvm::Error IsNotAllowedByMode(llvm::StringRef expr, Token token,
115 lldb::DILMode mode) {
116 switch (mode) {
117 case lldb::eDILModeSimple:
118 if (!token.IsOneOf({Token::identifier, Token::period, Token::eof})) {
119 return llvm::make_error<DILDiagnosticError>(
120 expr, llvm::formatv("{0} is not allowed in DIL simple mode", token),
121 token.GetLocation());
122 }
123 break;
124 case lldb::eDILModeLegacy:
125 if (!token.IsOneOf({Token::identifier, Token::integer_constant,
126 Token::period, Token::arrow, Token::star, Token::amp,
127 Token::l_square, Token::r_square, Token::eof})) {
128 return llvm::make_error<DILDiagnosticError>(
129 expr, llvm::formatv("{0} is not allowed in DIL legacy mode", token),
130 token.GetLocation());
131 }
132 break;
133 case lldb::eDILModeFull:
134 break;
135 }
136 return llvm::Error::success();
137}

139llvm::Expected<DILLexer> DILLexer::Create(llvm::StringRef expr,
140 lldb::DILMode mode) {
141 std::vector<Token> tokens;
142 llvm::StringRef remainder = expr;
143 do {
144 if (llvm::Expected<Token> t = Lex(expr, remainder)) {
145 Token token = *t;
146 if (llvm::Error error = IsNotAllowedByMode(expr, token, mode))
147 return error;
148 tokens.push_back(std::move(token));
149 } else {
150 return t.takeError();
151 }
152 } while (tokens.back().GetKind() != Token::eof);
153 return DILLexer(expr, std::move(tokens));
154}

156llvm::Expected<Token> DILLexer::Lex(llvm::StringRef expr,
157 llvm::StringRef &remainder) {
158 // Skip over whitespace (spaces).
159 remainder = remainder.ltrim();
160 llvm::StringRef::iterator cur_pos = remainder.begin();
161
162 // Check to see if we've reached the end of our input string.
163 if (remainder.empty())
164 return Token(Token::eof, "", (uint32_t)expr.size());
165
166 uint32_t position = cur_pos - expr.begin();
167 bool isFloat = false;
168 std::optional<llvm::StringRef> maybe_number = IsNumber(remainder, isFloat);
169 if (maybe_number) {
170 auto kind = isFloat ? Token::float_constant : Token::integer_constant;
171 return Token(kind, maybe_number->str(), position);
172 }
173 std::optional<llvm::StringRef> maybe_word = IsWord(expr, remainder);
174 if (maybe_word) {
175 llvm::StringRef word = *maybe_word;
176 Token::Kind kind = llvm::StringSwitch<Token::Kind>(word)
177 .Case("false", Token::kw_false)
178 .Case("true", Token::kw_true)
179 .Default(Token::identifier);
180 return Token(kind, word.str(), position);
181 }

183 constexpr std::pair<Token::Kind, const char *> operators[] = {
184 {Token::amp, "&"}, {Token::arrow, "->"}, {Token::coloncolon, "::"},
185 {Token::colon, ":"}, {Token::l_paren, "("}, {Token::l_square, "["},
186 {Token::minus, "-"}, {Token::period, "."}, {Token::plus, "+"},
187 {Token::r_paren, ")"}, {Token::r_square, "]"}, {Token::star, "*"},
188 };
189 for (auto [kind, str] : operators) {
190 if (remainder.consume_front(str))
191 return Token(kind, str, position);
192 }
193
194 // Unrecognized character(s) in string; unable to lex it.
195 return llvm::make_error<DILDiagnosticError>(expr, "unrecognized token",
196 position);
197}
