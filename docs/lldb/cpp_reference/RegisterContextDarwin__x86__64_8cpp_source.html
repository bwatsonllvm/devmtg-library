632bool RegisterContextDarwin_x86_64::ReadRegister(const RegisterInfo *reg_info,
633 RegisterValue &value) {
634 const uint32_t reg = reg_info->kinds[eRegisterKindLLDB];
635 int set = RegisterContextDarwin_x86_64::GetSetForNativeRegNum(reg);
636 if (set == -1)
637 return false;
638
639 if (ReadRegisterSet(set, false) != 0)
640 return false;
641
642 switch (reg) {
643 case gpr_rax:
644 case gpr_rbx:
645 case gpr_rcx:
646 case gpr_rdx:
647 case gpr_rdi:
648 case gpr_rsi:
