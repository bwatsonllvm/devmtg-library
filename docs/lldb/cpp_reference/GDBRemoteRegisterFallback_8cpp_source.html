82std::vector<DynamicRegisterInfo::Register>
83GetFallbackRegisters(const ArchSpec &arch_to_use) {
84 switch (arch_to_use.GetMachine()) {
85 case llvm::Triple::aarch64:
86 return GetRegisters_aarch64();
87 case llvm::Triple::msp430:
88 return GetRegisters_msp430();
89 case llvm::Triple::x86:
90 return GetRegisters_x86();
91 case llvm::Triple::x86_64:
92 return GetRegisters_x86_64();
93 default:
94 break;
95 }
96
97 return {};
98}
