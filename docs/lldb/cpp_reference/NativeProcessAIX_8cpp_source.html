39// Simple helper function to ensure flags are enabled on the given file
40// descriptor.
41static llvm::Error SetFDFlags(int fd, int flags) {
42 int status = fcntl(fd, F_GETFL);
43 if (status == -1)
44 return errorCodeToError(errnoAsErrorCode());
45 if (fcntl(fd, F_SETFL, status | flags) == -1)
46 return errorCodeToError(errnoAsErrorCode());
47 return Error::success();
48}

60llvm::Expected<std::unique_ptr<NativeProcessProtocol>>
61NativeProcessAIX::Manager::Launch(ProcessLaunchInfo &launch_info,
62 NativeDelegate &native_delegate) {
63 Log *log = GetLog(POSIXLog::Process);
64
65 Status status;
66 ::pid_t pid = ProcessLauncherPosixFork()
67 .LaunchProcess(launch_info, status)
68 .GetProcessId();
69 LLDB_LOG(log, "pid = {0:x}", pid);
70 if (status.Fail()) {
71 LLDB_LOG(log, "failed to launch process: {0}", status);
72 return status.ToError();
73 }
74
75 // Wait for the child process to trap on its call to execve.
76 int wstatus = 0;
77 ::pid_t wpid = llvm::sys::RetryAfterSignal(-1, ::waitpid, pid, &wstatus, 0);
78 assert(wpid == pid);
79 UNUSED_IF_ASSERT_DISABLED(wpid);
80 if (!WIFSTOPPED(wstatus)) {
81 LLDB_LOG(log, "Could not sync with inferior process: wstatus={1}",
82 WaitStatus::Decode(wstatus));
83 return llvm::make_error<StringError>("Could not sync with inferior process",
84 llvm::inconvertibleErrorCode());
85 }
86 LLDB_LOG(log, "inferior started, now in stopped state");
87
88 return std::unique_ptr<NativeProcessAIX>(new NativeProcessAIX(
89 pid, launch_info.GetPTY().ReleasePrimaryFileDescriptor(), native_delegate,
90 HostInfo::GetArchitecture(HostInfo::eArchKind64), *this, {pid}));
91}

112static std::optional<std::pair<lldb::pid_t, WaitStatus>> WaitPid() {
113 Log *log = GetLog(POSIXLog::Process);
114
115 int status;
116 ::pid_t wait_pid =
117 llvm::sys::RetryAfterSignal(-1, ::waitpid, -1, &status, WNOHANG);
118
119 if (wait_pid == 0)
120 return std::nullopt;
121
122 if (wait_pid == -1) {
123 Status error(errno, eErrorTypePOSIX);
124 LLDB_LOG(log, "waitpid(-1, &status, _) failed: {0}", error);
125 return std::nullopt;
126 }
127
128 WaitStatus wait_status = WaitStatus::Decode(status);
129
130 LLDB_LOG(log, "waitpid(-1, &status, _) = {0}, status = {1}", wait_pid,
131 wait_status);
132 return std::make_pair(wait_pid, wait_status);
133}

147NativeProcessAIX::NativeProcessAIX(::pid_t pid, int terminal_fd,
148 NativeDelegate &delegate,
149 const ArchSpec &arch, Manager &manager,
150 llvm::ArrayRef<::pid_t> tids)
151 : NativeProcessProtocol(pid, terminal_fd, delegate), m_manager(manager),
152 m_arch(arch) {

154 if (m_terminal_fd != -1)
155 cantFail(SetFDFlags(m_terminal_fd, O_NONBLOCK));
156
157 // Let our process instance know the thread has stopped.
158 SetCurrentThreadID(tids[0]);
159 SetState(StateType::eStateStopped, false);
160}

162llvm::Expected<std::vector<::pid_t>> NativeProcessAIX::Attach(::pid_t pid) {
163 Log *log = GetLog(POSIXLog::Process);
164 Status status;
165 if (llvm::Error err = PtraceWrapper(PT_ATTACH, pid).takeError())
166 return err;
167
168 int wpid = llvm::sys::RetryAfterSignal(-1, ::waitpid, pid, nullptr, WNOHANG);
169 if (wpid <= 0)
170 return llvm::errorCodeToError(errnoAsErrorCode());
171 LLDB_LOG(log, "adding pid = {0}", pid);
172
173 return std::vector<::pid_t>{pid};
174}

190Status NativeProcessAIX::Kill() {
191
192 Log *log = GetLog(POSIXLog::Process);
193 LLDB_LOG(log, "pid {0}", GetID());
194
195 Status error;
196
197 switch (m_state) {
198 case StateType::eStateInvalid:
199 case StateType::eStateExited:
200 case StateType::eStateCrashed:
201 case StateType::eStateDetached:
202 case StateType::eStateUnloaded:
203 // Nothing to do - the process is already dead.
204 LLDB_LOG(log, "ignored for PID {0} due to current state: {1}", GetID(),
205 m_state);
206 return error;
207
208 case StateType::eStateConnected:
209 case StateType::eStateAttaching:
210 case StateType::eStateLaunching:
211 case StateType::eStateStopped:
212 case StateType::eStateRunning:
213 case StateType::eStateStepping:
214 case StateType::eStateSuspended:
215 // We can try to kill a process in these states.
216 break;
217 }
218
219 llvm::Expected<int> result =
220 PtraceWrapper(PT_KILL, GetID(), nullptr, nullptr, 0);
221 if (!result) {
222 std::string error_string = std::string("Kill failed for process. error: ") +
223 llvm::toString(result.takeError());
224 error.FromErrorString(error_string.c_str());
225 }
226 return error;
227}

268llvm::Expected<int> NativeProcessAIX::PtraceWrapper(int req, lldb::pid_t pid,
269 void *addr, void *data,
270 size_t data_size) {
271 int ret;
272
273 Log *log = GetLog(POSIXLog::Ptrace);
274 switch (req) {
275 case PT_ATTACH:
276 case PT_DETACH:
277 case PT_KILL:
278 ret = ptrace64(req, pid, 0, 0, nullptr);
279 break;
280 default:
281 llvm_unreachable("PT_ request not supported yet.");
282 }
283
284 LLDB_LOG(log, "ptrace({0}, {1}, {2}, {3}, {4})={5:x}", req, pid, addr, data,
285 data_size, ret);
286
287 if (ret == -1) {
288 LLDB_LOG(log, "ptrace() failed");
289 return llvm::errorCodeToError(errnoAsErrorCode());
290 }
291 return ret;
292}
