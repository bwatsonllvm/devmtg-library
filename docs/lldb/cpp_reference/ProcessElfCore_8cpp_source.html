50lldb::ProcessSP ProcessElfCore::CreateInstance(lldb::TargetSP target_sp,
51 lldb::ListenerSP listener_sp,
52 const FileSpec *crash_file,
53 bool can_connect) {
54 lldb::ProcessSP process_sp;
55 if (crash_file && !can_connect) {
56 // Read enough data for an ELF32 header or ELF64 header Note: Here we care
57 // about e_type field only, so it is safe to ignore possible presence of
58 // the header extension.
59 const size_t header_size = sizeof(llvm::ELF::Elf64_Ehdr);
60
61 auto data_sp = FileSystem::Instance().CreateDataBuffer(
62 crash_file->GetPath(), header_size, 0);
63 if (data_sp && data_sp->GetByteSize() == header_size &&
64 elf::ELFHeader::MagicBytesMatch(data_sp->GetBytes())) {
65 elf::ELFHeader elf_header;
66 DataExtractor data(data_sp, lldb::eByteOrderLittle, 4);
67 lldb::offset_t data_offset = 0;
68 if (elf_header.Parse(data, &data_offset)) {
69 // Check whether we're dealing with a raw FreeBSD "full memory dump"
70 // ELF vmcore that needs to be handled via FreeBSDKernel plugin instead.
71 if (elf_header.e_ident[7] == 0xFF && elf_header.e_version == 0)
72 return process_sp;
73 if (elf_header.e_type == llvm::ELF::ET_CORE)
74 process_sp = std::make_shared<ProcessElfCore>(target_sp, listener_sp,
75 *crash_file);
76 }
77 }
78 }
79 return process_sp;
80}

115lldb::addr_t ProcessElfCore::AddAddressRangeFromLoadSegment(
116 const elf::ELFProgramHeader &header) {
117 const lldb::addr_t addr = header.p_vaddr;
118 FileRange file_range(header.p_offset, header.p_filesz);
119 VMRangeToFileOffset::Entry range_entry(addr, header.p_memsz, file_range);
120
121 // Only add to m_core_aranges if the file size is non zero. Some core files
122 // have PT_LOAD segments for all address ranges, but set f_filesz to zero for
123 // the .text sections since they can be retrieved from the object files.
124 if (header.p_filesz > 0) {
125 VMRangeToFileOffset::Entry *last_entry = m_core_aranges.Back();
126 if (last_entry && last_entry->GetRangeEnd() == range_entry.GetRangeBase() &&
127 last_entry->data.GetRangeEnd() == range_entry.data.GetRangeBase() &&
128 last_entry->GetByteSize() == last_entry->data.GetByteSize()) {
129 last_entry->SetRangeEnd(range_entry.GetRangeEnd());
130 last_entry->data.SetRangeEnd(range_entry.data.GetRangeEnd());
131 } else {
132 m_core_aranges.Append(range_entry);
133 }
134 }
135 // Keep a separate map of permissions that isn't coalesced so all ranges
136 // are maintained.
137 const uint32_t permissions =
138 ((header.p_flags & llvm::ELF::PF_R) ? lldb::ePermissionsReadable : 0u) |
139 ((header.p_flags & llvm::ELF::PF_W) ? lldb::ePermissionsWritable : 0u) |
140 ((header.p_flags & llvm::ELF::PF_X) ? lldb::ePermissionsExecutable : 0u);
141
142 m_core_range_infos.Append(
143 VMRangeToPermissions::Entry(addr, header.p_memsz, permissions));
144
145 return addr;
146}

160// Process Control
161Status ProcessElfCore::DoLoadCore() {
162 Status error;
163 if (!m_core_module_sp) {
164 error = Status::FromErrorString("invalid core module");
165 return error;
166 }
167
168 ObjectFileELF *core = (ObjectFileELF *)(m_core_module_sp->GetObjectFile());
169 if (core == nullptr) {
170 error = Status::FromErrorString("invalid core object file");
171 return error;
172 }
173
174 llvm::ArrayRef<elf::ELFProgramHeader> segments = core->ProgramHeaders();
175 if (segments.size() == 0) {
176 error = Status::FromErrorString("core file has no segments");
177 return error;
178 }
179
180 // Even if the architecture is set in the target, we need to override it to
181 // match the core file which is always single arch.
182 ArchSpec arch(m_core_module_sp->GetArchitecture());
183
184 ArchSpec target_arch = GetTarget().GetArchitecture();
185 ArchSpec core_arch(m_core_module_sp->GetArchitecture());
186 target_arch.MergeFrom(core_arch);
187 GetTarget().SetArchitecture(target_arch, /*set_platform*/ true);
188
189 SetUnixSignals(UnixSignals::Create(GetArchitecture()));
190
191 SetCanJIT(false);
192
193 m_thread_data_valid = true;
194
195 bool ranges_are_sorted = true;
196 lldb::addr_t vm_addr = 0;
197 lldb::addr_t tag_addr = 0;
198 /// Walk through segments and Thread and Address Map information.
199 /// PT_NOTE - Contains Thread and Register information
200 /// PT_LOAD - Contains a contiguous range of Process Address Space
201 /// PT_AARCH64_MEMTAG_MTE - Contains AArch64 MTE memory tags for a range of
202 /// Process Address Space.
203 for (const elf::ELFProgramHeader &H : segments) {
204 DataExtractor data = core->GetSegmentData(H);
205
206 // Parse thread contexts and auxv structure
207 if (H.p_type == llvm::ELF::PT_NOTE) {
208 if (llvm::Error error = ParseThreadContextsFromNoteSegment(H, data))
209 return Status::FromError(std::move(error));
210 }
211 // PT_LOAD segments contains address map
212 if (H.p_type == llvm::ELF::PT_LOAD) {
213 lldb::addr_t last_addr = AddAddressRangeFromLoadSegment(H);
214 if (vm_addr > last_addr)
215 ranges_are_sorted = false;
216 vm_addr = last_addr;
217 } else if (H.p_type == llvm::ELF::PT_AARCH64_MEMTAG_MTE) {
218 lldb::addr_t last_addr = AddAddressRangeFromMemoryTagSegment(H);
219 if (tag_addr > last_addr)
220 ranges_are_sorted = false;
221 tag_addr = last_addr;
222 }
223 }
224
225 if (!ranges_are_sorted) {
226 m_core_aranges.Sort();
227 m_core_range_infos.Sort();
228 m_core_tag_ranges.Sort();
229 }
230
231 // Ensure we found at least one thread that was stopped on a signal.
232 bool siginfo_signal_found = false;
233 bool prstatus_signal_found = false;
234 // Check we found a signal in a SIGINFO note.
235 for (const auto &thread_data : m_thread_data) {
236 if (!thread_data.siginfo_bytes.empty() || thread_data.signo != 0)
237 siginfo_signal_found = true;
238 if (thread_data.prstatus_sig != 0)
239 prstatus_signal_found = true;
240 }
241 if (!siginfo_signal_found) {
242 // If we don't have signal from SIGINFO use the signal from each threads
243 // PRSTATUS note.
244 if (prstatus_signal_found) {
245 for (auto &thread_data : m_thread_data)
246 thread_data.signo = thread_data.prstatus_sig;
247 } else if (m_thread_data.size() > 0) {
248 // If all else fails force the first thread to be SIGSTOP
249 m_thread_data.begin()->signo =
250 GetUnixSignals()->GetSignalNumberFromName("SIGSTOP");
251 }
252 }
253
254 // Try to find gnu build id before we load the executable.
255 UpdateBuildIdForNTFileEntries();
256
257 // Core files are useless without the main executable. See if we can locate
258 // the main executable using data we found in the core file notes.
259 lldb::ModuleSP exe_module_sp = GetTarget().GetExecutableModule();
260 if (!exe_module_sp) {
261 if (!m_nt_file_entries.empty()) {
262 std::string executable_path = GetMainExecutablePath();
263 ModuleSpec exe_module_spec;
264 exe_module_spec.GetArchitecture() = arch;
265 exe_module_spec.GetUUID() = FindModuleUUID(executable_path);
266 exe_module_spec.GetFileSpec().SetFile(executable_path,
267 FileSpec::Style::native);
268 if (exe_module_spec.GetFileSpec()) {
269 exe_module_sp =
270 GetTarget().GetOrCreateModule(exe_module_spec, true /* notify */);
271 if (!exe_module_sp) {
272 // Create an ELF file from memory for the main executable. The dynamic
273 // loader requires the main executable so that it can extract the
274 // DT_DEBUG key/value pair from the dynamic section and get the list
275 // of shared libraries.
276 std::optional<lldb::addr_t> exe_header_addr;
277
278 // We need to find its load address
279 for (const NT_FILE_Entry &file_entry : m_nt_file_entries) {
280 if (file_entry.path == executable_path) {
281 exe_header_addr = file_entry.start;
282 break;
283 }
284 }
285 if (exe_header_addr) {
286 if (llvm::Expected<lldb::ModuleSP> module_sp_or_err =
287 ReadModuleFromMemory(exe_module_spec.GetFileSpec(),
288 *exe_header_addr))
289 exe_module_sp = *module_sp_or_err;
290 else
291 llvm::consumeError(module_sp_or_err.takeError());
292 }
293 }
294 if (exe_module_sp)
295 GetTarget().SetExecutableModule(exe_module_sp, eLoadDependentsNo);
296 }
297 }
298 }
299 return error;
300}

302void ProcessElfCore::UpdateBuildIdForNTFileEntries() {
303 Log *log = GetLog(LLDBLog::Process);
304 m_uuids.clear();
305 for (NT_FILE_Entry &entry : m_nt_file_entries) {
306 UUID uuid = FindBuidIdInCoreMemory(entry.start);
307 if (uuid.IsValid()) {
308 // Assert that either the path is not in the map or the UUID matches
309 assert(m_uuids.count(entry.path) == 0 || m_uuids[entry.path] == uuid);
310 m_uuids[entry.path] = uuid;
311 if (log)
312 LLDB_LOGF(log, "%s found UUID @ %16.16" PRIx64 ": %s \"%s\"",
313 __FUNCTION__, entry.start, uuid.GetAsString().c_str(),
314 entry.path.c_str());
315 }
316 }

319std::string ProcessElfCore::GetMainExecutablePath() {
320 // Always try to read the program name from core file memory first via the
321 // AUXV_AT_EXECFN entry. This value is the address of a null terminated C
322 // string that contains the program path.
323 AuxVector aux_vector(m_auxv);
324 std::string execfn_str;
325 if (auto execfn = aux_vector.GetAuxValue(AuxVector::AUXV_AT_EXECFN)) {
326 Status error;
327 if (ReadCStringFromMemory(*execfn, execfn_str, error))
328 return execfn_str;
329 }
330
331 if (m_nt_file_entries.empty())
332 return {};
333
334 // The first entry in the NT_FILE might be our executable
335 std::string executable_path = m_nt_file_entries[0].path;
336 // Prefer the NT_FILE entry matching m_executable_name as main executable.
337 for (const NT_FILE_Entry &file_entry : m_nt_file_entries)
338 if (llvm::StringRef(file_entry.path).ends_with("/" + m_executable_name)) {
339 executable_path = file_entry.path;
340 break;
341 }
342 return executable_path;
343}

345UUID ProcessElfCore::FindModuleUUID(const llvm::StringRef path) {
346 // Lookup the UUID for the given path in the map.
347 // Note that this could be called by multiple threads so make sure
348 // we access the map in a thread safe way (i.e. don't use operator[]).
349 auto it = m_uuids.find(std::string(path));
350 if (it != m_uuids.end())
351 return it->second;
352 return UUID();
353}

395Status ProcessElfCore::DoGetMemoryRegionInfo(lldb::addr_t load_addr,
396 MemoryRegionInfo &region_info) {
397 region_info.Clear();
398 const VMRangeToPermissions::Entry *permission_entry =
399 m_core_range_infos.FindEntryThatContainsOrFollows(load_addr);
400 if (permission_entry) {
401 if (permission_entry->Contains(load_addr)) {
402 region_info.GetRange().SetRangeBase(permission_entry->GetRangeBase());
403 region_info.GetRange().SetRangeEnd(permission_entry->GetRangeEnd());
404 const Flags permissions(permission_entry->data);
405 region_info.SetReadable(permissions.Test(lldb::ePermissionsReadable)
406 ? MemoryRegionInfo::eYes
407 : MemoryRegionInfo::eNo);
408 region_info.SetWritable(permissions.Test(lldb::ePermissionsWritable)
409 ? MemoryRegionInfo::eYes
410 : MemoryRegionInfo::eNo);
411 region_info.SetExecutable(permissions.Test(lldb::ePermissionsExecutable)
412 ? MemoryRegionInfo::eYes
413 : MemoryRegionInfo::eNo);
414 region_info.SetMapped(MemoryRegionInfo::eYes);
415
416 // A region is memory tagged if there is a memory tag segment that covers
417 // the exact same range.
418 region_info.SetMemoryTagged(MemoryRegionInfo::eNo);
419 const VMRangeToFileOffset::Entry *tag_entry =
420 m_core_tag_ranges.FindEntryStartsAt(permission_entry->GetRangeBase());
421 if (tag_entry &&
422 tag_entry->GetRangeEnd() == permission_entry->GetRangeEnd())
423 region_info.SetMemoryTagged(MemoryRegionInfo::eYes);
424 } else if (load_addr < permission_entry->GetRangeBase()) {
425 region_info.GetRange().SetRangeBase(load_addr);
426 region_info.GetRange().SetRangeEnd(permission_entry->GetRangeBase());
427 region_info.SetReadable(MemoryRegionInfo::eNo);
428 region_info.SetWritable(MemoryRegionInfo::eNo);
429 region_info.SetExecutable(MemoryRegionInfo::eNo);
430 region_info.SetMapped(MemoryRegionInfo::eNo);
431 region_info.SetMemoryTagged(MemoryRegionInfo::eNo);
432 }
433 return Status();
434 }
435
436 region_info.GetRange().SetRangeBase(load_addr);
437 region_info.GetRange().SetRangeEnd(LLDB_INVALID_ADDRESS);
438 region_info.SetReadable(MemoryRegionInfo::eNo);
439 region_info.SetWritable(MemoryRegionInfo::eNo);
440 region_info.SetExecutable(MemoryRegionInfo::eNo);
441 region_info.SetMapped(MemoryRegionInfo::eNo);
442 region_info.SetMemoryTagged(MemoryRegionInfo::eNo);
443 return Status();
444}

446size_t ProcessElfCore::DoReadMemory(lldb::addr_t addr, void *buf, size_t size,
447 Status &error) {
448 ObjectFile *core_objfile = m_core_module_sp->GetObjectFile();
449
450 if (core_objfile == nullptr)
451 return 0;
452
453 // Get the address range
454 const VMRangeToFileOffset::Entry *address_range =
455 m_core_aranges.FindEntryThatContains(addr);
456 if (address_range == nullptr || address_range->GetRangeEnd() < addr) {
457 error = Status::FromErrorStringWithFormat(
458 "core file does not contain 0x%" PRIx64, addr);
459 return 0;
460 }
461
462 // Convert the address into core file offset
463 const lldb::addr_t offset = addr - address_range->GetRangeBase();
464 const lldb::addr_t file_start = address_range->data.GetRangeBase();
465 const lldb::addr_t file_end = address_range->data.GetRangeEnd();
466 size_t bytes_to_read = size; // Number of bytes to read from the core file
467 size_t bytes_copied = 0; // Number of bytes actually read from the core file
468 lldb::addr_t bytes_left =
469 0; // Number of bytes available in the core file from the given address
470
471 // Don't proceed if core file doesn't contain the actual data for this
472 // address range.
473 if (file_start == file_end)
474 return 0;
475
476 // Figure out how many on-disk bytes remain in this segment starting at the
477 // given offset
478 if (file_end > file_start + offset)
479 bytes_left = file_end - (file_start + offset);
480
481 if (bytes_to_read > bytes_left)
482 bytes_to_read = bytes_left;
483
484 // If there is data available on the core file read it
485 if (bytes_to_read)
486 bytes_copied =
487 core_objfile->CopyData(offset + file_start, bytes_to_read, buf);
488
489 return bytes_copied;
490}

492llvm::Expected<std::vector<lldb::addr_t>>
493ProcessElfCore::ReadMemoryTags(lldb::addr_t addr, size_t len) {
494 ObjectFile *core_objfile = m_core_module_sp->GetObjectFile();
495 if (core_objfile == nullptr)
496 return llvm::createStringError(llvm::inconvertibleErrorCode(),
497 "No core object file.");
498
499 llvm::Expected<const MemoryTagManager *> tag_manager_or_err =
500 GetMemoryTagManager();
501 if (!tag_manager_or_err)
502 return tag_manager_or_err.takeError();
503
504 // LLDB only supports AArch64 MTE tag segments so we do not need to worry
505 // about the segment type here. If you got here then you must have a tag
506 // manager (meaning you are debugging AArch64) and all the segments in this
507 // list will have had type PT_AARCH64_MEMTAG_MTE.
508 const VMRangeToFileOffset::Entry *tag_entry =
509 m_core_tag_ranges.FindEntryThatContains(addr);
510 // If we don't have a tag segment or the range asked for extends outside the
511 // segment.
512 if (!tag_entry || (addr + len) >= tag_entry->GetRangeEnd())
513 return llvm::createStringError(llvm::inconvertibleErrorCode(),
514 "No tag segment that covers this range.");
515
516 const MemoryTagManager *tag_manager = *tag_manager_or_err;
517 return tag_manager->UnpackTagsFromCoreFileSegment(
518 [core_objfile](lldb::offset_t offset, size_t length, void *dst) {
519 return core_objfile->CopyData(offset, length, dst);
520 },
521 tag_entry->GetRangeBase(), tag_entry->data.GetRangeBase(), addr, len);
522}

548// Parse a FreeBSD NT_PRSTATUS note - see FreeBSD sys/procfs.h for details.
549static void ParseFreeBSDPrStatus(ThreadData &thread_data,
550 const DataExtractor &data,
551 bool lp64) {
552 lldb::offset_t offset = 0;
553 int pr_version = data.GetU32(&offset);
554
555 Log *log = GetLog(LLDBLog::Process);
556 if (log) {
557 if (pr_version > 1)
558 LLDB_LOGF(log, "FreeBSD PRSTATUS unexpected version %d", pr_version);
559 }
560
561 // Skip padding, pr_statussz, pr_gregsetsz, pr_fpregsetsz, pr_osreldate
562 if (lp64)
563 offset += 32;
564 else
565 offset += 16;
566
567 thread_data.signo = data.GetU32(&offset); // pr_cursig
568 thread_data.tid = data.GetU32(&offset); // pr_pid
569 if (lp64)
570 offset += 4;
571
572 size_t len = data.GetByteSize() - offset;
573 thread_data.gpregset = DataExtractor(data, offset, len);
574}

576// Parse a FreeBSD NT_PRPSINFO note - see FreeBSD sys/procfs.h for details.
577static void ParseFreeBSDPrPsInfo(ProcessElfCore &process,
578 const DataExtractor &data,
579 bool lp64) {
580 lldb::offset_t offset = 0;
581 int pr_version = data.GetU32(&offset);
582
583 Log *log = GetLog(LLDBLog::Process);
584 if (log) {
585 if (pr_version > 1)
586 LLDB_LOGF(log, "FreeBSD PRPSINFO unexpected version %d", pr_version);
587 }
588
589 // Skip pr_psinfosz, pr_fname, pr_psargs
590 offset += 108;
591 if (lp64)
592 offset += 4;
593
594 process.SetID(data.GetU32(&offset)); // pr_pid
595}

597static llvm::Error ParseNetBSDProcInfo(const DataExtractor &data,
598 uint32_t &cpi_nlwps,
599 uint32_t &cpi_signo,
600 uint32_t &cpi_siglwp,
601 uint32_t &cpi_pid) {
602 lldb::offset_t offset = 0;
603
604 uint32_t version = data.GetU32(&offset);
605 if (version != 1)

613 "Error parsing NetBSD core(5) notes: Unsupported procinfo size",
614 llvm::inconvertibleErrorCode());

616 cpi_signo = data.GetU32(&offset); /* killing signal */
617
618 offset += NETBSD::NT_PROCINFO_CPI_SIGCODE_SIZE;
619 offset += NETBSD::NT_PROCINFO_CPI_SIGPEND_SIZE;
620 offset += NETBSD::NT_PROCINFO_CPI_SIGMASK_SIZE;
621 offset += NETBSD::NT_PROCINFO_CPI_SIGIGNORE_SIZE;
622 offset += NETBSD::NT_PROCINFO_CPI_SIGCATCH_SIZE;

624 offset += NETBSD::NT_PROCINFO_CPI_PPID_SIZE;
625 offset += NETBSD::NT_PROCINFO_CPI_PGRP_SIZE;
626 offset += NETBSD::NT_PROCINFO_CPI_SID_SIZE;
627 offset += NETBSD::NT_PROCINFO_CPI_RUID_SIZE;
628 offset += NETBSD::NT_PROCINFO_CPI_EUID_SIZE;
629 offset += NETBSD::NT_PROCINFO_CPI_SVUID_SIZE;
630 offset += NETBSD::NT_PROCINFO_CPI_RGID_SIZE;
631 offset += NETBSD::NT_PROCINFO_CPI_EGID_SIZE;
632 offset += NETBSD::NT_PROCINFO_CPI_SVGID_SIZE;

635 offset += NETBSD::NT_PROCINFO_CPI_NAME_SIZE;
636 cpi_siglwp = data.GetU32(&offset); /* LWP target of killing signal */

638 return llvm::Error::success();

728/// NetBSD specific Thread context from PT_NOTE segment
729///
730/// NetBSD ELF core files use notes to provide information about
731/// the process's state. The note name is "NetBSD-CORE" for
732/// information that is global to the process, and "NetBSD-CORE@nn",
733/// where "nn" is the lwpid of the LWP that the information belongs
734/// to (such as register state).
735///
736/// NetBSD uses the following note identifiers:
737///
738/// ELF_NOTE_NETBSD_CORE_PROCINFO (value 1)
739/// Note is a "netbsd_elfcore_procinfo" structure.
740/// ELF_NOTE_NETBSD_CORE_AUXV (value 2; since NetBSD 8.0)
741/// Note is an array of AuxInfo structures.
742///
743/// NetBSD also uses ptrace(2) request numbers (the ones that exist in
744/// machine-dependent space) to identify register info notes. The
745/// info in such notes is in the same format that ptrace(2) would
746/// export that information.
747///
748/// For more information see /usr/include/sys/exec_elf.h
749///
750llvm::Error ProcessElfCore::parseNetBSDNotes(llvm::ArrayRef<CoreNote> notes) {
751 ThreadData thread_data;
752 bool had_nt_regs = false;
753
754 // To be extracted from struct netbsd_elfcore_procinfo
755 // Used to sanity check of the LWPs of the process
756 uint32_t nlwps = 0;
757 uint32_t signo = 0; // killing signal
758 uint32_t siglwp = 0; // LWP target of killing signal
759 uint32_t pr_pid = 0;
760
761 for (const auto &note : notes) {
762 llvm::StringRef name = note.info.n_name;
763
764 if (name == "NetBSD-CORE") {
765 if (note.info.n_type == NETBSD::NT_PROCINFO) {
766 llvm::Error error = ParseNetBSDProcInfo(note.data, nlwps, signo,
767 siglwp, pr_pid);
768 if (error)
769 return error;
770 SetID(pr_pid);
771 } else if (note.info.n_type == NETBSD::NT_AUXV) {
772 m_auxv = note.data;
773 }
774 } else if (name.consume_front("NetBSD-CORE@")) {
775 lldb::tid_t tid;
776 if (name.getAsInteger(10, tid))
777 return llvm::make_error<llvm::StringError>(
778 "Error parsing NetBSD core(5) notes: Cannot convert LWP ID "
779 "to integer",
780 llvm::inconvertibleErrorCode());
781
782 switch (GetArchitecture().GetMachine()) {
783 case llvm::Triple::aarch64: {
784 // Assume order PT_GETREGS, PT_GETFPREGS
785 if (note.info.n_type == NETBSD::AARCH64::NT_REGS) {
786 // If this is the next thread, push the previous one first.
787 if (had_nt_regs) {
788 m_thread_data.push_back(thread_data);
789 thread_data = ThreadData();
790 had_nt_regs = false;
791 }
792
793 thread_data.gpregset = note.data;
794 thread_data.tid = tid;
795 if (thread_data.gpregset.GetByteSize() == 0)
796 return llvm::make_error<llvm::StringError>(
797 "Could not find general purpose registers note in core file.",
798 llvm::inconvertibleErrorCode());
799 had_nt_regs = true;
800 } else if (note.info.n_type == NETBSD::AARCH64::NT_FPREGS) {
801 if (!had_nt_regs || tid != thread_data.tid)
802 return llvm::make_error<llvm::StringError>(
803 "Error parsing NetBSD core(5) notes: Unexpected order "
804 "of NOTEs PT_GETFPREG before PT_GETREG",
805 llvm::inconvertibleErrorCode());
806 thread_data.notes.push_back(note);
807 }
808 } break;
809 case llvm::Triple::x86: {
810 // Assume order PT_GETREGS, PT_GETFPREGS
811 if (note.info.n_type == NETBSD::I386::NT_REGS) {
812 // If this is the next thread, push the previous one first.
813 if (had_nt_regs) {
814 m_thread_data.push_back(thread_data);
815 thread_data = ThreadData();
816 had_nt_regs = false;
817 }
818
819 thread_data.gpregset = note.data;
820 thread_data.tid = tid;
821 if (thread_data.gpregset.GetByteSize() == 0)
822 return llvm::make_error<llvm::StringError>(
823 "Could not find general purpose registers note in core file.",
824 llvm::inconvertibleErrorCode());
825 had_nt_regs = true;
826 } else if (note.info.n_type == NETBSD::I386::NT_FPREGS) {
827 if (!had_nt_regs || tid != thread_data.tid)
828 return llvm::make_error<llvm::StringError>(
829 "Error parsing NetBSD core(5) notes: Unexpected order "
830 "of NOTEs PT_GETFPREG before PT_GETREG",
831 llvm::inconvertibleErrorCode());
832 thread_data.notes.push_back(note);
833 }
834 } break;
835 case llvm::Triple::x86_64: {
836 // Assume order PT_GETREGS, PT_GETFPREGS
837 if (note.info.n_type == NETBSD::AMD64::NT_REGS) {
838 // If this is the next thread, push the previous one first.
839 if (had_nt_regs) {
840 m_thread_data.push_back(thread_data);
841 thread_data = ThreadData();
842 had_nt_regs = false;
843 }
844
845 thread_data.gpregset = note.data;
846 thread_data.tid = tid;
847 if (thread_data.gpregset.GetByteSize() == 0)
848 return llvm::make_error<llvm::StringError>(
849 "Could not find general purpose registers note in core file.",
850 llvm::inconvertibleErrorCode());
851 had_nt_regs = true;
852 } else if (note.info.n_type == NETBSD::AMD64::NT_FPREGS) {
853 if (!had_nt_regs || tid != thread_data.tid)
854 return llvm::make_error<llvm::StringError>(
855 "Error parsing NetBSD core(5) notes: Unexpected order "
856 "of NOTEs PT_GETFPREG before PT_GETREG",
857 llvm::inconvertibleErrorCode());
858 thread_data.notes.push_back(note);
859 }
860 } break;
861 default:
862 break;
863 }
864 }
865 }
866
867 // Push the last thread.
868 if (had_nt_regs)
869 m_thread_data.push_back(thread_data);
870
871 if (m_thread_data.empty())
872 return llvm::make_error<llvm::StringError>(
873 "Error parsing NetBSD core(5) notes: No threads information "
874 "specified in notes",
875 llvm::inconvertibleErrorCode());
876
877 if (m_thread_data.size() != nlwps)
878 return llvm::make_error<llvm::StringError>(
879 "Error parsing NetBSD core(5) notes: Mismatch between the number "
880 "of LWPs in netbsd_elfcore_procinfo and the number of LWPs specified "
881 "by MD notes",
882 llvm::inconvertibleErrorCode());
883
884 // Signal targeted at the whole process.
885 if (siglwp == 0) {
886 for (auto &data : m_thread_data)
887 data.signo = signo;
888 }
889 // Signal destined for a particular LWP.
890 else {
891 bool passed = false;
892
893 for (auto &data : m_thread_data) {
894 if (data.tid == siglwp) {
895 data.signo = signo;
896 passed = true;
897 break;
898 }
899 }
900
901 if (!passed)
902 return llvm::make_error<llvm::StringError>(
903 "Error parsing NetBSD core(5) notes: Signal passed to unknown LWP",
904 llvm::inconvertibleErrorCode());
905 }
906
907 return llvm::Error::success();
908}

942/// A description of a linux process usually contains the following NOTE
943/// entries:
944/// - NT_PRPSINFO - General process information like pid, uid, name, ...
945/// - NT_SIGINFO - Information about the signal that terminated the process
946/// - NT_AUXV - Process auxiliary vector
947/// - NT_FILE - Files mapped into memory
948///
949/// Additionally, for each thread in the process the core file will contain at
950/// least the NT_PRSTATUS note, containing the thread id and general purpose
951/// registers. It may include additional notes for other register sets (floating
952/// point and vector registers, ...). The tricky part here is that some of these
953/// notes have "CORE" in their owner fields, while other set it to "LINUX".
954llvm::Error ProcessElfCore::parseLinuxNotes(llvm::ArrayRef<CoreNote> notes) {
955 const ArchSpec &arch = GetArchitecture();
956 bool have_prstatus = false;
957 bool have_prpsinfo = false;
958 ThreadData thread_data;
959 for (const auto &note : notes) {
960 if (note.info.n_name != "CORE" && note.info.n_name != "LINUX")
961 continue;
962
963 if ((note.info.n_type == ELF::NT_PRSTATUS && have_prstatus) ||
964 (note.info.n_type == ELF::NT_PRPSINFO && have_prpsinfo)) {
965 assert(thread_data.gpregset.GetByteSize() > 0);
966 // Add the new thread to thread list
967 m_thread_data.push_back(thread_data);
968 thread_data = ThreadData();
969 have_prstatus = false;
970 have_prpsinfo = false;
971 }
972
973 switch (note.info.n_type) {
974 case ELF::NT_PRSTATUS: {
975 have_prstatus = true;
976 ELFLinuxPrStatus prstatus;
977 Status status = prstatus.Parse(note.data, arch);
978 if (status.Fail())
979 return status.ToError();
980 thread_data.prstatus_sig = prstatus.pr_cursig;
981 thread_data.tid = prstatus.pr_pid;
982 uint32_t header_size = ELFLinuxPrStatus::GetSize(arch);
983 size_t len = note.data.GetByteSize() - header_size;
984 thread_data.gpregset = DataExtractor(note.data, header_size, len);
985 break;
986 }
987 case ELF::NT_PRPSINFO: {
988 have_prpsinfo = true;
989 ELFLinuxPrPsInfo prpsinfo;
990 Status status = prpsinfo.Parse(note.data, arch);
991 if (status.Fail())
992 return status.ToError();
993 thread_data.name.assign (prpsinfo.pr_fname, strnlen (prpsinfo.pr_fname, sizeof (prpsinfo.pr_fname)));
994 SetID(prpsinfo.pr_pid);
995 m_executable_name = thread_data.name;
996 break;
997 }
998 case ELF::NT_SIGINFO: {
999 lldb::offset_t size = note.data.GetByteSize();
1000 lldb::offset_t offset = 0;
1001 const char *bytes =
1002 static_cast<const char *>(note.data.GetData(&offset, size));
1003 thread_data.siginfo_bytes = llvm::StringRef(bytes, size);
1004 break;
1005 }
1006 case ELF::NT_FILE: {
1007 m_nt_file_entries.clear();
1008 lldb::offset_t offset = 0;
1009 const uint64_t count = note.data.GetAddress(&offset);
1010 note.data.GetAddress(&offset); // Skip page size
1011 for (uint64_t i = 0; i < count; ++i) {
1012 NT_FILE_Entry entry;
1013 entry.start = note.data.GetAddress(&offset);
1014 entry.end = note.data.GetAddress(&offset);
1015 entry.file_ofs = note.data.GetAddress(&offset);
1016 m_nt_file_entries.push_back(entry);
1017 }
1018 for (uint64_t i = 0; i < count; ++i) {
1019 const char *path = note.data.GetCStr(&offset);
1020 if (path && path[0])
1021 m_nt_file_entries[i].path.assign(path);
1022 }
1023 break;
1024 }
1025 case ELF::NT_AUXV:
1026 m_auxv = note.data;
1027 break;
1028 default:
1029 thread_data.notes.push_back(note);
1030 break;
1031 }
1032 }
1033 // Add last entry in the note section
1034 if (have_prstatus)
1035 m_thread_data.push_back(thread_data);
1036 return llvm::Error::success();
1037}

1067 UUID invalid_uuid;
1068 const uint32_t addr_size = GetAddressByteSize();
1069 const size_t elf_header_size = addr_size == 4 ? sizeof(llvm::ELF::Elf32_Ehdr)
1070 : sizeof(llvm::ELF::Elf64_Ehdr);
1071
1072 std::vector<uint8_t> elf_header_bytes;
1073 elf_header_bytes.resize(elf_header_size);
1074 Status error;
1075 size_t byte_read =
1076 ReadMemory(address, elf_header_bytes.data(), elf_header_size, error);
1077 if (byte_read != elf_header_size ||
1078 !elf::ELFHeader::MagicBytesMatch(elf_header_bytes.data()))
1079 return invalid_uuid;

1081 GetByteOrder(), addr_size);
1082 lldb::offset_t offset = 0;
1083
1084 elf::ELFHeader elf_header;
1085 elf_header.Parse(elf_header_data, &offset);
1086
1087 const lldb::addr_t ph_addr = address + elf_header.e_phoff;
1088
1089 std::vector<uint8_t> ph_bytes;
1090 ph_bytes.resize(elf_header.e_phentsize);
1091 lldb::addr_t base_addr = 0;
1092 bool found_first_load_segment = false;
1093 for (unsigned int i = 0; i < elf_header.e_phnum; ++i) {
1094 byte_read = ReadMemory(ph_addr + i * elf_header.e_phentsize,
1095 ph_bytes.data(), elf_header.e_phentsize, error);
1096 if (byte_read != elf_header.e_phentsize)
1097 break;

1099 GetByteOrder(), addr_size);

1101 elf::ELFProgramHeader program_header;
1102 program_header.Parse(program_header_data, &offset);
1103 if (program_header.p_type == llvm::ELF::PT_LOAD &&
1104 !found_first_load_segment) {
1105 base_addr = program_header.p_vaddr;
1106 found_first_load_segment = true;
1107 }

1109 continue;

1112 note_bytes.resize(program_header.p_memsz);
1113
1114 // We need to slide the address of the p_vaddr as these values don't get
1115 // relocated in memory.
1116 const lldb::addr_t vaddr = program_header.p_vaddr + address - base_addr;

1118 ReadMemory(vaddr, note_bytes.data(), program_header.p_memsz, error);

1120 continue;

1122 GetByteOrder(), addr_size);
1123 auto notes_or_error = parseSegment(segment_data);
1124 if (!notes_or_error) {
1125 llvm::consumeError(notes_or_error.takeError());

1127 }
1128 for (const CoreNote &note : *notes_or_error) {

1132 note.data.ValidOffsetForDataOfSize(0, note.info.n_descsz))
1133 return UUID(note.data.GetData().take_front(note.info.n_descsz));
1134 }
1135 }
1136 return invalid_uuid;
