50ObjectContainer *ObjectContainerMachOFileset::CreateInstance(
51 const lldb::ModuleSP &module_sp, DataBufferSP &data_sp,
52 lldb::offset_t data_offset, const FileSpec *file,
53 lldb::offset_t file_offset, lldb::offset_t length) {
54 if (!data_sp)
55 return {};
56
57 DataExtractor extractor;
58 extractor.SetData(data_sp, data_offset, length);
59 if (!MagicBytesMatch(extractor))
60 return {};
61
62 auto container_up = std::make_unique<ObjectContainerMachOFileset>(
63 module_sp, data_sp, data_offset, file, file_offset, length);
64 if (!container_up->ParseHeader())
65 return {};
66
67 return container_up.release();

86static uint32_t MachHeaderSizeFromMagic(uint32_t magic) {
87 switch (magic) {
88 case MH_MAGIC:
89 case MH_CIGAM:
90 return sizeof(struct mach_header);
91 case MH_MAGIC_64:
92 case MH_CIGAM_64:
93 return sizeof(struct mach_header_64);
94 default:
95 return 0;
96 }
97}

99static std::optional<mach_header> ParseMachOHeader(DataExtractor &extractor) {
100 lldb::offset_t offset = 0;
101 mach_header header;
102 header.magic = extractor.GetU32(&offset);
103 switch (header.magic) {
104 case MH_MAGIC:
105 extractor.SetByteOrder(endian::InlHostByteOrder());
106 extractor.SetAddressByteSize(4);
107 break;
108 case MH_MAGIC_64:
109 extractor.SetByteOrder(endian::InlHostByteOrder());
110 extractor.SetAddressByteSize(8);
111 break;
112 case MH_CIGAM:
113 extractor.SetByteOrder(endian::InlHostByteOrder() == eByteOrderBig
114 ? eByteOrderLittle
115 : eByteOrderBig);
116 extractor.SetAddressByteSize(4);
117 break;
118 case MH_CIGAM_64:
119 extractor.SetByteOrder(endian::InlHostByteOrder() == eByteOrderBig
120 ? eByteOrderLittle
121 : eByteOrderBig);
122 extractor.SetAddressByteSize(8);
123 break;
124 default:
125 return {};
126 }
127
128 header.cputype = extractor.GetU32(&offset);
129 header.cpusubtype = extractor.GetU32(&offset);
130 header.filetype = extractor.GetU32(&offset);
131 header.ncmds = extractor.GetU32(&offset);
132 header.sizeofcmds = extractor.GetU32(&offset);
133 return header;
134}

136static bool
137ParseFileset(DataExtractor &extractor, mach_header header,
138 std::vector<ObjectContainerMachOFileset::Entry> &entries,
139 std::optional<lldb::addr_t> load_addr = std::nullopt) {
140 lldb::offset_t offset = MachHeaderSizeFromMagic(header.magic);
141 lldb::offset_t slide = 0;
142 for (uint32_t i = 0; i < header.ncmds; ++i) {
143 const lldb::offset_t load_cmd_offset = offset;
144 load_command lc = {};
145 if (extractor.GetU32(&offset, &lc.cmd, 2) == nullptr)
146 break;
147
148 // If we know the load address we can compute the slide.
149 if (load_addr) {
150 if (lc.cmd == llvm::MachO::LC_SEGMENT_64) {
151 segment_command_64 segment;
152 extractor.CopyData(load_cmd_offset, sizeof(segment_command_64),
153 &segment);
154 if (llvm::StringRef(segment.segname) == "__TEXT")
155 slide = *load_addr - segment.vmaddr;
156 }
157 }
158
159 if (lc.cmd == LC_FILESET_ENTRY) {
160 fileset_entry_command entry;
161 extractor.CopyData(load_cmd_offset, sizeof(fileset_entry_command),
162 &entry);
163 lldb::offset_t entry_id_offset = load_cmd_offset + entry.entry_id.offset;
164 if (const char *id = extractor.GetCStr(&entry_id_offset))
165 entries.emplace_back(entry.vmaddr + slide, entry.fileoff,
166 std::string(id));
167 }
168
169 offset = load_cmd_offset + lc.cmdsize;
170 }
171
172 return true;
173}

175bool ObjectContainerMachOFileset::ParseHeader(
176 DataExtractor &extractor, const lldb_private::FileSpec &file,
177 lldb::offset_t file_offset, std::vector<Entry> &entries) {
178 std::optional<mach_header> header = ParseMachOHeader(extractor);
179
180 if (!header)
181 return false;
182
183 const size_t header_size = MachHeaderSizeFromMagic(header->magic);
184 const size_t header_and_lc_size = header_size + header->sizeofcmds;
185
186 if (extractor.GetByteSize() < header_and_lc_size) {
187 DataBufferSP data_sp =
188 ObjectFile::MapFileData(file, header_and_lc_size, file_offset);
189 extractor.SetData(data_sp);
190 }
191
192 return ParseFileset(extractor, *header, entries);
193}

195bool ObjectContainerMachOFileset::ParseHeader() {
196 ModuleSP module_sp(GetModule());
197 if (!module_sp)
198 return false;
199
200 std::lock_guard<std::recursive_mutex> guard(module_sp->GetMutex());
201
202 std::optional<mach_header> header = ParseMachOHeader(*m_extractor_sp);
203 if (!header)
204 return false;
205
206 const size_t header_size = MachHeaderSizeFromMagic(header->magic);
207 const size_t header_and_lc_size = header_size + header->sizeofcmds;
208
209 if (m_extractor_sp->GetByteSize() < header_and_lc_size) {
210 ProcessSP process_sp(m_process_wp.lock());
211 DataBufferSP data_sp =
212 process_sp
213 ? ObjectFile::ReadMemory(process_sp, m_memory_addr,
214 header_and_lc_size)
215 : ObjectFile::MapFileData(m_file, header_and_lc_size, m_offset);
216 m_extractor_sp->SetData(data_sp);
217 }
218
219 return ParseFileset(*m_extractor_sp, *header, m_entries, m_memory_addr);
220}

222size_t ObjectContainerMachOFileset::GetModuleSpecifications(
223 const lldb_private::FileSpec &file, lldb::DataExtractorSP &extractor_sp,
224 lldb::offset_t data_offset, lldb::offset_t file_offset,
225 lldb::offset_t file_size, lldb_private::ModuleSpecList &specs) {
226 const size_t initial_count = specs.GetSize();
227 if (!extractor_sp)
228 return initial_count;
229
230 DataExtractorSP data_extractor_sp =
231 extractor_sp->GetSubsetExtractorSP(data_offset);
232 if (!data_extractor_sp)
233 return initial_count;
234 if (MagicBytesMatch(*data_extractor_sp)) {
235 std::vector<Entry> entries;
236 if (ParseHeader(*data_extractor_sp, file, file_offset, entries)) {
237 for (const Entry &entry : entries) {
238 const lldb::offset_t entry_offset = entry.fileoff + file_offset;
239 if (ObjectFile::GetModuleSpecifications(
240 file, entry_offset, file_size - entry_offset, specs)) {
241 ModuleSpec &spec = specs.GetModuleSpecRefAtIndex(specs.GetSize() - 1);
242 spec.GetObjectName() = ConstString(entry.id);
243 }
244 }
245 }
246 }
247 return specs.GetSize() - initial_count;
248}

258bool ObjectContainerMachOFileset::MagicBytesMatch(
259 const DataExtractor &extractor) {
260 lldb::offset_t offset = 0;
261 uint32_t magic = extractor.GetU32(&offset);
262 switch (magic) {
263 case MH_MAGIC:
264 case MH_CIGAM:
265 case MH_MAGIC_64:
266 case MH_CIGAM_64:
267 break;
268 default:
269 return false;
270 }
271 offset += 4; // cputype
272 offset += 4; // cpusubtype
273 uint32_t filetype = extractor.GetU32(&offset);
274 return filetype == MH_FILESET;
275}

277ObjectFileSP
278ObjectContainerMachOFileset::GetObjectFile(const lldb_private::FileSpec *file) {
279 ModuleSP module_sp(GetModule());
280 if (!module_sp)
281 return {};
282
283 ConstString object_name = module_sp->GetObjectName();
284 if (!object_name)
285 return {};
286
287 Entry *entry = FindEntry(object_name.GetCString());
288 if (!entry)
289 return {};
290
291 DataExtractorSP extractor_sp;
292 lldb::offset_t data_offset = 0;
293 return ObjectFile::FindPlugin(module_sp, file, m_offset + entry->fileoff,
294 m_extractor_sp->GetByteSize() - entry->fileoff,
295 extractor_sp, data_offset);
296}
