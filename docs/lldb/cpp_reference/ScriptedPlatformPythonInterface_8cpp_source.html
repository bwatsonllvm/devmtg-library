44StructuredData::DictionarySP ScriptedPlatformPythonInterface::ListProcesses() {
45 Status error;
46 StructuredData::DictionarySP dict_sp =
47 Dispatch<StructuredData::DictionarySP>("list_processes", error);
48
49 if (!dict_sp || !dict_sp->IsValid() || error.Fail()) {
50 return ScriptedInterface::ErrorWithMessage<StructuredData::DictionarySP>(
51 LLVM_PRETTY_FUNCTION,
52 llvm::Twine("Null or invalid object (" +
53 llvm::Twine(error.AsCString()) + llvm::Twine(")."))
54 .str(),
55 error);
56 }
57
58 return dict_sp;
59}
