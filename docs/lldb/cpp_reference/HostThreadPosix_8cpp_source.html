25Status HostThreadPosix::Join(lldb::thread_result_t *result) {
26 Status error;
27 if (IsJoinable()) {
28 int err = ::pthread_join(m_thread, result);
29 error = Status(err, lldb::eErrorTypePOSIX);
30 } else {
31 if (result)
32 *result = nullptr;
33 error = Status(EINVAL, eErrorTypePOSIX);
34 }
35
36 Reset();
37 return error;
38}
