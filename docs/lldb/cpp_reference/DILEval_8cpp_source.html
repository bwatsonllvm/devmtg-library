25lldb::ValueObjectSP
26GetDynamicOrSyntheticValue(lldb::ValueObjectSP value_sp,
27 lldb::DynamicValueType use_dynamic,
28 bool use_synthetic) {
29 if (!value_sp)
30 return nullptr;
31
32 if (use_dynamic != lldb::eNoDynamicValues) {
33 lldb::ValueObjectSP dynamic_sp = value_sp->GetDynamicValue(use_dynamic);
34 if (dynamic_sp)
35 value_sp = dynamic_sp;
36 }
37
38 if (use_synthetic) {
39 lldb::ValueObjectSP synthetic_sp = value_sp->GetSyntheticValue();
40 if (synthetic_sp)
41 value_sp = synthetic_sp;
42 }
43
44 return value_sp;
45}

67llvm::Expected<lldb::ValueObjectSP>
68Interpreter::UnaryConversion(lldb::ValueObjectSP valobj, uint32_t location) {
69 if (!valobj)
70 return llvm::make_error<DILDiagnosticError>(m_expr, "invalid value object",
71 location);
72 llvm::Expected<lldb::TypeSystemSP> type_system =
73 GetTypeSystemFromCU(m_exe_ctx_scope);
74 if (!type_system)
75 return type_system.takeError();
76
77 CompilerType in_type = valobj->GetCompilerType();
78 if (valobj->IsBitfield()) {
79 // Promote bitfields. If `int` can represent the bitfield value, it is
80 // converted to `int`. Otherwise, if `unsigned int` can represent it, it
81 // is converted to `unsigned int`. Otherwise, it is treated as its
82 // underlying type.
83 uint32_t bitfield_size = valobj->GetBitfieldBitSize();
84 // Some bitfields have undefined size (e.g. result of ternary operation).
85 // The AST's `bitfield_size` of those is 0, and no promotion takes place.
86 if (bitfield_size > 0 && in_type.IsInteger()) {
87 CompilerType int_type = GetBasicType(*type_system, lldb::eBasicTypeInt);
88 CompilerType uint_type =
89 GetBasicType(*type_system, lldb::eBasicTypeUnsignedInt);
90 llvm::Expected<uint64_t> int_bit_size =
91 int_type.GetBitSize(m_exe_ctx_scope.get());
92 if (!int_bit_size)
93 return int_bit_size.takeError();
94 llvm::Expected<uint64_t> uint_bit_size =
95 uint_type.GetBitSize(m_exe_ctx_scope.get());
96 if (!uint_bit_size)
97 return int_bit_size.takeError();
98 if (bitfield_size < *int_bit_size ||
99 (in_type.IsSigned() && bitfield_size == *int_bit_size))
100 return valobj->CastToBasicType(int_type);
101 if (bitfield_size <= *uint_bit_size)
102 return valobj->CastToBasicType(uint_type);
103 // Re-create as a const value with the same underlying type
104 Scalar scalar;
105 bool resolved = valobj->ResolveValue(scalar);
106 if (!resolved)
107 return llvm::createStringError("invalid scalar value");
108 return ValueObject::CreateValueObjectFromScalar(m_target, scalar, in_type,
109 "result");
110 }
111 }
112
113 if (in_type.IsArrayType())
114 valobj = ArrayToPointerConversion(*valobj, *m_exe_ctx_scope, "result");
115
116 if (valobj->GetCompilerType().IsInteger() ||
117 valobj->GetCompilerType().IsUnscopedEnumerationType()) {
118 llvm::Expected<CompilerType> promoted_type =
119 type_system.get()->DoIntegralPromotion(valobj->GetCompilerType(),
120 m_exe_ctx_scope.get());
121 if (!promoted_type)
122 return promoted_type.takeError();
123 if (!promoted_type->CompareTypes(valobj->GetCompilerType()))
124 return valobj->CastToBasicType(*promoted_type);
125 }
126
127 return valobj;
128}

130static lldb::VariableSP DILFindVariable(ConstString name,
131 VariableList &variable_list) {
132 lldb::VariableSP exact_match;
133 std::vector<lldb::VariableSP> possible_matches;
134
135 for (lldb::VariableSP var_sp : variable_list) {
136 llvm::StringRef str_ref_name = var_sp->GetName().GetStringRef();
137
138 str_ref_name.consume_front("::");
139 // Check for the exact same match
140 if (str_ref_name == name.GetStringRef())
141 return var_sp;
142
143 // Check for possible matches by base name
144 if (var_sp->NameMatches(name))
145 possible_matches.push_back(var_sp);
146 }
147
148 // If there's a non-exact match, take it.
149 if (possible_matches.size() > 0)
150 return possible_matches[0];
151
152 return nullptr;
153}

155lldb::ValueObjectSP LookupGlobalIdentifier(
156 llvm::StringRef name_ref, std::shared_ptr<StackFrame> stack_frame,
157 lldb::TargetSP target_sp, lldb::DynamicValueType use_dynamic) {
158 // Get a global variables list without the locals from the current frame
159 SymbolContext symbol_context =
160 stack_frame->GetSymbolContext(lldb::eSymbolContextCompUnit);
161 lldb::VariableListSP variable_list;
162 if (symbol_context.comp_unit)
163 variable_list = symbol_context.comp_unit->GetVariableList(true);
164
165 name_ref.consume_front("::");
166 lldb::ValueObjectSP value_sp;
167 if (variable_list) {
168 lldb::VariableSP var_sp =
169 DILFindVariable(ConstString(name_ref), *variable_list);
170 if (var_sp)
171 value_sp =
172 stack_frame->GetValueObjectForFrameVariable(var_sp, use_dynamic);
173 }
174
175 if (value_sp)
176 return value_sp;
177
178 // Check for match in modules global variables.
179 VariableList modules_var_list;
180 target_sp->GetImages().FindGlobalVariables(
181 ConstString(name_ref), std::numeric_limits<uint32_t>::max(),
182 modules_var_list);
183
184 if (!modules_var_list.Empty()) {
185 lldb::VariableSP var_sp =
186 DILFindVariable(ConstString(name_ref), modules_var_list);
187 if (var_sp)
188 value_sp = ValueObjectVariable::Create(stack_frame.get(), var_sp);
189
190 if (value_sp)
191 return value_sp;
192 }
193 return nullptr;
194}

196lldb::ValueObjectSP LookupIdentifier(llvm::StringRef name_ref,
197 std::shared_ptr<StackFrame> stack_frame,
198 lldb::DynamicValueType use_dynamic) {
199 // Support $rax as a special syntax for accessing registers.
200 // Will return an invalid value in case the requested register doesn't exist.
201 if (name_ref.consume_front("$")) {
202 lldb::RegisterContextSP reg_ctx(stack_frame->GetRegisterContext());
203 if (!reg_ctx)
204 return nullptr;
205
206 if (const RegisterInfo *reg_info = reg_ctx->GetRegisterInfoByName(name_ref))
207 return ValueObjectRegister::Create(stack_frame.get(), reg_ctx, reg_info);
208
209 return nullptr;
210 }
211
212 if (!name_ref.contains("::")) {
213 // Lookup in the current frame.
214 // Try looking for a local variable in current scope.
215 lldb::VariableListSP variable_list(
216 stack_frame->GetInScopeVariableList(false));
217
218 lldb::ValueObjectSP value_sp;
219 if (variable_list) {
220 lldb::VariableSP var_sp =
221 variable_list->FindVariable(ConstString(name_ref));
222 if (var_sp)
223 value_sp =
224 stack_frame->GetValueObjectForFrameVariable(var_sp, use_dynamic);
225 }
226
227 if (value_sp)
228 return value_sp;
229
230 // Try looking for an instance variable (class member).
231 SymbolContext sc = stack_frame->GetSymbolContext(
232 lldb::eSymbolContextFunction | lldb::eSymbolContextBlock);
233 llvm::StringRef ivar_name = sc.GetInstanceVariableName();
234 value_sp = stack_frame->FindVariable(ConstString(ivar_name));
235 if (value_sp)
236 value_sp = value_sp->GetChildMemberWithName(name_ref);
237
238 if (value_sp)
239 return value_sp;
240 }
241 return nullptr;
242}

253llvm::Expected<lldb::ValueObjectSP> Interpreter::Evaluate(const ASTNode &node) {
254 // Evaluate an AST.
255 auto value_or_error = node.Accept(this);
256 // Convert SP with a nullptr to an error.
257 if (value_or_error && !*value_or_error)
258 return llvm::make_error<DILDiagnosticError>(m_expr, "invalid value object",
259 node.GetLocation());
260 // Return the computed value-or-error. The caller is responsible for
261 // checking if an error occurred during the evaluation.
262 return value_or_error;
263}

265llvm::Expected<lldb::ValueObjectSP>
266Interpreter::EvaluateAndDereference(const ASTNode &node) {
267 auto valobj_or_err = Evaluate(node);
268 if (!valobj_or_err)
269 return valobj_or_err;
270 lldb::ValueObjectSP valobj = *valobj_or_err;
271
272 Status error;
273 if (valobj->GetCompilerType().IsReferenceType()) {
274 valobj = valobj->Dereference(error);
275 if (error.Fail())
276 return error.ToError();
277 }
278 return valobj;
279}

281llvm::Expected<lldb::ValueObjectSP>
282Interpreter::Visit(const IdentifierNode &node) {
283 lldb::DynamicValueType use_dynamic = m_use_dynamic;
284
285 lldb::ValueObjectSP identifier =
286 LookupIdentifier(node.GetName(), m_exe_ctx_scope, use_dynamic);
287
288 if (!identifier)
289 identifier = LookupGlobalIdentifier(node.GetName(), m_exe_ctx_scope,
290 m_target, use_dynamic);
291 if (!identifier) {
292 std::string errMsg =
293 llvm::formatv("use of undeclared identifier '{0}'", node.GetName());
294 return llvm::make_error<DILDiagnosticError>(
295 m_expr, errMsg, node.GetLocation(), node.GetName().size());
296 }
297
298 return identifier;
299}

301llvm::Expected<lldb::ValueObjectSP>
302Interpreter::Visit(const UnaryOpNode &node) {
303 Status error;
304 auto op_or_err = Evaluate(node.GetOperand());
305 if (!op_or_err)
306 return op_or_err;
307
308 lldb::ValueObjectSP operand = *op_or_err;
309
310 switch (node.GetKind()) {
311 case UnaryOpKind::Deref: {
312 lldb::ValueObjectSP dynamic_op = operand->GetDynamicValue(m_use_dynamic);
313 if (dynamic_op)
314 operand = dynamic_op;
315
316 lldb::ValueObjectSP child_sp = operand->Dereference(error);
317 if (!child_sp && m_use_synthetic) {
318 if (lldb::ValueObjectSP synth_obj_sp = operand->GetSyntheticValue()) {
319 error.Clear();
320 child_sp = synth_obj_sp->Dereference(error);
321 }
322 }
323 if (error.Fail())
324 return llvm::make_error<DILDiagnosticError>(m_expr, error.AsCString(),
325 node.GetLocation());
326
327 return child_sp;
328 }
329 case UnaryOpKind::AddrOf: {
330 Status error;
331 lldb::ValueObjectSP value = operand->AddressOf(error);
332 if (error.Fail())
333 return llvm::make_error<DILDiagnosticError>(m_expr, error.AsCString(),
334 node.GetLocation());
335
336 return value;
337 }
338 case UnaryOpKind::Minus: {
339 if (operand->GetCompilerType().IsReferenceType()) {
340 operand = operand->Dereference(error);
341 if (error.Fail())
342 return error.ToError();
343 }
344 llvm::Expected<lldb::ValueObjectSP> conv_op =
345 UnaryConversion(operand, node.GetOperand().GetLocation());
346 if (!conv_op)
347 return conv_op;
348 operand = *conv_op;
349 CompilerType operand_type = operand->GetCompilerType();
350 if (!operand_type.IsScalarType()) {
351 std::string errMsg =
352 llvm::formatv("invalid argument type '{0}' to unary expression",
353 operand_type.GetTypeName());
354 return llvm::make_error<DILDiagnosticError>(m_expr, errMsg,
355 node.GetLocation());
356 }
357 Scalar scalar;
358 bool resolved = operand->ResolveValue(scalar);
359 if (!resolved)
360 break;
361
362 bool negated = scalar.UnaryNegate();
363 if (negated)
364 return ValueObject::CreateValueObjectFromScalar(
365 m_target, scalar, operand->GetCompilerType(), "result");
366 break;
367 }
368 case UnaryOpKind::Plus: {
369 if (operand->GetCompilerType().IsReferenceType()) {
370 operand = operand->Dereference(error);
371 if (error.Fail())
372 return error.ToError();
373 }
374 llvm::Expected<lldb::ValueObjectSP> conv_op =
375 UnaryConversion(operand, node.GetOperand().GetLocation());
376 if (!conv_op)
377 return conv_op;
378 operand = *conv_op;
379 CompilerType operand_type = operand->GetCompilerType();
380 if (!operand_type.IsScalarType() &&
381 // Unary plus is allowed for pointers.
382 !operand_type.IsPointerType()) {
383 std::string errMsg =
384 llvm::formatv("invalid argument type '{0}' to unary expression",
385 operand_type.GetTypeName());
386 return llvm::make_error<DILDiagnosticError>(m_expr, errMsg,
387 node.GetLocation());
388 }
389 return operand;
390 }
391 }
392 return llvm::make_error<DILDiagnosticError>(m_expr, "invalid unary operation",
393 node.GetLocation());
394}

396llvm::Expected<lldb::ValueObjectSP>
397Interpreter::Visit(const MemberOfNode &node) {
398 auto base_or_err = Evaluate(node.GetBase());
399 if (!base_or_err)
400 return base_or_err;
401 bool expr_is_ptr = node.GetIsArrow();
402 lldb::ValueObjectSP base = *base_or_err;
403
404 // Perform some basic type & correctness checking.
405 if (node.GetIsArrow()) {
406 if (!m_fragile_ivar) {
407 // Make sure we aren't trying to deref an objective
408 // C ivar if this is not allowed
409 const uint32_t pointer_type_flags =
410 base->GetCompilerType().GetTypeInfo(nullptr);
411 if ((pointer_type_flags & lldb::eTypeIsObjC) &&
412 (pointer_type_flags & lldb::eTypeIsPointer)) {
413 // This was an objective C object pointer and it was requested we
414 // skip any fragile ivars so return nothing here
415 return lldb::ValueObjectSP();
416 }
417 }
418
419 // If we have a non-pointer type with a synthetic value then lets check
420 // if we have a synthetic dereference specified.
421 if (!base->IsPointerType() && base->HasSyntheticValue()) {
422 Status deref_error;
423 if (lldb::ValueObjectSP synth_deref_sp =
424 base->GetSyntheticValue()->Dereference(deref_error);
425 synth_deref_sp && deref_error.Success()) {
426 base = std::move(synth_deref_sp);
427 }
428 if (!base || deref_error.Fail()) {
429 std::string errMsg = llvm::formatv(
430 "Failed to dereference synthetic value: {0}", deref_error);
431 return llvm::make_error<DILDiagnosticError>(
432 m_expr, errMsg, node.GetLocation(), node.GetFieldName().size());
433 }
434
435 // Some synthetic plug-ins fail to set the error in Dereference
436 if (!base) {
437 std::string errMsg = "Failed to dereference synthetic value";
438 return llvm::make_error<DILDiagnosticError>(
439 m_expr, errMsg, node.GetLocation(), node.GetFieldName().size());
440 }
441 expr_is_ptr = false;
442 }
443 }
444
445 if (m_check_ptr_vs_member) {
446 bool base_is_ptr = base->IsPointerType();
447
448 if (expr_is_ptr != base_is_ptr) {
449 if (base_is_ptr) {
450 std::string errMsg =
451 llvm::formatv("member reference type {0} is a pointer; "
452 "did you mean to use '->'?",
453 base->GetCompilerType().TypeDescription());
454 return llvm::make_error<DILDiagnosticError>(
455 m_expr, errMsg, node.GetLocation(), node.GetFieldName().size());
456 } else {
457 std::string errMsg =
458 llvm::formatv("member reference type {0} is not a pointer; "
459 "did you mean to use '.'?",
460 base->GetCompilerType().TypeDescription());
461 return llvm::make_error<DILDiagnosticError>(
462 m_expr, errMsg, node.GetLocation(), node.GetFieldName().size());
463 }
464 }
465 }
466
467 lldb::ValueObjectSP field_obj =
468 base->GetChildMemberWithName(node.GetFieldName());
469 if (!field_obj) {
470 if (m_use_synthetic) {
471 field_obj = base->GetSyntheticValue();
472 if (field_obj)
473 field_obj = field_obj->GetChildMemberWithName(node.GetFieldName());
474 }
475
476 if (!m_use_synthetic || !field_obj) {
477 std::string errMsg = llvm::formatv(
478 "\"{0}\" is not a member of \"({1}) {2}\"", node.GetFieldName(),
479 base->GetTypeName().AsCString("<invalid type>"), base->GetName());
480 return llvm::make_error<DILDiagnosticError>(
481 m_expr, errMsg, node.GetLocation(), node.GetFieldName().size());
482 }
483 }
484
485 if (field_obj) {
486 if (m_use_dynamic != lldb::eNoDynamicValues) {
487 lldb::ValueObjectSP dynamic_val_sp =
488 field_obj->GetDynamicValue(m_use_dynamic);
489 if (dynamic_val_sp)
490 field_obj = dynamic_val_sp;
491 }
492 return field_obj;
493 }
494
495 CompilerType base_type = base->GetCompilerType();
496 if (node.GetIsArrow() && base->IsPointerType())
497 base_type = base_type.GetPointeeType();
498 std::string errMsg = llvm::formatv(
499 "\"{0}\" is not a member of \"({1}) {2}\"", node.GetFieldName(),
500 base->GetTypeName().AsCString("<invalid type>"), base->GetName());
501 return llvm::make_error<DILDiagnosticError>(
502 m_expr, errMsg, node.GetLocation(), node.GetFieldName().size());
503}

505llvm::Expected<lldb::ValueObjectSP>
506Interpreter::Visit(const ArraySubscriptNode &node) {
507 auto idx_or_err = EvaluateAndDereference(node.GetIndex());
508 if (!idx_or_err)
509 return idx_or_err;
510 lldb::ValueObjectSP idx = *idx_or_err;
511
512 if (!idx->GetCompilerType().IsIntegerOrUnscopedEnumerationType()) {
513 return llvm::make_error<DILDiagnosticError>(
514 m_expr, "array subscript is not an integer", node.GetLocation());
515 }
516
517 StreamString var_expr_path_strm;
518 uint64_t child_idx = idx->GetValueAsUnsigned(0);
519 lldb::ValueObjectSP child_valobj_sp;
520
521 auto base_or_err = Evaluate(node.GetBase());
522 if (!base_or_err)
523 return base_or_err;
524 lldb::ValueObjectSP base = *base_or_err;
525
526 CompilerType base_type = base->GetCompilerType().GetNonReferenceType();
527 base->GetExpressionPath(var_expr_path_strm);
528 bool is_incomplete_array = false;
529 if (base_type.IsPointerType()) {
530 bool is_objc_pointer = true;
531
532 if (base->GetCompilerType().GetMinimumLanguage() != lldb::eLanguageTypeObjC)
533 is_objc_pointer = false;
534 else if (!base->GetCompilerType().IsPointerType())
535 is_objc_pointer = false;
536
537 if (!m_use_synthetic && is_objc_pointer) {
538 std::string err_msg = llvm::formatv(
539 "\"({0}) {1}\" is an Objective-C pointer, and cannot be subscripted",
540 base->GetTypeName().AsCString("<invalid type>"),
541 var_expr_path_strm.GetData());
542 return llvm::make_error<DILDiagnosticError>(m_expr, std::move(err_msg),
543 node.GetLocation());
544 }
545 if (is_objc_pointer) {
546 lldb::ValueObjectSP synthetic = base->GetSyntheticValue();
547 if (!synthetic || synthetic == base) {
548 std::string err_msg =
549 llvm::formatv("\"({0}) {1}\" is not an array type",
550 base->GetTypeName().AsCString("<invalid type>"),
551 var_expr_path_strm.GetData());
552 return llvm::make_error<DILDiagnosticError>(m_expr, std::move(err_msg),
553 node.GetLocation());
554 }
555 if (static_cast<uint32_t>(child_idx) >=
556 synthetic->GetNumChildrenIgnoringErrors()) {
557 std::string err_msg = llvm::formatv(
558 "array index {0} is not valid for \"({1}) {2}\"", child_idx,
559 base->GetTypeName().AsCString("<invalid type>"),
560 var_expr_path_strm.GetData());
561 return llvm::make_error<DILDiagnosticError>(m_expr, std::move(err_msg),
562 node.GetLocation());
563 }
564 child_valobj_sp = synthetic->GetChildAtIndex(child_idx);
565 if (!child_valobj_sp) {
566 std::string err_msg = llvm::formatv(
567 "array index {0} is not valid for \"({1}) {2}\"", child_idx,
568 base->GetTypeName().AsCString("<invalid type>"),
569 var_expr_path_strm.GetData());
570 return llvm::make_error<DILDiagnosticError>(m_expr, std::move(err_msg),
571 node.GetLocation());
572 }
573 if (m_use_dynamic != lldb::eNoDynamicValues) {
574 if (auto dynamic_sp = child_valobj_sp->GetDynamicValue(m_use_dynamic))
575 child_valobj_sp = std::move(dynamic_sp);
576 }
577 return child_valobj_sp;
578 }
579
580 child_valobj_sp = base->GetSyntheticArrayMember(child_idx, true);
581 if (!child_valobj_sp) {
582 std::string err_msg = llvm::formatv(
583 "failed to use pointer as array for index {0} for "
584 "\"({1}) {2}\"",
585 child_idx, base->GetTypeName().AsCString("<invalid type>"),
586 var_expr_path_strm.GetData());
587 if (base_type.IsPointerToVoid())
588 err_msg = "subscript of pointer to incomplete type 'void'";
589 return llvm::make_error<DILDiagnosticError>(m_expr, std::move(err_msg),
590 node.GetLocation());
591 }
592 } else if (base_type.IsArrayType(nullptr, nullptr, &is_incomplete_array)) {
593 child_valobj_sp = base->GetChildAtIndex(child_idx);
594 if (!child_valobj_sp && (is_incomplete_array || m_use_synthetic))
595 child_valobj_sp = base->GetSyntheticArrayMember(child_idx, true);
596 if (!child_valobj_sp) {
597 std::string err_msg = llvm::formatv(
598 "array index {0} is not valid for \"({1}) {2}\"", child_idx,
599 base->GetTypeName().AsCString("<invalid type>"),
600 var_expr_path_strm.GetData());
601 return llvm::make_error<DILDiagnosticError>(m_expr, std::move(err_msg),
602 node.GetLocation());
603 }
604 } else if (base_type.IsScalarType()) {
605 child_valobj_sp =
606 base->GetSyntheticBitFieldChild(child_idx, child_idx, true);
607 if (!child_valobj_sp) {
608 std::string err_msg = llvm::formatv(
609 "bitfield range {0}:{1} is not valid for \"({2}) {3}\"", child_idx,
610 child_idx, base->GetTypeName().AsCString("<invalid type>"),
611 var_expr_path_strm.GetData());
612 return llvm::make_error<DILDiagnosticError>(m_expr, std::move(err_msg),
613 node.GetLocation(), 1);
614 }
615 } else {
616 lldb::ValueObjectSP synthetic = base->GetSyntheticValue();
617 if (!m_use_synthetic || !synthetic || synthetic == base) {
618 std::string err_msg =
619 llvm::formatv("\"{0}\" is not an array type",
620 base->GetTypeName().AsCString("<invalid type>"));
621 return llvm::make_error<DILDiagnosticError>(m_expr, std::move(err_msg),
622 node.GetLocation(), 1);
623 }
624 if (static_cast<uint32_t>(child_idx) >=
625 synthetic->GetNumChildrenIgnoringErrors(child_idx + 1)) {
626 std::string err_msg = llvm::formatv(
627 "array index {0} is not valid for \"({1}) {2}\"", child_idx,
628 base->GetTypeName().AsCString("<invalid type>"),
629 var_expr_path_strm.GetData());
630 return llvm::make_error<DILDiagnosticError>(m_expr, std::move(err_msg),
631 node.GetLocation(), 1);
632 }
633 child_valobj_sp = synthetic->GetChildAtIndex(child_idx);
634 if (!child_valobj_sp) {
635 std::string err_msg = llvm::formatv(
636 "array index {0} is not valid for \"({1}) {2}\"", child_idx,
637 base->GetTypeName().AsCString("<invalid type>"),
638 var_expr_path_strm.GetData());
639 return llvm::make_error<DILDiagnosticError>(m_expr, std::move(err_msg),
640 node.GetLocation(), 1);
641 }
642 }
643
644 if (child_valobj_sp) {
645 if (m_use_dynamic != lldb::eNoDynamicValues) {
646 if (auto dynamic_sp = child_valobj_sp->GetDynamicValue(m_use_dynamic))
647 child_valobj_sp = std::move(dynamic_sp);
648 }
649 return child_valobj_sp;
650 }
651
652 bool success;
653 int64_t signed_child_idx = idx->GetValueAsSigned(0, &success);
654 if (!success)
655 return llvm::make_error<DILDiagnosticError>(
656 m_expr, "could not get the index as an integer",
657 node.GetIndex().GetLocation());
658 return base->GetSyntheticArrayMember(signed_child_idx, true);
659}

661llvm::Expected<lldb::ValueObjectSP>
662Interpreter::Visit(const BitFieldExtractionNode &node) {
663 auto first_idx_or_err = EvaluateAndDereference(node.GetFirstIndex());
664 if (!first_idx_or_err)
665 return first_idx_or_err;
666 lldb::ValueObjectSP first_idx = *first_idx_or_err;
667 auto last_idx_or_err = EvaluateAndDereference(node.GetLastIndex());
668 if (!last_idx_or_err)
669 return last_idx_or_err;
670 lldb::ValueObjectSP last_idx = *last_idx_or_err;
671
672 if (!first_idx->GetCompilerType().IsIntegerOrUnscopedEnumerationType() ||
673 !last_idx->GetCompilerType().IsIntegerOrUnscopedEnumerationType()) {
674 return llvm::make_error<DILDiagnosticError>(
675 m_expr, "bit index is not an integer", node.GetLocation());
676 }
677
678 bool success_first, success_last;
679 int64_t first_index = first_idx->GetValueAsSigned(0, &success_first);
680 int64_t last_index = last_idx->GetValueAsSigned(0, &success_last);
681 if (!success_first || !success_last)
682 return llvm::make_error<DILDiagnosticError>(
683 m_expr, "could not get the index as an integer", node.GetLocation());
684
685 // if the format given is [high-low], swap range
686 if (first_index > last_index)
687 std::swap(first_index, last_index);
688
689 auto base_or_err = EvaluateAndDereference(node.GetBase());
690 if (!base_or_err)
691 return base_or_err;
692 lldb::ValueObjectSP base = *base_or_err;
693 lldb::ValueObjectSP child_valobj_sp =
694 base->GetSyntheticBitFieldChild(first_index, last_index, true);
695 if (!child_valobj_sp) {
696 std::string message = llvm::formatv(
697 "bitfield range {0}:{1} is not valid for \"({2}) {3}\"", first_index,
698 last_index, base->GetTypeName().AsCString("<invalid type>"),
699 base->GetName().AsCString());
700 return llvm::make_error<DILDiagnosticError>(m_expr, message,
701 node.GetLocation());
702 }
703 return child_valobj_sp;
704}

706llvm::Expected<CompilerType>
707Interpreter::PickIntegerType(lldb::TypeSystemSP type_system,
708 std::shared_ptr<ExecutionContextScope> ctx,
709 const IntegerLiteralNode &literal) {
710 // Binary, Octal, Hexadecimal and literals with a U suffix are allowed to be
711 // an unsigned integer.
712 bool unsigned_is_allowed = literal.IsUnsigned() || literal.GetRadix() != 10;
713 llvm::APInt apint = literal.GetValue();
714
715 llvm::SmallVector<std::pair<lldb::BasicType, lldb::BasicType>, 3> candidates;
716 if (literal.GetTypeSuffix() <= IntegerTypeSuffix::None)
717 candidates.emplace_back(lldb::eBasicTypeInt,
718 unsigned_is_allowed ? lldb::eBasicTypeUnsignedInt
719 : lldb::eBasicTypeInvalid);
720 if (literal.GetTypeSuffix() <= IntegerTypeSuffix::Long)
721 candidates.emplace_back(lldb::eBasicTypeLong,
722 unsigned_is_allowed ? lldb::eBasicTypeUnsignedLong
723 : lldb::eBasicTypeInvalid);
724 candidates.emplace_back(lldb::eBasicTypeLongLong,
725 lldb::eBasicTypeUnsignedLongLong);
726 for (auto [signed_, unsigned_] : candidates) {
727 CompilerType signed_type = type_system->GetBasicTypeFromAST(signed_);
728 if (!signed_type)
729 continue;
730 llvm::Expected<uint64_t> size = signed_type.GetBitSize(ctx.get());
731 if (!size)
732 return size.takeError();
733 if (!literal.IsUnsigned() && apint.isIntN(*size - 1))
734 return signed_type;
735 if (unsigned_ != lldb::eBasicTypeInvalid && apint.isIntN(*size))
736 return type_system->GetBasicTypeFromAST(unsigned_);
737 }
738
739 return llvm::make_error<DILDiagnosticError>(
740 m_expr,
741 "integer literal is too large to be represented in any integer type",
742 literal.GetLocation());
743}

745llvm::Expected<lldb::ValueObjectSP>
746Interpreter::Visit(const IntegerLiteralNode &node) {
747 llvm::Expected<lldb::TypeSystemSP> type_system =
748 GetTypeSystemFromCU(m_exe_ctx_scope);
749 if (!type_system)
750 return type_system.takeError();
751
752 llvm::Expected<CompilerType> type =
753 PickIntegerType(*type_system, m_exe_ctx_scope, node);
754 if (!type)
755 return type.takeError();
756
757 Scalar scalar = node.GetValue();
758 // APInt from StringRef::getAsInteger comes with just enough bitwidth to
759 // hold the value. This adjusts APInt bitwidth to match the compiler type.
760 llvm::Expected<uint64_t> type_bitsize =
761 type->GetBitSize(m_exe_ctx_scope.get());
762 if (!type_bitsize)
763 return type_bitsize.takeError();
764 scalar.TruncOrExtendTo(*type_bitsize, false);
765 return ValueObject::CreateValueObjectFromScalar(m_target, scalar, *type,
766 "result");
767}

797llvm::Expected<CastKind>
798Interpreter::VerifyArithmeticCast(CompilerType source_type,
799 CompilerType target_type, int location) {
800 if (source_type.IsPointerType() || source_type.IsNullPtrType()) {
801 // Cast from pointer to float/double is not allowed.
802 if (target_type.GetTypeInfo() & lldb::eTypeIsFloat) {
803 std::string errMsg = llvm::formatv("Cast from {0} to {1} is not allowed",
804 source_type.TypeDescription(),
805 target_type.TypeDescription());
806 return llvm::make_error<DILDiagnosticError>(
807 m_expr, std::move(errMsg), location,
808 source_type.TypeDescription().length());
809 }
810
811 // Casting from pointer to bool is always valid.
812 if (target_type.IsBoolean())
813 return CastKind::eArithmetic;
814
815 // Otherwise check if the result type is at least as big as the pointer
816 // size.
817 uint64_t type_byte_size = 0;
818 uint64_t rhs_type_byte_size = 0;
819 if (auto temp = target_type.GetByteSize(m_exe_ctx_scope.get())) {
820 type_byte_size = *temp;
821 } else {
822 std::string errMsg = llvm::formatv("unable to get byte size for type {0}",
823 target_type.TypeDescription());
824 return llvm::make_error<DILDiagnosticError>(
825 m_expr, std::move(errMsg), location,
826 target_type.TypeDescription().length());
827 }
828
829 if (auto temp = source_type.GetByteSize(m_exe_ctx_scope.get())) {
830 rhs_type_byte_size = *temp;
831 } else {
832 std::string errMsg = llvm::formatv("unable to get byte size for type {0}",
833 source_type.TypeDescription());
834 return llvm::make_error<DILDiagnosticError>(
835 m_expr, std::move(errMsg), location,
836 source_type.TypeDescription().length());
837 }
838
839 if (type_byte_size < rhs_type_byte_size) {
840 std::string errMsg = llvm::formatv(
841 "cast from pointer to smaller type {0} loses information",
842 target_type.TypeDescription());
843 return llvm::make_error<DILDiagnosticError>(
844 m_expr, std::move(errMsg), location,
845 source_type.TypeDescription().length());
846 }
847 } else if (!source_type.IsScalarType() && !source_type.IsEnumerationType()) {
848 // Otherwise accept only arithmetic types and enums.
849 std::string errMsg = llvm::formatv("cannot convert {0} to {1}",
850 source_type.TypeDescription(),
851 target_type.TypeDescription());
852
853 return llvm::make_error<DILDiagnosticError>(
854 m_expr, std::move(errMsg), location,
855 source_type.TypeDescription().length());
856 }
857 return CastKind::eArithmetic;
858}

860llvm::Expected<CastKind>
861Interpreter::VerifyCastType(lldb::ValueObjectSP operand,
862 CompilerType source_type, CompilerType target_type,
863 int location) {
864
865 if (target_type.IsScalarType())
866 return VerifyArithmeticCast(source_type, target_type, location);
867
868 if (target_type.IsEnumerationType()) {
869 // Cast to enum type.
870 if (!source_type.IsScalarType() && !source_type.IsEnumerationType()) {
871 std::string errMsg = llvm::formatv("Cast from {0} to {1} is not allowed",
872 source_type.TypeDescription(),
873 target_type.TypeDescription());
874
875 return llvm::make_error<DILDiagnosticError>(
876 m_expr, std::move(errMsg), location,
877 source_type.TypeDescription().length());
878 }
879 return CastKind::eEnumeration;
880 }
881
882 if (target_type.IsPointerType()) {
883 if (!source_type.IsInteger() && !source_type.IsEnumerationType() &&
884 !source_type.IsArrayType() && !source_type.IsPointerType() &&
885 !source_type.IsNullPtrType()) {
886 std::string errMsg = llvm::formatv(
887 "cannot cast from type {0} to pointer type {1}",
888 source_type.TypeDescription(), target_type.TypeDescription());
889
890 return llvm::make_error<DILDiagnosticError>(
891 m_expr, std::move(errMsg), location,
892 source_type.TypeDescription().length());
893 }
894 return CastKind::ePointer;
895 }
896
897 // Unsupported cast.
898 std::string errMsg = llvm::formatv(
899 "casting of {0} to {1} is not implemented yet",
900 source_type.TypeDescription(), target_type.TypeDescription());
901 return llvm::make_error<DILDiagnosticError>(
902 m_expr, std::move(errMsg), location,
903 source_type.TypeDescription().length());
904}

906llvm::Expected<lldb::ValueObjectSP> Interpreter::Visit(const CastNode &node) {
907 auto operand_or_err = Evaluate(node.GetOperand());
908
909 if (!operand_or_err)
910 return operand_or_err;
911
912 lldb::ValueObjectSP operand = *operand_or_err;
913 CompilerType op_type = operand->GetCompilerType();
914 CompilerType target_type = node.GetType();
915
916 if (op_type.IsReferenceType())
917 op_type = op_type.GetNonReferenceType();
918 if (target_type.IsScalarType() && op_type.IsArrayType()) {
919 operand = ArrayToPointerConversion(*operand, *m_exe_ctx_scope,
920 operand->GetName().GetStringRef());
921 op_type = operand->GetCompilerType();
922 }
923 auto type_or_err =
924 VerifyCastType(operand, op_type, target_type, node.GetLocation());
925 if (!type_or_err)
926 return type_or_err.takeError();
927
928 CastKind cast_kind = *type_or_err;
929 if (operand->GetCompilerType().IsReferenceType()) {
930 Status error;
931 operand = operand->Dereference(error);
932 if (error.Fail())
933 return llvm::make_error<DILDiagnosticError>(m_expr, error.AsCString(),
934 node.GetLocation());
935 }
936
937 switch (cast_kind) {
938 case CastKind::eEnumeration: {
939 // FIXME: is this correct for float vector types?
940 if (op_type.GetTypeInfo() & lldb::eTypeIsFloat || op_type.IsInteger() ||
941 op_type.IsEnumerationType())
942 return operand->CastToEnumType(target_type);
943 break;
944 }
945 case CastKind::eArithmetic: {
946 if (op_type.IsPointerType() || op_type.IsNullPtrType() ||
947 op_type.IsScalarType() || op_type.IsEnumerationType())
948 return operand->CastToBasicType(target_type);
949 break;
950 }
951 case CastKind::ePointer: {
952 uint64_t addr = op_type.IsArrayType()
953 ? operand->GetLoadAddress()
954 : (op_type.IsSigned() ? operand->GetValueAsSigned(0)
955 : operand->GetValueAsUnsigned(0));
956 llvm::StringRef name = "result";
957 ExecutionContext exe_ctx(m_target.get(), false);
958 return ValueObject::CreateValueObjectFromAddress(name, addr, exe_ctx,
959 target_type,
960 /* do_deref */ false);
961 }
962 case CastKind::eNone: {
963 return lldb::ValueObjectSP();
964 }
965 } // switch
966
967 std::string errMsg =
968 llvm::formatv("unable to cast from '{0}' to '{1}'",
969 op_type.TypeDescription(), target_type.TypeDescription());
970 return llvm::make_error<DILDiagnosticError>(m_expr, std::move(errMsg),
971 node.GetLocation());
972}
