224 inline uint64_t MemARead(EmulateInstruction::Context &context,
225 lldb::addr_t address, uint32_t size,
226 uint64_t fail_value, bool *success_ptr) {
227 // This is a stub function corresponding to "MemA[]" in the ARM manual
228 // pseudocode, for
229 // aligned reads from memory. Since we are not trying to write a full
230 // hardware simulator, and since
231 // we are running in User mode (rather than Kernel mode) and therefore won't
232 // have access to many of the
233 // system registers we would need in order to fully implement this function,
234 // we will just call
235 // ReadMemoryUnsigned from here. In the future, if we decide we do need to
236 // do more faithful emulation of
237 // the hardware, we can update this function appropriately.
238
239 return ReadMemoryUnsigned(context, address, size, fail_value, success_ptr);
240 }
