253static RegisterInfoInterface *
254CreateRegisterInfoInterface(const ArchSpec &target_arch) {
255 if (HostInfo::GetArchitecture().GetAddressByteSize() == 4) {
256 // 32-bit hosts run with a RegisterContextNetBSD_i386 context.
257 return new RegisterContextNetBSD_i386(target_arch);
258 } else {
259 assert((HostInfo::GetArchitecture().GetAddressByteSize() == 8) &&
260 "Register setting path assumes this is a 64-bit host");
261 // X86_64 hosts know how to work with 64-bit and 32-bit EXEs using the
262 // x86_64 register context.
263 return new RegisterContextNetBSD_x86_64(target_arch);
264 }
265}

267NativeRegisterContextNetBSD_x86::NativeRegisterContextNetBSD_x86(
268 const ArchSpec &target_arch, NativeThreadProtocol &native_thread)
269 : NativeRegisterContextRegisterInfo(
270 native_thread, CreateRegisterInfoInterface(target_arch)),
271 NativeRegisterContextDBReg_x86(native_thread), m_regset_offsets({0}) {
272 assert(m_gpr.size() == GetRegisterInfoInterface().GetGPRSize());
273 std::array<uint32_t, MaxRegularRegSet + 1> first_regnos;
274
275 switch (GetRegisterInfoInterface().GetTargetArchitecture().GetMachine()) {
276 case llvm::Triple::x86:
277 first_regnos[FPRegSet] = lldb_fctrl_i386;
278 first_regnos[DBRegSet] = lldb_dr0_i386;
279 break;
280 case llvm::Triple::x86_64:
281 first_regnos[FPRegSet] = lldb_fctrl_x86_64;
282 first_regnos[DBRegSet] = lldb_dr0_x86_64;
283 break;
284 default:
285 llvm_unreachable("Unhandled target architecture.");
286 }
287
288 for (int i : {FPRegSet, DBRegSet})
289 m_regset_offsets[i] = GetRegisterInfoInterface()
290 .GetRegisterInfo()[first_regnos[i]]
291 .byte_offset;
292}

310std::optional<NativeRegisterContextNetBSD_x86::RegSetKind>
311NativeRegisterContextNetBSD_x86::GetSetForNativeRegNum(
312 uint32_t reg_num) const {
313 switch (GetRegisterInfoInterface().GetTargetArchitecture().GetMachine()) {
314 case llvm::Triple::x86:
315 if (reg_num >= k_first_gpr_i386 && reg_num <= k_last_gpr_i386)
316 return GPRegSet;
317 if (reg_num >= k_first_fpr_i386 && reg_num <= k_last_fpr_i386)
318 return FPRegSet;
319 if (reg_num >= k_first_avx_i386 && reg_num <= k_last_avx_i386)
320 return YMMRegSet;
321 if (reg_num >= k_first_mpxr_i386 && reg_num <= k_last_mpxr_i386)
322 return std::nullopt; // MPXR
323 if (reg_num >= k_first_mpxc_i386 && reg_num <= k_last_mpxc_i386)
324 return std::nullopt; // MPXC
325 if (reg_num >= k_first_dbr_i386 && reg_num <= k_last_dbr_i386)
326 return DBRegSet; // DBR
327 break;
328 case llvm::Triple::x86_64:
329 if (reg_num >= k_first_gpr_x86_64 && reg_num <= k_last_gpr_x86_64)
330 return GPRegSet;
331 if (reg_num >= k_first_fpr_x86_64 && reg_num <= k_last_fpr_x86_64)
332 return FPRegSet;
333 if (reg_num >= k_first_avx_x86_64 && reg_num <= k_last_avx_x86_64)
334 return YMMRegSet;
335 if (reg_num >= k_first_mpxr_x86_64 && reg_num <= k_last_mpxr_x86_64)
336 return std::nullopt; // MPXR
337 if (reg_num >= k_first_mpxc_x86_64 && reg_num <= k_last_mpxc_x86_64)
338 return std::nullopt; // MPXC
339 if (reg_num >= k_first_dbr_x86_64 && reg_num <= k_last_dbr_x86_64)
340 return DBRegSet; // DBR
341 break;
342 default:
343 llvm_unreachable("Unhandled target architecture.");
344 }
345
346 llvm_unreachable("Register does not belong to any register set");
347}

349Status NativeRegisterContextNetBSD_x86::ReadRegisterSet(RegSetKind set) {
350 switch (set) {
351 case GPRegSet:
352 return DoRegisterSet(PT_GETREGS, m_gpr.data());
353 case DBRegSet:
354 return DoRegisterSet(PT_GETDBREGS, m_dbr.data());
355 case FPRegSet:
356 case YMMRegSet:
357 case MPXRegSet: {
358 struct iovec iov = {m_xstate.data(), m_xstate.size()};
359 Status ret = DoRegisterSet(PT_GETXSTATE, &iov);
360 assert(reinterpret_cast<xstate *>(m_xstate.data())->xs_rfbm & XCR0_X87);
361 return ret;
362 }
363 }
364 llvm_unreachable("NativeRegisterContextNetBSD_x86::ReadRegisterSet");
365}

367Status NativeRegisterContextNetBSD_x86::WriteRegisterSet(RegSetKind set) {
368 switch (set) {
369 case GPRegSet:
370 return DoRegisterSet(PT_SETREGS, m_gpr.data());
371 case DBRegSet:

379 }
380 llvm_unreachable("NativeRegisterContextNetBSD_x86::WriteRegisterSet");
381}

383Status
384NativeRegisterContextNetBSD_x86::ReadRegister(const RegisterInfo *reg_info,
385 RegisterValue &reg_value) {
386 Status error;
387
388 if (!reg_info) {
389 error = Status::FromErrorString("reg_info NULL");
390 return error;
391 }
392
393 uint32_t reg = reg_info->kinds[lldb::eRegisterKindLLDB];
394 if (reg == LLDB_INVALID_REGNUM) {
395 // This is likely an internal register for lldb use only and should not be
396 // directly queried.
397 error = Status::FromErrorStringWithFormat(
398 "register \"%s\" is an internal-only lldb "
399 "register, cannot read directly",
400 reg_info->name);
401 return error;
402 }
403
404 std::optional<RegSetKind> opt_set = GetSetForNativeRegNum(reg);
405 if (!opt_set) {
406 // This is likely an internal register for lldb use only and should not be
407 // directly queried.
408 error = Status::FromErrorStringWithFormat(
409 "register \"%s\" is in unrecognized set", reg_info->name);
410 return error;
411 }
412
413 RegSetKind set = *opt_set;
414 error = ReadRegisterSet(set);
415 if (error.Fail())
416 return error;
417
418 switch (set) {
419 case GPRegSet:
420 case FPRegSet:
421 case DBRegSet: {

424 offsetof(xstate, xs_fxsave));

427 AbridgedToFullTagWord(fpr->ftag, fpr->fstat, fpr->stmm));
428 else
429 reg_value.SetBytes(data, reg_info->byte_size, endian::InlHostByteOrder());
430 break;
431 }
432 case YMMRegSet: {
433 std::optional<YMMSplitPtr> ymm_reg = GetYMMSplitReg(reg);

436 "register \"%s\" not supported by CPU/kernel", reg_info->name);
437 } else {
438 YMMReg ymm = XStateToYMM(ymm_reg->xmm, ymm_reg->ymm_hi);
439 reg_value.SetBytes(ymm.bytes, reg_info->byte_size,
440 endian::InlHostByteOrder());
441 }
442 break;
443 }

451Status NativeRegisterContextNetBSD_x86::WriteRegister(
452 const RegisterInfo *reg_info, const RegisterValue &reg_value) {
453
454 Status error;
455
456 if (!reg_info) {
457 error = Status::FromErrorString("reg_info NULL");
458 return error;
459 }
460
461 uint32_t reg = reg_info->kinds[lldb::eRegisterKindLLDB];
462 if (reg == LLDB_INVALID_REGNUM) {
463 // This is likely an internal register for lldb use only and should not be
464 // directly queried.
465 error = Status::FromErrorStringWithFormat(
466 "register \"%s\" is an internal-only lldb "
467 "register, cannot read directly",
468 reg_info->name);
469 return error;
470 }
471
472 std::optional<RegSetKind> opt_set = GetSetForNativeRegNum(reg);
473 if (!opt_set) {
474 // This is likely an internal register for lldb use only and should not be
475 // directly queried.
476 error = Status::FromErrorStringWithFormat(
477 "register \"%s\" is in unrecognized set", reg_info->name);
478 return error;
479 }
480
481 RegSetKind set = *opt_set;
482 uint64_t new_xstate_bv = 0;
483
484 error = ReadRegisterSet(set);
485 if (error.Fail())
486 return error;
487
488 switch (set) {
489 case GPRegSet:
490 case DBRegSet:
491 ::memcpy(GetOffsetRegSetData(set, reg_info->byte_offset),
492 reg_value.GetBytes(), reg_value.GetByteSize());
493 break;
494 case FPRegSet: {
495 void *data = GetOffsetRegSetData(set, reg_info->byte_offset);
496 FXSAVE *fpr = reinterpret_cast<FXSAVE *>(m_xstate.data() +
497 offsetof(xstate, xs_fxsave));
498 if (data == &fpr->ftag) // ftag
499 fpr->ftag = FullToAbridgedTagWord(reg_value.GetAsUInt16());
500 else
501 ::memcpy(data, reg_value.GetBytes(), reg_value.GetByteSize());
502 if (data >= &fpr->xmm)
503 new_xstate_bv |= XCR0_SSE;
504 else if (data >= &fpr->mxcsr && data < &fpr->stmm)
505 new_xstate_bv |= XCR0_SSE;
506 else
507 new_xstate_bv |= XCR0_X87;
508 break;
509 }
510 case YMMRegSet: {
511 std::optional<YMMSplitPtr> ymm_reg = GetYMMSplitReg(reg);
512 if (!ymm_reg) {
513 error = Status::FromErrorStringWithFormat(
514 "register \"%s\" not supported by CPU/kernel", reg_info->name);
515 } else {
516 YMMReg ymm;
517 ::memcpy(ymm.bytes, reg_value.GetBytes(), reg_value.GetByteSize());
518 YMMToXState(ymm, ymm_reg->xmm, ymm_reg->ymm_hi);
519 new_xstate_bv |= XCR0_SSE | XCR0_YMM_Hi128;
520 }
521 break;
522 }
523 case MPXRegSet:
524 llvm_unreachable("MPX regset should have returned error");
525 }
526
527 if (new_xstate_bv != 0)
528 reinterpret_cast<xstate *>(m_xstate.data())->xs_xstate_bv |= new_xstate_bv;
529 return WriteRegisterSet(set);
530}
