37DynamicLoader *DynamicLoader::FindPlugin(Process *process,
38 llvm::StringRef plugin_name) {
39 DynamicLoaderCreateInstance create_callback = nullptr;
40 if (!plugin_name.empty()) {
41 create_callback =
42 PluginManager::GetDynamicLoaderCreateCallbackForPluginName(plugin_name);
43 if (create_callback) {
44 std::unique_ptr<DynamicLoader> instance_up(
45 create_callback(process, true));
46 if (instance_up)
47 return instance_up.release();
48 }
49 } else {
50 for (uint32_t idx = 0;
51 (create_callback =
52 PluginManager::GetDynamicLoaderCreateCallbackAtIndex(idx)) !=
53 nullptr;
54 ++idx) {
55 std::unique_ptr<DynamicLoader> instance_up(
56 create_callback(process, false));
57 if (instance_up)
58 return instance_up.release();
59 }
60 }
61 return nullptr;
62}

77ModuleSP DynamicLoader::GetTargetExecutable() {
78 Target &target = m_process->GetTarget();
79 ModuleSP executable = target.GetExecutableModule();
80
81 if (executable) {
82 if (FileSystem::Instance().Exists(executable->GetFileSpec())) {
83 ModuleSpec module_spec(executable->GetFileSpec(),
84 executable->GetArchitecture());
85 auto module_sp = std::make_shared<Module>(module_spec);
86 // If we're a coredump and we already have a main executable, we don't
87 // need to reload the module list that target already has
88 if (!m_process->IsLiveDebugSession()) {
89 return executable;
90 }
91 // Check if the executable has changed and set it to the target
92 // executable if they differ.
93 if (module_sp && module_sp->GetUUID().IsValid() &&
94 executable->GetUUID().IsValid()) {
95 if (module_sp->GetUUID() != executable->GetUUID())
96 executable.reset();
97 } else if (executable->FileHasChanged()) {
98 executable.reset();
99 }
100
101 if (!executable) {
102 executable = target.GetOrCreateModule(module_spec, true /* notify */);
103 if (executable.get() != target.GetExecutableModulePointer()) {
104 // Don't load dependent images since we are in dyld where we will
105 // know and find out about all images that are loaded
106 target.SetExecutableModule(executable, eLoadDependentsNo);
107 }
108 }
109 }
110 }
111 return executable;
112}

132void DynamicLoader::UnloadSectionsCommon(const ModuleSP module) {
133 Target &target = m_process->GetTarget();
134 const SectionList *sections = GetSectionListFromModule(module);
135
136 assert(sections && "SectionList missing from unloaded module.");
137
138 const size_t num_sections = sections->GetSize();
139 for (size_t i = 0; i < num_sections; ++i) {
140 SectionSP section_sp(sections->GetSectionAtIndex(i));
141 target.SetSectionUnloaded(section_sp);
142 }
143}

157ModuleSP DynamicLoader::FindModuleViaTarget(const FileSpec &file) {
158 Target &target = m_process->GetTarget();
159 ModuleSpec module_spec(file, target.GetArchitecture());
160 if (UUID uuid = m_process->FindModuleUUID(file.GetPath())) {
161 // Process may be able to augment the module_spec with UUID, e.g. ELF core.
162 module_spec.GetUUID() = uuid;
163 }
164
165 if (ModuleSP module_sp = target.GetImages().FindFirstModule(module_spec))
166 return module_sp;
167
168 if (ModuleSP module_sp =
169 target.GetOrCreateModule(module_spec, /*notify=*/false))
170 return module_sp;
171
172 return nullptr;
173}

175ModuleSP DynamicLoader::LoadModuleAtAddress(const FileSpec &file,
176 addr_t link_map_addr,
177 addr_t base_addr,
178 bool base_addr_is_offset) {
179 ModuleSP module_sp = FindModuleViaTarget(file);
180 // We have a core file, try to load the image from memory if we didn't find
181 // the module.
182 if (!module_sp && !m_process->IsLiveDebugSession()) {
183 llvm::Expected<ModuleSP> memory_module_sp_or_err =
184 m_process->ReadModuleFromMemory(file, base_addr);
185 if (auto err = memory_module_sp_or_err.takeError())
186 LLDB_LOG_ERROR(GetLog(LLDBLog::DynamicLoader), std::move(err),
187 "Failed to read module from memory: {0}");
188 else {
189 module_sp = *memory_module_sp_or_err;
190 m_process->GetTarget().GetImages().AppendIfNeeded(module_sp, false);
191 }
192 }
193 if (module_sp)
194 UpdateLoadedSections(module_sp, link_map_addr, base_addr,
195 base_addr_is_offset);
196 return module_sp;
197}

199static ModuleSP ReadUnnamedMemoryModule(Process *process, addr_t addr,
200 llvm::StringRef name) {
201 char namebuf[80];
202 if (name.empty()) {
203 snprintf(namebuf, sizeof(namebuf), "memory-image-0x%" PRIx64, addr);
204 name = namebuf;
205 }
206 llvm::Expected<ModuleSP> module_sp_or_err =
207 process->ReadModuleFromMemory(FileSpec(name), addr);
208 if (auto err = module_sp_or_err.takeError()) {
209 LLDB_LOG_ERROR(GetLog(LLDBLog::DynamicLoader), std::move(err),
210 "Failed to read module from memory: {0}");
211 return {};
212 }
213 return *module_sp_or_err;
214}

216ModuleSP DynamicLoader::LoadBinaryWithUUIDAndAddress(
217 Process *process, llvm::StringRef name, UUID uuid, addr_t value,
218 bool value_is_offset, bool force_symbol_search, bool notify,
219 bool set_address_in_target, bool allow_memory_image_last_resort) {
220 ModuleSP memory_module_sp;
221 ModuleSP module_sp;
222 PlatformSP platform_sp = process->GetTarget().GetPlatform();
223 Target &target = process->GetTarget();
224 Status error;
225
226 StreamString prog_str;
227 if (!name.empty()) {
228 prog_str << name.str() << " ";
229 }
230 if (uuid.IsValid())
231 prog_str << uuid.GetAsString();
232 if (value_is_offset == 0 && value != LLDB_INVALID_ADDRESS) {
233 prog_str << " at 0x";
234 prog_str.PutHex64(value);
235 }
236
237 if (!uuid.IsValid() && !value_is_offset) {
238 memory_module_sp = ReadUnnamedMemoryModule(process, value, name);
239
240 if (memory_module_sp) {
241 uuid = memory_module_sp->GetUUID();
242 if (uuid.IsValid()) {
243 prog_str << " ";
244 prog_str << uuid.GetAsString();
245 }
246 }
247 }
248 ModuleSpec module_spec;
249 module_spec.SetTarget(target.shared_from_this());
250 module_spec.GetUUID() = uuid;
251 FileSpec name_filespec(name);
252 if (FileSystem::Instance().Exists(name_filespec))
253 module_spec.GetFileSpec() = name_filespec;
254
255 if (uuid.IsValid()) {
256 Progress progress("Locating binary", prog_str.GetString().str());
257
258 // Has lldb already seen a module with this UUID?
259 // Or have external lookup enabled in DebugSymbols on macOS.
260 if (!module_sp)
261 error =
262 ModuleList::GetSharedModule(module_spec, module_sp, nullptr, nullptr);
263
264 // Can lldb's symbol/executable location schemes
265 // find an executable and symbol file.
266 if (!module_sp) {
267 FileSpecList search_paths = Target::GetDefaultDebugFileSearchPaths();
268 StatisticsMap symbol_locator_map;
269 module_spec.GetSymbolFileSpec() =
270 PluginManager::LocateExecutableSymbolFile(module_spec, search_paths,
271 symbol_locator_map);
272 ModuleSpec objfile_module_spec =
273 PluginManager::LocateExecutableObjectFile(module_spec,
274 symbol_locator_map);
275 module_spec.GetFileSpec() = objfile_module_spec.GetFileSpec();
276 if (FileSystem::Instance().Exists(module_spec.GetFileSpec()) &&
277 FileSystem::Instance().Exists(module_spec.GetSymbolFileSpec())) {
278 module_sp = std::make_shared<Module>(module_spec);
279 }
280
281 if (module_sp) {
282 module_sp->GetSymbolLocatorStatistics().merge(symbol_locator_map);
283 }
284 }
285
286 // If we haven't found a binary, or we don't have a SymbolFile, see
287 // if there is an external search tool that can find it.
288 if (!module_sp || !module_sp->GetSymbolFileFileSpec()) {
289 PluginManager::DownloadObjectAndSymbolFile(module_spec, error,
290 force_symbol_search);
291 if (FileSystem::Instance().Exists(module_spec.GetFileSpec())) {
292 module_sp = std::make_shared<Module>(module_spec);
293 } else if (force_symbol_search && error.AsCString("") &&
294 error.AsCString("")[0] != '\0') {
295 *target.GetDebugger().GetAsyncErrorStream() << error.AsCString();
296 }
297 }
298
299 // If we only found the executable, create a Module based on that.
300 if (!module_sp && FileSystem::Instance().Exists(module_spec.GetFileSpec()))
301 module_sp = std::make_shared<Module>(module_spec);
302 }
303
304 // If we couldn't find the binary anywhere else, as a last resort,
305 // read it out of memory.
306 if (allow_memory_image_last_resort && !module_sp.get() &&
307 value != LLDB_INVALID_ADDRESS && !value_is_offset) {
308 if (!memory_module_sp)
309 memory_module_sp = ReadUnnamedMemoryModule(process, value, name);
310 if (memory_module_sp)
311 module_sp = memory_module_sp;
312 }
313
314 Log *log = GetLog(LLDBLog::DynamicLoader);
315 if (module_sp.get()) {
316 // Ensure the Target has an architecture set in case
317 // we need it while processing this binary/eh_frame/debug info.
318 if (!target.GetArchitecture().IsValid())
319 target.SetArchitecture(module_sp->GetArchitecture());
320 target.GetImages().AppendIfNeeded(module_sp, false);
321
322 bool changed = false;
323 if (set_address_in_target) {
324 if (module_sp->GetObjectFile()) {
325 if (value != LLDB_INVALID_ADDRESS) {
326 LLDB_LOGF(log,
327 "DynamicLoader::LoadBinaryWithUUIDAndAddress Loading "
328 "binary %s UUID %s at %s 0x%" PRIx64,
329 name.str().c_str(), uuid.GetAsString().c_str(),
330 value_is_offset ? "offset" : "address", value);
331 module_sp->SetLoadAddress(target, value, value_is_offset, changed);
332 } else {
333 // No address/offset/slide, load the binary at file address,
334 // offset 0.
335 LLDB_LOGF(log,
336 "DynamicLoader::LoadBinaryWithUUIDAndAddress Loading "
337 "binary %s UUID %s at file address",
338 name.str().c_str(), uuid.GetAsString().c_str());
339 module_sp->SetLoadAddress(target, 0, true /* value_is_slide */,
340 changed);
341 }
342 } else {
343 // In-memory image, load at its true address, offset 0.
344 LLDB_LOGF(log,
345 "DynamicLoader::LoadBinaryWithUUIDAndAddress Loading binary "
346 "%s UUID %s from memory at address 0x%" PRIx64,
347 name.str().c_str(), uuid.GetAsString().c_str(), value);
348 module_sp->SetLoadAddress(target, 0, true /* value_is_slide */,
349 changed);
350 }
351 }
352
353 if (notify) {
354 ModuleList added_module;
355 added_module.Append(module_sp, false);
356 target.ModulesDidLoad(added_module);
357 }
358 } else {
359 if (force_symbol_search) {
360 lldb::StreamUP s = target.GetDebugger().GetAsyncErrorStream();
361 s->Printf("Unable to find file");
362 if (!name.empty())
363 s->Printf(" %s", name.str().c_str());
364 if (uuid.IsValid())
365 s->Printf(" with UUID %s", uuid.GetAsString().c_str());
366 if (value != LLDB_INVALID_ADDRESS) {
367 if (value_is_offset)
368 s->Printf(" with slide 0x%" PRIx64, value);
369 else
370 s->Printf(" at address 0x%" PRIx64, value);
371 }
372 s->Printf("\n");
373 }
374 LLDB_LOGF(log,
375 "Unable to find binary %s with UUID %s and load it at "
376 "%s 0x%" PRIx64,
377 name.str().c_str(), uuid.GetAsString().c_str(),
378 value_is_offset ? "offset" : "address", value);
379 }
380
381 return module_sp;
382}
