49template <typename T>
50Status consumeObject(llvm::ArrayRef<uint8_t> &Buffer, const T *&Object) {
51 if (Buffer.size() < sizeof(T))
52 return Status::FromErrorString("Insufficient buffer!");
53
54 Object = reinterpret_cast<const T *>(Buffer.data());
55 Buffer = Buffer.drop_front(sizeof(T));
56 return Status();
57}
