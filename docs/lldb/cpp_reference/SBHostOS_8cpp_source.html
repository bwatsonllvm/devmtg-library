53SBFileSpec SBHostOS::GetLLDBPath(lldb::PathType path_type) {
54 LLDB_INSTRUMENT_VA(path_type);
55
56 FileSpec fspec;
57 switch (path_type) {
58 case ePathTypeLLDBShlibDir:
59 fspec = HostInfo::GetShlibDir();
60 break;
61 case ePathTypeSupportExecutableDir:
62 fspec = HostInfo::GetSupportExeDir();
63 break;
64 case ePathTypeHeaderDir:
65 fspec = HostInfo::GetHeaderDir();
66 break;
67 case ePathTypePythonDir:
68 fspec = PluginManager::GetScriptInterpreterLibraryPath(
69 lldb::eScriptLanguagePython);
70 break;
71 case ePathTypeLLDBSystemPlugins:
72 fspec = HostInfo::GetSystemPluginDir();
73 break;
74 case ePathTypeLLDBUserPlugins:
75 fspec = HostInfo::GetUserPluginDir();
76 break;
77 case ePathTypeLLDBTempSystemDir:
78 fspec = HostInfo::GetProcessTempDir();
79 break;
80 case ePathTypeGlobalLLDBTempSystemDir:
81 fspec = HostInfo::GetGlobalTempDir();
82 break;
83 case ePathTypeClangDir:
84 fspec = GetClangResourceDir();
85 break;
86 }
87
88 SBFileSpec sb_fspec;
89 sb_fspec.SetFileSpec(fspec);
90 return sb_fspec;
91}
