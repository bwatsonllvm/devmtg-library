32/// An llvm::Error that represents an option parsing diagnostic.
33class OptionParseError
34 : public llvm::ErrorInfo<OptionParseError, DiagnosticError> {
35 std::vector<DiagnosticDetail> m_details;
36
37public:
38 using llvm::ErrorInfo<OptionParseError, DiagnosticError>::ErrorInfo;
39 OptionParseError(DiagnosticDetail detail)
40 : ErrorInfo(std::error_code(EINVAL, std::generic_category())),
41 m_details({detail}) {}

52 return std::make_unique<OptionParseError>(m_details[0]);
53 }

55 return m_details;
56 }
57 static char ID;
58};

82// Returns true is set_a is a subset of set_b; Otherwise returns false.
83
84bool Options::IsASubset(const OptionSet &set_a, const OptionSet &set_b) {
85 bool is_a_subset = true;
86 OptionSet::const_iterator pos_a;
87 OptionSet::const_iterator pos_b;
88
89 // set_a is a subset of set_b if every member of set_a is also a member of
90 // set_b
91
92 for (pos_a = set_a.begin(); pos_a != set_a.end() && is_a_subset; ++pos_a) {
93 pos_b = set_b.find(*pos_a);
94 if (pos_b == set_b.end())
95 is_a_subset = false;
96 }
97
98 return is_a_subset;
99}

101// Returns the set difference set_a - set_b, i.e. { x | ElementOf (x, set_a) &&
102// !ElementOf (x, set_b) }
103
104size_t Options::OptionsSetDiff(const OptionSet &set_a, const OptionSet &set_b,
105 OptionSet &diffs) {
106 size_t num_diffs = 0;
107 OptionSet::const_iterator pos_a;
108 OptionSet::const_iterator pos_b;
109
110 for (pos_a = set_a.begin(); pos_a != set_a.end(); ++pos_a) {
111 pos_b = set_b.find(*pos_a);
112 if (pos_b == set_b.end()) {
113 ++num_diffs;
114 diffs.insert(*pos_a);
115 }
116 }
117
118 return num_diffs;
119}

121// Returns the union of set_a and set_b. Does not put duplicate members into
122// the union.
123
124void Options::OptionsSetUnion(const OptionSet &set_a, const OptionSet &set_b,
125 OptionSet &union_set) {
126 OptionSet::const_iterator pos;
127 OptionSet::iterator pos_union;
128
129 // Put all the elements of set_a into the union.
130
131 for (pos = set_a.begin(); pos != set_a.end(); ++pos)
132 union_set.insert(*pos);
133
134 // Put all the elements of set_b that are not already there into the union.
135 for (pos = set_b.begin(); pos != set_b.end(); ++pos) {
136 pos_union = union_set.find(*pos);
137 if (pos_union == union_set.end())
138 union_set.insert(*pos);
139 }
140}

145void Options::BuildValidOptionSets() {
146 // Check to see if we already did this.
147 if (m_required_options.size() != 0)
148 return;
149
150 // Check to see if there are any options.
151 int num_options = NumCommandOptions();
152 if (num_options == 0)
153 return;
154
155 auto opt_defs = GetDefinitions();
156 m_required_options.resize(1);
157 m_optional_options.resize(1);
158
159 // First count the number of option sets we've got. Ignore
160 // LLDB_ALL_OPTION_SETS...
161
162 uint32_t num_option_sets = 0;
163
164 for (const auto &def : opt_defs) {
165 uint32_t this_usage_mask = def.usage_mask;
166 if (this_usage_mask == LLDB_OPT_SET_ALL) {
167 if (num_option_sets == 0)
168 num_option_sets = 1;
169 } else {
170 for (uint32_t j = 0; j < LLDB_MAX_NUM_OPTION_SETS; j++) {
171 if (this_usage_mask & (1 << j)) {
172 if (num_option_sets <= j)
173 num_option_sets = j + 1;
174 }
175 }
176 }
177 }
178
179 if (num_option_sets > 0) {
180 m_required_options.resize(num_option_sets);
181 m_optional_options.resize(num_option_sets);
182
183 for (const auto &def : opt_defs) {
184 for (uint32_t j = 0; j < num_option_sets; j++) {
185 if (def.usage_mask & 1 << j) {
186 if (def.required)
187 m_required_options[j].insert(def.short_option);
188 else
189 m_optional_options[j].insert(def.short_option);
190 }
191 }
192 }
193 }
194}

198Option *Options::GetLongOptions() {
199 // Check to see if this has already been done.
200 if (m_getopt_table.empty()) {
201 auto defs = GetDefinitions();
202 if (defs.empty())
203 return nullptr;
204
205 std::map<int, uint32_t> option_seen;
206
207 m_getopt_table.resize(defs.size() + 1);
208 for (size_t i = 0; i < defs.size(); ++i) {
209 const int short_opt = defs[i].short_option;
210
211 m_getopt_table[i].definition = &defs[i];
212 m_getopt_table[i].flag = nullptr;
213 m_getopt_table[i].val = short_opt;
214
215 auto [pos, inserted] = option_seen.try_emplace(short_opt, i);
216 if (!inserted && short_opt) {
217 m_getopt_table[i].val = 0;
218 StreamString strm;
219 if (defs[i].HasShortOption())
220 Debugger::ReportError(
221 llvm::formatv(
222 "option[{0}] --{1} has a short option -{2} that "
223 "conflicts with option[{3}] --{4}, short option won't "
224 "be used for --{5}",
225 i, defs[i].long_option, short_opt, pos->second,
226 m_getopt_table[pos->second].definition->long_option,
227 defs[i].long_option)
228 .str());
229 else
230 Debugger::ReportError(
231 llvm::formatv(
232 "option[{0}] --{1} has a short option {2:x} that "
233 "conflicts with option[{3}] --{4}, short option won't "
234 "be used for --{5}",
235 (int)i, defs[i].long_option, short_opt, pos->second,
236 m_getopt_table[pos->second].definition->long_option,
237 defs[i].long_option)
238 .str());
239 }
240 }
241
242 // getopt_long_only requires a NULL final entry in the table:
243
244 m_getopt_table.back().definition = nullptr;
245 m_getopt_table.back().flag = nullptr;
246 m_getopt_table.back().val = 0;
247 }
248
249 if (m_getopt_table.empty())
250 return nullptr;
251
252 return &m_getopt_table.front();
253}

255// This function takes INDENT, which tells how many spaces to output at the
256// front of each line; SPACES, which is a string containing 80 spaces; and
257// TEXT, which is the text that is to be output. It outputs the text, on
258// multiple lines if necessary, to RESULT, with INDENT spaces at the front of
259// each line. It breaks lines on spaces, tabs or newlines, shortening the line
260// if necessary to not break in the middle of a word. It assumes that each
261// output line should contain a maximum of OUTPUT_MAX_COLUMNS characters.
262
263void Options::OutputFormattedUsageText(Stream &strm,
264 const OptionDefinition &option_def,
265 uint32_t output_max_columns,
266 bool use_color) {
267 std::string actual_text;
268 if (option_def.validator) {
269 if (const char *condition = option_def.validator->ShortConditionString()) {
270 actual_text = "[";
271 actual_text.append(condition);
272 actual_text.append("] ");
273 }
274 }
275 actual_text.append(
276 ansi::FormatAnsiTerminalCodes(option_def.usage_text, use_color));
277
278 ansi::OutputWordWrappedLines(strm, actual_text, output_max_columns);
279}

281bool Options::SupportsLongOption(const char *long_option) {
282 if (!long_option || !long_option[0])
283 return false;
284
285 auto opt_defs = GetDefinitions();
286 if (opt_defs.empty())
287 return false;
288
289 const char *long_option_name = long_option;
290 if (long_option[0] == '-' && long_option[1] == '-')
291 long_option_name += 2;
292
293 for (auto &def : opt_defs) {
294 if (!def.long_option)
295 continue;
296
297 if (strcmp(def.long_option, long_option_name) == 0)
298 return true;
299 }
300
301 return false;
302}

310static bool PrintOption(const OptionDefinition &opt_def,
311 OptionDisplayType display_type, const char *header,
312 const char *footer, bool show_optional, Stream &strm) {
313 if (display_type == eDisplayShortOption && !opt_def.HasShortOption())
314 return false;
315
316 if (header && header[0])
317 strm.PutCString(header);
318
319 if (show_optional && !opt_def.required)
320 strm.PutChar('[');
321 const bool show_short_option =
322 opt_def.HasShortOption() && display_type != eDisplayLongOption;
323 if (show_short_option)
324 strm.Printf("-%c", opt_def.short_option);
325 else
326 strm.Printf("--%s", opt_def.long_option);
327 switch (opt_def.option_has_arg) {
328 case OptionParser::eNoArgument:
329 break;
330 case OptionParser::eRequiredArgument:
331 strm.Printf(" <%s>", CommandObject::GetArgumentName(opt_def.argument_type));
332 break;
333
334 case OptionParser::eOptionalArgument:
335 strm.Printf("%s[<%s>]", show_short_option ? "" : "=",
336 CommandObject::GetArgumentName(opt_def.argument_type));
337 break;
338 }
339 if (show_optional && !opt_def.required)
340 strm.PutChar(']');
341 if (footer && footer[0])
342 strm.PutCString(footer);
343 return true;
344}

346void Options::GenerateOptionUsage(Stream &strm, CommandObject &cmd,
347 uint32_t screen_width, bool use_color) {
348 auto opt_defs = GetDefinitions();
349 const uint32_t save_indent_level = strm.GetIndentLevel();
350 llvm::StringRef name = cmd.GetCommandName();
351 StreamString arguments_str;
352 cmd.GetFormattedCommandArguments(arguments_str);
353
354 const uint32_t num_options = NumCommandOptions();
355 if (num_options == 0)
356 return;
357
358 const bool only_print_args = cmd.IsDashDashCommand();
359 if (!only_print_args)
360 strm.PutCString("\nCommand Options Usage:\n");
361
362 strm.IndentMore(2);
363
364 // First, show each usage level set of options, e.g. <cmd> [options-for-
365 // level-0]
366 // <cmd>
367 // [options-for-level-1]
368 // etc.
369
370 if (!only_print_args) {
371 uint32_t num_option_sets = GetRequiredOptions().size();
372 for (uint32_t opt_set = 0; opt_set < num_option_sets; ++opt_set) {
373 if (opt_set > 0)
374 strm.Printf("\n");
375 strm.Indent(name);
376
377 // Different option sets may require different args.
378 StreamString args_str;
379 uint32_t opt_set_mask = 1 << opt_set;
380 cmd.GetFormattedCommandArguments(args_str, opt_set_mask);
381
382 // First go through and print all options that take no arguments as a
383 // single string. If a command has "-a" "-b" and "-c", this will show up
384 // as [-abc]
385
386 // We use a set here so that they will be sorted.
387 std::set<int> required_options;
388 std::set<int> optional_options;
389
390 for (auto &def : opt_defs) {
391 if (def.usage_mask & opt_set_mask && def.HasShortOption() &&
392 def.option_has_arg == OptionParser::eNoArgument) {
393 if (def.required) {
394 required_options.insert(def.short_option);
395 } else {
396 optional_options.insert(def.short_option);
397 }
398 }
399 }
400
401 if (!required_options.empty()) {
402 strm.PutCString(" -");
403 for (int short_option : required_options)
404 strm.PutChar(short_option);
405 }
406
407 if (!optional_options.empty()) {
408 strm.PutCString(" [-");
409 for (int short_option : optional_options)
410 strm.PutChar(short_option);
411 strm.PutChar(']');
412 }
413
414 // First go through and print the required options (list them up front).
415 for (auto &def : opt_defs) {
416 if (def.usage_mask & opt_set_mask && def.HasShortOption() &&
417 def.required && def.option_has_arg != OptionParser::eNoArgument)
418 PrintOption(def, eDisplayBestOption, " ", nullptr, true, strm);
419 }
420
421 // Now go through again, and this time only print the optional options.
422 for (auto &def : opt_defs) {
423 if (def.usage_mask & opt_set_mask && !def.required &&
424 def.option_has_arg != OptionParser::eNoArgument)
425 PrintOption(def, eDisplayBestOption, " ", nullptr, true, strm);
426 }
427
428 if (args_str.GetSize() > 0) {
429 if (cmd.WantsRawCommandString())
430 strm.Printf(" --");
431 strm << " " << args_str.GetString();
432 }
433 }
434 }
435
436 if ((only_print_args || cmd.WantsRawCommandString()) &&
437 arguments_str.GetSize() > 0) {
438 if (!only_print_args)
439 strm.PutChar('\n');
440 strm.Indent(name);
441 strm << " " << arguments_str.GetString();
442 }
443
444 if (!only_print_args) {
445 strm.Printf("\n\n");
446
447 // Now print out all the detailed information about the various options:
448 // long form, short form and help text:
449 // -short <argument> ( --long_name <argument> )
450 // help text
451
452 strm.IndentMore(5);
453
454 // Put the command options in a sorted container, so we can output
455 // them alphabetically by short_option.
456 std::multimap<int, uint32_t> options_ordered;
457 for (auto def : llvm::enumerate(opt_defs))
458 options_ordered.insert(
459 std::make_pair(def.value().short_option, def.index()));
460
461 // Go through each option, find the table entry and write out the detailed
462 // help information for that option.
463
464 bool first_option_printed = false;
465
466 for (auto pos : options_ordered) {
467 // Put a newline separation between arguments
468 if (first_option_printed)
469 strm.EOL();
470 else
471 first_option_printed = true;
472
473 OptionDefinition opt_def = opt_defs[pos.second];
474
475 strm.Indent();
476 if (opt_def.short_option && opt_def.HasShortOption()) {
477 PrintOption(opt_def, eDisplayShortOption, nullptr, nullptr, false,
478 strm);
479 PrintOption(opt_def, eDisplayLongOption, " ( ", " )", false, strm);
480 } else {
481 // Short option is not printable, just print long option
482 PrintOption(opt_def, eDisplayLongOption, nullptr, nullptr, false, strm);
483 }
484 strm.EOL();
485
486 strm.IndentMore(5);
487
488 if (opt_def.usage_text)
489 OutputFormattedUsageText(strm, opt_def, screen_width, use_color);
490 if (!opt_def.enum_values.empty()) {
491 strm.Indent();
492 strm.Printf("Values: ");
493 bool is_first = true;
494 for (const auto &enum_value : opt_def.enum_values) {
495 if (is_first) {
496 strm.Printf("%s", enum_value.string_value);
497 is_first = false;
498 }
499 else
500 strm.Printf(" | %s", enum_value.string_value);
501 }
502 strm.EOL();
503 }
504 strm.IndentLess(5);
505 }
506 }
507
508 // Restore the indent level
509 strm.SetIndentLevel(save_indent_level);
510}

512llvm::Error Options::VerifyOptions() {
513 bool options_are_valid = false;
514
515 int num_levels = GetRequiredOptions().size();
516 if (num_levels) {
517 for (int i = 0; i < num_levels && !options_are_valid; ++i) {
518 // This is the correct set of options if: 1). m_seen_options contains
519 // all of m_required_options[i] (i.e. all the required options at this
520 // level are a subset of m_seen_options); AND 2). { m_seen_options -
521 // m_required_options[i] is a subset of m_options_options[i] (i.e. all
522 // the rest of m_seen_options are in the set of optional options at this
523 // level.
524
525 // Check to see if all of m_required_options[i] are a subset of
526 // m_seen_options
527 if (IsASubset(GetRequiredOptions()[i], m_seen_options)) {
528 // Construct the set difference: remaining_options = {m_seen_options} -
529 // {m_required_options[i]}
530 OptionSet remaining_options;
531 OptionsSetDiff(m_seen_options, GetRequiredOptions()[i],
532 remaining_options);
533 // Check to see if remaining_options is a subset of
534 // m_optional_options[i]
535 if (IsASubset(remaining_options, GetOptionalOptions()[i]))
536 options_are_valid = true;
537 }
538 }
539 } else {
540 options_are_valid = true;
541 }
542
543 if (!options_are_valid)
544 return llvm::createStringError(
545 "invalid combination of options for the given command");
546
547 return llvm::Error::success();
548}
549
550// This function is called when we have been given a potentially incomplete set
551// of options, such as when an alias has been defined (more options might be
552// added at at the time the alias is invoked). We need to verify that the
553// options in the set m_seen_options are all part of a set that may be used
554// together, but m_seen_options may be missing some of the "required" options.
555llvm::Error Options::VerifyPartialOptions() {
556 bool options_are_valid = false;
557
558 int num_levels = GetRequiredOptions().size();
559 if (num_levels) {
560 for (int i = 0; i < num_levels && !options_are_valid; ++i) {
561 // In this case we are treating all options as optional rather than
562 // required. Therefore a set of options is correct if m_seen_options is a
563 // subset of the union of m_required_options and m_optional_options.
564 OptionSet union_set;
565 OptionsSetUnion(GetRequiredOptions()[i], GetOptionalOptions()[i],
566 union_set);
567 if (IsASubset(m_seen_options, union_set))
568 options_are_valid = true;
569 }
570 }
571
572 if (!options_are_valid)
573 return llvm::createStringError(
574 "invalid combination of options for the given command");
575
576 return llvm::Error::success();
577}

579bool Options::HandleOptionCompletion(CompletionRequest &request,
580 OptionElementVector &opt_element_vector,
581 CommandInterpreter &interpreter) {
582 // For now we just scan the completions to see if the cursor position is in
583 // an option or its argument. Otherwise we'll call HandleArgumentCompletion.
584 // In the future we can use completion to validate options as well if we
585 // want.
586
587 auto opt_defs = GetDefinitions();
588
589 llvm::StringRef cur_opt_str = request.GetCursorArgumentPrefix();
590 const bool use_color = interpreter.GetDebugger().GetUseColor();
591
592 for (size_t i = 0; i < opt_element_vector.size(); i++) {
593 size_t opt_pos = static_cast<size_t>(opt_element_vector[i].opt_pos);
594 size_t opt_arg_pos = static_cast<size_t>(opt_element_vector[i].opt_arg_pos);
595 int opt_defs_index = opt_element_vector[i].opt_defs_index;
596 if (opt_pos == request.GetCursorIndex()) {
597 // We're completing the option itself.
598
599 if (opt_defs_index == OptionArgElement::eBareDash) {
600 // We're completing a bare dash. That means all options are open.
601 // FIXME: We should scan the other options provided and only complete
602 // options
603 // within the option group they belong to.
604 std::string opt_str = "-a";
605
606 for (auto &def : opt_defs) {
607 if (!def.short_option)
608 continue;
609 opt_str[1] = def.short_option;
610 request.AddCompletion(opt_str, ansi::FormatAnsiTerminalCodes(
611 def.usage_text, use_color));
612 }
613
614 return true;
615 } else if (opt_defs_index == OptionArgElement::eBareDoubleDash) {
616 std::string full_name("--");
617 for (auto &def : opt_defs) {
618 if (!def.short_option)
619 continue;
620
621 full_name.erase(full_name.begin() + 2, full_name.end());
622 full_name.append(def.long_option);
623 request.AddCompletion(full_name, ansi::FormatAnsiTerminalCodes(
624 def.usage_text, use_color));
625 }
626 return true;
627 } else if (opt_defs_index != OptionArgElement::eUnrecognizedArg) {
628 // We recognized it, if it an incomplete long option, complete it
629 // anyway (getopt_long_only is happy with shortest unique string, but
630 // it's still a nice thing to do.) Otherwise return The string so the
631 // upper level code will know this is a full match and add the " ".
632 const OptionDefinition &opt = opt_defs[opt_defs_index];
633 llvm::StringRef long_option = opt.long_option;
634 if (cur_opt_str.starts_with("--") && cur_opt_str != long_option) {
635 request.AddCompletion(
636 "--" + long_option.str(),
637 ansi::FormatAnsiTerminalCodes(opt.usage_text, use_color));
638 return true;
639 } else
640 request.AddCompletion(request.GetCursorArgumentPrefix());
641 return true;
642 } else {
643 // FIXME - not handling wrong options yet:
644 // Check to see if they are writing a long option & complete it.
645 // I think we will only get in here if the long option table has two
646 // elements
647 // that are not unique up to this point. getopt_long_only does
648 // shortest unique match for long options already.
649 if (cur_opt_str.consume_front("--")) {
650 for (auto &def : opt_defs) {
651 llvm::StringRef long_option(def.long_option);
652 if (long_option.starts_with(cur_opt_str))
653 request.AddCompletion(
654 "--" + long_option.str(),
655 ansi::FormatAnsiTerminalCodes(def.usage_text, use_color));
656 }
657 }
658 return true;
659 }
660
661 } else if (opt_arg_pos == request.GetCursorIndex()) {
662 // Okay the cursor is on the completion of an argument. See if it has a
663 // completion, otherwise return no matches. Note, opt_defs_index == -1
664 // means we're after an option, but that option doesn't exist. We'll
665 // end up treating that as an argument. Not sure we can do much better.
666 if (opt_defs_index != -1) {
667 HandleOptionArgumentCompletion(request, opt_element_vector, i,
668 interpreter);
669 return true;
670 } else {
671 // No completion callback means no completions...
672 return true;
673 }
674
675 } else {
676 // Not the last element, keep going.
677 continue;
678 }
679 }
680 return false;
681}

683void Options::HandleOptionArgumentCompletion(
684 CompletionRequest &request, OptionElementVector &opt_element_vector,
685 int opt_element_index, CommandInterpreter &interpreter) {
686 auto opt_defs = GetDefinitions();
687 std::unique_ptr<SearchFilter> filter_up;
688
689 int opt_defs_index = opt_element_vector[opt_element_index].opt_defs_index;
690
691 // See if this is an enumeration type option, and if so complete it here:
692 const auto &enum_values = opt_defs[opt_defs_index].enum_values;
693 if (!enum_values.empty())
694 for (const auto &enum_value : enum_values)
695 request.TryCompleteCurrentArg(enum_value.string_value);
696
697 // If this is a source file or symbol type completion, and there is a -shlib
698 // option somewhere in the supplied arguments, then make a search filter for
699 // that shared library.
700 // FIXME: Do we want to also have an "OptionType" so we don't have to match
701 // string names?
702
703 uint32_t completion_mask = opt_defs[opt_defs_index].completion_type;
704
705 if (completion_mask == 0) {
706 lldb::CommandArgumentType option_arg_type =
707 opt_defs[opt_defs_index].argument_type;
708 if (option_arg_type != eArgTypeNone) {
709 const CommandObject::ArgumentTableEntry *arg_entry =
710 CommandObject::FindArgumentDataByType(
711 opt_defs[opt_defs_index].argument_type);
712 if (arg_entry)
713 completion_mask = arg_entry->completion_type;
714 }
715 }
716
717 if (completion_mask & lldb::eSourceFileCompletion ||
718 completion_mask & lldb::eSymbolCompletion) {
719 for (size_t i = 0; i < opt_element_vector.size(); i++) {
720 int cur_defs_index = opt_element_vector[i].opt_defs_index;
721
722 // trying to use <0 indices will definitely cause problems
723 if (cur_defs_index == OptionArgElement::eUnrecognizedArg ||
724 cur_defs_index == OptionArgElement::eBareDash ||
725 cur_defs_index == OptionArgElement::eBareDoubleDash)
726 continue;
727
728 int cur_arg_pos = opt_element_vector[i].opt_arg_pos;
729 const char *cur_opt_name = opt_defs[cur_defs_index].long_option;
730
731 // If this is the "shlib" option and there was an argument provided,
732 // restrict it to that shared library.
733 if (cur_opt_name && strcmp(cur_opt_name, "shlib") == 0 &&
734 cur_arg_pos != -1) {
735 const char *module_name =
736 request.GetParsedLine().GetArgumentAtIndex(cur_arg_pos);
737 if (module_name) {
738 FileSpec module_spec(module_name);
739 lldb::TargetSP target_sp =
740 interpreter.GetDebugger().GetSelectedTarget();
741 // Search filters require a target...
742 if (target_sp)
743 filter_up =
744 std::make_unique<SearchFilterByModule>(target_sp, module_spec);
745 }
746 break;
747 }
748 }
749 }
750
751 lldb_private::CommandCompletions::InvokeCommonCompletionCallbacks(
752 interpreter, completion_mask, request, filter_up.get());
753}

849// OptionParser permutes the arguments while processing them, so we create a
850// temporary array holding to avoid modification of the input arguments. The
851// options themselves are never modified, but the API expects a char * anyway,
852// hence the const_cast.
853static std::vector<char *> GetArgvForParsing(const Args &args) {
854 std::vector<char *> result;
855 // OptionParser always skips the first argument as it is based on getopt().
856 result.push_back(const_cast<char *>("<FAKE-ARG0>"));
857 for (const Args::ArgEntry &entry : args)
858 result.push_back(const_cast<char *>(entry.c_str()));
859 result.push_back(nullptr);
860 return result;
861}
862
863// Given a permuted argument, find it's position in the original Args vector.
864static Args::const_iterator FindOriginalIter(const char *arg,
865 const Args &original) {
866 return llvm::find_if(
867 original, [arg](const Args::ArgEntry &D) { return D.c_str() == arg; });
868}
869
870// Given a permuted argument, find it's index in the original Args vector.
871static size_t FindOriginalIndex(const char *arg, const Args &original) {
872 return std::distance(original.begin(), FindOriginalIter(arg, original));
873}
874
875// Construct a new Args object, consisting of the entries from the original
876// arguments, but in the permuted order.
877static Args ReconstituteArgsAfterParsing(llvm::ArrayRef<char *> parsed,
878 const Args &original) {
879 Args result;
880 for (const char *arg : parsed) {
881 auto pos = FindOriginalIter(arg, original);
882 assert(pos != original.end());
883 result.AppendArgument(pos->ref(), pos->GetQuoteChar());
884 }
885 return result;
886}

888static size_t FindArgumentIndexForOption(const Args &args,
889 const Option &long_option) {
890 std::string short_opt = llvm::formatv("-{0}", char(long_option.val)).str();
891 std::string long_opt =
892 std::string(llvm::formatv("--{0}", long_option.definition->long_option));
893 for (const auto &entry : llvm::enumerate(args)) {
894 if (entry.value().ref().starts_with(short_opt) ||
895 entry.value().ref().starts_with(long_opt))
896 return entry.index();
897 }
898
899 return size_t(-1);
900}

902static std::string BuildShortOptions(const Option *long_options) {
903 std::string storage;
904 llvm::raw_string_ostream sstr(storage);
905
906 // Leading : tells getopt to return a : for a missing option argument AND to
907 // suppress error messages.
908 sstr << ":";
909
910 for (size_t i = 0; long_options[i].definition != nullptr; ++i) {
911 if (long_options[i].flag == nullptr) {
912 sstr << (char)long_options[i].val;
913 switch (long_options[i].definition->option_has_arg) {
914 default:
915 case OptionParser::eNoArgument:
916 break;
917 case OptionParser::eRequiredArgument:
918 sstr << ":";
919 break;
920 case OptionParser::eOptionalArgument:
921 sstr << "::";
922 break;
923 }
924 }
925 }
926 return storage;
927}

929llvm::Expected<Args> Options::ParseAlias(const Args &args,
930 OptionArgVector *option_arg_vector,
931 std::string &input_line) {
932 Option *long_options = GetLongOptions();
933
934 if (long_options == nullptr) {
935 return llvm::createStringError("Invalid long options");
936 }
937
938 std::string short_options = BuildShortOptions(long_options);
939
940 Args args_copy = args;
941 std::vector<char *> argv = GetArgvForParsing(args);
942
943 std::unique_lock<std::mutex> lock;
944 OptionParser::Prepare(lock);
945 int val;
946 while (true) {
947 int long_options_index = -1;
948 val = OptionParser::Parse(argv, short_options, long_options,
949 &long_options_index);
950
951 if (val == ':') {
952 return llvm::createStringError(llvm::inconvertibleErrorCode(),
953 "last option requires an argument");
954 }
955
956 if (val == -1)
957 break;
958
959 if (val == '?') {
960 return llvm::createStringError("Unknown or ambiguous option");
961 }
962
963 if (val == 0)
964 continue;
965
966 OptionSeen(val);
967
968 // Look up the long option index
969 if (long_options_index == -1) {
970 for (int j = 0; long_options[j].definition || long_options[j].flag ||
971 long_options[j].val;
972 ++j) {
973 if (long_options[j].val == val) {
974 long_options_index = j;
975 break;
976 }
977 }
978 }
979
980 // See if the option takes an argument, and see if one was supplied.
981 if (long_options_index == -1) {
982 return llvm::createStringError(
983 llvm::formatv("Invalid option with value '{0}'.", char(val)).str());
984 }
985
986 StreamString option_str;
987 option_str.Printf("-%c", val);
988 const OptionDefinition *def = long_options[long_options_index].definition;
989 int has_arg =
990 (def == nullptr) ? OptionParser::eNoArgument : def->option_has_arg;
991
992 const char *option_arg = nullptr;
993 switch (has_arg) {
994 case OptionParser::eRequiredArgument:
995 if (OptionParser::GetOptionArgument() == nullptr) {
996 return llvm::createStringError(
997 llvm::formatv("Option '{0}' is missing argument specifier.",
998 option_str.GetString())
999 .str());
1000 }
1001 [[fallthrough]];
1002 case OptionParser::eOptionalArgument:
1003 option_arg = OptionParser::GetOptionArgument();
1004 [[fallthrough]];
1005 case OptionParser::eNoArgument:
1006 break;
1007 default:
1008 return llvm::createStringError(
1009 llvm::formatv("error with options table; invalid value in has_arg "
1010 "field for option '{0}'.",
1011 char(val))
1012 .str());
1013 }
1014 // Find option in the argument list; also see if it was supposed to take an
1015 // argument and if one was supplied. Remove option (and argument, if
1016 // given) from the argument list. Also remove them from the
1017 // raw_input_string, if one was passed in.
1018 // Note: We also need to preserve any option argument values that were
1019 // surrounded by backticks, as we lose track of them in the
1020 // option_args_vector.
1021 size_t idx =
1022 FindArgumentIndexForOption(args_copy, long_options[long_options_index]);
1023 std::string option_to_insert;
1024 if (option_arg) {
1025 if (idx != size_t(-1) && has_arg) {
1026 bool arg_has_backtick = args_copy[idx + 1].GetQuoteChar() == '`';
1027 if (arg_has_backtick)
1028 option_to_insert = "`";
1029 option_to_insert += option_arg;
1030 if (arg_has_backtick)
1031 option_to_insert += "`";
1032 } else
1033 option_to_insert = option_arg;
1034 } else
1035 option_to_insert = CommandInterpreter::g_no_argument;
1036
1037 option_arg_vector->emplace_back(std::string(option_str.GetString()),
1038 has_arg, option_to_insert);
1039
1040 if (idx == size_t(-1))
1041 continue;
1042
1043 if (!input_line.empty()) {
1044 llvm::StringRef tmp_arg = args_copy[idx].ref();
1045 size_t pos = input_line.find(tmp_arg);
1046 if (pos != std::string::npos)
1047 input_line.erase(pos, tmp_arg.size());
1048 }
1049 args_copy.DeleteArgumentAtIndex(idx);
1050 if ((option_to_insert != CommandInterpreter::g_no_argument) &&
1051 (OptionParser::GetOptionArgument() != nullptr) &&
1052 (idx < args_copy.GetArgumentCount()) &&
1053 (args_copy[idx].ref() == OptionParser::GetOptionArgument())) {
1054 if (input_line.size() > 0) {
1055 size_t pos = input_line.find(option_to_insert);
1056 if (pos != std::string::npos)
1057 input_line.erase(pos, option_to_insert.size());
1058 }
1059 args_copy.DeleteArgumentAtIndex(idx);
1060 }
1061 }
1062
1063 return std::move(args_copy);
1064}

1066OptionElementVector Options::ParseForCompletion(const Args &args,
1067 uint32_t cursor_index) {
1068 OptionElementVector option_element_vector;
1069 Option *long_options = GetLongOptions();
1070 option_element_vector.clear();
1071
1072 if (long_options == nullptr)
1073 return option_element_vector;
1074
1075 std::string short_options = BuildShortOptions(long_options);
1076
1077 std::unique_lock<std::mutex> lock;
1078 OptionParser::Prepare(lock);
1079 OptionParser::EnableError(false);
1080
1081 int val;
1082 auto opt_defs = GetDefinitions();
1083
1084 std::vector<char *> dummy_vec = GetArgvForParsing(args);
1085
1086 bool failed_once = false;
1087 uint32_t dash_dash_pos = -1;
1088
1089 while (true) {
1090 bool missing_argument = false;
1091 int long_options_index = -1;
1092
1093 val = OptionParser::Parse(dummy_vec, short_options, long_options,
1094 &long_options_index);
1095
1096 if (val == -1) {
1097 // When we're completing a "--" which is the last option on line,
1098 if (failed_once)
1099 break;
1100
1101 failed_once = true;
1102
1103 // If this is a bare "--" we mark it as such so we can complete it
1104 // successfully later. Handling the "--" is a little tricky, since that
1105 // may mean end of options or arguments, or the user might want to
1106 // complete options by long name. I make this work by checking whether
1107 // the cursor is in the "--" argument, and if so I assume we're
1108 // completing the long option, otherwise I let it pass to
1109 // OptionParser::Parse which will terminate the option parsing. Note, in
1110 // either case we continue parsing the line so we can figure out what
1111 // other options were passed. This will be useful when we come to
1112 // restricting completions based on what other options we've seen on the
1113 // line.
1114
1115 if (static_cast<size_t>(OptionParser::GetOptionIndex()) <
1116 dummy_vec.size() &&
1117 (strcmp(dummy_vec[OptionParser::GetOptionIndex() - 1], "--") == 0)) {
1118 dash_dash_pos = FindOriginalIndex(
1119 dummy_vec[OptionParser::GetOptionIndex() - 1], args);
1120 if (dash_dash_pos == cursor_index) {
1121 option_element_vector.push_back(
1122 OptionArgElement(OptionArgElement::eBareDoubleDash, dash_dash_pos,
1123 OptionArgElement::eBareDoubleDash));
1124 continue;
1125 } else
1126 break;
1127 } else
1128 break;
1129 } else if (val == '?') {
1130 option_element_vector.push_back(OptionArgElement(
1131 OptionArgElement::eUnrecognizedArg,
1132 FindOriginalIndex(dummy_vec[OptionParser::GetOptionIndex() - 1],
1133 args),
1134 OptionArgElement::eUnrecognizedArg));
1135 continue;
1136 } else if (val == 0) {
1137 continue;
1138 } else if (val == ':') {
1139 // This is a missing argument.
1140 val = OptionParser::GetOptionErrorCause();
1141 missing_argument = true;
1142 }
1143
1144 OptionSeen(val);
1145
1146 // Look up the long option index
1147 if (long_options_index == -1) {
1148 for (int j = 0; long_options[j].definition || long_options[j].flag ||
1149 long_options[j].val;
1150 ++j) {
1151 if (long_options[j].val == val) {
1152 long_options_index = j;
1153 break;
1154 }
1155 }
1156 }
1157
1158 // See if the option takes an argument, and see if one was supplied.
1159 if (long_options_index >= 0) {
1160 int opt_defs_index = -1;
1161 for (size_t i = 0; i < opt_defs.size(); i++) {
1162 if (opt_defs[i].short_option != val)
1163 continue;
1164 opt_defs_index = i;
1165 break;
1166 }
1167
1168 const OptionDefinition *def = long_options[long_options_index].definition;
1169 int has_arg =
1170 (def == nullptr) ? OptionParser::eNoArgument : def->option_has_arg;
1171 switch (has_arg) {
1172 case OptionParser::eNoArgument:
1173 option_element_vector.push_back(OptionArgElement(
1174 opt_defs_index,
1175 FindOriginalIndex(dummy_vec[OptionParser::GetOptionIndex() - 1],
1176 args),
1177 0));
1178 break;
1179 case OptionParser::eRequiredArgument:
1180 if (OptionParser::GetOptionArgument() != nullptr) {
1181 int arg_index;
1182 if (missing_argument)
1183 arg_index = -1;
1184 else
1185 arg_index = OptionParser::GetOptionIndex() - 2;
1186
1187 option_element_vector.push_back(OptionArgElement(
1188 opt_defs_index,
1189 FindOriginalIndex(dummy_vec[OptionParser::GetOptionIndex() - 2],
1190 args),
1191 arg_index));
1192 } else {
1193 option_element_vector.push_back(OptionArgElement(
1194 opt_defs_index,
1195 FindOriginalIndex(dummy_vec[OptionParser::GetOptionIndex() - 1],
1196 args),
1197 -1));
1198 }
1199 break;
1200 case OptionParser::eOptionalArgument:
1201 option_element_vector.push_back(OptionArgElement(
1202 opt_defs_index,
1203 FindOriginalIndex(dummy_vec[OptionParser::GetOptionIndex() - 2],
1204 args),
1205 FindOriginalIndex(dummy_vec[OptionParser::GetOptionIndex() - 1],
1206 args)));
1207 break;
1208 default:
1209 // The options table is messed up. Here we'll just continue
1210 option_element_vector.push_back(OptionArgElement(
1211 OptionArgElement::eUnrecognizedArg,
1212 FindOriginalIndex(dummy_vec[OptionParser::GetOptionIndex() - 1],
1213 args),
1214 OptionArgElement::eUnrecognizedArg));
1215 break;
1216 }
1217 } else {
1218 option_element_vector.push_back(OptionArgElement(
1219 OptionArgElement::eUnrecognizedArg,
1220 FindOriginalIndex(dummy_vec[OptionParser::GetOptionIndex() - 1],
1221 args),
1222 OptionArgElement::eUnrecognizedArg));
1223 }
1224 }
1225
1226 // Finally we have to handle the case where the cursor index points at a
1227 // single "-". We want to mark that in the option_element_vector, but only
1228 // if it is not after the "--". But it turns out that OptionParser::Parse
1229 // just ignores an isolated "-". So we have to look it up by hand here. We
1230 // only care if it is AT the cursor position. Note, a single quoted dash is
1231 // not the same as a single dash...
1232
1233 const Args::ArgEntry &cursor = args[cursor_index];
1234 if ((static_cast<int32_t>(dash_dash_pos) == -1 ||
1235 cursor_index < dash_dash_pos) &&
1236 !cursor.IsQuoted() && cursor.ref() == "-") {
1237 option_element_vector.push_back(
1238 OptionArgElement(OptionArgElement::eBareDash, cursor_index,
1239 OptionArgElement::eBareDash));
1240 }
1241 return option_element_vector;
1242}
1243
1244llvm::Expected<Args> Options::Parse(const Args &args,
1245 ExecutionContext *execution_context,
1246 lldb::PlatformSP platform_sp,
1247 bool require_validation) {
1248 Status error;
1249 Option *long_options = GetLongOptions();
1250 if (long_options == nullptr) {
1251 return llvm::createStringError("invalid long options");
1252 }
1253
1254 std::string short_options = BuildShortOptions(long_options);
1255 std::vector<char *> argv = GetArgvForParsing(args);
1256
1257 std::unique_lock<std::mutex> lock;
1258 OptionParser::Prepare(lock);
1259 while (true) {
1260 int long_options_index = -1;
1261 int val = OptionParser::Parse(argv, short_options, long_options,
1262 &long_options_index);
1263
1264 if (val == ':') {
1265 error = Status::FromErrorString("last option requires an argument");
1266 break;
1267 }
1268
1269 if (val == -1)
1270 break;
1271
1272 // Did we get an error?
1273 if (val == '?') {
1274 // Account for "argv[0]" and that it points to the next option.
1275 int idx = OptionParser::GetOptionIndex() - 2;
1276 if (idx >= 0 && (size_t)idx < args.GetArgumentCount())
1277 error = Status::FromError(llvm::make_error<OptionParseError>(
1278 args[idx], "unknown or ambiguous option"));
1279 else
1280 error = Status("unknown or ambiguous option");
1281
1282 break;
1283 }
1284 // The option auto-set itself
1285 if (val == 0)
1286 continue;
1287
1288 OptionSeen(val);
1289
1290 // Lookup the long option index
1291 if (long_options_index == -1) {
1292 for (int i = 0; long_options[i].definition || long_options[i].flag ||
1293 long_options[i].val;
1294 ++i) {
1295 if (long_options[i].val == val) {
1296 long_options_index = i;
1297 break;
1298 }
1299 }
1300 }
1301 // Call the callback with the option
1302 if (long_options_index >= 0 &&
1303 long_options[long_options_index].definition) {
1304 const OptionDefinition *def = long_options[long_options_index].definition;
1305
1306 if (!platform_sp) {
1307 // User did not pass in an explicit platform. Try to grab from the
1308 // execution context.
1309 TargetSP target_sp =
1310 execution_context ? execution_context->GetTargetSP() : TargetSP();
1311 platform_sp = target_sp ? target_sp->GetPlatform() : PlatformSP();
1312 }
1313 OptionValidator *validator = def->validator;
1314
1315 if (!platform_sp && require_validation) {
1316 // Caller requires validation but we cannot validate as we don't have
1317 // the mandatory platform against which to validate.
1318 return llvm::createStringError(
1319 "cannot validate options: no platform available");
1320 }
1321
1322 bool validation_failed = false;
1323 if (platform_sp) {
1324 // Ensure we have an execution context, empty or not.
1325 ExecutionContext dummy_context;
1326 ExecutionContext *exe_ctx_p =
1327 execution_context ? execution_context : &dummy_context;
1328 if (validator && !validator->IsValid(*platform_sp, *exe_ctx_p)) {
1329 validation_failed = true;
1330 error = Status::FromErrorStringWithFormat(
1331 "Option \"%s\" invalid. %s", def->long_option,
1332 def->validator->LongConditionString());
1333 }
1334 }
1335
1336 // As long as validation didn't fail, we set the option value.
1337 if (!validation_failed)
1338 error =
1339 SetOptionValue(long_options_index,
1340 (def->option_has_arg == OptionParser::eNoArgument)
1341 ? nullptr
1342 : OptionParser::GetOptionArgument(),
1343 execution_context);
1344 // If the Option setting returned an error, we should stop parsing
1345 // and return the error.
1346 if (error.Fail())
