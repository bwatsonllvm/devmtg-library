32void RegisterContext::InvalidateIfNeeded(bool force) {
33 ProcessSP process_sp(m_thread.GetProcess());
34 bool invalidate = force;
35 uint32_t process_stop_id = UINT32_MAX;
36
37 if (process_sp)
38 process_stop_id = process_sp->GetStopID();
39 else
40 invalidate = true;
41
42 if (!invalidate)
43 invalidate = process_stop_id != GetStopID();
44
45 if (invalidate) {
46 InvalidateAllRegisters();
47 SetStopID(process_stop_id);
48 }
49}

51const RegisterInfo *
52RegisterContext::GetRegisterInfoByName(llvm::StringRef reg_name,
53 uint32_t start_idx) {
54 if (reg_name.empty())
55 return nullptr;
56
57 // Generic register names take precedence over specific register names.
58 // For example, on x86 we want "sp" to refer to the complete RSP/ESP register
59 // rather than the 16-bit SP pseudo-register.
60 uint32_t generic_reg = Args::StringToGenericRegister(reg_name);
61 if (generic_reg != LLDB_INVALID_REGNUM) {
62 const RegisterInfo *reg_info =
63 GetRegisterInfo(eRegisterKindGeneric, generic_reg);
64 if (reg_info)
65 return reg_info;
66 }
67
68 const uint32_t num_registers = GetRegisterCount();
69 for (uint32_t reg = start_idx; reg < num_registers; ++reg) {
70 const RegisterInfo *reg_info = GetRegisterInfoAtIndex(reg);
71
72 if (reg_name.equals_insensitive(reg_info->name) ||
73 reg_name.equals_insensitive(reg_info->alt_name))
74 return reg_info;
75 }
76
77 return nullptr;
78}

95uint64_t RegisterContext::GetPC(uint64_t fail_value) {
96 uint32_t reg = ConvertRegisterKindToRegisterNumber(eRegisterKindGeneric,
97 LLDB_REGNUM_GENERIC_PC);
98 uint64_t pc = ReadRegisterAsUnsigned(reg, fail_value);
99
100 if (pc != fail_value) {
101 if (TargetSP target_sp = m_thread.CalculateTarget())
102 pc = target_sp->GetOpcodeLoadAddress(pc, AddressClass::eCode);
103 }
104
105 return pc;

114bool RegisterContext::SetPC(uint64_t pc) {
115 uint32_t reg = ConvertRegisterKindToRegisterNumber(eRegisterKindGeneric,
116 LLDB_REGNUM_GENERIC_PC);
117 bool success = WriteRegisterFromUnsigned(reg, pc);
118 if (success) {
119 StackFrameSP frame_sp(
120 m_thread.GetFrameWithConcreteFrameIndex(m_concrete_frame_idx));
121 if (frame_sp)
122 frame_sp->ChangePC(pc);
123 else
124 m_thread.ClearStackFrames();
125 }
126 return success;
127}

129bool RegisterContext::GetPCForSymbolication(Address &address) {
130 addr_t pc = GetPC(LLDB_INVALID_ADDRESS);
131 if (pc == LLDB_INVALID_ADDRESS)
132 return false;
133 TargetSP target_sp = m_thread.CalculateTarget();
134 if (!target_sp.get())
135 return false;
136
137 if (!BehavesLikeZerothFrame() && pc != 0)
138 pc--;
139 address.SetLoadAddress(pc, target_sp.get());
140 return true;
141}

223bool RegisterContext::CopyFromRegisterContext(lldb::RegisterContextSP context) {
224 uint32_t num_register_sets = context->GetRegisterSetCount();
225 // We don't know that two threads have the same register context, so require
226 // the threads to be the same.
227 if (context->GetThreadID() != GetThreadID())
228 return false;
229
230 if (num_register_sets != GetRegisterSetCount())
231 return false;
232
233 RegisterContextSP frame_zero_context = m_thread.GetRegisterContext();
234
235 for (uint32_t set_idx = 0; set_idx < num_register_sets; ++set_idx) {
236 const RegisterSet *const reg_set = GetRegisterSet(set_idx);
237
238 const uint32_t num_registers = reg_set->num_registers;
239 for (uint32_t reg_idx = 0; reg_idx < num_registers; ++reg_idx) {
240 const uint32_t reg = reg_set->registers[reg_idx];
241 const RegisterInfo *reg_info = GetRegisterInfoAtIndex(reg);
242 if (!reg_info || reg_info->value_regs)
243 continue;
244 RegisterValue reg_value;
245
246 // If we can reconstruct the register from the frame we are copying from,
247 // then do so, otherwise use the value from frame 0.
248 if (context->ReadRegister(reg_info, reg_value)) {
249 WriteRegister(reg_info, reg_value);
250 } else if (frame_zero_context->ReadRegister(reg_info, reg_value)) {
251 WriteRegister(reg_info, reg_value);
252 }
253 }
254 }
255 return true;
256}

267// Used when parsing DWARF and EH frame information and any other object file

269uint32_t
270RegisterContext::ConvertRegisterKindToRegisterNumber(lldb::RegisterKind kind,
271 uint32_t num) {
272 const uint32_t num_regs = GetRegisterCount();
273
274 assert(kind < kNumRegisterKinds);
275 for (uint32_t reg_idx = 0; reg_idx < num_regs; ++reg_idx) {
276 const RegisterInfo *reg_info = GetRegisterInfoAtIndex(reg_idx);
277
278 if (reg_info->kinds[kind] == num)
279 return reg_idx;
280 }
281
282 return LLDB_INVALID_REGNUM;
283}

300Status RegisterContext::ReadRegisterValueFromMemory(
301 const RegisterInfo *reg_info, lldb::addr_t src_addr, uint32_t src_len,
302 RegisterValue &reg_value) {
303 Status error;
304 if (!reg_info)
305 return Status::FromErrorString("invalid register info argument.");
306
307 // Moving from addr into a register
308 //
309 // Case 1: src_len == dst_len
310 //
311 // |AABBCCDD| Address contents
312 // |AABBCCDD| Register contents
313 //
314 // Case 2: src_len > dst_len
315 //
316 // Status! (The register should always be big enough to hold the data)
317 //
318 // Case 3: src_len < dst_len
319 //
320 // |AABB| Address contents
321 // |AABB0000| Register contents [on little-endian hardware]
322 // |0000AABB| Register contents [on big-endian hardware]
323 const uint32_t dst_len = reg_info->byte_size;
324
325 if (src_len > dst_len) {
326 return Status::FromErrorStringWithFormat(
327 "%u bytes is too big to store in register %s (%u bytes)", src_len,
328 reg_info->name, dst_len);
329 return error;
330 }
331
332 ProcessSP process_sp(m_thread.GetProcess());
333 if (process_sp) {
334 RegisterValue::BytesContainer src(src_len);
335
336 // Read the memory
337 const uint32_t bytes_read =
338 process_sp->ReadMemory(src_addr, src.data(), src_len, error);
339
340 // Make sure the memory read succeeded...
341 if (bytes_read != src_len) {
342 if (error.Success()) {
343 // This might happen if we read _some_ bytes but not all
344 return Status::FromErrorStringWithFormat("read %u of %u bytes",
345 bytes_read, src_len);
346 }
347 return error;
348 }
349
350 // We now have a memory buffer that contains the part or all of the
351 // register value. Set the register value using this memory data.
352 // TODO: we might need to add a parameter to this function in case the byte
353 // order of the memory data doesn't match the process. For now we are
354 // assuming they are the same.
355 reg_value.SetFromMemoryData(*reg_info, src.data(), src_len,
356 process_sp->GetByteOrder(), error);
357 } else
358 return Status::FromErrorString("invalid process");
359
360 return error;
361}

363Status RegisterContext::WriteRegisterValueToMemory(
364 const RegisterInfo *reg_info, lldb::addr_t dst_addr, uint32_t dst_len,
365 const RegisterValue &reg_value) {
366 Status error;
367 ProcessSP process_sp(m_thread.GetProcess());
368
369 if (!process_sp) {
370 return Status::FromErrorString("invalid process");
371 return error;
372 }
373
374 if (reg_info == nullptr) {
375 return Status::FromErrorString("Invalid register info argument.");
376 return error;
377 }
378
379 // TODO: we might need to add a parameter to this function in case the byte
380 // order of the memory data doesn't match the process. For now we are
381 // assuming they are the same.
382 RegisterValue::BytesContainer dst(dst_len);
383 const uint32_t bytes_copied = reg_value.GetAsMemoryData(
384 *reg_info, dst.data(), dst_len, process_sp->GetByteOrder(), error);
385
386 if (error.Success()) {
387 if (bytes_copied == 0) {
388 return Status::FromErrorString("byte copy failed.");
389 } else {
390 const uint32_t bytes_written =
391 process_sp->WriteMemory(dst_addr, dst.data(), bytes_copied, error);
392 if (bytes_written != bytes_copied) {
393 if (error.Success()) {
394 // This might happen if we read _some_ bytes but not all
395 return Status::FromErrorStringWithFormat("only wrote %u of %u bytes",
396 bytes_written, bytes_copied);
397 }
398 }
399 }
400 }
401
402 return error;
403}

449bool RegisterContext::ConvertBetweenRegisterKinds(lldb::RegisterKind source_rk,
450 uint32_t source_regnum,
451 lldb::RegisterKind target_rk,
452 uint32_t &target_regnum) {
453 const uint32_t num_registers = GetRegisterCount();
454 for (uint32_t reg = 0; reg < num_registers; ++reg) {
455 const RegisterInfo *reg_info = GetRegisterInfoAtIndex(reg);
456
457 if (reg_info->kinds[source_rk] == source_regnum) {
458 target_regnum = reg_info->kinds[target_rk];
459 return (target_regnum != LLDB_INVALID_REGNUM);
460 }
461 }
462 return false;
