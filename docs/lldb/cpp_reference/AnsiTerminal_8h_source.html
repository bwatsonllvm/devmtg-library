169 static const char tok_hdr[] = "${ansi.";
170
171 std::string fmt;
172 while (!format.empty()) {
173 llvm::StringRef left, right;
174 std::tie(left, right) = format.split(tok_hdr);
175
176 fmt += left;
177
178 if (left == format && right.empty()) {
179 // The header was not found. Just exit.
180 break;
181 }
182
183 bool found_code = false;
184 for (const auto &code : codes) {
185 if (!right.consume_front(code.name))
186 continue;
187
188 if (do_color)
189 fmt.append(code.value);
190 found_code = true;
191 break;
192 }
193 format = right;
194 // If we haven't found a valid replacement value, we just copy the string
195 // to the result without any modifications.
196 if (!found_code)
197 fmt.append(tok_hdr);
198 }
199 return fmt;

202inline std::tuple<llvm::StringRef, llvm::StringRef, llvm::StringRef>
203FindNextAnsiSequence(llvm::StringRef str) {
204 llvm::StringRef left;
205 llvm::StringRef right = str;
206
207 while (!right.empty()) {
208 const size_t start = right.find(ANSI_ESC_START);
209
210 // ANSI_ESC_START not found.
211 if (start == llvm::StringRef::npos)
212 return {str, {}, {}};
213
214 // Split the string around the current ANSI_ESC_START.
215 left = str.take_front(left.size() + start);
216 llvm::StringRef escape = right.substr(start);
217 right = right.substr(start + ANSI_ESC_START_LEN + 1);
218
219 const size_t end = right.find_first_not_of("0123456789;");
220
221 // ANSI_ESC_END found.
222 if (end < right.size() && (right[end] == 'm' || right[end] == 'G'))
223 return {left, escape.take_front(ANSI_ESC_START_LEN + 1 + end + 1),
224 right.substr(end + 1)};
225
226 // Maintain the invariant that str == left + right at the start of the loop.
227 left = str.take_front(left.size() + ANSI_ESC_START_LEN + 1);
228 }
229
230 return {str, {}, {}};
231}

233inline std::string StripAnsiTerminalCodes(llvm::StringRef str) {
234 std::string stripped;
235 while (!str.empty()) {
236 auto [left, escape, right] = FindNextAnsiSequence(str);
237 stripped += left;
238 str = right;
239 }
240 return stripped;
241}

243inline std::string TrimAndPad(llvm::StringRef str, size_t visible_length,
244 char padding = ' ') {
245 std::string result;
246 result.reserve(visible_length);
247 size_t result_visibile_length = 0;
248
249 // Trim the string to the given visible length.
250 while (!str.empty()) {
251 auto [left, escape, right] = FindNextAnsiSequence(str);
252 str = right;
253
254 // Compute the length of the string without escape codes. If it fits, append
255 // it together with the invisible escape code.
256 size_t column_width = llvm::sys::locale::columnWidth(left);
257 if (result_visibile_length + column_width <= visible_length) {
258 result.append(left).append(escape);
259 result_visibile_length += column_width;
260 continue;
261 }
262
263 // The string might contain unicode which means it's not safe to truncate.
264 // Repeatedly trim the string until it its valid unicode and fits.
265 llvm::StringRef trimmed = left;
266 while (!trimmed.empty()) {
267 // This relies on columnWidth returning -2 for invalid/partial unicode
268 // characters, which after conversion to size_t will be larger than the
269 // visible width.
270 column_width = llvm::sys::locale::columnWidth(trimmed);
271 if (result_visibile_length + column_width <= visible_length) {
272 result.append(trimmed);
273 result_visibile_length += column_width;
274 break;
275 }
276 trimmed = trimmed.drop_back();
277 }
278 }
279
280 // Pad the string.
281 if (result_visibile_length < visible_length)
282 result.append(visible_length - result_visibile_length, padding);
283
284 return result;
285}

287inline size_t ColumnWidth(llvm::StringRef str) {
288 std::string stripped = ansi::StripAnsiTerminalCodes(str);
289 return llvm::sys::locale::columnWidth(stripped);

292// Output text that may contain ANSI codes, word wrapped (wrapped at whitespace)
293// to the given stream. The indent level of the stream is counted towards the
294// output line length.
295// FIXME: This contains several bugs and does not handle unicode.
296inline void OutputWordWrappedLines(Stream &strm, llvm::StringRef text,
297 uint32_t output_max_columns) {
298 // We will indent using the stream, so leading whitespace is not significant.
299 text = text.ltrim();
300 if (text.size() == 0)
301 return;
302
303 const size_t visible_length = ansi::ColumnWidth(text);
304
305 // Will it all fit on one line, or is it a single word that we must not break?
306 if (static_cast<uint32_t>(visible_length + strm.GetIndentLevel()) <
307 output_max_columns ||
308 text.find_first_of(" \t\n") == llvm::StringRef::npos) {
309 // Output it as a single line.
310 strm.Indent(text);
311 strm.EOL();
312 return;
313 }
314
315 // We need to break it up into multiple lines. We can do this based on the
316 // formatted text because we know that:
317 // * We only break lines on whitespace, therefore we will not break in the
318 // middle of a Unicode character or escape code.
319 // * Escape codes are so far not applied to multiple words, so there is no
320 // risk of breaking up a phrase and the escape code being incorrectly
321 // applied to the indent too.
322
323 const int max_text_width = output_max_columns - strm.GetIndentLevel() - 1;
324 int start = 0;
325 int end = start;
326 const int final_end = visible_length;
327
328 while (end < final_end) {
329 // Don't start the 'text' on a space, since we're already outputting the
330 // indentation.
331 while ((start < final_end) && (text[start] == ' '))
332 start++;
333
334 end = start + max_text_width;
335 if (end > final_end)
336 end = final_end;
337
338 if (end != final_end) {
339 // If we're not at the end of the text, make sure we break the line on
340 // white space.
341 while (end > start && text[end] != ' ' && text[end] != '\t' &&
342 text[end] != '\n')
343 end--;
344 }
345
346 const int sub_len = end - start;
347 if (start != 0)
348 strm.EOL();
349 strm.Indent();
350 assert(start < final_end);
351 assert(start + sub_len <= final_end);
352 strm << text.substr(start, sub_len);
353 start = end + 1;
354 }
355 strm.EOL();
356}
