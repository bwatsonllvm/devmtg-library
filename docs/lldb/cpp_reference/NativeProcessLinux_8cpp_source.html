80static bool ProcessVmReadvSupported() {
81 static bool is_supported;
82 static llvm::once_flag flag;
83
84 llvm::call_once(flag, [] {
85 Log *log = GetLog(POSIXLog::Process);
86
87 uint32_t source = 0x47424742;
88 uint32_t dest = 0;
89
90 struct iovec local, remote;
91 remote.iov_base = &source;
92 local.iov_base = &dest;
93 remote.iov_len = local.iov_len = sizeof source;
94
95 // We shall try if cross-process-memory reads work by attempting to read a
96 // value from our own process.
97 ssize_t res = process_vm_readv(getpid(), &local, 1, &remote, 1, 0);
98 is_supported = (res == sizeof(source) && source == dest);
99 if (is_supported)
100 LLDB_LOG(log,
101 "Detected kernel support for process_vm_readv syscall. "
102 "Fast memory reads enabled.");
103 else
104 LLDB_LOG(log,
105 "syscall process_vm_readv failed (error: {0}). Fast memory "
106 "reads disabled.",
107 llvm::sys::StrError());
108 });
109
110 return is_supported;
111}

113static void MaybeLogLaunchInfo(const ProcessLaunchInfo &info) {
114 Log *log = GetLog(POSIXLog::Process);
115 if (!log)
116 return;
117
118 if (const FileAction *action = info.GetFileActionForFD(STDIN_FILENO))
119 LLDB_LOG(log, "setting STDIN to '{0}'", action->GetFileSpec());
120 else
121 LLDB_LOG(log, "leaving STDIN as is");
122
123 if (const FileAction *action = info.GetFileActionForFD(STDOUT_FILENO))
124 LLDB_LOG(log, "setting STDOUT to '{0}'", action->GetFileSpec());
125 else
126 LLDB_LOG(log, "leaving STDOUT as is");
127
128 if (const FileAction *action = info.GetFileActionForFD(STDERR_FILENO))
129 LLDB_LOG(log, "setting STDERR to '{0}'", action->GetFileSpec());
130 else
131 LLDB_LOG(log, "leaving STDERR as is");
132
133 int i = 0;
134 for (const char **args = info.GetArguments().GetConstArgumentVector(); *args;
135 ++args, ++i)
136 LLDB_LOG(log, "arg {0}: '{1}'", i, *args);
137}

148static void PtraceDisplayBytes(int &req, void *data, size_t data_size) {
149 Log *log = GetLog(POSIXLog::Ptrace);
150 if (!log)
151 return;
152 StreamString buf;
153
154 switch (req) {
155 case PTRACE_POKETEXT: {
156 DisplayBytes(buf, &data, 8);
157 LLDB_LOGV(log, "PTRACE_POKETEXT {0}", buf.GetData());
158 break;
159 }
160 case PTRACE_POKEDATA: {
161 DisplayBytes(buf, &data, 8);
162 LLDB_LOGV(log, "PTRACE_POKEDATA {0}", buf.GetData());
163 break;
164 }
165 case PTRACE_POKEUSER: {
166 DisplayBytes(buf, &data, 8);
167 LLDB_LOGV(log, "PTRACE_POKEUSER {0}", buf.GetData());
168 break;
169 }
170 case PTRACE_SETREGS: {
171 DisplayBytes(buf, data, data_size);
172 LLDB_LOGV(log, "PTRACE_SETREGS {0}", buf.GetData());
173 break;
174 }
175 case PTRACE_SETFPREGS: {
176 DisplayBytes(buf, data, data_size);
177 LLDB_LOGV(log, "PTRACE_SETFPREGS {0}", buf.GetData());
178 break;
179 }
180 case PTRACE_SETSIGINFO: {
181 DisplayBytes(buf, data, sizeof(siginfo_t));
182 LLDB_LOGV(log, "PTRACE_SETSIGINFO {0}", buf.GetData());
183 break;
184 }
185 case PTRACE_SETREGSET: {
186 // Extract iov_base from data, which is a pointer to the struct iovec
187 DisplayBytes(buf, *(void **)data, data_size);
188 LLDB_LOGV(log, "PTRACE_SETREGSET {0}", buf.GetData());
189 break;
190 }
191 default: {}
192 }
193}

199// Simple helper function to ensure flags are enabled on the given file
200// descriptor.
201static Status EnsureFDFlags(int fd, int flags) {
202 Status error;
203
204 int status = fcntl(fd, F_GETFL);
205 if (status == -1) {
206 error = Status::FromErrno();
207 return error;
208 }
209
210 if (fcntl(fd, F_SETFL, status | flags) == -1) {
211 error = Status::FromErrno();
212 return error;
213 }
214
215 return error;
216}

218static llvm::Error AddPtraceScopeNote(llvm::Error original_error) {
219 Expected<int> ptrace_scope = GetPtraceScope();
220 if (auto E = ptrace_scope.takeError()) {
221 LLDB_LOG_ERROR(GetLog(POSIXLog::Process), std::move(E),
222 "error reading value of ptrace_scope: {0}");
223
224 // The original error is probably more interesting than not being able to
225 // read or interpret ptrace_scope.
226 return original_error;
227 }
228
229 // We only have suggestions to provide for 1-3.
230 switch (*ptrace_scope) {
231 case 1:
232 case 2:
233 llvm::consumeError(std::move(original_error));
234 return llvm::createStringError(
235 std::error_code(errno, std::generic_category()),
236 "The current value of ptrace_scope is %d, which can cause ptrace to "
237 "fail to attach to a running process. To fix this, run:\n"
238 "\tsudo sysctl -w kernel.yama.ptrace_scope=0\n"
239 "For more information, see: "
240 "https://www.kernel.org/doc/Documentation/security/Yama.txt.",
241 *ptrace_scope);
242 case 3:
243 llvm::consumeError(std::move(original_error));
244 return llvm::createStringError(
245 std::error_code(errno, std::generic_category()),
246 "The current value of ptrace_scope is 3, which will cause ptrace to "
247 "fail to attach to a running process. This value cannot be changed "
248 "without rebooting.\n"
249 "For more information, see: "
250 "https://www.kernel.org/doc/Documentation/security/Yama.txt.");
251 case 0:
252 default:
253 return original_error;
254 }

265llvm::Expected<std::unique_ptr<NativeProcessProtocol>>
266NativeProcessLinux::Manager::Launch(ProcessLaunchInfo &launch_info,
267 NativeDelegate &native_delegate) {
268 Log *log = GetLog(POSIXLog::Process);
269
270 MaybeLogLaunchInfo(launch_info);
271
272 Status status;
273 ::pid_t pid = ProcessLauncherPosixFork()
274 .LaunchProcess(launch_info, status)
275 .GetProcessId();
276 LLDB_LOG(log, "pid = {0:x}", pid);
277 if (status.Fail()) {
278 LLDB_LOG(log, "failed to launch process: {0}", status);
279 return status.ToError();
280 }
281
282 // Wait for the child process to trap on its call to execve.
283 int wstatus = 0;
284 ::pid_t wpid = llvm::sys::RetryAfterSignal(-1, ::waitpid, pid, &wstatus, 0);
285 assert(wpid == pid);
286 UNUSED_IF_ASSERT_DISABLED(wpid);
287 if (!WIFSTOPPED(wstatus)) {
288 LLDB_LOG(log, "Could not sync with inferior process: wstatus={1}",
289 WaitStatus::Decode(wstatus));
290 return llvm::make_error<StringError>("Could not sync with inferior process",
291 llvm::inconvertibleErrorCode());
292 }
293 LLDB_LOG(log, "inferior started, now in stopped state");
294
295 status = SetDefaultPtraceOpts(pid);
296 if (status.Fail()) {
297 LLDB_LOG(log, "failed to set default ptrace options: {0}", status);
298 return status.ToError();
299 }
300
301 llvm::Expected<ArchSpec> arch_or =
302 NativeRegisterContextLinux::DetermineArchitecture(pid);
303 if (!arch_or)
304 return arch_or.takeError();
305
306 return std::unique_ptr<NativeProcessLinux>(new NativeProcessLinux(
307 pid, launch_info.GetPTY().ReleasePrimaryFileDescriptor(), native_delegate,
308 *arch_or, *this, {pid}));
309}

346static std::optional<std::pair<lldb::pid_t, WaitStatus>> WaitPid() {
347 Log *log = GetLog(POSIXLog::Process);
348
349 int status;
350 ::pid_t wait_pid = llvm::sys::RetryAfterSignal(
351 -1, ::waitpid, -1, &status, __WALL | __WNOTHREAD | WNOHANG);
352
353 if (wait_pid == 0)
354 return std::nullopt;
355
356 if (wait_pid == -1) {
357 Status error(errno, eErrorTypePOSIX);
358 LLDB_LOG(log, "waitpid(-1, &status, _) failed: {0}", error);
359 return std::nullopt;
360 }
361
362 WaitStatus wait_status = WaitStatus::Decode(status);
363
364 LLDB_LOG(log, "waitpid(-1, &status, _) = {0}, status = {1}", wait_pid,
365 wait_status);
366 return std::make_pair(wait_pid, wait_status);
367}

369void NativeProcessLinux::Manager::SigchldHandler() {
370 Log *log = GetLog(POSIXLog::Process);
371 while (true) {
372 auto wait_result = WaitPid();
373 if (!wait_result)
374 return;
375 lldb::pid_t pid = wait_result->first;
376 WaitStatus status = wait_result->second;
377
378 // Ask each process whether it wants to handle the event. Each event should
379 // be handled by exactly one process, but thread creation events require
380 // special handling.
381 // Thread creation consists of two events (one on the parent and one on the
382 // child thread) and they can arrive in any order nondeterministically. The
383 // parent event carries the information about the child thread, but not
384 // vice-versa. This means that if the child event arrives first, it may not
385 // be handled by any process (because it doesn't know the thread belongs to
386 // it).
387 bool handled = llvm::any_of(m_processes, [&](NativeProcessLinux *process) {
388 return process->TryHandleWaitStatus(pid, status);
389 });
390 if (!handled) {
391 if (status.type == WaitStatus::Stop && status.status == SIGSTOP) {
392 // Store the thread creation event for later collection.
393 m_unowned_threads.insert(pid);
394 } else {
395 LLDB_LOG(log, "Ignoring waitpid event {0} for pid {1}", status, pid);
396 }
397 }
398 }

401void NativeProcessLinux::Manager::CollectThread(::pid_t tid) {
402 Log *log = GetLog(POSIXLog::Process);
403
404 if (m_unowned_threads.erase(tid))
405 return; // We've encountered this thread already.
406
407 // The TID is not tracked yet, let's wait for it to appear.
408 int status = -1;
409 LLDB_LOG(log,
410 "received clone event for tid {0}. tid not tracked yet, "
411 "waiting for it to appear...",
412 tid);
413 ::pid_t wait_pid =
414 llvm::sys::RetryAfterSignal(-1, ::waitpid, tid, &status, __WALL);
415
416 // It's theoretically possible to get other events if the entire process was
417 // SIGKILLed before we got a chance to check this. In that case, we'll just
418 // clean everything up when we get the process exit event.
419
420 LLDB_LOG(log,
421 "waitpid({0}, &status, __WALL) => {1} (errno: {2}, status = {3})",
422 tid, wait_pid, errno, WaitStatus::Decode(status));
423}

449llvm::Expected<std::vector<::pid_t>> NativeProcessLinux::Attach(::pid_t pid) {
450 Log *log = GetLog(POSIXLog::Process);
451
452 Status status;
453 // Use a map to keep track of the threads which we have attached/need to
454 // attach.
455 Host::TidMap tids_to_attach;
456 while (Host::FindProcessThreads(pid, tids_to_attach)) {
457 for (Host::TidMap::iterator it = tids_to_attach.begin();
458 it != tids_to_attach.end();) {
459 if (it->second == false) {
460 lldb::tid_t tid = it->first;
461
462 // Attach to the requested process.
463 // An attach will cause the thread to stop with a SIGSTOP.
464 if ((status = PtraceWrapper(PTRACE_ATTACH, tid)).Fail()) {
465 // No such thread. The thread may have exited. More error handling
466 // may be needed.
467 if (status.GetError() == ESRCH) {
468 it = tids_to_attach.erase(it);
469 continue;
470 }
471 if (status.GetError() == EPERM) {
472 // Depending on the value of ptrace_scope, we can return a different
473 // error that suggests how to fix it.
474 return AddPtraceScopeNote(status.ToError());
475 }
476 return status.ToError();
477 }
478
479 int wpid =
480 llvm::sys::RetryAfterSignal(-1, ::waitpid, tid, nullptr, __WALL);
481 // Need to use __WALL otherwise we receive an error with errno=ECHLD At
482 // this point we should have a thread stopped if waitpid succeeds.
483 if (wpid < 0) {
484 // No such thread. The thread may have exited. More error handling
485 // may be needed.
486 if (errno == ESRCH) {
487 it = tids_to_attach.erase(it);
488 continue;
489 }
490 return llvm::errorCodeToError(
491 std::error_code(errno, std::generic_category()));
492 }
493
494 if ((status = SetDefaultPtraceOpts(tid)).Fail())
495 return status.ToError();
496
497 LLDB_LOG(log, "adding tid = {0}", tid);
498 it->second = true;
499 }
500
501 // move the loop forward
502 ++it;
503 }
504 }
505
506 size_t tid_count = tids_to_attach.size();
507 if (tid_count == 0)
508 return llvm::make_error<StringError>("No such process",
509 llvm::inconvertibleErrorCode());
510
511 std::vector<::pid_t> tids;
512 tids.reserve(tid_count);
513 for (const auto &p : tids_to_attach)
514 tids.push_back(p.first);
515 return std::move(tids);
516}

518Status NativeProcessLinux::SetDefaultPtraceOpts(lldb::pid_t pid) {
519 long ptrace_opts = 0;
520
521 // Have the child raise an event on exit. This is used to keep the child in
522 // limbo until it is destroyed.
523 ptrace_opts |= PTRACE_O_TRACEEXIT;
524
525 // Have the tracer trace threads which spawn in the inferior process.
526 ptrace_opts |= PTRACE_O_TRACECLONE;
527
528 // Have the tracer notify us before execve returns (needed to disable legacy
529 // SIGTRAP generation)
530 ptrace_opts |= PTRACE_O_TRACEEXEC;
531
532 // Have the tracer trace forked children.
533 ptrace_opts |= PTRACE_O_TRACEFORK;
534
535 // Have the tracer trace vforks.
536 ptrace_opts |= PTRACE_O_TRACEVFORK;
537
538 // Have the tracer trace vfork-done in order to restore breakpoints after
539 // the child finishes sharing memory.
540 ptrace_opts |= PTRACE_O_TRACEVFORKDONE;
541
542 return PtraceWrapper(PTRACE_SETOPTIONS, pid, nullptr, (void *)ptrace_opts);
543}

545bool NativeProcessLinux::TryHandleWaitStatus(lldb::pid_t pid,
546 WaitStatus status) {
547 if (pid == GetID() &&
548 (status.type == WaitStatus::Exit || status.type == WaitStatus::Signal)) {
549 // The process exited. We're done monitoring. Report to delegate.
550 SetExitStatus(status, true);
551 return true;
552 }
553 if (NativeThreadLinux *thread = GetThreadByID(pid)) {
554 MonitorCallback(*thread, status);
555 return true;
556 }
557 return false;

560void NativeProcessLinux::MonitorCallback(NativeThreadLinux &thread,
561 WaitStatus status) {
562 Log *log = GetLog(LLDBLog::Process);
563
564 // Certain activities differ based on whether the pid is the tid of the main
565 // thread.
566 const bool is_main_thread = (thread.GetID() == GetID());
567
568 // Handle when the thread exits.
569 if (status.type == WaitStatus::Exit || status.type == WaitStatus::Signal) {
570 LLDB_LOG(log,
571 "got exit status({0}) , tid = {1} ({2} main thread), process "
572 "state = {3}",
573 status, thread.GetID(), is_main_thread ? "is" : "is not",
574 GetState());
575
576 // This is a thread that exited. Ensure we're not tracking it anymore.
577 StopTrackingThread(thread);
578
579 assert(!is_main_thread && "Main thread exits handled elsewhere");
580 return;
581 }
582
583 siginfo_t info;
584 const auto info_err = GetSignalInfo(thread.GetID(), &info);
585
586 // Get details on the signal raised.
587 if (info_err.Success()) {
588 // We have retrieved the signal info. Dispatch appropriately.
589 if (info.si_signo == SIGTRAP)
590 MonitorSIGTRAP(info, thread);
591 else
592 MonitorSignal(info, thread);
593 } else {
594 if (info_err.GetError() == EINVAL) {
595 // This is a group stop reception for this tid. We can reach here if we
596 // reinject SIGSTOP, SIGSTP, SIGTTIN or SIGTTOU into the tracee,
597 // triggering the group-stop mechanism. Normally receiving these would
598 // stop the process, pending a SIGCONT. Simulating this state in a
599 // debugger is hard and is generally not needed (one use case is
600 // debugging background task being managed by a shell). For general use,
601 // it is sufficient to stop the process in a signal-delivery stop which
602 // happens before the group stop. This done by MonitorSignal and works
603 // correctly for all signals.
604 LLDB_LOG(log,
605 "received a group stop for pid {0} tid {1}. Transparent "
606 "handling of group stops not supported, resuming the "
607 "thread.",
608 GetID(), thread.GetID());
609 ResumeThread(thread, thread.GetState(), LLDB_INVALID_SIGNAL_NUMBER);
610 } else {
611 // ptrace(GETSIGINFO) failed (but not due to group-stop).
612
613 // A return value of ESRCH means the thread/process has died in the mean
614 // time. This can (e.g.) happen when another thread does an exit_group(2)
615 // or the entire process get SIGKILLed.
616 // We can't do anything with this thread anymore, but we keep it around
617 // until we get the WIFEXITED event.
618
619 LLDB_LOG(log,
620 "GetSignalInfo({0}) failed: {1}, status = {2}, main_thread = "
621 "{3}. Expecting WIFEXITED soon.",
622 thread.GetID(), info_err, status, is_main_thread);
623 }
624 }

627void NativeProcessLinux::MonitorSIGTRAP(const siginfo_t &info,
628 NativeThreadLinux &thread) {
629 Log *log = GetLog(POSIXLog::Process);
630 const bool is_main_thread = (thread.GetID() == GetID());
631
632 assert(info.si_signo == SIGTRAP && "Unexpected child signal!");
633
634 switch (info.si_code) {
635 case (SIGTRAP | (PTRACE_EVENT_FORK << 8)):
636 case (SIGTRAP | (PTRACE_EVENT_VFORK << 8)):
637 case (SIGTRAP | (PTRACE_EVENT_CLONE << 8)): {
638 // This can either mean a new thread or a new process spawned via
639 // clone(2) without SIGCHLD or CLONE_VFORK flag. Note that clone(2)
640 // can also cause PTRACE_EVENT_FORK and PTRACE_EVENT_VFORK if one
641 // of these flags are passed.
642
643 unsigned long event_message = 0;
644 if (GetEventMessage(thread.GetID(), &event_message).Fail()) {
645 LLDB_LOG(log,
646 "pid {0} received clone() event but GetEventMessage failed "
647 "so we don't know the new pid/tid",
648 thread.GetID());
649 ResumeThread(thread, thread.GetState(), LLDB_INVALID_SIGNAL_NUMBER);
650 } else {
651 MonitorClone(thread, event_message, info.si_code >> 8);
652 }
653
654 break;
655 }
656
657 case (SIGTRAP | (PTRACE_EVENT_EXEC << 8)): {
658 LLDB_LOG(log, "received exec event, code = {0}", info.si_code ^ SIGTRAP);
659
660 // Exec clears any pending notifications.
661 m_pending_notification_tid = LLDB_INVALID_THREAD_ID;
662
663 // Remove all but the main thread here. Linux fork creates a new process
664 // which only copies the main thread.
665 LLDB_LOG(log, "exec received, stop tracking all but main thread");
666
667 llvm::erase_if(m_threads, [&](std::unique_ptr<NativeThreadProtocol> &t) {
668 return t->GetID() != GetID();
669 });
670 assert(m_threads.size() == 1);
671 auto *main_thread = static_cast<NativeThreadLinux *>(m_threads[0].get());
672
673 SetCurrentThreadID(main_thread->GetID());
674 main_thread->SetStoppedByExec();
675
676 // Tell coordinator about the "new" (since exec) stopped main thread.
677 ThreadWasCreated(*main_thread);
678
679 // Let our delegate know we have just exec'd.
680 NotifyDidExec();
681
682 // Let the process know we're stopped.
683 StopRunningThreads(main_thread->GetID());
684
685 break;
686 }
687
688 case (SIGTRAP | (PTRACE_EVENT_EXIT << 8)): {
689 // The inferior process or one of its threads is about to exit. We don't
690 // want to do anything with the thread so we just resume it. In case we
691 // want to implement "break on thread exit" functionality, we would need to
692 // stop here.
693
694 unsigned long data = 0;
695 if (GetEventMessage(thread.GetID(), &data).Fail())
696 data = -1;
697
698 LLDB_LOG(log,
699 "received PTRACE_EVENT_EXIT, data = {0:x}, WIFEXITED={1}, "
700 "WIFSIGNALED={2}, pid = {3}, main_thread = {4}",
701 data, WIFEXITED(data), WIFSIGNALED(data), thread.GetID(),
702 is_main_thread);
703
704
705 StateType state = thread.GetState();
706 if (!StateIsRunningState(state)) {
707 // Due to a kernel bug, we may sometimes get this stop after the inferior
708 // gets a SIGKILL. This confuses our state tracking logic in
709 // ResumeThread(), since normally, we should not be receiving any ptrace
710 // events while the inferior is stopped. This makes sure that the
711 // inferior is resumed and exits normally.
712 state = eStateRunning;
713 }
714 ResumeThread(thread, state, LLDB_INVALID_SIGNAL_NUMBER);
715
716 if (is_main_thread) {
717 // Main thread report the read (WIFEXITED) event only after all threads in
718 // the process exit, so we need to stop tracking it here instead of in
719 // MonitorCallback
720 StopTrackingThread(thread);
721 }
722
723 break;
724 }
725
726 case (SIGTRAP | (PTRACE_EVENT_VFORK_DONE << 8)): {
727 if (bool(m_enabled_extensions & Extension::vfork)) {
728 thread.SetStoppedByVForkDone();
729 StopRunningThreads(thread.GetID());
730 }
731 else
732 ResumeThread(thread, thread.GetState(), LLDB_INVALID_SIGNAL_NUMBER);
733 break;
734 }
735
736 case 0:
737 case TRAP_TRACE: // We receive this on single stepping.
738 case TRAP_HWBKPT: // We receive this on watchpoint hit
739 {
740 // If a watchpoint was hit, report it
741 uint32_t wp_index;
742 Status error = thread.GetRegisterContext().GetWatchpointHitIndex(
743 wp_index, (uintptr_t)info.si_addr);
744 if (error.Fail())
745 LLDB_LOG(log,
746 "received error while checking for watchpoint hits, pid = "
747 "{0}, error = {1}",
748 thread.GetID(), error);
749 if (wp_index != LLDB_INVALID_INDEX32) {
750 MonitorWatchpoint(thread, wp_index);
751 break;
752 }
753
754 // If a breakpoint was hit, report it
755 uint32_t bp_index;
756 error = thread.GetRegisterContext().GetHardwareBreakHitIndex(
757 bp_index, (uintptr_t)info.si_addr);
758 if (error.Fail())
759 LLDB_LOG(log, "received error while checking for hardware "
760 "breakpoint hits, pid = {0}, error = {1}",
761 thread.GetID(), error);
762 if (bp_index != LLDB_INVALID_INDEX32) {
763 MonitorBreakpoint(thread);
764 break;
765 }
766
767 // Otherwise, report step over
768 MonitorTrace(thread);
769 break;
770 }
771
772 case SI_KERNEL:
773 case TRAP_BRKPT:
774 MonitorBreakpoint(thread);
775 break;
776
777 case SIGTRAP:
778 case (SIGTRAP | 0x80):
779 LLDB_LOG(
780 log,
781 "received unknown SIGTRAP stop event ({0}, pid {1} tid {2}, resuming",
782 info.si_code, GetID(), thread.GetID());
783
784 // Ignore these signals until we know more about them.
785 ResumeThread(thread, thread.GetState(), LLDB_INVALID_SIGNAL_NUMBER);
786 break;
787
788 default:
789 LLDB_LOG(log, "received unknown SIGTRAP stop event ({0}, pid {1} tid {2}",
790 info.si_code, GetID(), thread.GetID());
791 MonitorSignal(info, thread);
792 break;
793 }
794}

839void NativeProcessLinux::MonitorSignal(const siginfo_t &info,
840 NativeThreadLinux &thread) {
841 const int signo = info.si_signo;
842 const bool is_from_llgs = info.si_pid == getpid();
843
844 Log *log = GetLog(POSIXLog::Process);
845
846 // POSIX says that process behaviour is undefined after it ignores a SIGFPE,
847 // SIGILL, SIGSEGV, or SIGBUS *unless* that signal was generated by a kill(2)
848 // or raise(3). Similarly for tgkill(2) on Linux.
849 //
850 // IOW, user generated signals never generate what we consider to be a
851 // "crash".
852 //
853 // Similarly, ACK signals generated by this monitor.
854
855 // Handle the signal.
856 LLDB_LOG(log,
857 "received signal {0} ({1}) with code {2}, (siginfo pid = {3}, "
858 "waitpid pid = {4})",
859 Host::GetSignalAsCString(signo), signo, info.si_code, info.si_pid,
860 thread.GetID());
861
862 // Check for thread stop notification.
863 if (is_from_llgs && (info.si_code == SI_TKILL) && (signo == SIGSTOP)) {
864 // This is a tgkill()-based stop.
865 LLDB_LOG(log, "pid {0} tid {1}, thread stopped", GetID(), thread.GetID());
866
867 // Check that we're not already marked with a stop reason. Note this thread
868 // really shouldn't already be marked as stopped - if we were, that would
869 // imply that the kernel signaled us with the thread stopping which we
870 // handled and marked as stopped, and that, without an intervening resume,
871 // we received another stop. It is more likely that we are missing the
872 // marking of a run state somewhere if we find that the thread was marked
873 // as stopped.
874 const StateType thread_state = thread.GetState();
875 if (!StateIsStoppedState(thread_state, false)) {
876 // An inferior thread has stopped because of a SIGSTOP we have sent it.
877 // Generally, these are not important stops and we don't want to report
878 // them as they are just used to stop other threads when one thread (the
879 // one with the *real* stop reason) hits a breakpoint (watchpoint,
880 // etc...). However, in the case of an asynchronous Interrupt(), this
881 // *is* the real stop reason, so we leave the signal intact if this is
882 // the thread that was chosen as the triggering thread.
883 if (m_pending_notification_tid != LLDB_INVALID_THREAD_ID) {
884 if (m_pending_notification_tid == thread.GetID())
885 thread.SetStoppedBySignal(SIGSTOP, &info);
886 else
887 thread.SetStoppedWithNoReason();
888
889 SetCurrentThreadID(thread.GetID());
890 SignalIfAllThreadsStopped();
891 } else {
892 // We can end up here if stop was initiated by LLGS but by this time a
893 // thread stop has occurred - maybe initiated by another event.
894 Status error = ResumeThread(thread, thread.GetState(), 0);
895 if (error.Fail())
896 LLDB_LOG(log, "failed to resume thread {0}: {1}", thread.GetID(),
897 error);
898 }
899 } else {
900 LLDB_LOG(log,
901 "pid {0} tid {1}, thread was already marked as a stopped "
902 "state (state={2}), leaving stop signal as is",
903 GetID(), thread.GetID(), thread_state);
904 SignalIfAllThreadsStopped();
905 }
906
907 // Done handling.
908 return;
909 }
910
911 // Check if debugger should stop at this signal or just ignore it and resume
912 // the inferior.
913 if (m_signals_to_ignore.contains(signo)) {
914 ResumeThread(thread, thread.GetState(), signo);
915 return;
916 }
917
918 // This thread is stopped.
919 LLDB_LOG(log, "received signal {0}", Host::GetSignalAsCString(signo));
920 thread.SetStoppedBySignal(signo, &info);
921
922 // Send a stop to the debugger after we get all other threads to stop.
923 StopRunningThreads(thread.GetID());
924}

926bool NativeProcessLinux::MonitorClone(NativeThreadLinux &parent,
927 lldb::pid_t child_pid, int event) {
928 Log *log = GetLog(POSIXLog::Process);
929 LLDB_LOG(log, "parent_tid={0}, child_pid={1}, event={2}", parent.GetID(),
930 child_pid, event);
931
932 m_manager.CollectThread(child_pid);
933
934 switch (event) {
935 case PTRACE_EVENT_CLONE: {
936 // PTRACE_EVENT_CLONE can either mean a new thread or a new process.
937 // Try to grab the new process' PGID to figure out which one it is.
938 // If PGID is the same as the PID, then it's a new process. Otherwise,
939 // it's a thread.
940 auto tgid_ret = getPIDForTID(child_pid);
941 if (tgid_ret != child_pid) {
942 // A new thread should have PGID matching our process' PID.
943 assert(!tgid_ret || *tgid_ret == GetID());
944
945 NativeThreadLinux &child_thread = AddThread(child_pid, /*resume*/ true);
946 ThreadWasCreated(child_thread);
947
948 // Resume the parent.
949 ResumeThread(parent, parent.GetState(), LLDB_INVALID_SIGNAL_NUMBER);
950 break;
951 }
952 }
953 [[fallthrough]];
954 case PTRACE_EVENT_FORK:
955 case PTRACE_EVENT_VFORK: {
956 bool is_vfork = event == PTRACE_EVENT_VFORK;
957 std::unique_ptr<NativeProcessLinux> child_process{new NativeProcessLinux(
958 static_cast<::pid_t>(child_pid), m_terminal_fd, m_delegate, m_arch,
959 m_manager, {static_cast<::pid_t>(child_pid)})};
960 if (!is_vfork)
961 child_process->m_software_breakpoints = m_software_breakpoints;
962
963 Extension expected_ext = is_vfork ? Extension::vfork : Extension::fork;
964 if (bool(m_enabled_extensions & expected_ext)) {
965 m_delegate.NewSubprocess(this, std::move(child_process));
966 // NB: non-vfork clone() is reported as fork
967 parent.SetStoppedByFork(is_vfork, child_pid);
968 StopRunningThreads(parent.GetID());
969 } else {
970 child_process->Detach();
971 ResumeThread(parent, parent.GetState(), LLDB_INVALID_SIGNAL_NUMBER);
972 }
973 break;
974 }
975 default:
976 llvm_unreachable("unknown clone_info.event");
977 }
978
979 return true;
980}

989Status NativeProcessLinux::Resume(const ResumeActionList &resume_actions) {
990 Log *log = GetLog(POSIXLog::Process);
991 LLDB_LOG(log, "pid {0}", GetID());
992
993 NotifyTracersProcessWillResume();
994
995 bool software_single_step = !SupportHardwareSingleStepping();
996
997 if (software_single_step) {
998 for (const auto &thread : m_threads) {
999 assert(thread && "thread list should not contain NULL threads");
1000
1001 const ResumeAction *const action =
1002 resume_actions.GetActionForThread(thread->GetID(), true);
1003 if (action == nullptr)
1004 continue;
1005
1006 if (action->state == eStateStepping) {
1007 Status error = SetupSoftwareSingleStepping(
1008 static_cast<NativeThreadLinux &>(*thread));
1009 if (error.Fail())
1010 return error;
1011 }
1012 }
1013 }
1014
1015 for (const auto &thread : m_threads) {
1016 assert(thread && "thread list should not contain NULL threads");
1017
1018 const ResumeAction *const action =
1019 resume_actions.GetActionForThread(thread->GetID(), true);
1020
1021 if (action == nullptr) {
1022 LLDB_LOG(log, "no action specified for pid {0} tid {1}", GetID(),
1023 thread->GetID());
1024 continue;
1025 }
1026
1027 LLDB_LOG(log, "processing resume action state {0} for pid {1} tid {2}",
1028 action->state, GetID(), thread->GetID());
1029
1030 switch (action->state) {
1031 case eStateRunning:
1032 case eStateStepping: {
1033 // Run the thread, possibly feeding it the signal.
1034 const int signo = action->signal;
1035 Status error = ResumeThread(static_cast<NativeThreadLinux &>(*thread),
1036 action->state, signo);
1037 if (error.Fail())
1038 return Status::FromErrorStringWithFormat(
1039 "NativeProcessLinux::%s: failed to resume thread "
1040 "for pid %" PRIu64 ", tid %" PRIu64 ", error = %s",
1041 __FUNCTION__, GetID(), thread->GetID(), error.AsCString());
1042
1043 break;
1044 }
1045
1046 case eStateSuspended:
1047 case eStateStopped:
1048 break;
1049
1050 default:
1051 return Status::FromErrorStringWithFormat(
1052 "NativeProcessLinux::%s (): unexpected state %s specified "
1053 "for pid %" PRIu64 ", tid %" PRIu64,
1054 __FUNCTION__, StateAsCString(action->state), GetID(),
1055 thread->GetID());
1056 }
1057 }
1058
1059 return Status();
1060}

1107Status NativeProcessLinux::Interrupt() {
1108 // Pick a running thread (or if none, a not-dead stopped thread) as the
1109 // chosen thread that will be the stop-reason thread.
1110 Log *log = GetLog(POSIXLog::Process);
1111
1112 NativeThreadProtocol *running_thread = nullptr;
1113 NativeThreadProtocol *stopped_thread = nullptr;
1114
1115 LLDB_LOG(log, "selecting running thread for interrupt target");
1116 for (const auto &thread : m_threads) {
1117 // If we have a running or stepping thread, we'll call that the target of
1118 // the interrupt.
1119 const auto thread_state = thread->GetState();
1120 if (thread_state == eStateRunning || thread_state == eStateStepping) {
1121 running_thread = thread.get();
1122 break;
1123 } else if (!stopped_thread && StateIsStoppedState(thread_state, true)) {
1124 // Remember the first non-dead stopped thread. We'll use that as a
1125 // backup if there are no running threads.
1126 stopped_thread = thread.get();
1127 }
1128 }
1129
1130 if (!running_thread && !stopped_thread) {
1131 Status error("found no running/stepping or live stopped threads as target "
1132 "for interrupt");
1133 LLDB_LOG(log, "skipping due to error: {0}", error);
1134
1135 return error;
1136 }
1137
1138 NativeThreadProtocol *deferred_signal_thread =
1139 running_thread ? running_thread : stopped_thread;
1140
1141 LLDB_LOG(log, "pid {0} {1} tid {2} chosen for interrupt target", GetID(),
1142 running_thread ? "running" : "stopped",
1143 deferred_signal_thread->GetID());
1144
1145 StopRunningThreads(deferred_signal_thread->GetID());
1146
1147 return Status();
1148}

1150Status NativeProcessLinux::Kill() {
1151 Log *log = GetLog(POSIXLog::Process);
1152 LLDB_LOG(log, "pid {0}", GetID());
1153
1154 Status error;
1155
1156 switch (m_state) {
1157 case StateType::eStateInvalid:
1158 case StateType::eStateExited:
1159 case StateType::eStateCrashed:
1160 case StateType::eStateDetached:
1161 case StateType::eStateUnloaded:
1162 // Nothing to do - the process is already dead.
1163 LLDB_LOG(log, "ignored for PID {0} due to current state: {1}", GetID(),
1164 m_state);
1165 return error;
1166
1167 case StateType::eStateConnected:
1168 case StateType::eStateAttaching:
1169 case StateType::eStateLaunching:
1170 case StateType::eStateStopped:
1171 case StateType::eStateRunning:
1172 case StateType::eStateStepping:
1173 case StateType::eStateSuspended:
1174 // We can try to kill a process in these states.
1175 break;
1176 }
1177
1178 if (kill(GetID(), SIGKILL) != 0) {
1179 error = Status::FromErrno();
1180 return error;
1181 }
1182
1183 return error;
1184}

1186Status NativeProcessLinux::GetMemoryRegionInfo(lldb::addr_t load_addr,
1187 MemoryRegionInfo &range_info) {
1188 // FIXME review that the final memory region returned extends to the end of
1189 // the virtual address space,
1190 // with no perms if it is not mapped.
1191
1192 // Use an approach that reads memory regions from /proc/{pid}/maps. Assume
1193 // proc maps entries are in ascending order.
1194 // FIXME assert if we find differently.
1195
1196 if (m_supports_mem_region == LazyBool::eLazyBoolNo) {
1197 // We're done.
1198 return Status::FromErrorString("unsupported");
1199 }
1200
1201 Status error = PopulateMemoryRegionCache();
1202 if (error.Fail()) {
1203 return error;
1204 }
1205
1206 lldb::addr_t prev_base_address = 0;
1207
1208 // FIXME start by finding the last region that is <= target address using
1209 // binary search. Data is sorted.
1210 // There can be a ton of regions on pthreads apps with lots of threads.
1211 for (auto it = m_mem_region_cache.begin(); it != m_mem_region_cache.end();
1212 ++it) {
1213 MemoryRegionInfo &proc_entry_info = it->first;
1214
1215 // Sanity check assumption that /proc/{pid}/maps entries are ascending.
1216 assert((proc_entry_info.GetRange().GetRangeBase() >= prev_base_address) &&
1217 "descending /proc/pid/maps entries detected, unexpected");
1218 prev_base_address = proc_entry_info.GetRange().GetRangeBase();
1219 UNUSED_IF_ASSERT_DISABLED(prev_base_address);
1220
1221 // If the target address comes before this entry, indicate distance to next
1222 // region.
1223 if (load_addr < proc_entry_info.GetRange().GetRangeBase()) {
1224 range_info.GetRange().SetRangeBase(load_addr);
1225 range_info.GetRange().SetByteSize(
1226 proc_entry_info.GetRange().GetRangeBase() - load_addr);
1227 range_info.SetReadable(MemoryRegionInfo::OptionalBool::eNo);
1228 range_info.SetWritable(MemoryRegionInfo::OptionalBool::eNo);
1229 range_info.SetExecutable(MemoryRegionInfo::OptionalBool::eNo);
1230 range_info.SetMapped(MemoryRegionInfo::OptionalBool::eNo);
1231
1232 return error;
1233 } else if (proc_entry_info.GetRange().Contains(load_addr)) {
1234 // The target address is within the memory region we're processing here.
1235 range_info = proc_entry_info;
1236 return error;
1237 }
1238
1239 // The target memory address comes somewhere after the region we just
1240 // parsed.
1241 }
1242
1243 // If we made it here, we didn't find an entry that contained the given
1244 // address. Return the load_addr as start and the amount of bytes betwwen
1245 // load address and the end of the memory as size.
1246 range_info.GetRange().SetRangeBase(load_addr);
1247 range_info.GetRange().SetRangeEnd(LLDB_INVALID_ADDRESS);
1248 range_info.SetReadable(MemoryRegionInfo::OptionalBool::eNo);
1249 range_info.SetWritable(MemoryRegionInfo::OptionalBool::eNo);
1250 range_info.SetExecutable(MemoryRegionInfo::OptionalBool::eNo);
1251 range_info.SetMapped(MemoryRegionInfo::OptionalBool::eNo);
1252 return error;
1253}

1255Status NativeProcessLinux::PopulateMemoryRegionCache() {
1256 Log *log = GetLog(POSIXLog::Process);
1257
1258 // If our cache is empty, pull the latest. There should always be at least
1259 // one memory region if memory region handling is supported.
1260 if (!m_mem_region_cache.empty()) {
1261 LLDB_LOG(log, "reusing {0} cached memory region entries",
1262 m_mem_region_cache.size());
1263 return Status();
1264 }
1265
1266 Status Result;
1267 LinuxMapCallback callback = [&](llvm::Expected<MemoryRegionInfo> Info) {
1268 if (Info) {
1269 FileSpec file_spec(Info->GetName().GetCString());
1270 FileSystem::Instance().Resolve(file_spec);
1271 m_mem_region_cache.emplace_back(*Info, file_spec);
1272 return true;
1273 }
1274
1275 Result = Status::FromError(Info.takeError());
1276 m_supports_mem_region = LazyBool::eLazyBoolNo;
1277 LLDB_LOG(log, "failed to parse proc maps: {0}", Result);
1278 return false;
1279 };
1280
1281 // Linux kernel since 2.6.14 has /proc/{pid}/smaps
1282 // if CONFIG_PROC_PAGE_MONITOR is enabled
1283 auto BufferOrError = getProcFile(GetID(), GetCurrentThreadID(), "smaps");
1284 if (BufferOrError)
1285 ParseLinuxSMapRegions(BufferOrError.get()->getBuffer(), callback);
1286 else {
1287 BufferOrError = getProcFile(GetID(), GetCurrentThreadID(), "maps");
1288 if (!BufferOrError) {
1289 m_supports_mem_region = LazyBool::eLazyBoolNo;
1290 return BufferOrError.getError();
1291 }
1292
1293 ParseLinuxMapRegions(BufferOrError.get()->getBuffer(), callback);
1294 }
1295
1296 if (Result.Fail())
1297 return Result;
1298
1299 if (m_mem_region_cache.empty()) {
1300 // No entries after attempting to read them. This shouldn't happen if
1301 // /proc/{pid}/maps is supported. Assume we don't support map entries via
1302 // procfs.
1303 m_supports_mem_region = LazyBool::eLazyBoolNo;
1304 LLDB_LOG(log,
1305 "failed to find any procfs maps entries, assuming no support "
1306 "for memory region metadata retrieval");
1307 return Status::FromErrorString("not supported");
1308 }
1309
1310 LLDB_LOG(log, "read {0} memory region entries from /proc/{1}/maps",
1311 m_mem_region_cache.size(), GetID());
1312
1313 // We support memory retrieval, remember that.
1314 m_supports_mem_region = LazyBool::eLazyBoolYes;
1315 return Status();
1316}

1318void NativeProcessLinux::DoStopIDBumped(uint32_t newBumpId) {
1319 Log *log = GetLog(POSIXLog::Process);
1320 LLDB_LOG(log, "newBumpId={0}", newBumpId);
1321 LLDB_LOG(log, "clearing {0} entries from memory region cache",
1322 m_mem_region_cache.size());
1323 m_mem_region_cache.clear();
1324}

1326llvm::Expected<uint64_t>
1327NativeProcessLinux::Syscall(llvm::ArrayRef<uint64_t> args) {
1328 PopulateMemoryRegionCache();
1329 auto region_it = llvm::find_if(m_mem_region_cache, [](const auto &pair) {
1330 return pair.first.GetExecutable() == MemoryRegionInfo::eYes &&
1331 pair.first.GetShared() != MemoryRegionInfo::eYes;
1332 });
1333 if (region_it == m_mem_region_cache.end())
1334 return llvm::createStringError(llvm::inconvertibleErrorCode(),
1335 "No executable memory region found!");
1336
1337 addr_t exe_addr = region_it->first.GetRange().GetRangeBase();
1338
1339 NativeThreadLinux &thread = *GetCurrentThread();
1340 assert(thread.GetState() == eStateStopped);
1341 NativeRegisterContextLinux &reg_ctx = thread.GetRegisterContext();
1342
1343 NativeRegisterContextLinux::SyscallData syscall_data =
1344 *reg_ctx.GetSyscallData();
1345
1346 WritableDataBufferSP registers_sp;
1347 if (llvm::Error Err = reg_ctx.ReadAllRegisterValues(registers_sp).ToError())
1348 return std::move(Err);
1349 llvm::scope_exit restore_regs(
1350 [&] { reg_ctx.WriteAllRegisterValues(registers_sp); });
1351
1352 llvm::SmallVector<uint8_t, 8> memory(syscall_data.Insn.size());
1353 size_t bytes_read;
1354 if (llvm::Error Err =
1355 ReadMemory(exe_addr, memory.data(), memory.size(), bytes_read)
1356 .ToError()) {
1357 return std::move(Err);
1358 }
1359
1360 llvm::scope_exit restore_mem(
1361 [&] { WriteMemory(exe_addr, memory.data(), memory.size(), bytes_read); });
1362
1363 if (llvm::Error Err = reg_ctx.SetPC(exe_addr).ToError())
1364 return std::move(Err);
1365
1366 for (const auto &zip : llvm::zip_first(args, syscall_data.Args)) {
1367 if (llvm::Error Err =
1368 reg_ctx
1369 .WriteRegisterFromUnsigned(std::get<1>(zip), std::get<0>(zip))
1370 .ToError()) {
1371 return std::move(Err);
1372 }
1373 }
1374 if (llvm::Error Err = WriteMemory(exe_addr, syscall_data.Insn.data(),
1375 syscall_data.Insn.size(), bytes_read)
1376 .ToError())
1377 return std::move(Err);
1378
1379 m_mem_region_cache.clear();
1380
1381 // With software single stepping the syscall insn buffer must also include a
1382 // trap instruction to stop the process.
1383 int req = SupportHardwareSingleStepping() ? PTRACE_SINGLESTEP : PTRACE_CONT;
1384 if (llvm::Error Err =
1385 PtraceWrapper(req, thread.GetID(), nullptr, nullptr).ToError())
1386 return std::move(Err);
1387
1388 int status;
1389 ::pid_t wait_pid = llvm::sys::RetryAfterSignal(-1, ::waitpid, thread.GetID(),
1390 &status, __WALL);
1391 if (wait_pid == -1) {
1392 return llvm::errorCodeToError(
1393 std::error_code(errno, std::generic_category()));
1394 }
1395 assert((unsigned)wait_pid == thread.GetID());
1396
1397 uint64_t result = reg_ctx.ReadRegisterAsUnsigned(syscall_data.Result, -ESRCH);
1398
1399 // Values larger than this are actually negative errno numbers.
1400 uint64_t errno_threshold =
1401 (uint64_t(-1) >> (64 - 8 * m_arch.GetAddressByteSize())) - 0x1000;
1402 if (result > errno_threshold) {
1403 return llvm::errorCodeToError(
1404 std::error_code(-result & 0xfff, std::generic_category()));
1405 }
1406
1407 return result;
1408}

1410llvm::Expected<addr_t>
1411NativeProcessLinux::AllocateMemory(size_t size, uint32_t permissions) {
1412
1413 std::optional<NativeRegisterContextLinux::MmapData> mmap_data =
1414 GetCurrentThread()->GetRegisterContext().GetMmapData();
1415 if (!mmap_data)
1416 return llvm::make_error<UnimplementedError>();

1418 unsigned prot = PROT_NONE;
1419 assert((permissions & (ePermissionsReadable | ePermissionsWritable |
1420 ePermissionsExecutable)) == permissions &&
1421 "Unknown permission!");
1422 if (permissions & ePermissionsReadable)
1423 prot |= PROT_READ;
1424 if (permissions & ePermissionsWritable)
1425 prot |= PROT_WRITE;
1426 if (permissions & ePermissionsExecutable)
1427 prot |= PROT_EXEC;
1428
1429 llvm::Expected<uint64_t> Result =
1430 Syscall({mmap_data->SysMmap, 0, size, prot, MAP_ANONYMOUS | MAP_PRIVATE,
1431 uint64_t(-1), 0});
1432 if (Result)
1433 m_allocated_memory.try_emplace(*Result, size);
1434 return Result;
1435}

1457Status NativeProcessLinux::ReadMemoryTags(int32_t type, lldb::addr_t addr,
1458 size_t len,
1459 std::vector<uint8_t> &tags) {
1460 llvm::Expected<NativeRegisterContextLinux::MemoryTaggingDetails> details =
1461 GetCurrentThread()->GetRegisterContext().GetMemoryTaggingDetails(type);
1462 if (!details)
1463 return Status::FromError(details.takeError());
1464
1465 // Ignore 0 length read
1466 if (!len)
1467 return Status();
1468
1469 // lldb will align the range it requests but it is not required to by
1470 // the protocol so we'll do it again just in case.
1471 // Remove tag bits too. Ptrace calls may work regardless but that
1472 // is not a guarantee.
1473 MemoryTagManager::TagRange range(details->manager->RemoveTagBits(addr), len);
1474 range = details->manager->ExpandToGranule(range);
1475
1476 // Allocate enough space for all tags to be read
1477 size_t num_tags = range.GetByteSize() / details->manager->GetGranuleSize();
1478 tags.resize(num_tags * details->manager->GetTagSizeInBytes());
1479
1480 struct iovec tags_iovec;
1481 uint8_t *dest = tags.data();
1482 lldb::addr_t read_addr = range.GetRangeBase();
1483
1484 // This call can return partial data so loop until we error or
1485 // get all tags back.
1486 while (num_tags) {
1487 tags_iovec.iov_base = dest;
1488 tags_iovec.iov_len = num_tags;
1489
1490 Status error = NativeProcessLinux::PtraceWrapper(
1491 details->ptrace_read_req, GetCurrentThreadID(),
1492 reinterpret_cast<void *>(read_addr), static_cast<void *>(&tags_iovec),
1493 0, nullptr);
1494
1495 if (error.Fail()) {
1496 // Discard partial reads
1497 tags.resize(0);
1498 return error;
1499 }
1500
1501 size_t tags_read = tags_iovec.iov_len;
1502 assert(tags_read && (tags_read <= num_tags));
1503
1504 dest += tags_read * details->manager->GetTagSizeInBytes();
1505 read_addr += details->manager->GetGranuleSize() * tags_read;
1506 num_tags -= tags_read;
1507 }
1508
1509 return Status();
1510}
1511
1512Status NativeProcessLinux::WriteMemoryTags(int32_t type, lldb::addr_t addr,
1513 size_t len,
1514 const std::vector<uint8_t> &tags) {
1515 llvm::Expected<NativeRegisterContextLinux::MemoryTaggingDetails> details =
1516 GetCurrentThread()->GetRegisterContext().GetMemoryTaggingDetails(type);
1517 if (!details)
1518 return Status::FromError(details.takeError());
1519
1520 // Ignore 0 length write
1521 if (!len)
1522 return Status();
1523
1524 // lldb will align the range it requests but it is not required to by
1525 // the protocol so we'll do it again just in case.
1526 // Remove tag bits too. Ptrace calls may work regardless but that
1527 // is not a guarantee.
1528 MemoryTagManager::TagRange range(details->manager->RemoveTagBits(addr), len);
1529 range = details->manager->ExpandToGranule(range);
1530
1531 // Not checking number of tags here, we may repeat them below
1532 llvm::Expected<std::vector<lldb::addr_t>> unpacked_tags_or_err =
1533 details->manager->UnpackTagsData(tags);
1534 if (!unpacked_tags_or_err)
1535 return Status::FromError(unpacked_tags_or_err.takeError());
1536
1537 llvm::Expected<std::vector<lldb::addr_t>> repeated_tags_or_err =
1538 details->manager->RepeatTagsForRange(*unpacked_tags_or_err, range);
1539 if (!repeated_tags_or_err)
1540 return Status::FromError(repeated_tags_or_err.takeError());
1541
1542 // Repack them for ptrace to use
1543 llvm::Expected<std::vector<uint8_t>> final_tag_data =
1544 details->manager->PackTags(*repeated_tags_or_err);
1545 if (!final_tag_data)
1546 return Status::FromError(final_tag_data.takeError());
1547
1548 struct iovec tags_vec;
1549 uint8_t *src = final_tag_data->data();
1550 lldb::addr_t write_addr = range.GetRangeBase();
1551 // unpacked tags size because the number of bytes per tag might not be 1
1552 size_t num_tags = repeated_tags_or_err->size();
1553
1554 // This call can partially write tags, so we loop until we
1555 // error or all tags have been written.
1556 while (num_tags > 0) {
1557 tags_vec.iov_base = src;
1558 tags_vec.iov_len = num_tags;
1559
1560 Status error = NativeProcessLinux::PtraceWrapper(
1561 details->ptrace_write_req, GetCurrentThreadID(),
1562 reinterpret_cast<void *>(write_addr), static_cast<void *>(&tags_vec), 0,
1563 nullptr);
1564
1565 if (error.Fail()) {
1566 // Don't attempt to restore the original values in the case of a partial
1567 // write
1568 return error;
1569 }
1570
1571 size_t tags_written = tags_vec.iov_len;
1572 assert(tags_written && (tags_written <= num_tags));
1573
1574 src += tags_written * details->manager->GetTagSizeInBytes();
1575 write_addr += details->manager->GetGranuleSize() * tags_written;
1576 num_tags -= tags_written;
1577 }
1578
1579 return Status();
1580}

1627Status NativeProcessLinux::ReadMemory(lldb::addr_t addr, void *buf, size_t size,
1628 size_t &bytes_read) {
1629 Log *log = GetLog(POSIXLog::Memory);
1630 LLDB_LOG(log, "addr = {0}, buf = {1}, size = {2}", addr, buf, size);
1631
1632 bytes_read = 0;
1633 if (ProcessVmReadvSupported()) {
1634 // The process_vm_readv path is about 50 times faster than ptrace api. We
1635 // want to use this syscall if it is supported.
1636
1637 struct iovec local_iov, remote_iov;
1638 local_iov.iov_base = buf;
1639 local_iov.iov_len = size;
1640 remote_iov.iov_base = reinterpret_cast<void *>(addr);
1641 remote_iov.iov_len = size;
1642
1643 ssize_t read_result = process_vm_readv(GetCurrentThreadID(), &local_iov, 1,
1644 &remote_iov, 1, 0);
1645 int error = 0;
1646 if (read_result < 0)
1647 error = errno;
1648 else
1649 bytes_read = read_result;
1650
1651 LLDB_LOG(log,
1652 "process_vm_readv({0}, [iovec({1}, {2})], [iovec({3:x}, {2})], 1, "
1653 "0) => {4} ({5})",
1654 GetCurrentThreadID(), buf, size, addr, read_result,
1655 error > 0 ? llvm::sys::StrError(errno) : "sucesss");
1656 }
1657
1658 unsigned char *dst = static_cast<unsigned char *>(buf);
1659 size_t remainder;
1660 long data;
1661
1662 for (; bytes_read < size; bytes_read += remainder) {
1663 Status error = NativeProcessLinux::PtraceWrapper(
1664 PTRACE_PEEKDATA, GetCurrentThreadID(),
1665 reinterpret_cast<void *>(addr + bytes_read), nullptr, 0, &data);
1666 if (error.Fail())
1667 return error;
1668
1669 remainder = size - bytes_read;
1670 remainder = remainder > k_ptrace_word_size ? k_ptrace_word_size : remainder;
1671
1672 // Copy the data into our buffer
1673 memcpy(dst + bytes_read, &data, remainder);
1674 }
1675 return Status();
1676}

1678Status NativeProcessLinux::WriteMemory(lldb::addr_t addr, const void *buf,
1679 size_t size, size_t &bytes_written) {
1680 const unsigned char *src = static_cast<const unsigned char *>(buf);
1681 size_t remainder;
1682 Status error;
1683
1684 Log *log = GetLog(POSIXLog::Memory);
1685 LLDB_LOG(log, "addr = {0}, buf = {1}, size = {2}", addr, buf, size);
1686
1687 for (bytes_written = 0; bytes_written < size; bytes_written += remainder) {
1688 remainder = size - bytes_written;
1689 remainder = remainder > k_ptrace_word_size ? k_ptrace_word_size : remainder;
1690
1691 if (remainder == k_ptrace_word_size) {
1692 unsigned long data = 0;
1693 memcpy(&data, src, k_ptrace_word_size);
1694
1695 LLDB_LOG(log, "[{0:x}]:{1:x}", addr, data);
1696 error = NativeProcessLinux::PtraceWrapper(
1697 PTRACE_POKEDATA, GetCurrentThreadID(), (void *)addr, (void *)data);
1698 if (error.Fail())
1699 return error;
1700 } else {
1701 unsigned char buff[8];
1702 size_t bytes_read;
1703 error = ReadMemory(addr, buff, k_ptrace_word_size, bytes_read);
1704 if (error.Fail())
1705 return error;
1706
1707 memcpy(buff, src, remainder);
1708
1709 size_t bytes_written_rec;
1710 error = WriteMemory(addr, buff, k_ptrace_word_size, bytes_written_rec);
1711 if (error.Fail())
1712 return error;
1713
1714 LLDB_LOG(log, "[{0:x}]:{1:x} ({2:x})", addr, *(const unsigned long *)src,
1715 *(unsigned long *)buff);
1716 }
1717
1718 addr += k_ptrace_word_size;
1719 src += k_ptrace_word_size;
1720 }
1721 return error;
1722}

1871Status NativeProcessLinux::ResumeThread(NativeThreadLinux &thread,
1872 lldb::StateType state, int signo) {
1873 Log *const log = GetLog(POSIXLog::Thread);
1874 LLDB_LOG(log, "tid: {0}", thread.GetID());
1875
1876 // Before we do the resume below, first check if we have a pending stop
1877 // notification that is currently waiting for all threads to stop. This is
1878 // potentially a buggy situation since we're ostensibly waiting for threads
1879 // to stop before we send out the pending notification, and here we are
1880 // resuming one before we send out the pending stop notification.
1881 if (m_pending_notification_tid != LLDB_INVALID_THREAD_ID) {
1882 LLDB_LOG(log,
1883 "about to resume tid {0} per explicit request but we have a "
1884 "pending stop notification (tid {1}) that is actively "
1885 "waiting for this thread to stop. Valid sequence of events?",
1886 thread.GetID(), m_pending_notification_tid);
1887 }
1888
1889 // Request a resume. We expect this to be synchronous and the system to
1890 // reflect it is running after this completes.
1891 switch (state) {
1892 case eStateRunning: {
1893 Status resume_result = thread.Resume(signo);
1894 if (resume_result.Success())
1895 SetState(eStateRunning, true);
1896 return resume_result;
1897 }
1898 case eStateStepping: {
1899 Status step_result = thread.SingleStep(signo);
1900 if (step_result.Success())
1901 SetState(eStateRunning, true);
1902 return step_result;
1903 }
1904 default:
1905 LLDB_LOG(log, "Unhandled state {0}.", state);
1906 llvm_unreachable("Unhandled state for resume");
1907 }
1908}

1912void NativeProcessLinux::StopRunningThreads(const lldb::tid_t triggering_tid) {
1913 Log *const log = GetLog(POSIXLog::Thread);
1914 LLDB_LOG(log, "about to process event: (triggering_tid: {0})",
1915 triggering_tid);
1916
1917 m_pending_notification_tid = triggering_tid;
1918
1919 // Request a stop for all the thread stops that need to be stopped and are
1920 // not already known to be stopped.
1921 for (const auto &thread : m_threads) {
1922 if (StateIsRunningState(thread->GetState()))
1923 static_cast<NativeThreadLinux *>(thread.get())->RequestStop();
1924 }
1925
1926 SignalIfAllThreadsStopped();
1927 LLDB_LOG(log, "event processing done");
1928}
1929
1930void NativeProcessLinux::SignalIfAllThreadsStopped() {
1931 if (m_pending_notification_tid == LLDB_INVALID_THREAD_ID)
1932 return; // No pending notification. Nothing to do.
1933
1934 for (const auto &thread_sp : m_threads) {
1935 if (StateIsRunningState(thread_sp->GetState()))
1936 return; // Some threads are still running. Don't signal yet.
1937 }
1938

1940 Log *log = GetLog(LLDBLog::Process | LLDBLog::Breakpoints);
1941
1942 // Clear any temporary breakpoints we used to implement software single
1943 // stepping.
1944 for (const auto &thread_info : m_threads_stepping_with_breakpoint) {
1945 for (auto &&bp_addr : thread_info.second) {
1946 Status error = RemoveBreakpoint(bp_addr);
1947 if (error.Fail())
1948 LLDB_LOG(log, "pid = {0} remove stepping breakpoint: {1}",
1949 thread_info.first, error);
1950 }
1951 }
1952 m_threads_stepping_with_breakpoint.clear();
1953
1954 // Notify the delegate about the stop
1955 SetCurrentThreadID(m_pending_notification_tid);
1956 SetState(StateType::eStateStopped, true);
1957 m_pending_notification_tid = LLDB_INVALID_THREAD_ID;
1958}

1973// errno on error because -1 can be a valid result (i.e. for PTRACE_PEEK*)
1974Status NativeProcessLinux::PtraceWrapper(int req, lldb::pid_t pid, void *addr,
1975 void *data, size_t data_size,
1976 long *result) {
1977 Status error;
1978 long int ret;
1979
1980 Log *log = GetLog(POSIXLog::Ptrace);
1981
1982 PtraceDisplayBytes(req, data, data_size);
1983
1984 errno = 0;
1985 if (req == PTRACE_GETREGSET || req == PTRACE_SETREGSET)
1986 ret = ptrace(static_cast<__ptrace_request>(req), static_cast<::pid_t>(pid),
1987 *(unsigned int *)addr, data);
1988 else
1989 ret = ptrace(static_cast<__ptrace_request>(req), static_cast<::pid_t>(pid),
1990 addr, data);
1991
1992 if (ret == -1)
1993 error = Status::FromErrno();
1994
1995 if (result)
1996 *result = ret;
1997
1998 LLDB_LOG(log, "ptrace({0}, {1}, {2}, {3}, {4})={5:x}", req, pid, addr, data,
1999 data_size, ret);
2000
2001 PtraceDisplayBytes(req, data, data_size);
2002
2003 if (error.Fail())
2004 LLDB_LOG(log, "ptrace() failed: {0}", error);
2005
2006 return error;
