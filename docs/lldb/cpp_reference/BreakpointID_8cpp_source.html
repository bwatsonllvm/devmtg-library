66std::optional<BreakpointID>
67BreakpointID::ParseCanonicalReference(llvm::StringRef input) {
68 break_id_t bp_id;
69 break_id_t loc_id = LLDB_INVALID_BREAK_ID;
70
71 if (input.empty())
72 return std::nullopt;
73
74 // If it doesn't start with an integer, it's not valid.
75 if (input.consumeInteger(0, bp_id))
76 return std::nullopt;
77
78 // period is optional, but if it exists, it must be followed by a number.
79 if (input.consume_front(".")) {
80 if (input.consumeInteger(0, loc_id))
81 return std::nullopt;
82 }
83
84 // And at the end, the entire string must have been consumed.
85 if (!input.empty())
86 return std::nullopt;
87
88 return BreakpointID(bp_id, loc_id);
89}

91bool BreakpointID::StringIsBreakpointName(llvm::StringRef str, Status &error) {
92 error.Clear();
93 if (str.empty())
94 {
95 error = Status::FromErrorString("Empty breakpoint names are not allowed");
96 return false;
97 }
98
99 // First character must be a letter or _
100 if (!isalpha(str[0]) && str[0] != '_')
101 {
102 error =
103 Status::FromErrorStringWithFormatv("Breakpoint names must start with a "
104 "character or underscore: {0}",
105 str);
106 return false;
107 }
108
109 // Cannot contain ., -, or space.
110 if (str.find_first_of(".- ") != llvm::StringRef::npos) {
111 error =
112 Status::FromErrorStringWithFormatv("Breakpoint names cannot contain "
113 "'.' or '-' or spaces: \"{0}\"",
114 str);
115 return false;
116 }
117
118 return true;
119}
