39lldb::ProcessSP ProcessFreeBSDKernelCore::CreateInstance(
40 lldb::TargetSP target_sp, ListenerSP listener_sp,
41 const FileSpec *crash_file, bool can_connect) {
42 ModuleSP executable = target_sp->GetExecutableModule();
43 if (crash_file && !can_connect && executable) {
44 kvm_t *kvm =
45 kvm_open2(executable->GetFileSpec().GetPath().c_str(),
46 crash_file->GetPath().c_str(), O_RDONLY, nullptr, nullptr);
47 if (kvm)
48 return std::make_shared<ProcessFreeBSDKernelCore>(target_sp, listener_sp,
49 kvm, *crash_file);
50 }
51 return nullptr;
52}

93bool ProcessFreeBSDKernelCore::DoUpdateThreadList(ThreadList &old_thread_list,
94 ThreadList &new_thread_list) {
95 if (old_thread_list.GetSize(false) == 0) {
96 // Make up the thread the first time this is called so we can set our one
97 // and only core thread state up.
98
99 // We cannot construct a thread without a register context as that crashes
100 // LLDB but we can construct a process without threads to provide minimal
101 // memory reading support.
102 switch (GetTarget().GetArchitecture().GetMachine()) {
103 case llvm::Triple::aarch64:
104 case llvm::Triple::x86:
105 case llvm::Triple::x86_64:
106 break;
107 default:
108 return false;
109 }
110
111 Status error;
112
113 // struct field offsets are written as symbols so that we don't have
114 // to figure them out ourselves
115 int32_t offset_p_list = ReadSignedIntegerFromMemory(
116 FindSymbol("proc_off_p_list"), 4, -1, error);
117 int32_t offset_p_pid =
118 ReadSignedIntegerFromMemory(FindSymbol("proc_off_p_pid"), 4, -1, error);
119 int32_t offset_p_threads = ReadSignedIntegerFromMemory(
120 FindSymbol("proc_off_p_threads"), 4, -1, error);
121 int32_t offset_p_comm = ReadSignedIntegerFromMemory(
122 FindSymbol("proc_off_p_comm"), 4, -1, error);
123
124 int32_t offset_td_tid = ReadSignedIntegerFromMemory(
125 FindSymbol("thread_off_td_tid"), 4, -1, error);
126 int32_t offset_td_plist = ReadSignedIntegerFromMemory(
127 FindSymbol("thread_off_td_plist"), 4, -1, error);
128 int32_t offset_td_pcb = ReadSignedIntegerFromMemory(
129 FindSymbol("thread_off_td_pcb"), 4, -1, error);
130 int32_t offset_td_oncpu = ReadSignedIntegerFromMemory(
131 FindSymbol("thread_off_td_oncpu"), 4, -1, error);
132 int32_t offset_td_name = ReadSignedIntegerFromMemory(
133 FindSymbol("thread_off_td_name"), 4, -1, error);
134
135 // Fail if we were not able to read any of the offsets.
136 if (offset_p_list == -1 || offset_p_pid == -1 || offset_p_threads == -1 ||
137 offset_p_comm == -1 || offset_td_tid == -1 || offset_td_plist == -1 ||
138 offset_td_pcb == -1 || offset_td_oncpu == -1 || offset_td_name == -1)
139 return false;
140
141 // dumptid contains the thread-id of the crashing thread
142 // dumppcb contains its PCB
143 int32_t dumptid =
144 ReadSignedIntegerFromMemory(FindSymbol("dumptid"), 4, -1, error);
145 lldb::addr_t dumppcb = FindSymbol("dumppcb");
146
147 // stoppcbs is an array of PCBs on all CPUs.
148 // Each element is of size pcb_size.
149 int32_t pcbsize =
150 ReadSignedIntegerFromMemory(FindSymbol("pcb_size"), 4, -1, error);
151 lldb::addr_t stoppcbs = FindSymbol("stoppcbs");
152
153 // from FreeBSD sys/param.h
154 constexpr size_t fbsd_maxcomlen = 19;
155
156 // Iterate through a linked list of all processes. New processes are added
157 // to the head of this list. Which means that earlier PIDs are actually at
158 // the end of the list, so we have to walk it backwards. First collect all
159 // the processes in the list order.
160 std::vector<lldb::addr_t> process_addrs;
161 for (lldb::addr_t proc =
162 ReadPointerFromMemory(FindSymbol("allproc"), error);
163 proc != 0 && proc != LLDB_INVALID_ADDRESS;
164 proc = ReadPointerFromMemory(proc + offset_p_list, error)) {
165 process_addrs.push_back(proc);
166 }
167
168 // Processes are in the linked list in descending PID order, so we must walk
169 // them in reverse to get ascending PID order.
170 for (auto proc_it = process_addrs.rbegin(); proc_it != process_addrs.rend();
171 ++proc_it) {
172 lldb::addr_t proc = *proc_it;
173 int32_t pid =
174 ReadSignedIntegerFromMemory(proc + offset_p_pid, 4, -1, error);
175 // process' command-line string
176 char comm[fbsd_maxcomlen + 1];
177 ReadCStringFromMemory(proc + offset_p_comm, comm, sizeof(comm), error);
178
179 // Iterate through a linked list of all process' threads
180 // the initial thread is found in process' p_threads, subsequent
181 // elements are linked via td_plist field
182 for (lldb::addr_t td =
183 ReadPointerFromMemory(proc + offset_p_threads, error);
184 td != 0; td = ReadPointerFromMemory(td + offset_td_plist, error)) {
185 int32_t tid =
186 ReadSignedIntegerFromMemory(td + offset_td_tid, 4, -1, error);
187 lldb::addr_t pcb_addr =
188 ReadPointerFromMemory(td + offset_td_pcb, error);
189 // whether process was on CPU (-1 if not, otherwise CPU number)
190 int32_t oncpu =
191 ReadSignedIntegerFromMemory(td + offset_td_oncpu, 4, -2, error);
192 // thread name
193 char thread_name[fbsd_maxcomlen + 1];
194 ReadCStringFromMemory(td + offset_td_name, thread_name,
195 sizeof(thread_name), error);
196
197 // If we failed to read TID, ignore this thread.
198 if (tid == -1)
199 continue;
200
201 std::string thread_desc = llvm::formatv("(pid {0}) {1}", pid, comm);
202 if (*thread_name && strcmp(thread_name, comm)) {
203 thread_desc += '/';
204 thread_desc += thread_name;
205 }
206
207 // Roughly:
208 // 1. if the thread crashed, its PCB is going to be at "dumppcb"
209 // 2. if the thread was on CPU, its PCB is going to be on the CPU
210 // 3. otherwise, its PCB is in the thread struct
211 if (tid == dumptid) {
212 // NB: dumppcb can be LLDB_INVALID_ADDRESS if reading it failed
213 pcb_addr = dumppcb;
214 thread_desc += " (crashed)";
215 } else if (oncpu != -1) {
216 // If we managed to read stoppcbs and pcb_size, use them to find
217 // the correct PCB.
218 if (stoppcbs != LLDB_INVALID_ADDRESS && pcbsize > 0)
219 pcb_addr = stoppcbs + oncpu * pcbsize;
220 else
221 pcb_addr = LLDB_INVALID_ADDRESS;
222 thread_desc += llvm::formatv(" (on CPU {0})", oncpu);
223 }
224
225 auto thread =
226 new ThreadFreeBSDKernelCore(*this, tid, pcb_addr, thread_desc);
227
228 if (tid == dumptid)
229 thread->SetIsCrashedThread(true);
230
231 new_thread_list.AddThread(static_cast<ThreadSP>(thread));
232 }
233 }
234 } else {
235 const uint32_t num_threads = old_thread_list.GetSize(false);
236 for (uint32_t i = 0; i < num_threads; ++i)
237 new_thread_list.AddThread(old_thread_list.GetThreadAtIndex(i, false));
238 }
239 return new_thread_list.GetSize(false) > 0;
240}

242size_t ProcessFreeBSDKernelCore::DoReadMemory(lldb::addr_t addr, void *buf,
243 size_t size, Status &error) {
244 ssize_t rd = 0;
245 rd = kvm_read2(m_kvm, addr, buf, size);
246 if (rd < 0 || static_cast<size_t>(rd) != size) {
247 error = Status::FromErrorStringWithFormat("Reading memory failed: %s",
248 GetError());
249 return rd > 0 ? rd : 0;
250 }
251 return rd;
252}

255 ModuleSP mod_sp = GetTarget().GetExecutableModule();
256 const Symbol *sym = mod_sp->FindFirstSymbolWithNameAndType(ConstString(name));

258}

260void ProcessFreeBSDKernelCore::PrintUnreadMessage() {
261 Target &target = GetTarget();
262 Debugger &debugger = target.GetDebugger();
263
264 if (!debugger.GetCommandInterpreter().IsInteractive())
265 return;
266
267 Status error;
268
269 // Find msgbufp symbol (pointer to message buffer)
270 lldb::addr_t msgbufp_addr = FindSymbol("msgbufp");
271 if (msgbufp_addr == LLDB_INVALID_ADDRESS)
272 return;
273
274 // Read the pointer value
275 lldb::addr_t msgbufp = ReadPointerFromMemory(msgbufp_addr, error);
276 if (!error.Success() || msgbufp == LLDB_INVALID_ADDRESS)
277 return;
278
279 // Get the type information for struct msgbuf from DWARF
280 TypeQuery query("msgbuf");
281 TypeResults results;
282 target.GetImages().FindTypes(nullptr, query, results);
283
284 uint64_t offset_msg_ptr = 0;
285 uint64_t offset_msg_size = 0;
286 uint64_t offset_msg_wseq = 0;
287 uint64_t offset_msg_rseq = 0;
288
289 if (results.GetTypeMap().GetSize() > 0) {
290 // Found type info - use it to get field offsets
291 CompilerType msgbuf_type =
292 results.GetTypeMap().GetTypeAtIndex(0)->GetForwardCompilerType();
293
294 uint32_t num_fields = msgbuf_type.GetNumFields();
295 int field_found = 0;
296 for (uint32_t i = 0; i < num_fields; i++) {
297 std::string field_name;
298 uint64_t field_offset = 0;
299
300 msgbuf_type.GetFieldAtIndex(i, field_name, &field_offset, nullptr,
301 nullptr);
302
303 if (field_name == "msg_ptr") {
304 offset_msg_ptr = field_offset / 8; // Convert bits to bytes
305 field_found++;
306 } else if (field_name == "msg_size") {
307 offset_msg_size = field_offset / 8;
308 field_found++;
309 } else if (field_name == "msg_wseq") {
310 offset_msg_wseq = field_offset / 8;
311 field_found++;
312 } else if (field_name == "msg_rseq") {
313 offset_msg_rseq = field_offset / 8;
314 field_found++;
315 }
316 }
317
318 if (field_found != 4) {
319 LLDB_LOGF(
320 GetLog(LLDBLog::Object),
321 "FreeBSD-Kernel-Core: Could not find all required fields for msgbuf");
322 return;
323 }
324 } else {
325 // Fallback: use hardcoded offsets based on struct layout
326 // struct msgbuf layout (from sys/sys/msgbuf.h):
327 // char *msg_ptr; - offset 0
328 // u_int msg_magic; - offset ptr_size
329 // u_int msg_size; - offset ptr_size + 4
330 // u_int msg_wseq; - offset ptr_size + 8
331 // u_int msg_rseq; - offset ptr_size + 12
332 uint32_t ptr_size = GetAddressByteSize();
333 offset_msg_ptr = 0;
334 offset_msg_size = ptr_size + 4;
335 offset_msg_wseq = ptr_size + 8;
336 offset_msg_rseq = ptr_size + 12;
337 }
338
339 // Read struct msgbuf fields
340 lldb::addr_t bufp = ReadPointerFromMemory(msgbufp + offset_msg_ptr, error);
341 if (!error.Success() || bufp == LLDB_INVALID_ADDRESS)
342 return;
343
344 uint32_t size =
345 ReadUnsignedIntegerFromMemory(msgbufp + offset_msg_size, 4, 0, error);
346 if (!error.Success() || size == 0)
347 return;
348
349 uint32_t wseq =
350 ReadUnsignedIntegerFromMemory(msgbufp + offset_msg_wseq, 4, 0, error);
351 if (!error.Success())
352 return;
353
354 uint32_t rseq =
355 ReadUnsignedIntegerFromMemory(msgbufp + offset_msg_rseq, 4, 0, error);
356 if (!error.Success())
357 return;
358
359 // Convert sequences to positions
360 // MSGBUF_SEQ_TO_POS macro in FreeBSD: ((seq) % (size))
361 uint32_t rseq_pos = rseq % size;
362 uint32_t wseq_pos = wseq % size;
363
364 if (rseq_pos == wseq_pos)
365 return;
366
367 // Print crash info at once using stream
368 lldb::StreamSP stream_sp = debugger.GetAsyncOutputStream();
369 if (!stream_sp)
370 return;
371
372 stream_sp->PutCString("\nUnread portion of the kernel message buffer:\n");
373
374 // Read ring buffer in at most two chunks
375 if (rseq_pos < wseq_pos) {
376 // No wrap: read from rseq_pos to wseq_pos
377 size_t len = wseq_pos - rseq_pos;
378 std::string buf(len, '\0');
379 size_t bytes_read = ReadMemory(bufp + rseq_pos, &buf[0], len, error);
380 if (error.Success() && bytes_read > 0) {
381 buf.resize(bytes_read);
382 *stream_sp << buf;
383 }
384 } else {
385 // Wrap around: read from rseq_pos to end, then from start to wseq_pos
386 size_t len1 = size - rseq_pos;
387 std::string buf1(len1, '\0');
388 size_t bytes_read1 = ReadMemory(bufp + rseq_pos, &buf1[0], len1, error);
389 if (error.Success() && bytes_read1 > 0) {
390 buf1.resize(bytes_read1);
391 *stream_sp << buf1;
392 }
393
394 if (wseq_pos > 0) {
395 std::string buf2(wseq_pos, '\0');
396 size_t bytes_read2 = ReadMemory(bufp, &buf2[0], wseq_pos, error);
397 if (error.Success() && bytes_read2 > 0) {
398 buf2.resize(bytes_read2);
399 *stream_sp << buf2;
400 }
401 }
402 }
403
404 stream_sp->PutChar('\n');
405 stream_sp->Flush();
406}
