163// For context_flags. These values indicate the type of
164// context stored in the structure. The high 24 bits identify the CPU, the
165// low 8 bits identify the type of context saved.
166LLVM_ENABLE_BITMASK_ENUMS_IN_NAMESPACE();

168enum class MinidumpContext_x86_64_Flags : uint32_t {
169 x86_64_Flag = 0x00100000,
170 Control = x86_64_Flag | 0x00000001,
171 Integer = x86_64_Flag | 0x00000002,
172 Segments = x86_64_Flag | 0x00000004,
173 FloatingPoint = x86_64_Flag | 0x00000008,
174 DebugRegisters = x86_64_Flag | 0x00000010,
175 XState = x86_64_Flag | 0x00000040,
176 LLDBSpecific = x86_64_Flag | 0x80000000,
177
178 Full = Control | Integer | FloatingPoint,
179 All = Full | Segments | DebugRegisters | LLDBSpecific,
180
181 LLVM_MARK_AS_BITMASK_ENUM(/* LargestValue = */ All)
182};
