27/// Create a UTF-16 environment block to use with CreateProcessW.
28///
29/// The buffer is a sequence of null-terminated UTF-16 strings, followed by an
30/// extra L'\0' (two bytes of 0). An empty environment must have one
31/// empty string, followed by an extra L'\0'.
32///
33/// The keys are sorted to comply with the CreateProcess API calling convention.
34///
35/// Ensure that the resulting buffer is used in conjunction with
36/// CreateProcessW and be sure that dwCreationFlags includes
37/// CREATE_UNICODE_ENVIRONMENT.
38///
39/// \param env The Environment object to convert.
40/// \returns The sorted sequence of environment variables and their values,
41/// separated by null terminators. The vector is guaranteed to never be empty.
42static std::vector<wchar_t> CreateEnvironmentBufferW(const Environment &env) {
43 std::vector<std::wstring> env_entries;
44 for (const auto &KV : env) {
45 std::wstring wentry;
46 if (llvm::ConvertUTF8toWide(Environment::compose(KV), wentry))
47 env_entries.push_back(std::move(wentry));
48 }
49 std::sort(env_entries.begin(), env_entries.end(),
50 [](const std::wstring &a, const std::wstring &b) {
51 return _wcsicmp(a.c_str(), b.c_str()) < 0;
52 });
53
54 std::vector<wchar_t> buffer;
55 for (const auto &env_entry : env_entries) {
56 buffer.insert(buffer.end(), env_entry.begin(), env_entry.end());
57 buffer.push_back(L'\0');
58 }
59
60 if (buffer.empty())
61 buffer.push_back(L'\0'); // If there are no environment variables, we have
62 // to ensure there are 4 zero bytes in the buffer.
63 buffer.push_back(L'\0');
64
65 return buffer;
66}

68/// Flattens an Args object into a Windows command-line wide string.
69///
70/// Returns an empty string if args is empty.
71///
72/// \param args The Args object to flatten.
73/// \returns A wide string containing the flattened command line.
74static llvm::ErrorOr<std::wstring>
75GetFlattenedWindowsCommandStringW(Args args) {
76 if (args.empty())
77 return L"";
78
79 std::vector<llvm::StringRef> args_ref;
80 for (auto &entry : args.entries())
81 args_ref.push_back(entry.ref());
82
83 return llvm::sys::flattenWindowsCommandLine(args_ref);
84}

86llvm::ErrorOr<ProcThreadAttributeList>
87ProcThreadAttributeList::Create(STARTUPINFOEXW &startupinfoex) {
88 SIZE_T attributelist_size = 0;
89 InitializeProcThreadAttributeList(/*lpAttributeList=*/nullptr,
90 /*dwAttributeCount=*/1, /*dwFlags=*/0,
91 &attributelist_size);
92
93 startupinfoex.lpAttributeList =
94 static_cast<LPPROC_THREAD_ATTRIBUTE_LIST>(malloc(attributelist_size));
95
96 if (!startupinfoex.lpAttributeList)
97 return llvm::mapWindowsError(ERROR_OUTOFMEMORY);
98
99 if (!InitializeProcThreadAttributeList(startupinfoex.lpAttributeList,
100 /*dwAttributeCount=*/1,
101 /*dwFlags=*/0, &attributelist_size)) {
102 free(startupinfoex.lpAttributeList);
103 return llvm::mapWindowsError(GetLastError());
104 }
105
106 return ProcThreadAttributeList(startupinfoex.lpAttributeList);
107}

118HostProcess
119ProcessLauncherWindows::LaunchProcess(const ProcessLaunchInfo &launch_info,
120 Status &error) {
121 error.Clear();
122
123 STARTUPINFOEXW startupinfoex = {};
124 startupinfoex.StartupInfo.cb = sizeof(STARTUPINFOEXW);
125 startupinfoex.StartupInfo.dwFlags |= STARTF_USESTDHANDLES;
126
127 bool use_pty = launch_info.ShouldUsePTY();
128
129 HANDLE stdin_handle = GetStdioHandle(launch_info, STDIN_FILENO);
130 HANDLE stdout_handle = GetStdioHandle(launch_info, STDOUT_FILENO);
131 HANDLE stderr_handle = GetStdioHandle(launch_info, STDERR_FILENO);
132 llvm::scope_exit close_handles([&] {
133 if (stdin_handle)
134 ::CloseHandle(stdin_handle);
135 if (stdout_handle)
136 ::CloseHandle(stdout_handle);
137 if (stderr_handle)
138 ::CloseHandle(stderr_handle);
139 });
140
141 auto attributelist_or_err = ProcThreadAttributeList::Create(startupinfoex);
142 if (!attributelist_or_err) {
143 error = attributelist_or_err.getError();
144 return HostProcess();
145 }
146 ProcThreadAttributeList attributelist = std::move(*attributelist_or_err);
147
148 std::vector<HANDLE> inherited_handles;
149 if (use_pty) {
150 HPCON hPC = launch_info.GetPTY().GetPseudoTerminalHandle();
151 if (auto err = attributelist.SetupPseudoConsole(hPC)) {
152 error = Status::FromError(std::move(err));
153 return HostProcess();
154 }
155 } else {
156 auto inherited_handles_or_err = GetInheritedHandles(
157 launch_info, startupinfoex, stdout_handle, stderr_handle, stdin_handle);
158 if (!inherited_handles_or_err) {
159 error = Status(inherited_handles_or_err.getError());
160 return HostProcess();
161 }
162 inherited_handles = std::move(*inherited_handles_or_err);
163 }
164
165 const char *hide_console_var =
166 getenv("LLDB_LAUNCH_INFERIORS_WITHOUT_CONSOLE");
167 if (hide_console_var &&
168 llvm::StringRef(hide_console_var).equals_insensitive("true")) {
169 startupinfoex.StartupInfo.dwFlags |= STARTF_USESHOWWINDOW;
170 startupinfoex.StartupInfo.wShowWindow = SW_HIDE;
171 }
172
173 DWORD flags = CREATE_NEW_CONSOLE | CREATE_UNICODE_ENVIRONMENT |
174 EXTENDED_STARTUPINFO_PRESENT;
175 if (launch_info.GetFlags().Test(eLaunchFlagDebug))
176 flags |= DEBUG_ONLY_THIS_PROCESS;
177
178 if (launch_info.GetFlags().Test(eLaunchFlagDisableSTDIO) || use_pty)
179 flags &= ~CREATE_NEW_CONSOLE;
180
181 std::vector<wchar_t> environment =
182 CreateEnvironmentBufferW(launch_info.GetEnvironment());
183
184 auto wcommandLineOrErr =
185 GetFlattenedWindowsCommandStringW(launch_info.GetArguments());
186 if (!wcommandLineOrErr) {
187 error = Status(wcommandLineOrErr.getError());
188 return HostProcess();
189 }
190 std::wstring wcommandLine = *wcommandLineOrErr;
191 // If the command line is empty, it's best to pass a null pointer to tell
192 // CreateProcessW to use the executable name as the command line. If the
193 // command line is not empty, its contents may be modified by CreateProcessW.
194 WCHAR *pwcommandLine = wcommandLine.empty() ? nullptr : &wcommandLine[0];
195
196 std::wstring wexecutable, wworkingDirectory;
197 llvm::ConvertUTF8toWide(launch_info.GetExecutableFile().GetPath(),
198 wexecutable);
199 llvm::ConvertUTF8toWide(launch_info.GetWorkingDirectory().GetPath(),
200 wworkingDirectory);
201
202 PROCESS_INFORMATION pi = {};
203
204 BOOL result = ::CreateProcessW(
205 wexecutable.c_str(), pwcommandLine, NULL, NULL,
206 /*bInheritHandles=*/!inherited_handles.empty() || use_pty, flags,
207 environment.data(),
208 wworkingDirectory.size() == 0 ? NULL : wworkingDirectory.c_str(),
209 reinterpret_cast<STARTUPINFOW *>(&startupinfoex), &pi);
210
211 if (!result) {
212 // Call GetLastError before we make any other system calls.
213 error = Status(::GetLastError(), eErrorTypeWin32);
214 // Note that error 50 ("The request is not supported") will occur if you
215 // try debug a 64-bit inferior from a 32-bit LLDB.
216 }
217
218 if (result) {
219 // Do not call CloseHandle on pi.hProcess, since we want to pass that back
220 // through the HostProcess.
221 ::CloseHandle(pi.hThread);
222 }
223
224 if (!result)
225 return HostProcess();
226
227 return HostProcess(pi.hProcess);
228}

230llvm::ErrorOr<std::vector<HANDLE>> ProcessLauncherWindows::GetInheritedHandles(
231 const ProcessLaunchInfo &launch_info, STARTUPINFOEXW &startupinfoex,
232 HANDLE stdout_handle, HANDLE stderr_handle, HANDLE stdin_handle) {
233 std::vector<HANDLE> inherited_handles;
234
235 startupinfoex.StartupInfo.hStdError =
236 stderr_handle ? stderr_handle : GetStdHandle(STD_ERROR_HANDLE);
237 startupinfoex.StartupInfo.hStdInput =
238 stdin_handle ? stdin_handle : GetStdHandle(STD_INPUT_HANDLE);
239 startupinfoex.StartupInfo.hStdOutput =
240 stdout_handle ? stdout_handle : GetStdHandle(STD_OUTPUT_HANDLE);
241
242 if (startupinfoex.StartupInfo.hStdError)
243 inherited_handles.push_back(startupinfoex.StartupInfo.hStdError);
244 if (startupinfoex.StartupInfo.hStdInput)
245 inherited_handles.push_back(startupinfoex.StartupInfo.hStdInput);
246 if (startupinfoex.StartupInfo.hStdOutput)
247 inherited_handles.push_back(startupinfoex.StartupInfo.hStdOutput);
248
249 for (size_t i = 0; i < launch_info.GetNumFileActions(); ++i) {
250 const FileAction *act = launch_info.GetFileActionAtIndex(i);
251 if (act->GetAction() == FileAction::eFileActionDuplicate &&
252 act->GetFD() == act->GetActionArgument())
253 inherited_handles.push_back(reinterpret_cast<HANDLE>(act->GetFD()));
254 }
255
256 if (inherited_handles.empty())
257 return inherited_handles;
258
259 if (!UpdateProcThreadAttribute(
260 startupinfoex.lpAttributeList, /*dwFlags=*/0,
261 PROC_THREAD_ATTRIBUTE_HANDLE_LIST, inherited_handles.data(),
262 inherited_handles.size() * sizeof(HANDLE),
263 /*lpPreviousValue=*/nullptr, /*lpReturnSize=*/nullptr))
264 return llvm::mapWindowsError(::GetLastError());
265
266 return inherited_handles;
267}

269HANDLE
270ProcessLauncherWindows::GetStdioHandle(const ProcessLaunchInfo &launch_info,
271 int fd) {
272 const FileAction *action = launch_info.GetFileActionForFD(fd);
273 if (action == nullptr)
274 return NULL;
275 SECURITY_ATTRIBUTES secattr = {};
276 secattr.nLength = sizeof(SECURITY_ATTRIBUTES);
277 secattr.bInheritHandle = TRUE;
278
279 DWORD access = 0;
280 DWORD share = FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE;
281 DWORD create = 0;
282 DWORD flags = 0;
283 if (fd == STDIN_FILENO) {
284 access = GENERIC_READ;
285 create = OPEN_EXISTING;
286 flags = FILE_ATTRIBUTE_READONLY;
287 }
288 if (fd == STDOUT_FILENO || fd == STDERR_FILENO) {
289 access = GENERIC_WRITE;
290 create = CREATE_ALWAYS;
291 if (fd == STDERR_FILENO)
292 flags = FILE_FLAG_WRITE_THROUGH;
293 }
294
295 const std::string path = action->GetFileSpec().GetPath();
296 std::wstring wpath;
297 llvm::ConvertUTF8toWide(path, wpath);
298 HANDLE result = ::CreateFileW(wpath.c_str(), access, share, &secattr, create,
299 flags, NULL);
300 return (result == INVALID_HANDLE_VALUE) ? NULL : result;
301}
